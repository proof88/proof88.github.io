131 void PRREPosUpTarget::SetRotation(TPRREfloat x, TPRREfloat y, TPRREfloat z)
132 {
133  const PRREVector vDefTarget(0, 0, 1);
134  const PRREVector vDefPos(0, 0, 0);
135  // Note: we set absolute rotation here so our initial Target vec is the default in +Z direction and Pos vec is zero.
136  // To implement Rotate() instead of SetRotation(), use the current vTarget and vPos vectors to calculate tmpView.
137  PRREVector tmpView(vDefTarget - vDefPos);
138  tmpView.Normalize();
139 
140  PRRETransformMatrix matRotX, matRotY, matRotZ;
141  matRotX.SetRotationX(x);
142  matRotY.SetRotationY(y);
143  matRotZ.SetRotationZ(z);
144  // todo: 3 matrices, are u kidding me?
145 
146  tmpView *= matRotX * matRotY * matRotZ;
147  tmpView.Normalize();
148  vTarget = vPos + tmpView;
149 }
