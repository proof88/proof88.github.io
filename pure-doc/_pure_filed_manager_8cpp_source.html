260PureFiledManaged* PureFiledManager::PureFiledManagerImpl::getByFilename(const char* filename) const
261{
262 for (int i = 0; i < _pOwner->getCount(); i++)
263 {
264 PureFiledManaged* managed = (PureFiledManaged*) _pOwner->getAttachedAt(i);
265 if ( managed != PGENULL )
266 {
267 if ( strcmp(managed->getFilename().c_str(), filename) == 0 )
268 return managed;
269 }
270 }
271 return PGENULL;
272} // getByFilename()
