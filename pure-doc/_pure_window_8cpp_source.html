226LRESULT __stdcall PuredefaultWindowProc(HWND hWindow, UINT wndMsg, WPARAM wParam, LPARAM lParam)
227{
228 PureWindowImpl* const wnd = (PureWindowImpl*) GetWindowLongA(hWindow, GWL_USERDATA);
229
230 if ( wnd == NULL )
231 return DefWindowProc(hWindow, wndMsg, wParam, lParam);
232
233 if (ImGui_ImplWin32_WndProcHandler(hWindow, wndMsg, wParam, lParam))
234 return true;
235
236 switch ( wndMsg )
237 {
238 case WM_ACTIVATE : return wnd->onActivate((LOWORD(wParam) == WA_ACTIVE) || (LOWORD(wParam) == WA_CLICKACTIVE));
239 case WM_SYSCOMMAND : return wnd->onSysCommand(wParam, lParam);
240 case WM_CLOSE : return wnd->onClose();
241 case WM_DESTROY : return wnd->onDestroy();
242 case WM_INPUT : return wnd->onInput( (HRAWINPUT) lParam);
243 case WM_KEYDOWN : return wnd->onKeyDown(wParam);
244 case WM_KEYUP : return wnd->onKeyUp(wParam);
245 case WM_MOUSEMOVE : return wnd->onMouseMove(GET_X_LPARAM(lParam), GET_Y_LPARAM(lParam));
246 case WM_MOUSEWHEEL : return wnd->onMouseWheel( HIWORD(wParam) );
247 // Note: Quake3::win_wndproc.c states that Win32 may accumulate multiple events into wParam ... I havent experienced this yet though ...
248 case WM_LBUTTONDOWN: return wnd->onMouseDown(PGEInputMouse::MouseButton::MBTN_LEFT, wParam, GET_X_LPARAM(lParam), GET_Y_LPARAM(lParam));
249 case WM_LBUTTONUP : return wnd->onMouseUp(PGEInputMouse::MouseButton::MBTN_LEFT, wParam, GET_X_LPARAM(lParam), GET_Y_LPARAM(lParam));
250 case WM_MBUTTONDOWN: return wnd->onMouseDown(PGEInputMouse::MouseButton::MBTN_MIDDLE, wParam, GET_X_LPARAM(lParam), GET_Y_LPARAM(lParam));
251 case WM_MBUTTONUP : return wnd->onMouseUp(PGEInputMouse::MouseButton::MBTN_MIDDLE, wParam, GET_X_LPARAM(lParam), GET_Y_LPARAM(lParam));
252 case WM_RBUTTONDOWN: return wnd->onMouseDown(PGEInputMouse::MouseButton::MBTN_RIGHT, wParam, GET_X_LPARAM(lParam), GET_Y_LPARAM(lParam));
253 case WM_RBUTTONUP : return wnd->onMouseUp(PGEInputMouse::MouseButton::MBTN_RIGHT, wParam, GET_X_LPARAM(lParam), GET_Y_LPARAM(lParam));
254 case WM_REDRAW_REQUIRED: return wnd->onRedrawRequired();
255 case WM_WINDOWPOSCHANGED: return wnd->onPosChanged( (LPWINDOWPOS) lParam );
256 //case WM_G3T_PING: return wnd->onG3tCommand(wndMsg, wParam, lParam);
257 default: return DefWindowProc(hWindow, wndMsg, wParam, lParam);
258 } // switch()

312TPureBool PureWindowImpl::initialize(TPureUInt width, TPureUInt height, TPureBool bFullscreen, const char* caption, HWND window)
313{
314 if ( hWindow != NULL )
315 return false;
316
317 if ( width == 0 || height == 0 )
318 return false;
319
320 getConsole().OIOLn("PureWindow(%d, %d, %b, \"%s\", con) ...", width, height, bFullscreen, caption);
321 getConsole().OI();
322
323 SetPropertiesBeforeInitialize();
324 ResetDefaults();
325
326 if ( window == NULL )
327 {
328 if ( !setupWindowClass(clWindow) )
329 {
330 getConsole().OO();
331 getConsole().OO();
332 return false;
333 }
334
335 if ( !actualCreateWindow(width, height, bFullscreen, caption, clWindow) )
336 {
337 getConsole().OO();
338 getConsole().OO();
339 UnregisterClass(clWindow.lpszClassName, clWindow.hInstance);
340 SetPropertiesBeforeInitialize();
341 return false;
342 }
343 }
344 else
345 {
346 getConsole().OLn("Skipped creating new window as an already existing window has been specified!");
347 hWindow = window;
348 bExternalOwner = true;
349 }
350
351 GetClientRect(hWindow, &rectWindowClient);
352 getConsole().OLn("GetClientRect(): %d x %d",
353 rectWindowClient.right - rectWindowClient.left,
354 rectWindowClient.bottom - rectWindowClient.top);
355
356 dwStyle = GetWindowLong(hWindow, GWL_STYLE);
357 dwStyleEx = GetWindowLong(hWindow, GWL_EXSTYLE);
358
359 if ( !bExternalOwner )
360 {
361 SetWindowLong(hWindow, GWL_USERDATA, (LONG) this);
362 SetCaption(caption);
363 }
364
365 ProcessMessages(); // let the window keep up the pace
366
367 // triggering an onPosChanged() event to retrieve additional geometric details
368 tagWINDOWPOS tmpStuff;
369 tmpStuff.cx = rectWindowFull.right - rectWindowFull.left;
370 tmpStuff.cy = rectWindowFull.bottom - rectWindowFull.top;
371 tmpStuff.x = rectWindowFull.left;
372 tmpStuff.y = rectWindowFull.top;
373 onPosChanged(&tmpStuff);
374
375 getConsole().OOSOLn("> done!");
376 getConsole().OO();
377
378 return true;
379} // Initialize()

385void PureWindowImpl::Deinitialize()
386{
387 getConsole().OLn("PureWindow::Deinitialize()");
388 if ( hWindow == NULL )
389 return;
390
391 getConsole().OI();
392
393 if ( !bExternalOwner )
394 {
395 if ( DestroyWindow( hWindow ) )
396 getConsole().SOLn("DestroyWindow() OK!");
397 else
398 getConsole().EOLn("ERROR: DestroyWindow() failed!");
399
400 if ( UnregisterClass(clWindow.lpszClassName, clWindow.hInstance) )
401 getConsole().SOLn("UnregisterClass() OK!");
402 else
403 getConsole().EOLn("ERROR: UnregisterClass() failed!");
404 }
405 else
406 {
407 getConsole().OLn("Skipped destroying the window as it was already created for us at init!");
408 }
409
410 SetPropertiesBeforeInitialize();
411
412 getConsole().OO();
413} // Deinitialize()

494void PureWindowImpl::SetClientWidth(TPureUInt newwidth)
495{
496 if ( bExternalOwner )
497 return;
498
499 if ( !m_bFullscreen && (hWindow != NULL) )
500 {
501 rectWindowFull.right = rectWindowFull.left + newwidth;
502 const RECT old = rectWindowFull;
503 rectWindowFull.bottom = rectWindowFull.top + getClientHeight();
504 AdjustWindowRectEx(&rectWindowFull, dwStyle, false, dwStyleEx);
505 rectWindowFull.top = old.top;
506 rectWindowFull.bottom = old.bottom;
507 const LONG diff = old.left - rectWindowFull.left;
508 rectWindowFull.left = old.left;
509 rectWindowFull.right += diff;
510 SetWindowPos(hWindow, NULL, rectWindowFull.left, rectWindowFull.top,
511 rectWindowFull.right - rectWindowFull.left, rectWindowFull.bottom - rectWindowFull.top,
512 SWP_NOZORDER);
513 SendMessage(hWindow, WM_REDRAW_REQUIRED, 0, 0);
514 }

1529TPureBool PureWindowImpl::actualCreateWindow(TPureUInt width, TPureUInt height, TPureBool bFullscreen, const char* caption, const WNDCLASS& clWnd)
1530{

1532 m_bFullscreen = bFullscreen;
1533 bBtnMin = !bFullscreen;
1534 bBorder = bSysMenu = bVisibleCursor = enabled = true;
1535 /*
1536 "WS_CLIPCHILDREN affects the drawing of the parent window.
1537 More specifically, it prevents it from drawing in the areas occupied by its children.
1538 So if you're re-drawing the parent on a regular basis and WS_CLIPCHILDREN isn't set, you'll end up stomping on the child window's display..."
1539 https://stackoverflow.com/questions/2399216/why-might-ws-clipchildren-be-necessary-for-the-display-of-directshow-window-on-x
1540
1541 AFAIK WS_CLIPSIBLINGS is only for child windows, but specifying here doesn't cause any issues anyway so I'm using it.
1542
1543 Another topic is how to force fullscreen exclusive mode.
1544 But, I don't need that. The reason is described below.
1545 It cannot be explicitly set, and if you try to set it then it might or might not be applied.
1546 Based on the following article, must have things are: WS_POPUP + DPI Awareness.
1547 https://www.gamedeveloper.com/programming/fullscreen-exclusive-is-a-lie-sort-of-
1548 But still there is not 100% chance it will be exclusive.
1549
1550 Also, from Windows 10, it will be exclusive automatically based on the OS windowing system:
1551 "we enhanced the DWM to recognize when a game is running in a borderless full screen window with no other applications on the screen.
1552 In this circumstance, the DWM gives control of the display and almost all the CPU/GPU power to the game.
1553 Which in turn allows equivalent performance to running a game in FSE.
1554 Fullscreen Optimizations is essentially FSE with the flexibility to go back to DWM composition in a simple manner."
1555 source: https://devblogs.microsoft.com/directx/demystifying-full-screen-optimizations/
1556 */
1557 dwStyle = bFullscreen ? WS_POPUP | WS_CLIPCHILDREN | WS_CLIPSIBLINGS : WS_POPUPWINDOW | WS_CAPTION | WS_MINIMIZEBOX | WS_CLIPCHILDREN | WS_CLIPSIBLINGS;

1559 nClientWidthOrig = width;
1560 nClientHeightOrig = height;
1561 rectWindowFull.left = bFullscreen ? 0 : GetSystemMetrics(SM_CXSCREEN) / 2 - nClientWidthOrig / 2;
1562 rectWindowFull.top = bFullscreen ? 0 : GetSystemMetrics(SM_CYSCREEN) / 2 - nClientHeightOrig / 2;
1563 rectWindowFull.right = rectWindowFull.left + nClientWidthOrig;
1564 rectWindowFull.bottom = rectWindowFull.top + nClientHeightOrig;
1565 if ( !bFullscreen )
1566 AdjustWindowRectEx(&rectWindowFull, dwStyle, false, dwStyleEx);
1567
1568 hWindow = CreateWindowEx(dwStyleEx, clWnd.lpszClassName, caption, dwStyle,
1569 rectWindowFull.left, rectWindowFull.top,
1570 rectWindowFull.right - rectWindowFull.left, rectWindowFull.bottom - rectWindowFull.top,
1571 HWND_DESKTOP, NULL, clWnd.hInstance, NULL);
1572
1573 if ( hWindow == NULL )
1574 {
1575 getConsole().EOLn("CreateWindowEx() failed!");

1578 getConsole().SOLn("CreateWindowEx() ok!");
