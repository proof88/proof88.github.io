236 void PRRETransformMatrix::SetFrustum(float left, float right, float bottom, float top, float n, float f)
237 {
238  if ( (n <= 0.0f) || (n >= f) )
239  return;
240 
241  if ( (right <= left) || (top <= bottom) )
242  return;
243 
244  SetZero();
245 
246  /* 1st row */
247  mat[0][0] = 2*n / (right-left);
248  mat[0][2] = (right+left) / (right-left);
249 
250  /* 2nd row */
251  mat[1][1] = 2*n / (top-bottom);
252  mat[1][2] = (top+bottom) / (top-bottom);
253 
254  /* 3rd row */
255  mat[2][2] = (f+n) / (f-n);
256  mat[2][3] = -2*f*n / (f-n);
257 
258  /* 4th row */
259  mat[3][2] = 1.0f;
260  mat[3][3] = 0.0f; /* because SetZero had set it as 1 */
261 }

275 void PRRETransformMatrix::SetFrustumByFovX(float fovx, float aspect, float zNear, float zFar)
276 {
277  if ( (fovx <= 0.0f) || (aspect <= 0.0f) || (zNear <= 0.0f) || (zNear >= zFar) )
278  return;
279 
280  fovx = fovx * PFL::PI/180;

282  const float left = -(zNear * tan(fovx / 2)); /* left needs to be negative for sure */

284 
285  const float top = right / aspect;

328 void PRRETransformMatrix::SetLookAt(const PRREVector& vPos,
329  const PRREVector& vTarget,
330  const PRREVector& vUp)
331 {
332  /*
333 
334  My implementation doesn't equal to any of these:
335  https://www.scratchapixel.com/lessons/mathematics-physics-for-computer-graphics/lookat-function
336  https://www.khronos.org/opengl/wiki/GluLookAt_code
337  https://github.com/g-truc/glm/blob/master/glm/gtc/matrix_transform.inl#L756
338 
339  And above implementations differ from each other a bit. But still good material to read and learn.
340  */
341 
342  PRREVector vForwardFromTargetAndPos = vTarget - vPos;
343  vForwardFromTargetAndPos.Normalize();
344 
345  PRREVector vAltUp = vUp;
346 
347  // this is to detect when our calculated forward vector is parallel with Y-axis and that would make cross product zero right vector ...
348  // todo: switch to quaternions
349  if ( (abs(vForwardFromTargetAndPos.getX()) < PFL::E) && (abs(vForwardFromTargetAndPos.getZ()) < PFL::E) )
350  {
351  if ( vForwardFromTargetAndPos.getY() >= 0.0f )
352  vAltUp.Set(0.0f, 0.0f, 1.0f);
353  else
354  vAltUp.Set(0.0f, 0.0f, -1.0f);
355  }
356 
357  PRREVector vRightFromForwardAndUp = vAltUp ^ vForwardFromTargetAndPos;
358  vRightFromForwardAndUp.Normalize();
359 
360  PRREVector vUpFromRightAndForward = vForwardFromTargetAndPos ^ vRightFromForwardAndUp;
361  vUpFromRightAndForward.Normalize();
362 
363  mat[0][0] = vRightFromForwardAndUp.getX();
364  mat[1][0] = vRightFromForwardAndUp.getY();
365  mat[2][0] = vRightFromForwardAndUp.getZ();
366  mat[3][0] = 0.0f;
367 
368  mat[0][1] = vUpFromRightAndForward.getX();
369  mat[1][1] = vUpFromRightAndForward.getY();
370  mat[2][1] = vUpFromRightAndForward.getZ();
371  mat[3][1] = 0.0f;
372 
373  mat[0][2] = vForwardFromTargetAndPos.getX();
374  mat[1][2] = vForwardFromTargetAndPos.getY();
375  mat[2][2] = vForwardFromTargetAndPos.getZ();
376  mat[3][2] = 0.0f;
377 
378  /*
379  Normally we should use negative vPos because this is camera's coords so we should translate in reverse direction,
380  but we call invert() at the end anyway so let's stick to positive vPos now and let invert make the translation to reverse direction.
381  */
382  mat[0][3] = vPos.getX();
383  mat[1][3] = vPos.getY();
384  mat[2][3] = vPos.getZ();
385  mat[3][3] = 1.0f;
386 
387  /*
388  First I didn't understand why inverting the matrix was needed here. Originally I didn't even invert.
389  But it didn't work well ... position of camera was fine as -vPos but the angles given by Right, Forward and Up vectors calculated above were bad!
390  Then I realized during debugging they were giving inverse angles! When target was right to the camera, the camera was rotating to left side, etc.
391  So what I had to do was to invert the matrix. This gave good angles, I have verified matrix content during debugging.
392  But the position of the camera was bad this time... because I had set the last column of the matrix to -vPos because I knew that I needed
393  to transform the camera in negative direction. But since invert makes translation to inverse direction, no need to specify negative vPos anymore!
394  */
395  invert();
396 }
