267  void DiscoverTextureCompressionAvailability()
268  {
269  // bSuppTexCompression is already set by DiscoverOpenGL_X_Y_availability()
270  if ( bSuppTexCompression )
271  {
272  getConsole().SOLn("Texture compression supported!");
273  glGetIntegerv(GL_NUM_COMPRESSED_TEXTURE_FORMATS_ARB, tmpRet);
274  if ( !PRREGLsnippets::isGLerrorPresent() )
275  {
276  getConsole().OLn(" %d compressed texture formats available:", tmpRet[0]);
277  CheckAgainstStandardMinValue(tmpRet[0], GL_STANDARD_MIN_COMPRESSED_TEXTURE_FORMATS);
278  try
279  {
280  GLint* compressedFormats = new GLint[tmpRet[0]];
281  glGetIntegerv(GL_COMPRESSED_TEXTURE_FORMATS_ARB, compressedFormats);
282  if ( !PRREGLsnippets::isGLerrorPresent() )
283  {
284  getConsole().OI();
285  for (GLint i = 0; i < tmpRet[0]; i++)
286  if ( vTexComprFmts.find( compressedFormats[i] ) == vTexComprFmts.end() )
287  getConsole().OLn(" Unknown: %d", compressedFormats[i]);
288  else
289  getConsole().OLn(" %s", vTexComprFmts[ compressedFormats[i] ].c_str());
290  getConsole().OO();
291  }
292  else
293  {
294  getConsole().EOLn(" glGetIntegerv() failed: %s", PRREGLsnippets::getGLerrorTextFromEnum( PRREGLsnippets::getLastSavedGLerror() ));
295  }
296  delete[] compressedFormats;
297  }
298  catch (const std::bad_alloc&)
299  {
300  getConsole().EOLn(" Compressed texture formats available: ? (error: failed to allocate array!)");
301  }
302  }
303  else
304  getConsole().EOLn(" Compressed texture formats available: ? (error: %s)", PRREGLsnippets::getGLerrorTextFromEnum( PRREGLsnippets::getLastSavedGLerror() ));
305 
306  if ( bSuppVTC = isExtensionSupported("GL_NV_texture_compression_vtc") )
307  getConsole().SOLn(" VTC also supported.");
308  if ( bSupp3Dc = isExtensionSupported("GL_ATI_texture_compression_3dc") )
309  getConsole().SOLn(" 3Dc also supported.");
310  if ( bSuppLATC = isExtensionSupported("GL_EXT_texture_compression_latc") || isExtensionSupported("GL_NV_texture_compression_latc") )
311  getConsole().SOLn(" LATC also supported.");
312  }
313  else
314  getConsole().EOLn("Texture compression not supported.");
315  } // DiscoverTextureCompressionAvailability()
