252void PureRendererHWfixedPipeImpl::LastFrameStats::Update(const PureObject3D& obj, const TPURE_RENDER_PASS& iRenderPass)
253{
254 switch (iRenderPass)
255 {
256 case PURE_RPASS_NORMAL:
257 {
258 if ( obj.isRenderingAllowed() )
259 {
260 nLastFrameObjectsVisible++;
261 }
262
263 if ( obj.isOccluder() )
264 {
265 nLastFrameObjectsOccluders++;
266 }
267 else
268 {
269 if ( obj.isOcclusionTested() )
270 {
271 nLastFrameObjectsOcclusionTested++;
272 if ( obj.isOccluded() )
273 {
274 nLastFrameObjectsOccluded++;
275 }
276 else
277 {
278 nLastFrameObjectsNonOccluded++;
279 if ( !obj.isRenderingAllowed() )
280 {
281 nLastFrameObjectsNonOccludedButNotVisible++;
282 }
283 }
284 }
285 else
286 {
287 nLastFrameObjectsNonOcclusionTested++;
288 }
289 }
290 } // case PURE_RPASS_NORMAL
291 // fall-through
292 case PURE_RPASS_START_OCCLUSION_QUERY:
293 case PURE_RPASS_BOUNDING_BOX_DEBUG_FOR_OCCLUSION_QUERY:
294 case PURE_RPASS_Z_ONLY:
295 default:
296 nLastFrameVertices += obj.getLastTransferredVertexCount();
297 nLastFrameTriangles += obj.getLastTransferredTriangleCount();
298 } // switch
299} // Update()

356TPureUInt PureRendererHWfixedPipeImpl::initialize(
357 TPureUInt width, TPureUInt height,
358 TPURE_DISPLAY_MODES dmode, TPureUInt freq,
359 TPureInt cdepth, TPureInt zdepth, TPureInt stencil,
360 TPureInt samples,
361 HWND window)
362{
363 TPureInt st = stencil;
364
365 getConsole().OLn("PureRendererHWfixedPipe::initialize(%d, %d, dmode, %d, %d, %d, %d, %d)",
366 width, height, freq, cdepth, zdepth, st, samples);
367 if ( isInitialized() )
368 {
369 getConsole().SOLn(" > already initialized!");
370 return 1;
371 }
372
373 getConsole().OI();
374
375 // we accept only 0 or 8 for stencil bits
376 if ( st != 0 )
377 {
378 st = 8;
379 getConsole().OLn("WARNING: forcing non-0 stencil to %d !", st);
380 }
381
382 TPureBool bRequestAdvancedPixelFormat = false; // will be set to true in the loop AFTER 1st GL init if antialiasing is requested AND if advanced formats available
383 TPureBool bLoopNeeded = true; // will be set to false in the loop AFTER 1st GL init if antialiasing is NOT requested OR advanced formats are NOT available
384
385 /*
386 Initialize OpenGL once or twice.
387 2nd init is needed if advanced pixel format is needed, eg. antialiasing is requested. That's the reason of the loop here.
388 This comes from the nature of OpenGL: it must be initialized to query available advanced pixel formats, then it may be reinitialized again with an
389 advanced pixel format, thus a renderer must handle screen, window and hwinfo instances init during its init.
390 */
391 while ( bLoopNeeded )
392 {
393 getConsole().OLn("");
394 getConsole().OLn("#### STAGE 1/3 ####");
395
396 getConsole().OLn("screen settings ...");
397 screen.SetResolution(width, height);
398 screen.SetFullScreened( dmode != PURE_WINDOWED ); // == PURE_FULLSCREEN nem j, mert lesz 3. FS md is ...
399 screen.SetFreq( freq );
400 screen.SetColorBits( cdepth );
401 screen.SetDepthBits( zdepth );
402 screen.SetStencilBits( st );
403 screen.SetFSAAlevel( samples );
404
405 getConsole().OLn("new PureWindow...");
406 /*
407 A window may be created only once, even if advanced pixel format is requested, since a DC is required to be created only once.
408 This would mean that window instance can be created before renderer initialization. But based on SetPixelFormat() documentation, a
409 pixel format shouldn't be set to a window multiple times. This means that we actually have to recreate a window if an advanced pixel
410 format is to be set!
411 */
412 if ( !wnd.initialize(screen.getResWidth(), screen.getResHeight(), dmode != PURE_WINDOWED, "Window", window) )
413 {
414 getConsole().EOLnOO("ERROR: wnd.initialize() failed!");
415 shutdown();
416 return 1;
417 }
418 getConsole().SOLn("done!");
419
420 if ( !screen.applyDisplaySettings(wnd.getWndDC(), bRequestAdvancedPixelFormat ? PURE_SCREEN_PF_ADVANCED : PURE_SCREEN_PF_SIMPLE) )
421 {
422 getConsole().EOLnOO("ERROR: applyDisplaySettings() failed!");
423 shutdown();
424 return 1;
425 }
426
427 getConsole().OLn("");
428 getConsole().OLn("#### STAGE 2/3 ####");
429 if ( !initializeOpenGL(wnd.getWndDC()) )
430 {
431 getConsole().OO();
432 shutdown();
433 return 1;
434 }
435
436 hwInfo.Initialize(
437 rc, wnd.getWndDC(),
438 screen.getResWidth(), screen.getResHeight(),
439 screen.getColorBits(), screen.getDepthBits(),
440 screen.getStencilBits(), screen.getFSAAlevel() );
441
442 if ( hwInfo.getVideo().isAdvancedPixelFormatSetSupported() )
443 {
444 if ( (samples > 0) && !bRequestAdvancedPixelFormat )
445 {
446 getConsole().OLn("Advanced pixel format requested, trying ...");
447 // following call can be removed after screen.applyDisplaySettings() is modified to automatically
448 // query advanced pixel formats at first call ... see PGE-History.txt's relevant RFR task!
449 screen.applyDisplaySettings( wnd.getWndDC() );
450 bRequestAdvancedPixelFormat = true;
451 shutdown();
452 }
453 else
454 {
455 bLoopNeeded = false;
456 getConsole().OLn("Could set advanced pixel format but not requested or already set.");
457 }
458 }
459 else
460 {
461 getConsole().OLn("Advanced pixel formats not available.");
462 bLoopNeeded = false;
463 }
464 } // while bLoopNeeded
465
466 getConsole().OLn("");
467 getConsole().OLn("#### STAGE 3/3 ####");
468 getConsole().OLn("");
469
470 printOGLerrorFull();
471 SetBasicThingsInOpenGL();
472 printOGLerrorFull();
473 PureGLsnippets::Init();
474
475 timeDurationStart.tv_sec = 0;
476 timeDurationStart.tv_usec = 0;
477
478 if (m_cfgProfiles.getVars()[PureScreen::CVAR_GFX_VSYNC].getAsString().empty())
479 {
480 // GFX card drivers' default setting in 2015: off (if undefined by application), so we also set it to false initially
481 getConsole().OLn("V-Sync default: %b", false);
482 screen.setVSyncEnabled(false);
483 m_cfgProfiles.getVars()[PureScreen::CVAR_GFX_VSYNC].Set(false);
484 }
485 else
486 {
487 const bool bVSyncConfig = m_cfgProfiles.getVars()[PureScreen::CVAR_GFX_VSYNC].getAsBool();
488 getConsole().O("Trying V-Sync from config: %b ... ", bVSyncConfig);
489 const bool bVSyncSetRet = screen.setVSyncEnabled(bVSyncConfig);
490 m_cfgProfiles.getVars()[PureScreen::CVAR_GFX_VSYNC].Set(bVSyncSetRet);
491 if (bVSyncConfig == bVSyncSetRet)
492 {
493 getConsole().SOLn("OK!");
494 }
495 else
496 {
497 getConsole().EOLn("FAILED!");
498 }
499 }
500
501 getConsole().OLn("");
502 getConsole().SOLn("> Fixed Pipeline HW Renderer initialized");
503 getConsole().OLnOO("");
504
505 return 0;
506
507} // initialize()

566void PureRendererHWfixedPipeImpl::RenderScene()
567{
568 if ( !isInitialized() )
569 {
570 return;
571 }
572
573 PFL::timeval timeFrameStart, timeFrameEnd;
574 PFL::gettimeofday(&timeFrameStart, 0);
575
576 if ( (timeDurationStart.tv_sec == 0) && (timeDurationStart.tv_usec == 0) )
577 { // mark beginning of new stats duration
578 PFL::gettimeofday(&timeDurationStart, 0);
579 }
580
581 lastFrameStats.Reset();
582
583 BeginRendering();
584
585 SwitchToPerspectiveProjection();
586 const TPureUInt iRenderPath = BITF_READ(stats[stats.size()-1].renderHints, PURE_RH_RENDER_PATH_BITS, 3);
587 switch (iRenderPath)
588 {
589 case PURE_RH_RP_LEGACY_PR00FPS:
590 Draw3DObjects_Legacy(*this);
591 break;
592 case PURE_RH_RP_DISTANCE_ORDERED:
593 Draw3DObjects_Ordered_Containers(*this);
594 break;
595 case PURE_RH_RP_OCCLUSION_CULLED:
596 {
597 const TPureUInt iOcclusionQueryMethod = BITF_READ(stats[stats.size()-1].renderHints, PURE_RH_OQ_METHOD_BITS, 2);
598 switch (iOcclusionQueryMethod)
599 {
600 case PURE_RH_OQ_METHOD_SYNC:
601 Draw3DObjects_OcclusionQuery(*this, false);
602 break;
603 case PURE_RH_OQ_METHOD_ASYNC:
604 Draw3DObjects_OcclusionQuery(*this, true);
605 break;
606 default:
607 // later more OQ methods will come ...
608 getConsole().EOLn("%s() ERROR: unhandled iOcclusionQueryMethod!", __FUNCTION__);
609 assert(false);
610 } // switch iOcclusionQueryMethod
611 }
612 break;
613 default:
614 // later more renderpaths will come ...
615 getConsole().EOLn("%s() ERROR: unhandled iRenderPath!", __FUNCTION__);
616 assert(false);
617 } // switch iRenderPath
618
619 SwitchToOrtographicProjection();
620 Draw2DObjects(*this);
621
622 FinishRendering();
623
624 // we don't really check for GL errors in above functions, so check them here
625 printOGLerrorBrief();
626
627 stats[stats.size()-1].nRenderedFrames++;
628 PFL::gettimeofday(&timeFrameEnd, 0);
629 PFL::updateForMinDuration(stats[stats.size()-1].timeMinFrameTime, timeFrameStart, timeFrameEnd);
630 PFL::updateForMaxDuration(stats[stats.size()-1].timeMaxFrameTime, timeFrameStart, timeFrameEnd);
631
632} // RenderScene()

669 }

674 getConsole().OLn(" bNewOcclusionDrawBoundingBoxes : %b", bNewOcclusionDrawBoundingBoxes);
675 }

1223struct CompareByZdistance
1224{
1225 enum Order
1226 {
1227 NEAREST_TO_FAREST, /* front-to-back */
1228 FAREST_TO_NEAREST /* back-to-front */
1229 };
1230
1231 PureVector vPosCam;
1232 PureVector vFwdCam;
1233 Order order;
1234 bool log;
1235
1236 // std::sort() uses this to determine if a is less than b
1237 bool operator()(PureObject3D* a, PureObject3D* b) const
1238 {
1239 const PureVector vDistanceToCam_a = a->getPosVec() - vPosCam;
1240 const PureVector vDistanceToCam_b = b->getPosVec() - vPosCam;
1241 const TPureFloat fZdistanceToCam_a = vDistanceToCam_a.getDotProduct(vFwdCam);
1242 const TPureFloat fZdistanceToCam_b = vDistanceToCam_b.getDotProduct(vFwdCam);
1243 if ( log )
1244 {
1245 CConsole::getConsoleInstance().OLn("Comparator(%s, %s)", a->getName().c_str(), b->getName().c_str());
1246 CConsole::getConsoleInstance().OLn(" pos a: [%f, %f, %f], pos b: [%f, %f, %f], zdist a: %f, zdist b: %f",
1247 a->getPosVec().getX(), a->getPosVec().getY(), a->getPosVec().getZ(),
1248 b->getPosVec().getX(), b->getPosVec().getY(), b->getPosVec().getZ(),
1249 fZdistanceToCam_a, fZdistanceToCam_b);
1250 }
1251 // std::sort() sorts in non-descending order, if comparator also returns true if element 'a' is less than element 'b',
1252 // so in our case it is nearest to farest ordering
1253 return (order == NEAREST_TO_FAREST) ? (fZdistanceToCam_a < fZdistanceToCam_b) : (fZdistanceToCam_a > fZdistanceToCam_b);
1254 }
1255}; // struct CompareByZdistance
