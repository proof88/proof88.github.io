256TPureBool PureTexture::PureTextureImpl::setAnisoFilteringMode(TPURE_ANISO_TEX_FILTERING filtering)
257{
258 _pOwner->getManagedConsole().OIOLn("PureTexture::setAnisoFilteringMode(%fX)", PureTextureManager::getFloatFromPureanisoTexFilteringName(filtering));
259
260 if ( _pOwner->getManager() == PGENULL )
261 return false;
262
263 if ( !PureHwInfo::get().getVideo().isAcceleratorDetected() )
264 {
265 _pOwner->getManagedConsole().EOLnOO("WARNING: anisotropic filtering is not supported without acceleration!");
266 return false;
267 }
268
269 bool result = false;
270 const TPureFloat newFilteringF = ((PureTextureManager*)_pOwner->getManager())->getFloatFromPureanisoTexFilteringName(filtering);
271 const TPureFloat availMaxFilteringF = ((PureTextureManager*)_pOwner->getManager())->getFloatFromPureanisoTexFilteringName(
272 ((PureTextureManager*)_pOwner->getManager())->getMaximumAnisoFiltering() );
273 // in Release mode, although getMaximumAnisoFiltering() returns ANISO_16X, availMaxFilteringF becomes 0.
274 // This can be fixed by adding a console printout to getFloatFromPureanisoTexFilteringName(), so probably this is some
275 // compiler optimization bug which doesnt happen in Debug mode. Would be good to play with the Release mode compiler options
276 // to see exactly which option causes this. This is weird because getFloatFromPureanisoTexFilteringName() can never ever be 0.
277 // _pOwner->getManagedConsole().OLn( "availMaxFilteringF: %f", availMaxFilteringF);
278 if ( availMaxFilteringF > 1.0f )
279 {
280 if ( (nInternalNum == 0) || (!pglBindTexture(GL_TEXTURE_2D, nInternalNum)) )
281 {
282 _pOwner->getManagedConsole().EOLnOO("ERROR: nInternalNum is 0 or pglBindTexture() failed!");
283 return false;
284 }
285 if ( availMaxFilteringF >= newFilteringF )
286 {
287 if ( pglTexParameterf(GL_TEXTURE_2D, GL_TEXTURE_MAX_ANISOTROPY_EXT, (GLfloat) newFilteringF) )
288 {
289 filtAniso = filtering;
290 result = true;
291 }
292 }
293 else
294 {
295 _pOwner->getManagedConsole().EOLn("WARNING: selected mode not supported, setting max supported: %fX!", availMaxFilteringF);
296 if ( pglTexParameterf(GL_TEXTURE_2D, GL_TEXTURE_MAX_ANISOTROPY_EXT, (GLfloat) availMaxFilteringF) )
297 {
298 filtAniso = PureTextureManager::getPureanisoTexFilteringNameFromFloat(availMaxFilteringF);
299 result = true;
300 }
301 }
302 }
303 else
304 {
305 _pOwner->getManagedConsole().EOLn("ERROR: anisotropic filtering is not supported");
306 }
307 _pOwner->getManagedConsole().OO();
308 return result;
309} // setAnisoFilteringMode()

387TPureBool PureTexture::PureTextureImpl::uploadPixels()
388{
389 _pOwner->getManagedConsole().OIOLn("PureTexture::uploadPixels()");
390
391 if ( _pOwner->getPixels() == PGENULL )
392 {
393 _pOwner->getManagedConsole().EOLnOO("ERROR: getPixels() returned NULL!");
394 return false;
395 }
396
397 if ( _pOwner->getManager() == PGENULL )
398 {
399 _pOwner->getManagedConsole().EOLnOO("ERROR: getManager() returned NULL!");
400 return false;
401 }
402
403 if ( strcmp(typeid(*_pOwner->getManager()).raw_name(), typeid(PureTextureManager&).raw_name()) )
404 {
405 _pOwner->getManagedConsole().EOLnOO("ERROR: getManager() type is not PureTextureManager!");
406 return false;
407 }
408
409 if ( PureHwInfo::get().getVideo().isAcceleratorDetected() )
410 {
411 if (!pglBindTexture(GL_TEXTURE_2D, nInternalNum))
412 {
413 return false;
414 }
415 }
416
417 const TPURE_PIXEL_COMPONENT_ORDER oldco = _pOwner->getPixelComponentOrder();
418 const GLenum glTexFormat = getTransformedSourceFormat(oldco); // might invoke setPixelComponentOrder()!
419 if ( glTexFormat == GL_INVALID_ENUM )
420 {
421 _pOwner->getManagedConsole().EOLnOO("ERROR: getTransformedSourceFormat() failed!");
422 // try reset original pixel component order
423 if ( !_pOwner->setPixelComponentOrder( oldco ) )
424 {
425 _pOwner->getManagedConsole().EOLnOO("ERROR: recovery setPixelComponentOrder(oldco) failed!");
426 }
427 return false;
428 }
429
430 const GLint internalfmt = getTargetInternalFormat();
431
432 if ( !actualUploadProc(internalfmt, glTexFormat) )
433 {
434 _pOwner->getManagedConsole().EOLnOO("ERROR: actualUploadProc() failed!");
435 return false;
436 }
437
438 DescribeTexFormatAndSize(internalfmt);
439
440 if ( _pOwner->getPixelComponentOrder() != oldco )
441 {
442 if ( !_pOwner->setPixelComponentOrder( oldco ) )
443 {
444 _pOwner->getManagedConsole().EOLnOO("ERROR: finished creating new texture, but setPixelComponentOrder(oldco) failed!");
445 // intentionally let function still return true
446 }
447 }
448
449 _pOwner->getManagedConsole().OO();
450
451 return true;
452} // uploadPixels()

521GLenum PureTexture::PureTextureImpl::getTransformedSourceFormat(TPURE_PIXEL_COMPONENT_ORDER oldco)
522{
523 const PureTextureManager& parent = *((PureTextureManager*) _pOwner->getManager());
524 const PureHwInfo& hwInfo = PureHwInfo::get();
525
526 if ( oldco == PURE_BGR )
527 {
528 if ( hwInfo.getVideo().isNativeDIBFormatSupported() && parent.isNativeDIBFormatSupportEnabled() )
529 return GL_BGR_EXT;
530 else
531 {
532 if ( !_pOwner->setPixelComponentOrder(PURE_RGB) )
533 return GL_INVALID_ENUM;
534 return GL_RGB;
535 }
536 }
537 else if ( oldco == PURE_BGRA )
538 {
539 if ( hwInfo.getVideo().isNativeDIBFormatSupported() && parent.isNativeDIBFormatSupportEnabled() )
540 return GL_BGRA_EXT;
541 else
542 {
543 if ( !_pOwner->setPixelComponentOrder(PURE_RGBA) )
544 return GL_INVALID_ENUM;
545 return GL_RGBA;
546 }
547 }
548 else if ( _pOwner->getPixelComponentOrder() % 2 == 0 )
549 {
550 if ( oldco != PURE_RGB )
551 {
552 if ( !_pOwner->setPixelComponentOrder(PURE_RGB) )
553 return GL_INVALID_ENUM;
554 }
555 return GL_RGB;
556 }
557 else
558 {
559 if ( oldco != PURE_RGBA )
560 {
561 if ( !_pOwner->setPixelComponentOrder(PURE_RGBA) )
562 return GL_INVALID_ENUM;
563 }
564 return GL_RGBA;
565 }
566} // getTransformedSourceFormat()

575GLint PureTexture::PureTextureImpl::getTargetInternalFormat()
576{
577 const PureTextureManager& parent = *((PureTextureManager*) _pOwner->getManager());
578 const PureHwInfo& hwInfo = PureHwInfo::get();
579
580 if ( hwInfo.getVideo().isTextureCompressionSupported() )
581 {
582 switch ( parent.getDefaultCompressionMode() )
583 {
584 case PURE_TC_NONE: return _pOwner->getPixelComponentOrder() % 2 == 0 ? 3 : 4;
585 case PURE_TC_S3TC_RGB_DXT1 : return GL_COMPRESSED_RGB_S3TC_DXT1_EXT;
586 case PURE_TC_S3TC_RGBA_DXT1: return GL_COMPRESSED_RGBA_S3TC_DXT1_EXT;
587 case PURE_TC_S3TC_RGBA_DXT3: return GL_COMPRESSED_RGBA_S3TC_DXT3_EXT;
588 case PURE_TC_S3TC_RGBA_DXT5: return GL_COMPRESSED_RGBA_S3TC_DXT5_EXT;
589 case PURE_TC_3DFX_RGB_FXT1 : return GL_COMPRESSED_RGB_FXT1_3DFX;
590 case PURE_TC_3DFX_RGBA_FXT1: return GL_COMPRESSED_RGBA_FXT1_3DFX;
591 /* PURE_TC_AUTO causes the driver to select any of the above */
592 default: /* for PURE_TC_AUTO */ return _pOwner->getPixelComponentOrder() % 2 == 0 ? GL_COMPRESSED_RGB_ARB : GL_COMPRESSED_RGBA_ARB;
593 }
594 }
595 else
596 return _pOwner->getPixelComponentOrder() % 2 == 0 ? 3 : 4;
597} // getTargetInternalFormat()

612TPureBool PureTexture::PureTextureImpl::actualUploadProc(GLint internalfmt, GLenum glTexFormat)
613{
614 const PureTextureManager& parent = *((PureTextureManager*) _pOwner->getManager());
615 const PureHwInfo& hwInfo = PureHwInfo::get();
616
617 if ( !hwInfo.getVideo().isAcceleratorDetected() )
618 {
619 _pOwner->getManagedConsole().OLn("no need to upload texture, non-accelerated rendering is active");
620 return true;
621 }
622
623 if ( nMIPmapCount > 1 )
624 {
625 _pOwner->getManagedConsole().OLn("mipmapping on");
626 if ( hwInfo.getVideo().isHardwareMipMapGenerationSupported() && parent.isHardwareMipMapGenerationEnabled() )
627 {
628 _pOwner->getManagedConsole().OLn("auto-generating mipmaps with HW-support.");
629 if (!pglTexParameteri(GL_TEXTURE_2D, GL_GENERATE_MIPMAP_SGIS, GL_TRUE) )
630 {
631 return false;
632 }
633
634 // WARNING: up to OpenGL 2.x, border is allowed to be 0 or 1. However from GL 3.0 it must be 0.
635 // WARNING: if GL version does not support NPOT sizes, width must be 2n+2(border) for some integer n.
636 // WARNING: if GL version does not support NPOT sizes, height must be 2m+2(border) for some integer m.
637 // TODO: So in the future when I add support for NPOT textures, I need to consider border value when specifying width and height.
638 return pglTexImage2D(GL_TEXTURE_2D, 0, internalfmt, _pOwner->getWidth(), _pOwner->getHeight(), bBorder ? (GLint) 1 : (GLint) 0, glTexFormat, GL_UNSIGNED_BYTE, _pOwner->getPixels());
639 }
640 else
641 {
642 _pOwner->getManagedConsole().OLn("auto-generating mipmaps with SW-only.");
643 const GLint errCode = gluBuild2DMipmaps(GL_TEXTURE_2D, internalfmt, _pOwner->getWidth(), _pOwner->getHeight(), glTexFormat, GL_UNSIGNED_BYTE, _pOwner->getPixels());
644 if ( errCode != GL_NO_ERROR )
645 {
646 _pOwner->getManagedConsole().EOLn("PureTextureImpl::actualUploadProc() ERROR: %d, %s", errCode, gluErrorString(errCode));
647 return false;
648 }
649 return true;
650 }
651 }
652 else
653 {
654 _pOwner->getManagedConsole().OLn("mipmapping off");
655
656 // WARNING: up to OpenGL 2.x, border is allowed to be 0 or 1. However from GL 3.0 it must be 0.
657 // WARNING: if GL version does not support NPOT sizes, width must be 2n+2(border) for some integer n.
658 // WARNING: if GL version does not support NPOT sizes, height must be 2m+2(border) for some integer m.
659 // TODO: So in the future when I add support for NPOT textures, I need to consider border value when specifying width and height.
660 return pglTexImage2D(GL_TEXTURE_2D, 0, internalfmt, _pOwner->getWidth(), _pOwner->getHeight(), bBorder ? (GLint) 1 : (GLint) 0, glTexFormat, GL_UNSIGNED_BYTE, _pOwner->getPixels());
661 }
662} // actualUploadProc()

670void PureTexture::PureTextureImpl::DescribeTexFormatAndSize(GLint internalfmt)
671{
672 nSize = _pOwner->getWidth() * _pOwner->getHeight() * 4;
673 texFormat = PURE_TF_RGBA;
674
675 if ( !PureHwInfo::get().getVideo().isAcceleratorDetected() )
676 {
677 return;
678 }
679
680 if ( (internalfmt != 3) && (internalfmt != 4) )
681 {
682 GLint tmpret = 0;
683 glGetTexLevelParameteriv(GL_TEXTURE_2D, 0, GL_TEXTURE_COMPRESSED, &tmpret);
684 if ( tmpret == GL_TRUE )
685 {
686 glGetTexLevelParameteriv(GL_TEXTURE_2D, 0, GL_TEXTURE_INTERNAL_FORMAT, &tmpret);
687 switch ( tmpret )
688 {
689 case GL_RGBA:
690 _pOwner->getManagedConsole().OLn("final internal texFormat is GL_RGBA");
691 break;
692 case GL_COMPRESSED_RGB_S3TC_DXT1_EXT:
693 texFormat = PURE_TF_DXT1;
694 _pOwner->getManagedConsole().OLn("final internal texFormat is GL_COMPRESSED_RGB_S3TC_DXT1_EXT");
695 break;
696 case GL_COMPRESSED_RGBA_S3TC_DXT1_EXT:
697 texFormat = PURE_TF_DXT1A;
698 _pOwner->getManagedConsole().OLn("final internal texFormat is GL_COMPRESSED_RGBA_S3TC_DXT1_EXT");
699 break;
700 case GL_COMPRESSED_RGBA_S3TC_DXT3_EXT:
701 texFormat = PURE_TF_DXT3A;
702 _pOwner->getManagedConsole().OLn("final internal texFormat is GL_COMPRESSED_RGBA_S3TC_DXT3_EXT");
703 break;
704 case GL_COMPRESSED_RGBA_S3TC_DXT5_EXT:
705 texFormat = PURE_TF_DXT5A;
706 _pOwner->getManagedConsole().OLn("final internal texFormat is GL_COMPRESSED_RGBA_S3TC_DXT5_EXT");
707 break;
708 case GL_COMPRESSED_RGB_FXT1_3DFX:
709 texFormat = PURE_TF_FXT1;
710 _pOwner->getManagedConsole().OLn("final internal texFormat is GL_COMPRESSED_RGB_FXT1_3DFX");
711 break;
712 case GL_COMPRESSED_RGBA_FXT1_3DFX:
713 texFormat = PURE_TF_FXT1A;
714 _pOwner->getManagedConsole().OLn("final internal texFormat is GL_COMPRESSED_RGBA_FXT1_3DFX");
715 break;
716 /* In 14.12 AMD Catalyst Omega Software, requesting auto-compression at glTexImage2D() returns general compression here instead of any if the specific values above. */
717 case GL_COMPRESSED_RGB_ARB:
718 texFormat = PURE_TF_UNSURE;
719 _pOwner->getManagedConsole().OLn("final internal texFormat is GL_COMPRESSED_RGB_ARB");
720 break;
721 case GL_COMPRESSED_RGBA_ARB:
722 texFormat = PURE_TF_UNSURE;
723 _pOwner->getManagedConsole().OLn("final internal texFormat is GL_COMPRESSED_RGBA_ARB");
724 break;
725 default:
726 _pOwner->getManagedConsole().EOLn("final internal texFormat ERROR: glGetTexLevelParameteriv() returned %d for GL_TEXTURE_INTERNAL_FORMAT", tmpret);
727 break;
728 }
729 glGetTexLevelParameteriv(GL_TEXTURE_2D, 0, GL_TEXTURE_COMPRESSED_IMAGE_SIZE, &tmpret);
730 nSize = tmpret;
731 }
732 }
733 if ( nMIPmapCount > 1 )
734 {
735 nSize = PureTextureManager::getMIPmappedTexSize(nSize, _pOwner->getWidth(), _pOwner->getHeight());
736 }
737} // DescribeTexFormatAndSize()
