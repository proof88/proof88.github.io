506 void PRRERendererSWincrementalImpl::VertexProcessing(
507  const TXYZ* pVertices,
508  TPRREuint nVertices_h,
509  bool bIndexed,
510  const void* pVertexIndices,
511  TPRREuint nVertexIndices_h,
512  unsigned int nIndicesType,
513  TPRRE_TRANSFORMED_VERTEX* pVerticesTransf,
514  const TXYZ& pos,
515  const TXYZ& rot,
516  const TXYZ& scaling,
517  const TXYZ& campos,
518  const TXYZ& camrot
519 )
520 {
521  if ( pVertices == NULL || nVertices_h == 0 )
522  return;
523 
524  if ( bIndexed )
525  {
526  if ( pVertexIndices == NULL || nVertexIndices_h == 0 )
527  return;
528  }
529 
530  if ( pVerticesTransf == NULL )
531  return;
532 
533  /* we need completely defined triangles */
534  if ( nVertexIndices_h % 3 != 0 )
535  return;
536 
537  for (TPRREuint i = 0; i < nVertexIndices_h; i++)
538  {
539  const TXYZ& vertex = bIndexed ? pVertices[ PRREGLsnippets::getVertexIndex(pVertexIndices, i, nIndicesType) ] : pVertices[i];
540  PRREVector transfdVertex(vertex.x, vertex.y, vertex.z);
541 
542  /* Model -> World */
543  PRRETransformMatrix mModelTr;
544  mModelTr.SetTranslation(pos.x, pos.y, pos.z);
545 
546  PRRETransformMatrix mScalingTr;
547  mScalingTr.SetScale(scaling.x, scaling.y, scaling.z);
548  transfdVertex = mScalingTr * transfdVertex;
549  // Note that we need to detect negative scaling factor and change front face winding order on-the-fly, see details in PRREObject3D::PRREObject3DImpl::Draw_ApplyTransformations()
550 
551  PRRETransformMatrix mRotation;
552  // TODO: we should set rotation on all axes with 1 matrix but unfortunately currently we just have separate setrotation methods for each axes
553  // so we need 3 different rotation matrices for each axis, and this also introduces the rotation order problem.
554 
555 
556  mRotation.SetRotationX(rot.x);
557  transfdVertex = mRotation * transfdVertex;
558  mRotation.SetRotationY(abs(rot.y-360)); // TODO: in proofps, weapon is rotating in other direction ... probably my SetRotationY() is bad from LH/RH point of view ???
559  transfdVertex = mRotation * transfdVertex;
560  mRotation.SetRotationZ(rot.z);
561  transfdVertex = mRotation * transfdVertex;
562 
563  transfdVertex = mModelTr * transfdVertex;
564 
565  /* World -> View/Eye */
566  PRRETransformMatrix mViewTr;
567  mViewTr.SetLookAt(pCamera->getPosVec(), pCamera->getTargetVec(), pCamera->getUpVec());
568  transfdVertex = mViewTr * transfdVertex;
569 
570  /* View/Eye -> Clip */
571  /* The pyramidal frustum in eye space becomes a cube, therefore, the clipping plane equations in clip space are: X + Wc = 0, Y + Wc = 0, Z + Wc = 0, etc. */
572  PRRETransformMatrix mProjTr;
573  mProjTr.SetFrustumByFovY(pCamera->getFieldOfView(), pCamera->getAspectRatio(), pCamera->getNearPlane(), pCamera->getFarPlane());
574  transfdVertex = mProjTr * transfdVertex;
575 
576  pVerticesTransf[i].coords = transfdVertex.getXYZW();
577  }
578 }

587 void PRRERendererSWincrementalImpl::PrimitiveAssembly(
588  TPRRE_TRANSFORMED_VERTEX* pVerticesTransf,
589  TPRREuint nVertexIndices_h,
590  const TXYZ& pos,
591  const TXYZ& rot,
592  const TXYZ& campos,
593  const TXYZ& camrot,
594  TPRREint vp_x,
595  TPRREint vp_y,
596  TPRREuint vp_w,
597  TPRREuint vp_h
598 )
599 {
600  if ( pVerticesTransf == NULL || nVertexIndices_h == 0 )
601  return;
602 
603  /* we need completely defined triangles */
604  if ( nVertexIndices_h % 3 != 0 )
605  return;
606 
607  /* We are in clip space now, meaning that clip planes are now parallel with axes, so frustum culling / clipping is very easy to do. */
608  /* The clip coordinates, Xc, Yc and Zc are tested by comparing with Wc. If any clip coordinate is less than -Wc, or greater than Wc, then the vertex is outside of the view frustum. */
609  /* Proposed WoW: for each triangle, check the location of all 3 vertices to the view frustum:
610  - if all vertices are inside, we are fine;
611  - if all vertices are outside, throw away the triangle somehow at this stage to decrease the workload on Render Stage;
612  - if the triangle is partially inside and outside the frustum, it needs to be clipped, that will be taken care in Render Stage with some 2D-scissoring. */
613  /* Current WoW: regardless of to which triangle a vertex is related, we mark each vertex if it is inside the view frustum or not (draw bool variable).
614  Then Render Stage checks if all vertices of a triangle is outside: if so, triangle will be dropped there. */
615  // TODO: Note that clipping could be also done in 3D-space now, maybe sometime in future I implement it.
616  // It would be nicer because I think responsibility of Primitive Assembly is to output proper triangles for the rasterizer and if needed, insert extra triangles. */
617 
618  for (TPRREuint i = 0; i < (nVertexIndices_h / 3); i++)
619  {
620  for (TPRREuint j = 0; j < 3; j++)
621  {
622  const TPRREuint vi = i*3 + j;
623  if ( (pVerticesTransf[vi].coords.x >= -pVerticesTransf[vi].coords.w) && (pVerticesTransf[vi].coords.x <= pVerticesTransf[vi].coords.w)
624  &&
625  (pVerticesTransf[vi].coords.y >= -pVerticesTransf[vi].coords.w) && (pVerticesTransf[vi].coords.y <= pVerticesTransf[vi].coords.w)
626  &&
627  (pVerticesTransf[vi].coords.z >= -pVerticesTransf[vi].coords.w) && (pVerticesTransf[vi].coords.z <= pVerticesTransf[vi].coords.w) )
628  {
629  /* triangle is partially inside the view frustum, do not throw it away */
630  pVerticesTransf[vi].draw = true;
631  }
632  else
633  {
634  pVerticesTransf[vi].draw = false;
635  }
636  }
637  }
638 
639  /* perspective divide: input clip coordinates, output normalized device coordinates */
640  for (TPRREuint i = 0; i < nVertexIndices_h; i++)
641  {
642  pVerticesTransf[i].coords.x = pVerticesTransf[i].coords.x / pVerticesTransf[i].coords.w;
643  pVerticesTransf[i].coords.y = pVerticesTransf[i].coords.y / pVerticesTransf[i].coords.w;
644  pVerticesTransf[i].coords.z = pVerticesTransf[i].coords.z / pVerticesTransf[i].coords.w;
645  }
646 
647  const float DEPTH_RANGE_MIN = 0.0f;
648  const float DEPTH_RANGE_MAX = 1.0f;
649 
650  /* viewport transf: input normalized device coordinates, output screen coordinates */
651  for (TPRREuint i = 0; i < nVertexIndices_h; i++)
652  {
653  pVerticesTransf[i].coords.x = vp_w / 2 * pVerticesTransf[i].coords.x + vp_x + vp_w / 2;
654  pVerticesTransf[i].coords.y = vp_h / 2 * pVerticesTransf[i].coords.y + vp_y + vp_h / 2;
655  pVerticesTransf[i].coords.z = (DEPTH_RANGE_MAX - DEPTH_RANGE_MIN) / 2 * pVerticesTransf[i].coords.z + (DEPTH_RANGE_MAX + DEPTH_RANGE_MIN) / 2;
656  }
657 
658  // TODO: face culling (here or sometime earlier in view space)
659 }
