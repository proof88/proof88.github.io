269 void DiscoverTextureCompressionAvailability()
270 {
271 // bSuppTexCompression is already set by DiscoverOpenGL_X_Y_availability()
272 if ( bSuppTexCompression )
273 {
274 getConsole().SOLn("Texture compression supported!");
275 glGetIntegerv(GL_NUM_COMPRESSED_TEXTURE_FORMATS_ARB, tmpRet);
276 if ( !PureGLsnippets::isGLErrorPresent() )
277 {
278 getConsole().OLn(" %d compressed texture formats available:", tmpRet[0]);
279 CheckAgainstStandardMinValue(tmpRet[0], GL_STANDARD_MIN_COMPRESSED_TEXTURE_FORMATS);
280 try
281 {
282 GLint* compressedFormats = new GLint[tmpRet[0]];
283 glGetIntegerv(GL_COMPRESSED_TEXTURE_FORMATS_ARB, compressedFormats);
284 if ( !PureGLsnippets::isGLErrorPresent() )
285 {
286 getConsole().OI();
287 for (GLint i = 0; i < tmpRet[0]; i++)
288 if ( vTexComprFmts.find( compressedFormats[i] ) == vTexComprFmts.end() )
289 getConsole().OLn(" Unknown: %d", compressedFormats[i]);
290 else
291 getConsole().OLn(" %s", vTexComprFmts[ compressedFormats[i] ].c_str());
292 getConsole().OO();
293 }
294 else
295 {
296 getConsole().EOLn(" glGetIntegerv() failed: %s", PureGLsnippets::getGLErrorTextFromEnum( PureGLsnippets::getLastSavedGLError() ));
297 }
298 delete[] compressedFormats;
299 }
300 catch (const std::bad_alloc&)
301 {
302 getConsole().EOLn(" Compressed texture formats available: ? (error: failed to allocate array!)");
303 }
304 }
305 else
306 getConsole().EOLn(" Compressed texture formats available: ? (error: %s)", PureGLsnippets::getGLErrorTextFromEnum( PureGLsnippets::getLastSavedGLError() ));
307
308 if ( bSuppVTC = isExtensionSupported("GL_NV_texture_compression_vtc") )
309 getConsole().SOLn(" VTC also supported.");
310 if ( bSupp3Dc = isExtensionSupported("GL_ATI_texture_compression_3dc") )
311 getConsole().SOLn(" 3Dc also supported.");
312 if ( bSuppLATC = isExtensionSupported("GL_EXT_texture_compression_latc") || isExtensionSupported("GL_NV_texture_compression_latc") )
313 getConsole().SOLn(" LATC also supported.");
314 }
315 else
316 getConsole().EOLn("Texture compression not supported.");
317 } // DiscoverTextureCompressionAvailability()

353 nMSAAmaxLevel = 0;
354 memset(bMSAAlevels, 0, MSAA_SUPP_LEVELS_ARRAY_SIZE);

385 {
