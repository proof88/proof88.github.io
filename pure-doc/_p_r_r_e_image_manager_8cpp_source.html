80 PRREImage* PRREImageManager::PRREImageManagerImpl::loadBMP(const char* filename)
81 {
82  BITMAPFILEHEADER file_header;
83  BITMAPINFOHEADER info_header;
84  DWORD bitmaplength;
85  RGBQUAD* palette = NULL;
86  DWORD palettesize = 0;
87  DWORD bytesread;
88  PRREImage* pNewImage = NULL;
89 
90  HANDLE bitmapfile = CreateFileA(filename, GENERIC_READ, FILE_SHARE_READ, NULL, OPEN_EXISTING, 0, 0);
91 
92  _pOwner->getConsole().OLnOI("PRREImageManager::loadBMP(\"%s\")", filename);
93  if ( bitmapfile == INVALID_HANDLE_VALUE )
94  return loadBMPfail(bitmapfile, pNewImage, palette, "ERROR: bitmapfile == INVALID_HANDLE_VALUE, returning PGENULL!");
95 
96  ReadFile(bitmapfile, &file_header, sizeof(BITMAPFILEHEADER), &bytesread, NULL);
97  _pOwner->getConsole().OLn("Read %d bytes of total %d bytes BITMAPFILEHEADER", bytesread, sizeof(BITMAPFILEHEADER));
98  ReadFile(bitmapfile, &info_header, sizeof(BITMAPINFOHEADER), &bytesread, NULL);
99  _pOwner->getConsole().OLn("Read %d bytes of total %d bytes BITMAPINFOHEADER", bytesread, sizeof(BITMAPINFOHEADER));
100 
101  if ( info_header.biCompression != BI_RGB )
102  return loadBMPfail(bitmapfile, pNewImage, palette, "ERROR: info_header.biCompression != BI_RGB, returning PGENULL!");
103 
104  _pOwner->getConsole().OLn("W x H x Bpp: %d x %d x %d", info_header.biWidth, info_header.biHeight, info_header.biBitCount);
105 
106  // Once I managed to save a bmp in Photoshop where biHeight was negative, so we need to make sure everything we have is positive
107  info_header.biWidth = abs(info_header.biWidth);
108  info_header.biHeight = abs(info_header.biHeight);
109  info_header.biBitCount = (WORD) abs((int) info_header.biBitCount);
110 
111  if ( (bitmaplength = info_header.biWidth * info_header.biHeight * (info_header.biBitCount < 32 ? 3 : 4)) == 0 )
112  return loadBMPfail(bitmapfile, pNewImage, palette, "ERROR: bitmaplength == 0, returning returning PGENULL!");
113 
114  try
115  {
116  if ( info_header.biClrUsed > 0 )
117  {
118  palettesize = info_header.biClrUsed * sizeof(RGBQUAD);
119  palette = new RGBQUAD[palettesize];
120  ReadFile(bitmapfile, palette, palettesize, &bytesread, NULL);
121  _pOwner->getConsole().OLn("Read %d bytes of total %d bytes of RGBQUAD array for palette", bytesread, palettesize);
122  if ( bytesread != palettesize )
123  {
124  return loadBMPfail(bitmapfile, pNewImage, palette, "ERROR: bytesread != palettesize, returning PGENULL!");
125  }
126  }
127 
128  pNewImage = new PRREImage();
129  // PRRE_RGB is just a fake value at this point
130  if ( !pNewImage->pImpl->initMembers(info_header.biBitCount, info_header.biWidth, info_header.biHeight,
131  PRRE_RGB, PRRE_RGB, true, false, PGENULL, bitmaplength) )
132  return loadBMPfail(bitmapfile, pNewImage, palette, "ERROR: initMembers() failed, returning PGENULL!");
133 
134  _pOwner->getConsole().OLn("offset: %d", file_header.bfOffBits);
135 
136  if ( !pNewImage->pImpl->readBMPpixels(bitmapfile, palette, info_header.biBitCount) )
137  return loadBMPfail(bitmapfile, pNewImage, palette, "ERROR: readBMPpixels() failed!");
138 
139  CloseHandle(bitmapfile);
140  delete[] palette;
141 
142  _pOwner->getConsole().SOLnOO("> bitmap successfully loaded into memory, using %d bytes (%d kbytes)",
143  sizeof(pNewImage)+pNewImage->getPixelsSize(), (sizeof(*pNewImage)+pNewImage->getPixelsSize())/1024);
144  } // try
145  catch (const std::bad_alloc&)
146  {
147  return loadBMPfail(bitmapfile, pNewImage, palette, "ERROR: failed to allocated palette!");
148  }
149 
150  return pNewImage;
151 
152 } // loadBMP()

228 TPRREuint PRREImageManager::getColorConversionSwapCount(TPIXCOMPORD from, TPIXCOMPORD to)
229 {
230  if ( (from % 2) == 1 )
231  from = (TPIXCOMPORD)(from-1);
232  if ( (to % 2) == 1 )
233  to = (TPIXCOMPORD)(to-1);
234  if ( from == to )
235  return 0;
236 
237  switch (from)
238  {
239  case PRRE_RGB: return (to == PRRE_GBR) || (to == PRRE_BRG) ? 2 : 1;
240  case PRRE_RBG: return (to == PRRE_GRB) || (to == PRRE_BGR) ? 2 : 1;
241  case PRRE_GRB: return (to == PRRE_RBG) || (to == PRRE_BGR) ? 2 : 1;
242  case PRRE_BGR: return (to == PRRE_RBG) || (to == PRRE_GRB) ? 2 : 1;
243  case PRRE_GBR: return (to == PRRE_RGB) || (to == PRRE_BRG) ? 2 : 1;
244  default /* BRG*/: return (to == PRRE_RGB) || (to == PRRE_GBR) ? 2 : 1;
245  }
246 } // getColorConversionSwapCount()

326 PRREImage* PRREImageManager::createFromFile(const char* filename)
327 {
328  getConsole().OLnOI("PRREImageManager::createFromFile(\"%s\")", filename);
329 
330  if ( filename == NULL )
331  return pImpl->createFromFileFail("ERROR: NULLPOINTER!");
332 
333  if ( !PFL::fileExists(filename) )
334  return pImpl->createFromFileFail("ERROR: file doesn't exist!");
335 
336  string sFileExt = PFL::getExtension(filename);
337  if ( sFileExt == "" )
338  return pImpl->createFromFileFail("ERROR: no file extension!");
339 
340  transform(sFileExt.begin(), sFileExt.end(), sFileExt.begin(), ::toupper);
341  getConsole().O("ext: .%s, ", sFileExt.c_str());
342  PRREImage* pNewImage;
343 
344  if ( sFileExt == "BMP" )
345  {
346  getConsole().OLn("calling loadBMP()... ");
347  pNewImage = pImpl->loadBMP(filename);
348  }
349  else return pImpl->createFromFileFail("ERROR: unsupported extension!");
350 
351  if ( pNewImage == PGENULL )
352  return pImpl->createFromFileFail("ERROR: loadXXX() returned PGENULL!");
353 
354  pNewImage->SetName("Image " + std::to_string(pImpl->nRunningCounter++));
355  pNewImage->SetFilename( filename );
356  Attach( *pNewImage );
357 
358  getConsole().SOLnOO("> Image loaded, name: %s!", pNewImage->getName().c_str());
359  getConsole().OLn("");
360 
361  return pNewImage;
362 } // createFromFile()

374 PRREImage* PRREImageManager::createBlank(TPRREuint width, TPRREuint height, TPRREuint bpp)
375 {
376  if ( (width == 0) || (height == 0) || (bpp != 24) )
377  return PGENULL;
378 
379  PRREImage* pNewImage = PGENULL;
380  try
381  {
382  pNewImage = new PRREImage();
383 
384  // ppxls = PGENULL means it won't try to copy anything to the allocated buffer so the content of the buffer will be undefined.
385  if ( !pNewImage->pImpl->initMembers(bpp, width, height, PRRE_RGB, PRRE_RGB, true, false, PGENULL, width * height * bpp/8) )
386  {
387  delete pNewImage;
388  getConsole().EOLn("ERROR: Failed initMembers(%d, %d, %d, ...)", bpp, width, height);
389  getConsole().OLn("");
390  return PGENULL;
391  }
392 
393  Attach( *pNewImage );
394 
395  pNewImage->SetName("Image " + std::to_string(pImpl->nRunningCounter++));
396 
397  getConsole().SOLnOO("> Blank Image created, name: %s!", pNewImage->getName().c_str());
398  getConsole().OLn("");
399  }
400  catch (const std::bad_alloc&)
401  {
402  getConsole().EOLn("ERROR: failed to instantiate PRREImage!");
403  getConsole().OLn("");
404  }
405 
406  return pNewImage;
407 }
