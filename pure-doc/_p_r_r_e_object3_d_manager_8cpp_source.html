689 PRREObject3D* PRREObject3DManager::createFromFile(
690  const char* filename,
691  TPRRE_VERTEX_MODIFYING_HABIT vmod,
692  TPRRE_VERTEX_REFERENCING_MODE vref,
693  TPRREbool bForceUseClientMemory )
694 {
695  if ( !pImpl->isInitialized() )
696  {
697  return PGENULL;
698  }
699 
700  getConsole().OLnOI("PRREObject3DManager::createFromFile(\"%s\")", filename);
701 
702  if ( filename == NULL )
703  {
704  getConsole().EOLnOO("ERROR: input was NULL, returning PGENULL");
705  getConsole().OLn("");
706  return PGENULL;
707  }
708 
709  string sFileExt = PFL::getExtension(filename);
710  if ( sFileExt == "" )
711  {
712  getConsole().EOLnOO("ERROR: no file extension!");
713  getConsole().OLn("");
714  return PGENULL;
715  }
716  if ( !PFL::fileExists(filename) )
717  {
718  getConsole().EOLnOO("ERROR: file doesn't exist!");
719  getConsole().OLn("");
720  return PGENULL;
721  }
722 
723  PRREObject3D* obj = PGENULL;
724  PRREObject3D* subobject = PGENULL;
725  PRREMesh3D* tmpMesh = PGENULL;
726  transform(sFileExt.begin(), sFileExt.end(), sFileExt.begin(), ::toupper);
727  getConsole().OLn("ext: .%s", sFileExt.c_str());
728  if ( sFileExt == "OBJ" )
729  {
730  try
731  {
732  getConsole().OI();
733  tmpMesh = PRREMesh3DManager::createFromFile(filename);
734  if ( !tmpMesh )
735  {
736  getConsole().OO();
737  getConsole().EOLnOO("ERROR: failed to create tmpMesh!");
738  getConsole().OLn("");
739  return PGENULL;
740  }
741  obj = new PRREObject3D(pImpl->materialMgr, vmod, vref, bForceUseClientMemory);
742  if ( !((PRREMesh3D*)obj)->pImpl->cannibalize(*tmpMesh) )
743  {
744  const std::string sErrMsg = "cannibalize() failed for level-1 obj!";
745  throw std::runtime_error(sErrMsg);
746  }
747  obj->SetName("Object3D " + std::to_string(pImpl->nRunningCounter++));
748 
749  TPRREuint nVerticesTotalTmpMesh = 0;
750  for (TPRREint i = 0; i < tmpMesh->getCount(); i++)
751  {
752  subobject = new PRREObject3D(pImpl->materialMgr, vmod, vref, bForceUseClientMemory);
753  obj->Attach( *subobject );
754  if ( !((PRREMesh3D*)subobject)->pImpl->cannibalize(*(PRREMesh3D*)(tmpMesh->getAttachedAt(i))) )
755  {
756  const std::string sErrMsg = "cannibalize() failed for level-2 subobj " + std::to_string(i) + "!";
757  throw std::runtime_error(sErrMsg);
758  }
759  subobject->pImpl->pVerticesTransf = new TPRRE_TRANSFORMED_VERTEX[subobject->getVerticesCount()];
760  nVerticesTotalTmpMesh += subobject->getVertexIndicesCount();
761 
762  // Legacy tmcsgfxlib behavior: auto-load textures for subobjects where pipe character is present in name
763  const std::string::size_type nPipePos = subobject->getName().find('|');
764  if ( nPipePos != std::string::npos )
765  {
766  const std::string sTexName = subobject->getName().substr(nPipePos+1);
767  getConsole().OLn("Legacy: loading texture implicitly by submodel name: %s", sTexName.c_str());

769  if ( tex )
770  {
