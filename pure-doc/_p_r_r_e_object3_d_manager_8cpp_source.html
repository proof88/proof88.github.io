424 PRREObject3D* PRREObject3DManager::createFromFile(
425  const char* filename,
426  TPRRE_VERTEX_MODIFYING_HABIT vmod,
427  TPRRE_VERTEX_REFERENCING_MODE vref,
428  TPRREbool bForceUseClientMemory )
429 {
430  if ( !pImpl->isInitialized() )
431  {
432  return PGENULL;
433  }
434 
435  getConsole().OLnOI("PRREObject3DManager::createFromFile(\"%s\")", filename);
436 
437  if ( filename == NULL )
438  {
439  getConsole().EOLnOO("ERROR: input was NULL, returning PGENULL");
440  getConsole().OLn("");
441  return PGENULL;
442  }
443 
444  string sFileExt = PFL::getExtension(filename);
445  if ( sFileExt == "" )
446  {
447  getConsole().EOLnOO("ERROR: no file extension!");
448  getConsole().OLn("");
449  return PGENULL;
450  }
451  if ( !PFL::fileExists(filename) )
452  {
453  getConsole().EOLnOO("ERROR: file doesn't exist!");
454  getConsole().OLn("");
455  return PGENULL;
456  }
457 
458  PRREObject3D* obj = PGENULL;
459  PRREObject3D* subobject = PGENULL;
460  PRREMesh3D* tmpMesh = PGENULL;
461  transform(sFileExt.begin(), sFileExt.end(), sFileExt.begin(), ::toupper);
462  getConsole().OLn("ext: .%s", sFileExt.c_str());
463  if ( sFileExt == "OBJ" )
464  {
465  try
466  {
467  getConsole().OI();
468  tmpMesh = PRREMesh3DManager::createFromFile(filename);
469  if ( !tmpMesh )
470  {
471  getConsole().OO();
472  getConsole().EOLnOO("ERROR: failed to create tmpMesh!");
473  getConsole().OLn("");
474  return PGENULL;
475  }
476  obj = new PRREObject3D(pImpl->materialMgr, vmod, vref, bForceUseClientMemory);
477  if ( !((PRREMesh3D*)obj)->pImpl->cannibalize(*tmpMesh) )
478  {
479  const std::string sErrMsg = "cannibalize() failed for level-1 obj!";
480  throw std::runtime_error(sErrMsg);
481  }
482 
483  for (TPRREint i = 0; i < tmpMesh->getCount(); i++)
484  {
485  subobject = new PRREObject3D(pImpl->materialMgr, vmod, vref, bForceUseClientMemory);
486  obj->Attach( *subobject );
487  if ( !((PRREMesh3D*)subobject)->pImpl->cannibalize(*(PRREMesh3D*)(tmpMesh->getAttachedAt(i))) )
488  {
489  const std::string sErrMsg = "cannibalize() failed for level-2 subobj " + std::to_string(i) + "!";
490  throw std::runtime_error(sErrMsg);
491  }
492  subobject->pImpl->pVerticesTransf = new TPRRE_TRANSFORMED_VERTEX[subobject->getVerticesCount()];
493 
494  // Legacy tmcsgfxlib behavior: auto-load textures for subobjects where pipe character is present in name
495  const std::string::size_type nPipePos = subobject->getName().find('|');
496  if ( nPipePos != std::string::npos )
497  {
498  const std::string sTexName = subobject->getName().substr(nPipePos+1);
499  getConsole().OLn("Legacy: loading texture implicitly by submodel name: %s", sTexName.c_str());
500  PRRETexture* const tex = pImpl->textureMgr.createFromFile( PFL::getDirectory(filename).append(sTexName).c_str() );
501  if ( tex )
502  {
503  if ( !subobject->getMaterial().setTexture(tex) )
504  {
505  getConsole().EOLn("ERROR: failed to set texture for subobject! Continuing ...");
506  }
507  }
508  else
509  {
510  getConsole().EOLn("ERROR: PRREObject3DManager::createFromFile() failed to load texture: %s! Continuing ...", sTexName.c_str());
511  }
512  // since we have loaded texture from submodelname, we can get rid of the texture filename part of it
513  subobject->SetName( subobject->getName().substr(0, nPipePos) );
514  }
515  }
516  this->DeleteAttachedInstance(*tmpMesh);
517  tmpMesh = PGENULL;
518 
519  obj->SetFilename(filename);
520  getConsole().OO();
521  Attach( *obj );
522 
523  // although PPP has already selected the vtransmode, we set it again
524  // to actually allocate the needed resources for the geometry
525  getConsole().OI();
526  if ( !obj->setVertexTransferMode( obj->getVertexTransferMode() ) )
527  {
528  getConsole().OO();
529  DeleteAttachedInstance( *obj );
530  getConsole().EOLnOO("ERROR: setVertexTransferMode() failed, returning NULL now!");
531  getConsole().OLn("");
532  return PGENULL;
533  }
534  getConsole().OO();
535  } // try
536  catch (const std::exception& e)
537  {
538  // note: textures won't be deallocated cause we don't track what textures were loaded for this object!
539  delete tmpMesh;
540  delete subobject; // will invoke obj->Detach()
541  delete obj; // would delete subobj, except if it is not yet attached, so we explicitly delete subobj above
542  getConsole().OO();
543  getConsole().EOLnOO("ERROR: failed to instantiate: %s!", e.what());
544  getConsole().OLn("");
545  return PGENULL;
546  } // catch
547  }
548  else
549  {
550  getConsole().EOLnOO("ERROR: unsupported extension!");
551  getConsole().OLn("");
552  return PGENULL;
553  }
554 
555  obj->SetName("Object3D " + std::to_string(pImpl->nRunningCounter++));
556 
557  getConsole().SOLnOO("> Object loaded successfully, name: %s!", obj->getName().c_str());
558  getConsole().OLn("");
559  return obj;
560 } // createFromFile()
