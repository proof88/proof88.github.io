31 Vertex Culling (Object-Space)
32 *****************************
33
34 GL_EXT_cull_vertex: https://www.khronos.org/registry/OpenGL/extensions/EXT/EXT_cull_vertex.txt
35 - 1996
36 - "Culling a polygon by examining its vertexes in object space can be more efficient than screen space polygon culling since
37 the transformation to screen space (which may include a division by w) can be avoided for culled vertexes. Also, vertex culling
38 can be computed before vertexes are assembled into primitives. This is a useful property when drawing meshes with shared vertexes,
39 since a vertex can be culled once, and the resulting state can be used for all primitives which share the vertex."
40 - This extension was most probably introduced for optimisation before HW T&L spread. Lack of HW T&L was present in the pre-Geforce era,
41 i.e. TNT, etc.
42 Desktop: was mostly supported by early integrated Intel VGA.
43 Mobile: no support at all!
44 - Verdict: don't use it if neither 3dfx nor other major vendors supported it before TnL!

78PureObject3D::PureObject3DImpl::~PureObject3DImpl()
79{
80 _pOwner->getManagedConsole().OLnOI("~PureObject3D() ...");
81
82 delete[] pVerticesTransf;
83 delete[] pFbBuffer;
84 pVerticesTransf = PGENULL;
85 pFbBuffer = PGENULL;
86
87 // Remember: we cannot use isLevel1() or isLevel2() here because isLevel2() might return true for
88 // a level-1 object that was originally a cloned object but not referring to any object anymore after
89 // the original object has been deleted!
90 // We might check for actual vertex data because level-1 objects never have vertex data, but I think
91 // in the future even level-2 objects might also lose vertex data in their client memory if it is
92 // already uploaded to host memory.
93 // So, this dtor must execute operations keeping mind that the same code is executed for both level-1
94 // and level-2 objects!
95 // On the long run, maybe we change behavior and in case of deleting an object referred to by other
96 // objects, all referrers are also get deleted automatically.
97
98 if ( pRefersto )
99 {
100 pRefersto->pImpl->referrers.erase(_pOwner);
101 }
102 else if ( referrers.size() > 0 )
103 {
104 // referrers now become zombie objects: they do not refer but they dont have any subobjects either!
105 // user's responsibility to get rid of them if original object is removed!
106 for (auto& referrer : referrers)
107 referrer->pImpl->pRefersto = PGENULL;
108 }
109
110 SetOcclusionTested(false);
111
112 // these are way unnecessary things if we are a level-2 object, but as explained above, we have to live
113 // with it - at least these are all harmless code even when executed for a level-2.
114 std::set< std::deque<PureObject3D*>* > mapsEraseObjectFrom;
115 // would be much nicer with Cpp11 initializer list
116 mapsEraseObjectFrom.insert(&occluders);
117 mapsEraseObjectFrom.insert(&occludees_opaque);
118 mapsEraseObjectFrom.insert(&occludees_blended);
119 mapsEraseObjectFrom.insert(&occludees_2d_opaque);
120 mapsEraseObjectFrom.insert(&occludees_2d_blended);
121
122 for (auto pMapEraseFrom : mapsEraseObjectFrom )
123 {
124 auto occ_it = std::find(pMapEraseFrom->begin(), pMapEraseFrom->end(), _pOwner);
125 while ( occ_it != pMapEraseFrom->end() )
126 {
127 pMapEraseFrom->erase(occ_it);
128 occ_it = std::find(pMapEraseFrom->begin(), pMapEraseFrom->end(), _pOwner);
129 }
130 } // end for pMapEraseFrom
131
132 _pOwner->getManagedConsole().OI();
133 _pOwner->DeleteAll();
134 _pOwner->getManagedConsole().OO();
135
136 _pOwner->getManagedConsole().SOLnOO("Done!");
137} // ~PureObject3DImpl()

488void PureObject3D::PureObject3DImpl::SetOccluder(TPureBool value)
489{
490 if ( !_pOwner->isLevel1() )
491 {
492 _pOwner->getManagedConsole().EOLn("PureObject3D::%s() attempt for level-2 object is ignored!", __FUNCTION__);
493 return;
494 }
495
496 if ( _pOwner->getManager() == PGENULL )
497 {
498 _pOwner->getConsole().EOLn("%s() ERROR: Object3D does not have manager associated!", __FUNCTION__);
499 return;
500 }
501
502 const TPureBool bBlended = PureMaterial::isBlendFuncReallyBlending(_pOwner->getMaterial(false).getSourceBlendFunc(), _pOwner->getMaterial(false).getDestinationBlendFunc());
503
504 std::set< std::deque<PureObject3D*>* > mapsEraseObjectFrom;
505
506 if ( value )
507 {
508 if ( !isRenderingAllowed() ||
509 isStickedToScreen() ||
510 !(isAffectingZBuffer()) ||
511 isWireframed() ||
512 bBlended
513 )
514 {
515 _pOwner->getConsole().EOLn("%s() ERROR: cannot set occluder state to true due to conflicting property!", __FUNCTION__);
516 return;
517 }
518
519 // At this point we are setting this object to be occluder, however due to rare issue we have to make sure we wait for
520 // its occlusion query to finish AND set its occluded state to false because occluders are always meant to be rendered.
521 // This fixes an issue that can be also seen in EV2008P03 sample, when the user moves the mouse changing camera angle
522 // within the 1st few frames, during when the renderer automatically decides which objects should be occluders by UpdateOccluderStates().
523 ForceFinishOcclusionTest();
524
525 auto occ_it = std::find(occluders.begin(), occluders.end(), _pOwner);
526 if ( occ_it == occluders.end() )
527 {
528 occluders.push_back(_pOwner);
529 }
530 mapsEraseObjectFrom.insert(&occludees_opaque);
531 mapsEraseObjectFrom.insert(&occludees_blended);
532 } // value true
533 else
534 {
535 mapsEraseObjectFrom.insert(&occluders);
536
537 if ( bBlended )
538 {
539 if ( isStickedToScreen() )
540 {
541 auto occ_it = std::find(occludees_2d_blended.begin(), occludees_2d_blended.end(), _pOwner);
542 if ( occ_it == occludees_2d_blended.end() )
543 {
544 occludees_2d_blended.push_back(_pOwner);
545 }
546 mapsEraseObjectFrom.insert(&occludees_blended);
547 }
548 else
549 {
550 auto occ_it = std::find(occludees_blended.begin(), occludees_blended.end(), _pOwner);
551 if ( occ_it == occludees_blended.end() )
552 {
553 occludees_blended.push_back(_pOwner);
554 }
555 mapsEraseObjectFrom.insert(&occludees_2d_blended);
556 }
557
558 mapsEraseObjectFrom.insert(&occludees_opaque);
559 mapsEraseObjectFrom.insert(&occludees_2d_opaque);
560 } // bBlended true
561 else
562 {
563 if ( isStickedToScreen() )
564 {
565 auto occ_it = std::find(occludees_2d_opaque.begin(), occludees_2d_opaque.end(), _pOwner);
566 if ( occ_it == occludees_2d_opaque.end() )
567 {
568 occludees_2d_opaque.push_back(_pOwner);
569 }
570 mapsEraseObjectFrom.insert(&occludees_opaque);
571 }
572 else
573 {
574 auto occ_it = std::find(occludees_opaque.begin(), occludees_opaque.end(), _pOwner);
575 if ( occ_it == occludees_opaque.end() )
576 {
577 occludees_opaque.push_back(_pOwner);
578 }
579 mapsEraseObjectFrom.insert(&occludees_2d_opaque);
580 }
581
582 mapsEraseObjectFrom.insert(&occludees_blended);
583 mapsEraseObjectFrom.insert(&occludees_2d_blended);
584 } // bBlended false
585 } // value false
586
587 for (auto pMapEraseFrom : mapsEraseObjectFrom )
588 {
589 auto occ_it = std::find(pMapEraseFrom->begin(), pMapEraseFrom->end(), _pOwner);
590 while ( occ_it != pMapEraseFrom->end() )
591 {
592 pMapEraseFrom->erase(occ_it);
593 occ_it = std::find(pMapEraseFrom->begin(), pMapEraseFrom->end(), _pOwner);
594 }
595 } // end for pMapEraseFrom
596
597 bOccluder = value;
598} // SetOccluder()

613void PureObject3D::PureObject3DImpl::SetOcclusionTested(TPureBool state)
614{
615 // It can happen at rare cases e.g. at engine shutdown that isLevel1() returns false due to we are cloned object NOT having the referred object anymore,
616 // because it has just been deleted prior to deleting us (SetOcclusionTested(false) is being invoked by our dtor).
617 // This case should be somehow identified to NOT log it since that is misleading to log this call as we were really a level-2 object!
618 // I could not find a way to detect at this point that isLevel1() returns false just because our referred object has been already deleted.
619 // So, instead of just checking isLevel1() right at the beginning of the function:
620 // - check for isLevel1() only when state is true, because that path we definitely don't want to be executed for level-2 objects, it must be logged as error;
621 // - and it is definitely deleting objects case, since dtors set state to false, not true!
622 // - do not check for isLevel1() when given state is false because it that code path doesn't have any effect when invoked for a real level-2 object.
623
624 if ( !PureHwInfo::get().getVideo().isOcclusionQuerySupported() )
625 return;
626
627 if ( state )
628 {
629 if ( PGENULL == _pOwner->getManager() )
630 {
631 throw std::runtime_error("no manager!");
632 }
633
634 if ( !_pOwner->isLevel1() )
635 {
636 _pOwner->getManagedConsole().EOLn("PureObject3D::%s(true) attempt for level-2 object is ignored!", __FUNCTION__);
637 return;
638 }
639
640 if ( !pglGenQueries(1, &nOcclusionQuery) )
641 {
642 _pOwner->getConsole().EOLn("%s() ERROR: Could not generate occlusion query, this object won't be tested for occlusion!", __FUNCTION__);
643 nOcclusionQuery = 0; // just in case pglGenQueries() changed it to something else ...
644 return;
645 }
646
647 _pOwner->getConsole().OLn("Occlusion query ID: %d", nOcclusionQuery);
648
649 if ( getReferredObject() )
650 {
651 if ( getReferredObject()->pImpl->pBoundingBox )
652 {
653 // if we are cloned object, we should NOT create bounding box object if referred already has since we can use the box of the referred object
654 // we can do this because in case of cloned object, we expect the same geometry, etc.
655 _pOwner->getConsole().OLn("We are cloned and referred already has bounding box, so will use that");
656 return;
657 }
658 else
659 {
660 _pOwner->getConsole().OLn("We are cloned but referred does not have bounding box, so creating it now for cloned ...");
661 }
662 }
663
664 // here we specify forcing bounding box geometry in client memory because we alter vertex positions with rel pos vec, and then we upload it to host memory
665 pBoundingBox = ((PureObject3DManager*)_pOwner->getManager())->createBox(_pOwner->getSizeVec().getX(), _pOwner->getSizeVec().getY(), _pOwner->getSizeVec().getZ(), PURE_VMOD_DYNAMIC, PURE_VREF_INDEXED, true);
666 if ( PGENULL == pBoundingBox )
667 {
668 const std::string sErrMsg = "failed to create pBoundingBox!";
669 throw std::runtime_error(sErrMsg);
670 }
671
672 pBoundingBox->SetName("Bounding Box for " + _pOwner->getName());
673 pBoundingBox->Hide();

702

761TPureUInt PureObject3D::PureObject3DImpl::draw(const TPURE_RENDER_PASS& renderPass, TPureBool bASyncQuery, TPureBool bRenderIfQueryPending)
762{
763 // caller renderer is expected to check for GL errors, probably only once per frame, so we don't check them here
764
765 if ( _pOwner->isLevel1() )
766 {
767 _pOwner->ResetLastTransferredCounts();
768 }
769
770 if ( !bVisible )
771 return 0;
772
773 if ( !PureHwInfo::get().getVideo().isAcceleratorDetected() )
774 {
775 return draw_DrawSW();
776 }
777
778 // see if we are parent or referring to another object i.e. we are cloned object
779 if ( _pOwner->isLevel1() )
780 {
781 // Currently 3D objects are 2-level entities:
782 // first level (parent) has no geometry, owns subobjects, sets basic things,
783 // while second level (subobjects) own geometry, inherit basic things set by parent.
784
785 // So transformations and other basic things are set by parent objects having subobjects.
786 // Or by cloned objects which refer to another original objects but still have their own position, angle, etc.
787 if ( renderPass == PURE_RPASS_NORMAL )
788 {
789 if ( (nOcclusionQuery != 0) && draw_OcclusionQuery_Finish(bASyncQuery, bRenderIfQueryPending) )
790 {
791 return 0;
792 }
793
794 Draw_ApplyTransformations();
795 Draw_PrepareGLBeforeDrawNormal(false);
796 // continue with drawing our subobjects
797 }
798 else if ( renderPass == PURE_RPASS_START_OCCLUSION_QUERY )
799 {
800 if ( nOcclusionQuery == 0 )
801 {
802 // there is nothing to do in this pass
803 return 0;
804 }
805 Draw_ApplyTransformations();
806 return draw_OcclusionQuery_Start(bASyncQuery);
807 }
808 else if ( renderPass == PURE_RPASS_BOUNDING_BOX_DEBUG_FOR_OCCLUSION_QUERY )
809 {
810 if ( nOcclusionQuery == 0 )
811 {
812 // there is nothing to do in this pass
813 return 0;
814 }
815 Draw_ApplyTransformations();
816 return draw_RenderBoundingBox();
817 }
818 else if ( renderPass == PURE_RPASS_Z_ONLY )
819 {
820 Draw_ApplyTransformations();
821 // continue with drawing our subobjects
822 }
823
824 // take care of attached objects (subobjects)
825 // note that either we have our own subobjects, OR we are just a cloned object and we need to draw original object's subobjects
826 PureObject3D* pWhichParent = getReferredObject() ? getReferredObject() : _pOwner;
827
828 pWhichParent->pImpl->bParentInitiatedOperation = true;
829 TPureUInt nRetSum = 0;
830 for (TPureInt i = 0; i < pWhichParent->getCount(); i++)
831 nRetSum += ((PureObject3D*)pWhichParent->getAttachedAt(i))->draw(renderPass, bASyncQuery, bRenderIfQueryPending);
832 pWhichParent->pImpl->bParentInitiatedOperation = false;
833 return nRetSum;
834 }
835
836 // if we reach this point then either our parent is drawing us as its subobject, or a cloned object is drawing us on behalf of our parent
837 PureObject3D* pObjLevel1 = (PureObject3D*)_pOwner->getManager();
838 const TPureBool bObjLevel1Blended = PureMaterial::isBlendFuncReallyBlending(pObjLevel1->getMaterial(false).getSourceBlendFunc(), pObjLevel1->getMaterial(false).getDestinationBlendFunc());
839
840 // subobject must ignore draw if its Draw() was not called by its parent level-1 object but someone else from outside ...
841 if ( !pObjLevel1->pImpl->bParentInitiatedOperation )
842 {
843 _pOwner->getManagedConsole().EOLn("Draw() of subobject called outside of its level-1 parent object, ignoring draw!");
844 return 0;
845 }
846
847 if ( renderPass == PURE_RPASS_NORMAL )
848 {
849 PureGLsnippets::glLoadTexturesAndSetBlendState(&(_pOwner->getMaterial(false)), pObjLevel1->isStickedToScreen(), bObjLevel1Blended);
850 }
851
852 Draw_FeedbackBuffer_Start();
853 const TPureUInt nTransferredVertexCount = _pOwner->transferVertices();
854 Draw_FeedbackBuffer_Finish();
855
856 return nTransferredVertexCount;
857} // Draw()

936void PureObject3D::PureObject3DImpl::Draw_FeedbackBuffer_Start()
937{
938 assert(_pOwner->isLevel2());
939
940 /* We create this buffer only once at first run. */
941 /* We intentionally switch to feedback mode only once since now we just use it for debugging. */
942 if ( pFbBuffer != NULL )
943 return;
944
945 /* num of values can be stored in this buffer */
946 /* we need to store not only the transformed values but also some extra values for grammar */
947 /* in this case, the items placed in this array are polygon items (GL_POLYGON_TOKEN) which look like this:
948 GL_POLYGON_TOKEN 3 (value value value value (value value value value) (value value value value))
949 So for a standard cube where each side has 2 triangles, it is 12 triangles in total, meaning
950 12 polygons. GL_POLYGON_TOKEN itself is 1 value, 3 itself is also 1 value, and then 3*12 values for 3 vertices.
951 So the total num of values is: numOfTriangles * (2+3*12)
952 Example with some real data is in later comment with debugbuffer[]. */
953
954 nFbBuffer_h = GLsizei(ceil((_pOwner->getVerticesCount() / 3.0f))) * (2+3*12);
955 try
956 {
957 /* TODO: probably in future we should rather use pVerticesTransf here as well since we already have it for that purpose, right? ;) */
958 pFbBuffer = new GLfloat[nFbBuffer_h];
959
960 /* unfortunately only the most detailed option GL_4D_COLOR_TEXTURE will give us the w-coord of vertices so we need to use that */
961 if ( pglFeedbackBuffer(nFbBuffer_h, GL_4D_COLOR_TEXTURE, pFbBuffer) )
962 {
963 // If you want all vertices to be transformed and catched in feedback mode then dont forget to disable culling and depth testing (maybe only 1 is needed to be disabled).
964 //glDisable(GL_DEPTH_TEST);
965 glDisable(GL_CULL_FACE); /* otherwise only the front facing side of cube would be written to feedback buffer */
966
967 glRenderMode(GL_FEEDBACK);
968 }
969 }
970 catch (const std::bad_alloc&)
971 {
972 _pOwner->getManagedConsole().EOLn("ERROR: PureObject3D::PureObject3DImpl::Draw() failed to allocate pFbBuffer!");

1038void PureObject3D::PureObject3DImpl::Draw_ApplyTransformations() const
1039{
1040 assert(_pOwner->isLevel1());
1041
1042 /* Make sure we don't call GL functions when no accelerator is present */
1043 if ( !PureHwInfo::get().getVideo().isAcceleratorDetected() )
1044 return;
1045
1046 glTranslatef(_pOwner->getPosVec().getX(), _pOwner->getPosVec().getY(), _pOwner->getPosVec().getZ());
1047
1048 switch ( getRotationOrder() )
1049 {
1050 case PURE_XYZ:
1051 glRotatef(getAngleVec().getX(), 1,0,0);
1052 glRotatef(getAngleVec().getY(), 0,1,0);
1053 glRotatef(getAngleVec().getZ(), 0,0,1);
1054 break;
1055 case PURE_XZY:
1056 glRotatef(getAngleVec().getX(), 1,0,0);
1057 glRotatef(getAngleVec().getZ(), 0,0,1);
1058 glRotatef(getAngleVec().getY(), 0,1,0);
1059 break;
1060 case PURE_YXZ:
1061 glRotatef(getAngleVec().getY(), 0,1,0);
1062 glRotatef(getAngleVec().getX(), 1,0,0);
1063 glRotatef(getAngleVec().getZ(), 0,0,1);
1064 break;
1065 case PURE_YZX:
1066 glRotatef(getAngleVec().getY(), 0,1,0);
1067 glRotatef(getAngleVec().getZ(), 0,0,1);
1068 glRotatef(getAngleVec().getX(), 1,0,0);
1069 break;
1070 case PURE_ZXY:
1071 glRotatef(getAngleVec().getZ(), 0,0,1);
1072 glRotatef(getAngleVec().getX(), 1,0,0);
1073 glRotatef(getAngleVec().getY(), 0,1,0);
1074 break;
1075 case PURE_ZYX:
1076 glRotatef(getAngleVec().getZ(), 0,0,1);
1077 glRotatef(getAngleVec().getY(), 0,1,0);
1078 glRotatef(getAngleVec().getX(), 1,0,0);
1079 break;
1080 } // switch
1081
1082 glScalef(getScaling().getX(), getScaling().getY(), getScaling().getZ());
1083 /*
1084 Well, mirror-transforming an object (scaling by negative value) makes its faces flip:
1085 https://www.gamedev.net/forums/topic/640616-negative-scaling-flips-face-winding-affects-backface-culling/
1086 I wanted to X-mirror objects loaded by tmcsgfxlib-to-prre wrapper for legacy projects, because I noticed they appeared mirrored compared to proofps engine behavior.
1087 So I noticed when setting negative scale on X axis, faces also flipped, so objects became inside-out.
1088 This is expected since vertices' winding order also flips because we still feed triangle vertices in same order even though triangles become mirrored.
1089 But I don't treat this as a legacy-project-only issue, since even a new project with the new engine could come up for any reason with negative scaling for any mirroring reason.
1090 So, this is rather a feature: we need to detect negative scaling factor and change front face winding order on-the-fly.
1091 */
1092 if ( getScaling().getX() * getScaling().getY() * getScaling().getZ() <= (0.0f-PFL::E) )
1093 glFrontFace(GL_CW);
1094 else
1095 glFrontFace(GL_CCW);
1096} // Draw_ApplyTransformations()

1314TPureUInt PureObject3D::PureObject3DImpl::draw_OcclusionQuery_Start(TPureBool async)
1315{
1316 assert(_pOwner->isLevel1());
1317
1318 if ( nOcclusionQuery == 0 )
1319 {
1320 return 0;
1321 }
1322
1323 PureObject3D* const pWhichBoundingBox = pBoundingBox ? pBoundingBox : (getReferredObject() ? getReferredObject()->pImpl->pBoundingBox : PGENULL);
1324
1325 assert(pWhichBoundingBox != PGENULL);
1326 assert(pWhichBoundingBox->getCount() > 0);
1327
1328 pWhichBoundingBox->ResetLastTransferredCounts();
1329
1330 if ( bOcclusionQueryStarted )
1331 {
1332 // obviously we don't start it if it is already started, doesn't matter if it is sync or async,
1333 // however since all sync occlusion queries must finish in the same frame, it cannot happen it is already started when we are just invoked to start it!
1334 if ( !async )
1335 {
1336 _pOwner->getConsole().EOLn("%s() ERROR: occlusion query is already started but we are in SYNC query mode!", __FUNCTION__);
1337 }
1338 assert(async);
1339 return 0;
1340 }
1341
1342 if ( async )
1343 {
1344 if ( bOccluded )
1345 {
1346 if ( nFramesWithoutOcclusionTest < OQ_MAX_FRAMES_WO_START_QUERY_WHEN_OCCLUDED )
1347 {
1348 nFramesWithoutOcclusionTest++;
1349 return 0;
1350 }
1351 }
1352 else
1353 {
1354 if ( nFramesWithoutOcclusionTest < OQ_MAX_FRAMES_WO_START_QUERY_WHEN_VISIBLE )
1355 {
1356 nFramesWithoutOcclusionTest++;
1357 return 0;
1358 }
1359 }
1360 } // async
1361
1362 glBeginOcclusionQuery();
1363 const TPureUInt nRet = ((PureObject3D*)(pWhichBoundingBox->getAttachedAt(0)))->transferVertices();
1364 glEndOcclusionQuery();
1365
1366 bOcclusionQueryStarted = true;
1367 nFramesWaitedForOcclusionTestResult = 0; // for async query
1368
1369 return nRet;
1370} // draw_OcclusionQuery_Start()

1386TPureBool PureObject3D::PureObject3DImpl::draw_OcclusionQuery_Finish(TPureBool bASyncQuery, TPureBool bRenderIfQueryPending)
1387{
1388 assert(_pOwner->isLevel1());
1389
1390 if ( nOcclusionQuery == 0 )
1391 {
1392 return false; // let it be rendered
1393 }
1394
1395 if ( !bOcclusionQueryStarted )
1396 {
1397 // we always end up here with legacy renderer which never starts query for any objects but
1398 // still invokes this function (i.e. Object3D::Draw() invokes it)
1399
1400 if ( bASyncQuery )
1401 {
1402 if ( bOccluded && (OQ_MAX_FRAMES_WO_START_QUERY_WHEN_OCCLUDED == 0) )
1403 { // if query is not started, but it should had been started if constant value says it should not wait for another frame,
1404 // then this is a programmer error condition
1405 if ( !isOccluder() )
1406 { // abort only if this is really an occludee: because renderer won't start occlusion query for occluders beforehand!
1407 assert(false); // this just cannot happen
1408 }
1409 }
1410 return bOccluded; // let it be rendered if last result says not occluded
1411 }
1412 else
1413 {
1414 // it is ok to render with either legacy renderer or when occlusion query is in sync
1415 // since in latter case bOcclusionQueryStarted is false only when it is not tested for occlusion
1416 return false; // let it be rendered
1417 }
1418 }
1419
1420 // this should be done rather with chrono after we have C++11
1421 PFL::timeval begin, end;
1422 PFL::gettimeofday(&begin, 0);
1423
1424 GLint queryResultAvailable;
1425 do {
1426 glGetQueryObjectivARB(nOcclusionQuery, GL_QUERY_RESULT_AVAILABLE_ARB, &queryResultAvailable);
1427 } while (!queryResultAvailable && !bASyncQuery);
1428
1429 // update sync statistics related to wait time
1430 if ( !bASyncQuery )
1431 {
1432 PFL::gettimeofday(&end, 0);
1433 if ( PFL::updateForMaxDuration(timeLongestWaitForSyncQueryFinish, begin, end) )
1434 {
1435 PFL::updateForMaxDuration(PureObject3D::PureObject3DImpl::stats[PureObject3D::PureObject3DImpl::stats.size()-1].timeLongestGlobalWaitForSyncQueryFinish, begin, end);
1436 }
1437 } // sync stats update
1438
1439 if ( queryResultAvailable == GL_FALSE )
1440 {
1441 // this is async query
1442 nFramesWaitedForOcclusionTestResult++;
1443 return bRenderIfQueryPending ? false : bOccluded;
1444 }
1445
1446 bOcclusionQueryStarted = false;
1447
1448 if ( PureHwInfo::get().getVideo().isBooleanOcclusionQuerySupported() )
1449 {
1450 GLint sampleBoolean;
1451 glGetQueryObjectivARB(nOcclusionQuery, GL_QUERY_RESULT_ARB, &sampleBoolean);
1452 bOccluded = ( sampleBoolean == GL_FALSE );
1453 }
1454 else
1455 {
1456 GLuint sampleCount;
1457 glGetQueryObjectuivARB(nOcclusionQuery, GL_QUERY_RESULT_ARB, &sampleCount);
1458 bOccluded = ( sampleCount == 0 );
1459 }
1460
1461 // update async statistics
1462 if ( bASyncQuery )
1463 {
1464 if ( nFramesWaitedForOcclusionTestResult < nFramesWaitedForOcclusionTestResultMin )
1465 {
1466 nFramesWaitedForOcclusionTestResultMin = nFramesWaitedForOcclusionTestResult;
1467 if ( nFramesWaitedForOcclusionTestResultMin < PureObject3D::PureObject3DImpl::stats[PureObject3D::PureObject3DImpl::stats.size()-1].nFramesWaitedForOcclusionTestResultGlobalMin )
1468 {
1469 PureObject3D::PureObject3DImpl::stats[PureObject3D::PureObject3DImpl::stats.size()-1].nFramesWaitedForOcclusionTestResultGlobalMin = nFramesWaitedForOcclusionTestResultMin;
1470 }
1471 }
1472 if ( nFramesWaitedForOcclusionTestResult > nFramesWaitedForOcclusionTestResultMax )
1473 {
1474 nFramesWaitedForOcclusionTestResultMax = nFramesWaitedForOcclusionTestResult;
1475 if ( nFramesWaitedForOcclusionTestResultMax > PureObject3D::PureObject3DImpl::stats[PureObject3D::PureObject3DImpl::stats.size()-1].nFramesWaitedForOcclusionTestResultGlobalMax )
1476 {
1477 PureObject3D::PureObject3DImpl::stats[PureObject3D::PureObject3DImpl::stats.size()-1].nFramesWaitedForOcclusionTestResultGlobalMax = nFramesWaitedForOcclusionTestResultMax;
1478 }
1479 }
1480 nFramesWaitedForOcclusionTestResult = 0;
1481 nFramesWithoutOcclusionTest = 0;
1482 }
1483
1484 return bOccluded;
1485} // draw_OcclusionQuery_Finish()

1492{
1493 if ( !bVisible )
1494 return 0;
1495
1496 // see if we are parent or referring to another object i.e. we are cloned object
1497 if ( _pOwner->isLevel1() )

1499 // Currently 3D objects are 2-level things: first level (parent) has no geometry, owns subobjects, sets basic things,
1500 // while second level (subobjects) own geometry, inherit basic things set by parent.
1501

1503 // Or by cloned objects which refer to another original objects but still have their own position, angle, etc.

1506 Draw_ApplyTransformations();

1511 TPureUInt nRetSum = 0;
1512 for (TPureInt i = 0; i < pWhichParent->getCount(); i++)
1513 nRetSum += ((PureObject3D*)pWhichParent->getAttachedAt(i))->draw(PURE_RPASS_NORMAL, false, false);
1514 return nRetSum;
1515 }

1524TPureFloat PureObject3D::PureObject3DImpl::recalculateBiggestAreaScaled()
1525{
1526 assert(_pOwner->isLevel1());
1527
1528 const PureVector vScaledSizeVec = _pOwner->getScaledSizeVec();
1529 const TPureFloat fAreaXY = vScaledSizeVec.getX() * vScaledSizeVec.getY();
1530 const TPureFloat fAreaXZ = vScaledSizeVec.getX() * vScaledSizeVec.getZ();

1536 }
