114 TPRREuint PRRERendererHWfixedPipeImpl::initialize(
115  TPRREuint width, TPRREuint height,
116  TPRRE_DISPLAY_MODES dmode, TPRREuint freq,
117  TPRREint cdepth, TPRREint zdepth, TPRREint stencil,
118  TPRREint samples,
119  HWND window)
120 {
121  TPRREint st = stencil;
122 
123  getConsole().OLn("PRRERendererHWfixedPipe::initialize(%d, %d, dmode, %d, %d, %d, %d, %d)",
124  width, height, freq, cdepth, zdepth, st, samples);
125  if ( isInitialized() )
126  {
127  getConsole().SOLn(" > already initialized!");
128  return 1;
129  }
130 
131  getConsole().OI();
132 
133  // we accept only 0 or 8 for stencil bits
134  if ( st != 0 )
135  {
136  st = 8;
137  getConsole().OLn("WARNING: forcing non-0 stencil to %d !", st);
138  }
139 
140  TPRREbool bRequestAdvancedPixelFormat = false; // will be set to true in the loop AFTER 1st GL init if antialiasing is requested AND if advanced formats available
141  TPRREbool bLoopNeeded = true; // will be set to false in the loop AFTER 1st GL init if antialiasing is NOT requested OR advanced formats are NOT available
142 
143  /*
144  Initialize OpenGL once or twice.
145  2nd init is needed if advanced pixel format is needed, eg. antialiasing is requested. That's the reason of the loop here.
146  This comes from the nature of OpenGL: it must be initialized to query available advanced pixel formats, then it may be reinitialized again with an
147  advanced pixel format, thus a renderer must handle screen, window and hwinfo instances init during its init.
148  */
149  while ( bLoopNeeded )
150  {
151  getConsole().OLn("");
152  getConsole().OLn("#### STAGE 1/3 ####");
153 
154  getConsole().OLn("screen settings ...");
155  screen.SetResolution(width, height);
156  screen.SetFullScreened( dmode != PRRE_WINDOWED ); // == PRRE_FULLSCREEN nem jó, mert lesz 3. FS mód is ...
157  screen.SetFreq( freq );
158  screen.SetColorBits( cdepth );
159  screen.SetDepthBits( zdepth );
160  screen.SetStencilBits( st );
161  screen.SetFSAAlevel( samples );
162 
163  getConsole().OLn("new PRREWindow...");
164  /*
165  A window may be created only once, even if advanced pixel format is requested, since a DC is required to be created only once.
166  This would mean that window instance can be created before renderer initialization. But based on SetPixelFormat() documentation, a
167  pixel format shouldn't be set to a window multiple times. This means that we actually have to recreate a window if an advanced pixel
168  format is to be set!
169  */
170  if ( !wnd.initialize(screen.getResWidth(), screen.getResHeight(), dmode != PRRE_WINDOWED, "Window", window) )
171  {
172  getConsole().EOLnOO("ERROR: wnd.initialize() failed!");
173  shutdown();
174  return 1;
175  }
176  getConsole().SOLn("done!");
177 
178  if ( !screen.applyDisplaySettings(wnd.getWndDC(), bRequestAdvancedPixelFormat ? PRRE_SCREEN_PF_ADVANCED : PRRE_SCREEN_PF_SIMPLE) )
179  {
180  getConsole().EOLnOO("ERROR: applyDisplaySettings() failed!");
181  shutdown();
182  return 1;
183  }
184 
185  getConsole().OLn("");
186  getConsole().OLn("#### STAGE 2/3 ####");
187  if ( !initializeOpenGL(wnd.getWndDC()) )
188  {
189  getConsole().OO();
190  shutdown();
191  return 1;
192  }
193 
194  hwInfo.Initialize(
195  rc, wnd.getWndDC(),
196  screen.getResWidth(), screen.getResHeight(),
197  screen.getColorBits(), screen.getDepthBits(),
198  screen.getStencilBits(), screen.getFSAAlevel() );
199 
200  if ( hwInfo.getVideo().isAdvancedPixelFormatSetSupported() )
201  {
202  if ( (samples > 0) && !bRequestAdvancedPixelFormat )
203  {
204  getConsole().OLn("Advanced pixel format requested, trying ...");
205  // following call can be removed after screen.applyDisplaySettings() is modified to automatically
206  // query advanced pixel formats at first call ... see PGE-History.txt's relevant RFR task!
207  screen.applyDisplaySettings( wnd.getWndDC() );
208  bRequestAdvancedPixelFormat = true;
209  shutdown();
210  }
211  else
212  {
213  bLoopNeeded = false;
214  getConsole().OLn("Could set advanced pixel format but not requested or already set.");
215  }
216  }
217  else
218  {
219  getConsole().OLn("Advanced pixel formats not available.");
220  bLoopNeeded = false;
221  }
222  } // while bLoopNeeded
223 
224  getConsole().OLn("");
225  getConsole().OLn("#### STAGE 3/3 ####");
226  getConsole().OLn("");
227 
228  printOGLerrorFull();
229  SetBasicThingsInOpenGL();
230  printOGLerrorFull();
231 
232  // GFX card drivers' default setting in 2015: off (if undefined by application), so we also set it to false initially
233  screen.SetVSyncEnabled(false);
234 
235  getConsole().OLn("");
236  getConsole().SOLn("> Fixed Pipeline HW Renderer initialized");
237  getConsole().OLnOO("");
238 
239  return 0;
240 
241 } // initialize()

481 void PRRERendererHWfixedPipeImpl::SetBasicThingsInOpenGL()
482 {
483  glEnable(GL_DEPTH_TEST); // some objects like sticked objects or 2-sided AND wireframed objects may temporarily modify this
484  glClearDepth(1.0f);
485  glDepthFunc(GL_LESS); // permanent setting, don't change in middle of rendering to get use of depth buffer optimisations
486  glDepthRange(0, 1); // permanent
487  glCullFace(GL_BACK); // 2-sided objects may temporarily modify this
488  glFrontFace(GL_CCW); // usually this is true, however some negative-scaled objects may temporarily modify this
489  glEnable(GL_SCISSOR_TEST);

491  glShadeModel(GL_SMOOTH);
492  //glShadeModel(GL_FLAT);

494  glEnable(GL_LIGHTING);
495  glEnable(GL_LIGHT0);
496  GLfloat white[4] = {1.0f, 1.0f, 1.0f, 1.0};
497  glMaterialf(GL_FRONT_AND_BACK, GL_SHININESS, 127);
498  glMaterialfv(GL_FRONT_AND_BACK, GL_SPECULAR, white);
499  glColorMaterial(GL_FRONT_AND_BACK, GL_AMBIENT_AND_DIFFUSE);

591 void PRRERendererHWfixedPipeImpl::Draw3DObjects(PRREIRenderer& renderer)
592 {
593  glEnable(GL_CULL_FACE);
594  glPolygonMode(GL_FRONT_AND_BACK, GL_FILL);
595  glEnable(GL_DEPTH_TEST);
596 
597  bool blended = true;
598  glPushMatrix();
599 
600  for (int i2 = 1; i2 < 3; i2++)
601  {
602  blended = !blended;
603  if ( blended )
604  {
605  glEnable(GL_BLEND);
606  glDepthMask(GL_FALSE);
607  }
608  else
609  {
610  glPushMatrix();
611  glDisable(GL_BLEND);
612  }
613  for (int i = 0; i < pObject3DMgr->getSize(); i++)
614  {
615 
616  PRREObject3D* const obj = (PRREObject3D*) pObject3DMgr->getAttachedAt(i);
617 
618  if ( obj == PGENULL )
619  continue;
620 
621  if ( (blended == PRREObject3DManager::isBlendFuncBlends(obj->getMaterial().getSourceBlendFunc(), obj->getMaterial().getDestinationBlendFunc()))
622  &&
623  ( obj->isVisible() )
624  &&
625  ( !obj->isStickedToScreen() )
626  )
627  {
628  glPushMatrix();
629  obj->Draw(bLighting);
630  glPopMatrix();
631  }
632  } // for i
633 
634  if ( !blended )
635  glPopMatrix();
636  else
637  glDepthMask(GL_TRUE);
638 
639  } // for i2
640 
641  glPopMatrix();
642 
643 }
