354 TPRREuint PRRERendererHWfixedPipeImpl::initialize(
355  TPRREuint width, TPRREuint height,
356  TPRRE_DISPLAY_MODES dmode, TPRREuint freq,
357  TPRREint cdepth, TPRREint zdepth, TPRREint stencil,
358  TPRREint samples,
359  HWND window)
360 {
361  TPRREint st = stencil;
362 
363  getConsole().OLn("PRRERendererHWfixedPipe::initialize(%d, %d, dmode, %d, %d, %d, %d, %d)",
364  width, height, freq, cdepth, zdepth, st, samples);
365  if ( isInitialized() )
366  {
367  getConsole().SOLn(" > already initialized!");
368  return 1;
369  }
370 
371  getConsole().OI();
372 
373  // we accept only 0 or 8 for stencil bits
374  if ( st != 0 )
375  {
376  st = 8;
377  getConsole().OLn("WARNING: forcing non-0 stencil to %d !", st);
378  }
379 
380  TPRREbool bRequestAdvancedPixelFormat = false; // will be set to true in the loop AFTER 1st GL init if antialiasing is requested AND if advanced formats available
381  TPRREbool bLoopNeeded = true; // will be set to false in the loop AFTER 1st GL init if antialiasing is NOT requested OR advanced formats are NOT available
382 
383  /*
384  Initialize OpenGL once or twice.
385  2nd init is needed if advanced pixel format is needed, eg. antialiasing is requested. That's the reason of the loop here.
386  This comes from the nature of OpenGL: it must be initialized to query available advanced pixel formats, then it may be reinitialized again with an
387  advanced pixel format, thus a renderer must handle screen, window and hwinfo instances init during its init.
388  */
389  while ( bLoopNeeded )
390  {
391  getConsole().OLn("");
392  getConsole().OLn("#### STAGE 1/3 ####");
393 
394  getConsole().OLn("screen settings ...");
395  screen.SetResolution(width, height);
396  screen.SetFullScreened( dmode != PRRE_WINDOWED ); // == PRRE_FULLSCREEN nem j, mert lesz 3. FS md is ...
397  screen.SetFreq( freq );
398  screen.SetColorBits( cdepth );
399  screen.SetDepthBits( zdepth );
400  screen.SetStencilBits( st );
401  screen.SetFSAAlevel( samples );
402 
403  getConsole().OLn("new PRREWindow...");
404  /*
405  A window may be created only once, even if advanced pixel format is requested, since a DC is required to be created only once.
406  This would mean that window instance can be created before renderer initialization. But based on SetPixelFormat() documentation, a
407  pixel format shouldn't be set to a window multiple times. This means that we actually have to recreate a window if an advanced pixel
408  format is to be set!
409  */
410  if ( !wnd.initialize(screen.getResWidth(), screen.getResHeight(), dmode != PRRE_WINDOWED, "Window", window) )
411  {
412  getConsole().EOLnOO("ERROR: wnd.initialize() failed!");
413  shutdown();
414  return 1;
415  }
416  getConsole().SOLn("done!");
417 
418  if ( !screen.applyDisplaySettings(wnd.getWndDC(), bRequestAdvancedPixelFormat ? PRRE_SCREEN_PF_ADVANCED : PRRE_SCREEN_PF_SIMPLE) )
419  {
420  getConsole().EOLnOO("ERROR: applyDisplaySettings() failed!");
421  shutdown();
422  return 1;
423  }
424 
425  getConsole().OLn("");
426  getConsole().OLn("#### STAGE 2/3 ####");
427  if ( !initializeOpenGL(wnd.getWndDC()) )
428  {
429  getConsole().OO();
430  shutdown();
431  return 1;
432  }
433 
434  hwInfo.Initialize(
435  rc, wnd.getWndDC(),
436  screen.getResWidth(), screen.getResHeight(),
437  screen.getColorBits(), screen.getDepthBits(),
438  screen.getStencilBits(), screen.getFSAAlevel() );
439 
440  if ( hwInfo.getVideo().isAdvancedPixelFormatSetSupported() )
441  {
442  if ( (samples > 0) && !bRequestAdvancedPixelFormat )
443  {
444  getConsole().OLn("Advanced pixel format requested, trying ...");
445  // following call can be removed after screen.applyDisplaySettings() is modified to automatically
446  // query advanced pixel formats at first call ... see PGE-History.txt's relevant RFR task!
447  screen.applyDisplaySettings( wnd.getWndDC() );
448  bRequestAdvancedPixelFormat = true;
449  shutdown();
450  }
451  else
452  {
453  bLoopNeeded = false;
454  getConsole().OLn("Could set advanced pixel format but not requested or already set.");
455  }
456  }
457  else
458  {
459  getConsole().OLn("Advanced pixel formats not available.");
460  bLoopNeeded = false;
461  }
462  } // while bLoopNeeded
463 
464  getConsole().OLn("");
465  getConsole().OLn("#### STAGE 3/3 ####");
466  getConsole().OLn("");
467 
468  printOGLerrorFull();
469  SetBasicThingsInOpenGL();
470  printOGLerrorFull();
471  PRREGLsnippets::Init();
472 
473  timeDurationStart.tv_sec = 0;
474  timeDurationStart.tv_usec = 0;
475 
476  // GFX card drivers' default setting in 2015: off (if undefined by application), so we also set it to false initially
477  screen.SetVSyncEnabled(false);
478 
479  getConsole().OLn("");
480  getConsole().SOLn("> Fixed Pipeline HW Renderer initialized");
481  getConsole().OLnOO("");
482 
483  return 0;
484 
485 } // initialize()

629

1185 {
1186  enum Order
1187  {
1188  NEAREST_TO_FAREST, /* front-to-back */
1189  FAREST_TO_NEAREST /* back-to-front */
1190  };
1191

1194  Order order;
1195  bool log;
1196 
1197  // std::sort() uses this to determine if a is less than b
1198  bool operator()(PRREObject3D* a, PRREObject3D* b) const
1199  {
1200  const PRREVector vDistanceToCam_a = a->getPosVec() - vPosCam;
1201  const PRREVector vDistanceToCam_b = b->getPosVec() - vPosCam;
1202  const TPRREfloat fZdistanceToCam_a = vDistanceToCam_a.getDotProduct(vFwdCam);
1203  const TPRREfloat fZdistanceToCam_b = vDistanceToCam_b.getDotProduct(vFwdCam);
1204  if ( log )
1205  {
1206  CConsole::getConsoleInstance().OLn("Comparator(%s, %s)", a->getName().c_str(), b->getName().c_str());
1207  CConsole::getConsoleInstance().OLn(" pos a: [%f, %f, %f], pos b: [%f, %f, %f], zdist a: %f, zdist b: %f",
1208  a->getPosVec().getX(), a->getPosVec().getY(), a->getPosVec().getZ(),
1209  b->getPosVec().getX(), b->getPosVec().getY(), b->getPosVec().getZ(),
1210  fZdistanceToCam_a, fZdistanceToCam_b);
