507 TPRREint PRREManager::PRREManagerImpl::preAlloc(TPRREint count)
508 {
509  _pOwner->getConsole().OLn("PRREManager::preAlloc(%d) ...", count);
510  if ( count > 0 )
511  {
512  TPRREint prev_h = nManagedAllocated;
513  nManagedAllocated += count;
514  PRREManaged** pNewStorage = PGENULL;
515  try
516  {
517  pNewStorage = new PRREManaged*[nManagedAllocated];
518  }
519  catch (const std::bad_alloc&)
520  {
521  _pOwner->getConsole().EOLnOO("ERROR: failed to alloc new storage!");
522  return -1;
523  }
524  memcpy(pNewStorage, pManageds, prev_h*sizeof(PRREManaged*));
525  delete[] pManageds;
526  pManageds = pNewStorage;
527  for (TPRREint i = prev_h; i < nManagedAllocated; i++)
528  pManageds[i] = PGENULL;
529  return prev_h;
530  }
531 
532  _pOwner->getConsole().EOLn("PRREManager::preAlloc(%d) ERROR!", count);
533  return -1;
534 } // preAlloc()
