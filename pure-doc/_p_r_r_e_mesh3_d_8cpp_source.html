190 void PRREMesh3D::PRREMesh3DImpl::RecalculateSize()
191 {
192  if ( isLevel1() )
193  {
194  // we are parent, ask our submeshes to determine their sizes and object-local positions
195  for (TPRREint i = 0; i < _pOwner->getCount(); i++)
196  ((PRREMesh3D*)_pOwner->getAttachedAt(i))->RecalculateSize();
197 
198  // now we can determine our own size as parent based on min and max subobject positions and size
199  TXYZ minvals, maxvals;
200  minvals.x = ((PRREMesh3D*)_pOwner->getAttachedAt(0))->getPosVec().getX() - ((PRREMesh3D*)_pOwner->getAttachedAt(0))->getSizeVec().getX() / 2.0f;
201  minvals.y = ((PRREMesh3D*)_pOwner->getAttachedAt(0))->getPosVec().getY() - ((PRREMesh3D*)_pOwner->getAttachedAt(0))->getSizeVec().getY() / 2.0f;
202  minvals.z = ((PRREMesh3D*)_pOwner->getAttachedAt(0))->getPosVec().getZ() - ((PRREMesh3D*)_pOwner->getAttachedAt(0))->getSizeVec().getZ() / 2.0f;
203  maxvals.x = -minvals.x;
204  maxvals.y = -minvals.y;
205  maxvals.z = -minvals.z;
206 
207  for (TPRREint i = 1; i < _pOwner->getCount(); i++)
208  {
209  if ( minvals.x > ((PRREMesh3D*)_pOwner->getAttachedAt(i))->getPosVec().getX() - ((PRREMesh3D*)_pOwner->getAttachedAt(i))->getSizeVec().getX() / 2.0f )
210  minvals.x = ((PRREMesh3D*)_pOwner->getAttachedAt(i))->getPosVec().getX() - ((PRREMesh3D*)_pOwner->getAttachedAt(i))->getSizeVec().getX() / 2.0f;
211  if ( maxvals.x < ((PRREMesh3D*)_pOwner->getAttachedAt(i))->getPosVec().getX() + ((PRREMesh3D*)_pOwner->getAttachedAt(i))->getSizeVec().getX() / 2.0f )
212  maxvals.x = ((PRREMesh3D*)_pOwner->getAttachedAt(i))->getPosVec().getX() + ((PRREMesh3D*)_pOwner->getAttachedAt(i))->getSizeVec().getX() / 2.0f;
213  if ( minvals.y > ((PRREMesh3D*)_pOwner->getAttachedAt(i))->getPosVec().getY() - ((PRREMesh3D*)_pOwner->getAttachedAt(i))->getSizeVec().getY() / 2.0f )
214  minvals.y = ((PRREMesh3D*)_pOwner->getAttachedAt(i))->getPosVec().getY() - ((PRREMesh3D*)_pOwner->getAttachedAt(i))->getSizeVec().getY() / 2.0f;
215  if ( maxvals.y < ((PRREMesh3D*)_pOwner->getAttachedAt(i))->getPosVec().getY() + ((PRREMesh3D*)_pOwner->getAttachedAt(i))->getSizeVec().getY() / 2.0f )
216  maxvals.y = ((PRREMesh3D*)_pOwner->getAttachedAt(i))->getPosVec().getY() + ((PRREMesh3D*)_pOwner->getAttachedAt(i))->getSizeVec().getY() / 2.0f;
217  if ( minvals.z > ((PRREMesh3D*)_pOwner->getAttachedAt(i))->getPosVec().getZ() - ((PRREMesh3D*)_pOwner->getAttachedAt(i))->getSizeVec().getZ() / 2.0f )
218  minvals.z = ((PRREMesh3D*)_pOwner->getAttachedAt(i))->getPosVec().getZ() - ((PRREMesh3D*)_pOwner->getAttachedAt(i))->getSizeVec().getZ() / 2.0f;
219  if ( maxvals.z < ((PRREMesh3D*)_pOwner->getAttachedAt(i))->getPosVec().getZ() + ((PRREMesh3D*)_pOwner->getAttachedAt(i))->getSizeVec().getZ() / 2.0f )
220  maxvals.z = ((PRREMesh3D*)_pOwner->getAttachedAt(i))->getPosVec().getZ() + ((PRREMesh3D*)_pOwner->getAttachedAt(i))->getSizeVec().getZ() / 2.0f;
221  }
222 
223  vSize.SetX( abs( minvals.x - maxvals.x ) );
224  vSize.SetY( abs( minvals.y - maxvals.y ) );
225  vSize.SetZ( abs( minvals.z - maxvals.z ) );
226 
227  // since we are parent we dont need to modify our world-space position at all based on our subobjects
228  }
229 
230  // as level-2 in theory we have geometry but check it for sure
231  if ( (nVertices_h == 0) || (pVertices == PGENULL) )
232  {
233  _pOwner->getManagedConsole().EOLn("PRREMesh3D::PRREMesh3DImpl::RecalculateSize() in level-2 but vertex data is not present!");
234  return;
235  }
236 
237  // we are subobject, so we need to calculate our own size and object-local position based on min and max vertex positions
238 
239  TXYZ minvals, maxvals;
240  minvals.x = pVertices[0].x;
241  minvals.y = pVertices[0].y;
242  minvals.z = pVertices[0].z;
243  maxvals.x = minvals.x;
244  maxvals.y = minvals.y;
245  maxvals.z = minvals.z;
246 
247  for (TPRREuint i = 1; i < nVertices_h; i++)
248  {
249  if ( minvals.x > pVertices[i].x ) minvals.x = pVertices[i].x;
250  if ( maxvals.x < pVertices[i].x ) maxvals.x = pVertices[i].x;
251  if ( minvals.y > pVertices[i].y ) minvals.y = pVertices[i].y;
252  if ( maxvals.y < pVertices[i].y ) maxvals.y = pVertices[i].y;
253  if ( minvals.z > pVertices[i].z ) minvals.z = pVertices[i].z;
254  if ( maxvals.z < pVertices[i].z ) maxvals.z = pVertices[i].z;
255  }
256 
257  vSize.SetX( abs( minvals.x - maxvals.x ) );
258  vSize.SetY( abs( minvals.y - maxvals.y ) );
259  vSize.SetZ( abs( minvals.z - maxvals.z ) );
260 
261  vPos.SetX( minvals.x + vSize.getX() / 2.0f );
262  vPos.SetY( minvals.y + vSize.getY() / 2.0f );
263  vPos.SetZ( minvals.z + vSize.getZ() / 2.0f );
264 
265 } // RecalculateSize()

305 TPRREbool PRREMesh3D::PRREMesh3DImpl::cannibalize(PRREMesh3D& victim)
306 {
307  // Note that we cannot easily copy victim's Impl since copy ctor and assignment operators are empty!
308  // Thus the following assignment cannot be used: *(this->pImpl) = *(victim.pImpl)
309  // We wouldn't even want to use copy ctor or assignment here since we do not want any array copy.
310  // That is why this function is written, to "steal" the arrays of the victim so victim will be "empty".
311  // In C++11 this could be done with a move ctor.
312 
313  // Note: this function could be recursive and be called for level-1 mesh only, it would get the job
314  // done for all the submeshes too. But now it was easier to implement this silly way so it needs to be
315  // invoked manually for all submeshes too.
316 
317  // we should not have any submesh at this point, and no geometry yet
318  if ( (_pOwner->getCount() != 0) || (pVertices != PGENULL) || (nVertices_h > 0) )
319  {
320  _pOwner->getManagedConsole().EOLn(
321  "PRREMesh3D::cannibalize() ERROR: cannibalizer has non-zero managed count (%d) or has geometry (%d)!", _pOwner->getCount(), nVertices_h);
322  return false;
323  }
324 
325  // at this point we do not need to deal with the Manager or FiledManager parts
326 
327  _pOwner->SetName(victim.getName()); // copy the Managed part
328  _pOwner->SetFilename(victim.getFilename()); // copy the FiledManaged part
329 
330  primitiveFormat = victim.pImpl->primitiveFormat;
331 
332  pVertices = victim.pImpl->pVertices;
333  pNormals = victim.pImpl->pNormals;
334  pVertexIndices = victim.pImpl->pVertexIndices;
335  nMinVertexIndex = victim.pImpl->nMinVertexIndex;
336  nMaxVertexIndex = victim.pImpl->nMaxVertexIndex;
337  nIndicesType = victim.pImpl->nIndicesType;
338 
339  nVertices_h = victim.pImpl->nVertices_h;
340  nVertexIndices_h = victim.pImpl->nVertexIndices_h;
341  nFaces_h = victim.pImpl->nFaces_h;
342 
343  vPos = victim.pImpl->vPos;
344  vSize = victim.pImpl->vSize;
345 
346  pMaterial = victim.pImpl->pMaterial;
347 
348  victim.pImpl->pVertices = PGENULL;
349  victim.pImpl->pNormals = PGENULL;
350  victim.pImpl->pVertexIndices = PGENULL;
351  victim.pImpl->pMaterial = PGENULL;
352  victim.pImpl->nMinVertexIndex = UINT_MAX;
353  victim.pImpl->nMaxVertexIndex = 0;
354  victim.pImpl->nIndicesType = GL_UNSIGNED_BYTE;
355  victim.pImpl->nVertices_h = 0;
356  victim.pImpl->nVertexIndices_h = 0;
357  victim.pImpl->nFaces_h = 0;
358 
359  return true;
360 } // cannibalize()
