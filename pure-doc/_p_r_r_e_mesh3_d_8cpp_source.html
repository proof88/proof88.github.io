268 void PRREMesh3D::PRREMesh3DImpl::RecalculateSize()
269 {
270  if ( isLevel1() )
271  {
272  // we are parent, ask our submeshes to determine their sizes and object-local positions
273  for (TPRREint i = 0; i < _pOwner->getCount(); i++)
274  ((PRREMesh3D*)_pOwner->getAttachedAt(i))->RecalculateSize();
275 
276  // now we can determine our own size as parent based on min and max subobject positions and size
277  TXYZ minvals, maxvals;
278  const PRREMesh3D* subMesh = (PRREMesh3D*)_pOwner->getAttachedAt(0);
279  minvals.x = subMesh->getPosVec().getX() - subMesh->getSizeVec().getX() / 2.0f;
280  minvals.y = subMesh->getPosVec().getY() - subMesh->getSizeVec().getY() / 2.0f;
281  minvals.z = subMesh->getPosVec().getZ() - subMesh->getSizeVec().getZ() / 2.0f;
282  maxvals.x = subMesh->getPosVec().getX() + subMesh->getSizeVec().getX() / 2.0f;
283  maxvals.y = subMesh->getPosVec().getY() + subMesh->getSizeVec().getY() / 2.0f;
284  maxvals.z = subMesh->getPosVec().getZ() + subMesh->getSizeVec().getZ() / 2.0f;
285 
286  for (TPRREint i = 1; i < _pOwner->getCount(); i++)
287  {
288  subMesh = (PRREMesh3D*)_pOwner->getAttachedAt(i);
289  if ( minvals.x > subMesh->getPosVec().getX() - subMesh->getSizeVec().getX() / 2.0f )
290  minvals.x = subMesh->getPosVec().getX() - subMesh->getSizeVec().getX() / 2.0f;
291  if ( maxvals.x < subMesh->getPosVec().getX() + subMesh->getSizeVec().getX() / 2.0f )
292  maxvals.x = subMesh->getPosVec().getX() + subMesh->getSizeVec().getX() / 2.0f;
293  if ( minvals.y > subMesh->getPosVec().getY() - subMesh->getSizeVec().getY() / 2.0f )
294  minvals.y = subMesh->getPosVec().getY() - subMesh->getSizeVec().getY() / 2.0f;
295  if ( maxvals.y < subMesh->getPosVec().getY() + subMesh->getSizeVec().getY() / 2.0f )
296  maxvals.y = subMesh->getPosVec().getY() + subMesh->getSizeVec().getY() / 2.0f;
297  if ( minvals.z > subMesh->getPosVec().getZ() - subMesh->getSizeVec().getZ() / 2.0f )
298  minvals.z = subMesh->getPosVec().getZ() - subMesh->getSizeVec().getZ() / 2.0f;
299  if ( maxvals.z < subMesh->getPosVec().getZ() + subMesh->getSizeVec().getZ() / 2.0f )
300  maxvals.z = subMesh->getPosVec().getZ() + subMesh->getSizeVec().getZ() / 2.0f;
301  }
302 
303  vSize.SetX( abs( minvals.x - maxvals.x ) );
304  vSize.SetY( abs( minvals.y - maxvals.y ) );
305  vSize.SetZ( abs( minvals.z - maxvals.z ) );
306 
307  // since we are parent we dont need to modify our world-space position at all based on our subobjects
308 
309  vRelPos.SetX( minvals.x + vSize.getX() / 2.0f );
310  vRelPos.SetY( minvals.y + vSize.getY() / 2.0f );
311  vRelPos.SetZ( minvals.z + vSize.getZ() / 2.0f );
312 
313  return;
314  }
315 
316  // as level-2 in theory we have geometry but check it for sure
317  if ( (nVertices_h == 0) || (pVertices == PGENULL) )
318  {
319  _pOwner->getManagedConsole().EOLn("PRREMesh3D::PRREMesh3DImpl::RecalculateSize() in level-2 but vertex data is not present!");
320  return;
321  }
322 
323  // we are subobject, so we need to calculate our own size and object-local position based on min and max vertex positions
324 
325  TXYZ minvals, maxvals;
326  minvals.x = pVertices[0].x;
327  minvals.y = pVertices[0].y;
328  minvals.z = pVertices[0].z;
329  maxvals.x = minvals.x;
330  maxvals.y = minvals.y;
331  maxvals.z = minvals.z;
332 
333  for (TPRREuint i = 1; i < nVertices_h; i++)
334  {
335  if ( minvals.x > pVertices[i].x ) minvals.x = pVertices[i].x;
336  if ( maxvals.x < pVertices[i].x ) maxvals.x = pVertices[i].x;
337  if ( minvals.y > pVertices[i].y ) minvals.y = pVertices[i].y;
338  if ( maxvals.y < pVertices[i].y ) maxvals.y = pVertices[i].y;
339  if ( minvals.z > pVertices[i].z ) minvals.z = pVertices[i].z;
340  if ( maxvals.z < pVertices[i].z ) maxvals.z = pVertices[i].z;
341  }
342 
343  vSize.SetX( abs( minvals.x - maxvals.x ) );
344  vSize.SetY( abs( minvals.y - maxvals.y ) );
345  vSize.SetZ( abs( minvals.z - maxvals.z ) );
346 
347  vPos.SetX( minvals.x + vSize.getX() / 2.0f );
348  vPos.SetY( minvals.y + vSize.getY() / 2.0f );
349  vPos.SetZ( minvals.z + vSize.getZ() / 2.0f );
350 
351  // vRelPos is not updated for level-2, as default [0,0,0] is always valid since we set position above based on minvals and size!
352 
353 } // RecalculateSize()

393 TPRREbool PRREMesh3D::PRREMesh3DImpl::cannibalize(PRREMesh3D& victim)
394 {
395  // Note that we cannot easily copy victim's Impl since copy ctor and assignment operators are empty!
396  // Thus the following assignment cannot be used: *(this->pImpl) = *(victim.pImpl)
397  // We wouldn't even want to use copy ctor or assignment here since we do not want any array copy.
398  // That is why this function is written, to "steal" the arrays of the victim so victim will be "empty".
399  // In C++11 this could be done with a move ctor.
400 
401  // Note: this function could be recursive and be called for level-1 mesh only, it would get the job
402  // done for all the submeshes too. But now it was easier to implement this silly way so it needs to be
403  // invoked manually for all submeshes too.
404 
405  // we should not have any submesh at this point, and no geometry yet
406  if ( (_pOwner->getCount() != 0) || (pVertices != PGENULL) || (nVertices_h > 0) )
407  {
408  _pOwner->getManagedConsole().EOLn(
409  "PRREMesh3D::cannibalize() ERROR: cannibalizer has non-zero managed count (%d) or has geometry (%d)!", _pOwner->getCount(), nVertices_h);
410  return false;
411  }
412 
413  // at this point we do not need to deal with the Manager or FiledManager parts
414 
415  _pOwner->SetName(victim.getName()); // copy the Managed part
416  _pOwner->SetFilename(victim.getFilename()); // copy the FiledManaged part
417 
418  primitiveFormat = victim.pImpl->primitiveFormat;
419 
420  pVertices = victim.pImpl->pVertices;
421  pNormals = victim.pImpl->pNormals;
422  pVertexIndices = victim.pImpl->pVertexIndices;
423  nMinVertexIndex = victim.pImpl->nMinVertexIndex;
424  nMaxVertexIndex = victim.pImpl->nMaxVertexIndex;
425  nIndicesType = victim.pImpl->nIndicesType;
426 
427  nVertices_h = victim.pImpl->nVertices_h;
428  nVertexIndices_h = victim.pImpl->nVertexIndices_h;
429  nFaces_h = victim.pImpl->nFaces_h;
430 
431  vPos = victim.pImpl->vPos;
432  vRelPos = victim.pImpl->vRelPos;
433  vSize = victim.pImpl->vSize;
434 
435  pMaterial = victim.pImpl->pMaterial;
436  pMaterial->SetUtiliser(this->_pOwner);
437 
438  victim.pImpl->pVertices = PGENULL;
439  victim.pImpl->pNormals = PGENULL;
440  victim.pImpl->pVertexIndices = PGENULL;
441  victim.pImpl->pMaterial = PGENULL;
442  victim.pImpl->nMinVertexIndex = UINT_MAX;
443  victim.pImpl->nMaxVertexIndex = 0;
444  victim.pImpl->nIndicesType = GL_UNSIGNED_BYTE;
445  victim.pImpl->nVertices_h = 0;
446  victim.pImpl->nVertexIndices_h = 0;
447  victim.pImpl->nFaces_h = 0;
448 
449  return true;
450 } // cannibalize()
