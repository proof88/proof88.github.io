177 unsigned long PRREuiText::calcHash(const std::string& text, int x, int y, int height)
178 {
179  unsigned long hash = 5381;
180 
181  for (std::string::size_type i = 0; i < text.length(); i++)
182  hash = ((hash << 5) + hash) + (int)(text[i]) + x*3000 + y + height; /* hash * 33 + c + x + y + height*/
183
