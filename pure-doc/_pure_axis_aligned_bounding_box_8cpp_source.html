69TPureBool PureAxisAlignedBoundingBox::isInside(const PureVector& p) const
70{
71 if ( (size.getX() == 0.f) || (size.getY() == 0.f) || (size.getZ() == 0.f) )
72 return false;
73
74 return ( (p.getX() >= (pos.getX() - size.getX()/2.f)) && (p.getX() <= (pos.getX() + size.getX()/2.f)) ) &&
75 ( (p.getY() >= (pos.getY() - size.getY()/2.f)) && (p.getY() <= (pos.getY() + size.getY()/2.f)) ) &&
76 ( (p.getZ() >= (pos.getZ() - size.getZ()/2.f)) && (p.getZ() <= (pos.getZ() + size.getZ()/2.f)) );
77}

107void PureAxisAlignedBoundingBox::ExtendBy(const PureVector& p)
108{
109 if ( (size.getX() == 0.f) || (size.getY() == 0.f) || (size.getZ() == 0.f) )
110 {
111 return;
112 }
113
114 const PureVector oldPos = pos;
115
116 if ( p.getX() < pos.getX() - size.getX()/2.f ) {
117 pos.SetX( (p.getX() + pos.getX()+size.getX()/2.f) / 2.f );
118 size.SetX( ( oldPos.getX() + size.getX()/2.f ) - p.getX());
119 } else if ( p.getX() > pos.getX() + size.getX()/2.f ) {
120 pos.SetX( (p.getX() + pos.getX()-size.getX()/2.f) / 2.f );
121 size.SetX( p.getX() - ( oldPos.getX() - size.getX()/2.f ));
122 }
123
124 if ( p.getY() < pos.getY() - size.getY()/2.f ) {
125 pos.SetY( (p.getY() + pos.getY()+size.getY()/2.f) / 2.f );

127 } else if ( p.getY() > pos.getY() + size.getY()/2.f ) {
