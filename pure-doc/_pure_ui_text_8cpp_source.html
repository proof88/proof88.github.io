151void PureUiText::PrintText() const
152{
153 /* SW rendering currently not supported here */
154 if ( !PureHwInfo::get().getVideo().isAcceleratorDetected() )
155 return;
156
157 if ( uiFont->getListBase() == 0 )
158 return;
159
160 if (bDropShadow)
161 {
162 glColor4f(0.f, 0.f, 0.f, clr.getAlphaAsFloat());
163 glRasterPos2f((GLfloat)(_x+1), (GLfloat)(_y-1));
164 glPushAttrib(GL_LIST_BIT);
165
166 glListBase(uiFont->getListBase());
167 glCallLists(sText.length(), GL_UNSIGNED_BYTE, sText.c_str());
168
169 glPopAttrib();
170 }
171
172 glColor4f(clr.getRedAsFloat(), clr.getGreenAsFloat(), clr.getBlueAsFloat(), clr.getAlphaAsFloat());
173 glRasterPos2f((GLfloat)_x, (GLfloat)_y);
174
175 glPushAttrib(GL_LIST_BIT);
176
177 glListBase(uiFont->getListBase());
178 glCallLists(sText.length(), GL_UNSIGNED_BYTE, sText.c_str());
179
180 glPopAttrib();
181}

201unsigned long PureUiText::calcHash(const std::string& text, int x, int y, int height)
202{
203 unsigned long hash = 5381;
204
205 for (std::string::size_type i = 0; i < text.length(); i++)
206 hash = ((hash << 5) + hash) + (int)(text[i]) + x*3000 + y + height; /* hash * 33 + c + x + y + height*/
207
208 return hash;
209}
