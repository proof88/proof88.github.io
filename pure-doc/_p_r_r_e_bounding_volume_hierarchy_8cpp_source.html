58 PRREBoundingVolumeHierarchy* PRREBoundingVolumeHierarchy::insertObject(const PRREObject3D& obj)
59 {
60  // Note 1:
61  // most of the time dynamic_cast should be used for downcasting, however in this specific case,
62  // we can guarantee that the returned Octree node is actually a derived PRREBoundingVolumeHierarchy node,
63  // since we are in PRREBoundingVolumeHierarchy method. So we can use static_cast for faster cast.
64 
65  // Note 2: although we call ancestor's insertObject(), due to polymorphism it will still call our
66  // derived Subdivide() method, so children will be PRREBoundingVolumeHierarchy instances!
67  PRREBoundingVolumeHierarchy* node = static_cast<PRREBoundingVolumeHierarchy*>(PRREOctree::insertObject(obj));
68  if ( PGENULL == node )
69  {
70  return PGENULL;
71  }
72

74  node->aabb.ExtendBy(obj_bbox);
75 
76  PRREBoundingVolumeHierarchy* prevChild = node;
77  PRREBoundingVolumeHierarchy* p = static_cast<PRREBoundingVolumeHierarchy*>(node->parent);
78  while (p != PGENULL)

80  p->aabb.ExtendBy(prevChild->aabb);

145 {

147  try

149  for (TPRREuint i = 0; i < 8; i++)
