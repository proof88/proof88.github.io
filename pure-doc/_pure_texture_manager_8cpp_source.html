348PureTextureManager::PureTextureManagerImpl::PureTextureManagerImpl(PureTextureManager* owner)
349{
350 _pOwner = owner;
351 _pOwner->getConsole().OLnOI("PureTextureManager() ...");
352 bInited = false;
353 nRunningCounter = 0;
354 filtDefIsoMin = PURE_ISO_LINEAR;
355 filtDefIsoMag = PURE_ISO_LINEAR;
356 filtDefAniso = PURE_ANISO_1X;
357 bAlwaysCreateNewFromFile = false;
358 bKeepPixelsInSysRAM = pHWInfo.isInitialized() ? ( pHWInfo.getVideo().isAcceleratorDetected() ? false : true ) : false;
359 nAnisoFilterMax = pHWInfo.isInitialized() ? pHWInfo.getVideo().getMaximumAnisoFiltering() : 1.0f;
360 bAllowHWMIPmapGen = pHWInfo.isInitialized() ? pHWInfo.getVideo().isHardwareMipMapGenerationSupported() : false;
361 bAllowBGRAext = pHWInfo.isInitialized() ? pHWInfo.getVideo().isNativeDIBFormatSupported() : false;
362 if ( pHWInfo.isInitialized() )
363 {
364 if ( pHWInfo.getVideo().isTextureCompressionSupported() )
365 {
366 pglHint(GL_TEXTURE_COMPRESSION_HINT_ARB, GL_NICEST);
367 }
368 }
369 texComprDef = PURE_TC_NONE;
370 bDefBorder = false;
371 twDefS = twDefT = PURE_TW_REPEAT;
372
373 if ( !pHWInfo.isInitialized() )
374 {
375 _pOwner->getConsole().EOLnOO("No HardwareInfo!");
376 return;
377 }
378
379 bInited = true;
380 _pOwner->getConsole().SOLnOO("Done!");
381}

396TPureBool PureTextureManager::PureTextureManagerImpl::generateAndUploadTexture(PureTexture& texture)
397{
398 if ( (getDefaultMinFilteringMode() == PURE_ISO_NEAREST) || (getDefaultMinFilteringMode() == PURE_ISO_LINEAR) )
399 texture.pImpl->nMIPmapCount = 1;
400 else
401 texture.pImpl->nMIPmapCount = PureTextureManager::getMIPmapCount( texture.getWidth(), texture.getHeight() );
402
403 if ( pHWInfo.getVideo().isAcceleratorDetected() )
404 {
405 if ( !pglGenTextures(1, &(texture.pImpl->nInternalNum)) )
406 {
407 return false;
408 }
409
410 if ( !pglBindTexture(GL_TEXTURE_2D, texture.pImpl->nInternalNum) )
411 {
412 return false;
413 }
414
415 if ( !pglTexEnvi(GL_TEXTURE_ENV, GL_TEXTURE_ENV_MODE, GL_MODULATE) )
416 {
417 return false;
418 }
419 }
420
421 bool b = true;
422 b &= texture.setMagFilteringMode( getDefaultMagFilteringMode() );
423 b &= texture.setMinFilteringMode( getDefaultMinFilteringMode() );
424 b &= texture.setAnisoFilteringMode( getDefaultAnisoFilteringMode() );
425 b &= texture.setTextureWrappingMode( getDefaultTextureWrappingModeS(), getDefaultTextureWrappingModeT() );
426 b &= texture.setBorder( getDefaultBorder() );
427 b &= texture.setBorderColor( getDefaultBorderColor() );
428 if ( !b )
429 {
430 _pOwner->getConsole().EOLn("ERROR: failed to set an attribute to the texture!");
431 return false;
432 }
433
434 if ( !texture.uploadPixels() )
435 {
436 _pOwner->getConsole().EOLn("ERROR: uploadPixels() failed!");
437 return false;
438 }
439
440 return true;
441}

789PureTexture* PureTextureManager::createTextureFromImage(const PureImage& img)
790{
791 getConsole().OLnOI("PureTextureManager::createTextureFromImage(...)");
792
793 if ( !isInitialized() )
794 {
795 getConsole().OO();
796 return PGENULL;
797 }
798
799 PureTexture* texture = isLazyInstancingEnabled() ? (PureTexture*) getByFilename( img.getFilename().c_str() ) : PGENULL;
800
801 if ( texture != PGENULL )
802 { // we already have a texture with this filename
803 getConsole().SOLn("> Found loaded texture, returning.");
804 getConsole().OLnOO("");
805 return texture;
806 }
807
808 getConsole().OLn("texture == PGENULL");
809 try
810 {
811 texture = new PureTexture(img);
812 }
813 catch (const std::bad_alloc&)
814 {
815 getConsole().EOLn("ERROR: Failed to instantiate new PureTexture!");
816 getConsole().OLnOO("");
817 return PGENULL;
818 }
819
820 Attach(*texture);
821
822 if ( !pImpl->generateAndUploadTexture(*texture) )
823 {
824 getConsole().EOLn("ERROR: generateAndUploadTexture() failed!");
825 getConsole().OLnOO("");
826 DeleteAttachedInstance(*texture);
827 return PGENULL;
828 }
829
830 if ( !isPixelPreservingEnabled() )
831 texture->FlushResources();
832
833 texture->SetName("Texture " + std::to_string(pImpl->nRunningCounter++));
834
835 getConsole().OLn("Texture nWidth and nHeight: %d %d", texture->getWidth(), texture->getHeight());
836 getConsole().OLn("final nSize is %d Byte(s).", texture->pImpl->nSize);
837 getConsole().OLn("internalnum = %d", texture->pImpl->nInternalNum);
838 getConsole().OLn("");
839 getConsole().SOLn("> Texture created, name: %s!", texture->getName().c_str());
840 getConsole().OOOLn("");
841 return texture;
842} // createTextureFromImage()

859PureTexture* PureTextureManager::createFromFile(const char* filename)
860{
861 if ( !isInitialized() )
862 {
863 return PGENULL;
864 }
865
866 getConsole().OLnOI("PureTextureManager::createFromFile(\"%s\")", filename);
867
868 if ( filename == PGENULL )
869 {
870 getConsole().EOLnOO("ERROR: input was NULL, returning PGENULL");
871 getConsole().OLn("");
872 return PGENULL;
873 }
874
875 PureTexture* texture = isLazyInstancingEnabled() ? (PureTexture*) getByFilename( filename ) : PGENULL;
876
877 if ( texture != PGENULL )
878 { // we already have a texture with this filename
879 getConsole().SOLnOO("> Found loaded texture, returning: %s", texture->getName().c_str());
880 getConsole().OLn("");
881 return texture;
882 }
883
884 getConsole().OI();
885 PureImage* tmpImg = PureImageManager::createFromFile(filename);
886 getConsole().OO();
887 if ( tmpImg == PGENULL )
888 {
889 getConsole().EOLnOO("ERROR: failed to create Image from file!");
890 getConsole().OLn("");
891 return PGENULL;
892 }
893
894 try
895 {
896 texture = new PureTexture(*tmpImg);
897 PureImage* textureAsImage = (PureImage*)texture;
898 if ( !textureAsImage->pImpl->cannibalize(*tmpImg) )
899 {
900 const std::string sErrMsg = "cannibalize() failed!";
901 throw std::runtime_error(sErrMsg);
902 }
903 }
904 catch (const std::exception& e)
905 {
906 getConsole().EOLn("ERROR: Failed to create or fill new PureTexture: %s!", e.what());
907 DeleteAttachedInstance(*tmpImg);
908 getConsole().OLnOO("");
909 return PGENULL;
910 }
911 DeleteAttachedInstance(*tmpImg);
912
913 Attach(*texture);
914
915 if ( !pImpl->generateAndUploadTexture(*texture) )
916 {
917 getConsole().EOLn("ERROR: generateAndUploadTexture() failed!");
918 getConsole().OLnOO("");
919 DeleteAttachedInstance(*texture);
920 return PGENULL;
921 }
922
923 if ( !isPixelPreservingEnabled() )
924 texture->FlushResources();
925
926 texture->SetName("Texture " + std::to_string(pImpl->nRunningCounter++));
