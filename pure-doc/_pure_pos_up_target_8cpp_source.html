150void PurePosUpTarget::SetRotation(TPureFloat x, TPureFloat y, TPureFloat z)
151{
152 const PureVector vDefTarget(0, 0, 1);
153 const PureVector vDefPos(0, 0, 0);
154 // Note: we set absolute rotation here so our initial Target vec is the default in +Z direction and Pos vec is zero.
155 // To implement Rotate() instead of SetRotation(), use the current vTarget and vPos vectors to calculate tmpView.
156 PureVector tmpView(vDefTarget - vDefPos);
157 tmpView.Normalize();
158
159 PureTransformMatrix matRotX, matRotY, matRotZ;
160 matRotX.SetRotationX(x);
161 matRotY.SetRotationY(y);
162 matRotZ.SetRotationZ(z);
163 // todo: 3 matrices, are u kidding me?
164
165 tmpView *= matRotX * matRotY * matRotZ;
166 tmpView.Normalize();
167 vTarget = vPos + tmpView;
168}
