781TPureBool PureHwVideoImpl::initializeBase()
782{
783 EnumDisplayDevicesA(NULL, 0, &dispdev, NULL);
784 sVidNameWin = dispdev.DeviceString;
785
786 if ( hGLRC == NULL )
787 {
788 // app may be using software renderer
789 getConsole().OLn("GFX card: %s", sVidNameWin.c_str());
790 sharedSettings.Set(PURE_SSET_VSYNC_SUPPORTED, false);
791 getConsole().OO();
792 getConsole().OO();
793
794 return true;
795 }
796
797 if (bAlreadyInitializedOnceOGL)
798 {
799 getConsole().OLn("This process already initialized OpenGL during a previous initialization, no need to fetch pointers again!");
800 }
801 else
802 {
803 PureGLgetFunctionPointers();
804
805 getConsole().OLn("");
806 getConsole().OLn("GFX card: %s", sVidNameWin.c_str());
807
808 const GLubyte* ptmp = glGetString(GL_RENDERER);
809 if ( ptmp )
810 {
811 sVidNameOGL = (char*) ptmp;
812 getConsole().OLn("glRenderer = %s", sVidNameOGL.c_str());
813 }
814 else
815 {
816 getConsole().EOLn("ERROR: failed to query GL_RENDERER!");
817 return false;
818 }
819
820 if ( ptmp = glGetString(GL_VENDOR) )
821 {
822 sVidVendorOGL = (char*) ptmp;
823 getConsole().OLn("glVendor = %s", sVidVendorOGL.c_str());
824 }
825 else
826 {
827 getConsole().EOLn("ERROR: failed to query GL_VENDOR!");
828 return false;
829 }
830
831 if ( ptmp = glGetString(GL_VERSION) )
832 {
833 sVidVersionOGL = (char*) ptmp;
834 getConsole().OLn("glVersion = %s", sVidVersionOGL.c_str());
835 }
836 else
837 {
838 getConsole().EOLn("ERROR: failed to query GL_VERSION!");
839 return false;
840 }
841
842 if ( ptmp = glGetString(GL_SHADING_LANGUAGE_VERSION) )
843 {
844 std::string sVidVersionGLSL = (char*) ptmp;
845 sscanf((const char*) ptmp, "%d.%d", &nVidVersionGLSLmaj, &nVidVersionGLSLmin);
846 getConsole().OLn("glslVersion = %s (%d.%d)", sVidVersionGLSL.c_str(), nVidVersionGLSLmaj, nVidVersionGLSLmin);
847 }
848 else
849 {
850 getConsole().EOLn("WARNING: failed to query GL_SHADING_LANGUAGE_VERSION!");
851 }
852
853 // OGL error may be generated due to querying GL_SHADING_LANGUAGE_VERSION which may not be supported by all implementations
854 PureGLsnippets::ClearGLError();
855
856 if ( ptmp = glGetString(GL_EXTENSIONS) )
857 {
858 sVidFeaturesOGL = (char*) ptmp;
859 }
860 else
861 {
862 getConsole().EOLn("ERROR: failed to query GL_EXTENSIONS!");
863 return false;

1028void PureHwVideoImpl::printExtensionList(const std::string& features, const char* endstring)
1029{
1030#ifdef PGE_CCONSOLE_IS_ENABLED
1031 stringstream ssFeatures(features);
1032 string tmpSpacesString;
1033 int i = 0;
1034 int numspaces = 0;
1035
1036 const int nIndentOriginal = getConsole().getIndent();
1037 const int nLineLeftMargin = 2; // each line starts at this position
1038 const int nLineMax = 80; // assuming we have this number of chars available per line
1039 const int nLineEndMargin = 2; // need to leave this number of characters empty at end of line
1040 const int nLeftColMax = nLineMax/2 - nLineLeftMargin - nLineEndMargin; // left column have this width
1041
1042 // temporarily reset console indentation level so we have enough space
1043 getConsole().SetIndent(nLineLeftMargin);
1044
1045 // we print extensions in 2 columns
1046 do
1047 {
1048 string sub;
1049 ssFeatures >> sub;
1050
1051 // sub is empty when no more extension left in ssFeatures
1052 if ( sub.empty() )
1053 continue;
1054
1055 if ( i % 2 == 0 )
1056 {
1057 // left column in table
1058 // O(): CConsole known issue B) but we still use it here, reason is written a few lines later below ...
1059 getConsole().O("%s", sub.c_str());
1060 numspaces = nLeftColMax - sub.length();
1061 }
1062 else
1063 {
1064 // right column in table
1065 if ( (sub.length() <= (nLeftColMax+2)) && (numspaces > 0) )
1066 {
1067 tmpSpacesString = "";
1068 for (int j = 0; j < numspaces; j++)
1069 tmpSpacesString += ' ';
1070 // although usage of O() is not recommended when multiple threads are allowed to log due to
1071 // CConsole known issue B), I still leave this here ... because it is complicated to refactor
1072 // this part, because of another known issue D) which is about problem with how nbsp chars are
1073 // put into output html log file.
1074 getConsole().O(tmpSpacesString.c_str());
1075 }
1076 else
1077 {
1078 // too long extension needs to be written to next line (left column) to avoid line wrap
1079 // we fell here due to one of 2 reasons:
1080 // - sub.length() > 40 means our current string originally targeted to right column is too long
1081 // - numspaces <= 0 means the previous string already printed to left column is too long, did not leave enough space for us
1082 // either reason we need to jump to next line
1083 getConsole().OLn("");
1084 }
1085 getConsole().OLn("%s", sub.c_str());
1086 }
1087 i++;
1088 } while (ssFeatures);
1089
1090 getConsole().SetIndent(nIndentOriginal);
1091
1092 if ( i % 2 == 1 )
1093 getConsole().OLn("");
1094 getConsole().OLn("");
1095 getConsole().OLn(endstring, i);
1096 getConsole().OLn("");
1097#endif
1098} // printExtensionList()
