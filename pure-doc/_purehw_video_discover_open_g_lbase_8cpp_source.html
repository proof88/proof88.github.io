138bool PureHwVideoDiscoverOpenGLbase::printExtensionSupported(const char* title, const char* name, std::vector<void*>& functions)
139{
140 bool result;
141
142 if ( result = isExtensionSupported(name) )
143 {
144 for (std::size_t i = 0; (i < functions.size()) && result; i++)
145 result &= (functions[i] != PGENULL);
146
147 if ( result )
148 getConsole().SOLn("%sOK!", title);
149 else
150 getConsole().EOLn("%sNOK (func)!", title);
151 }
152 else
153 {
154 getConsole().EOLn("%sNOK (ext)!", title);
155 }
156
157 functions.clear();
158 return result;
159}
