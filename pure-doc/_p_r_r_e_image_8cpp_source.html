69 TPRREbool PRREImage::PRREImageImpl::setPixelComponentOrder(TPRRE_PIXEL_COMPONENT_ORDER cord)
70 {
71  _pOwner->getManagedConsole().OLn("PRREImage::setPixelComponentOrder(%PXCO)", cord);
72 
73  if ( pPixels == PGENULL )
74  return false;
75 
76  if ( cord == clrCompOrder )
77  {
78  _pOwner->getManagedConsole().OLn(" WARNING: nothing to do, equal component orders!");
79  _pOwner->getManagedConsole().OLn(" clrCompOrder == %PXCO", clrCompOrder);
80  return true;
81  }
82 
83  bool retVal = true;
84  if ( (abs(cord-clrCompOrder) > 1) ||

86  ((cord-clrCompOrder == -1) && (cord % 2 == 1)) ) )

88  bChanged = true;

91  while ( swapcount > 0 )
92  {
93  _pOwner->getManagedConsole().OI();
94  retVal &= swapColors(clrCompOrder, cord, (TPRREbyte) swapcount);
95  swapcount--;
96  _pOwner->getManagedConsole().OO();
97  }
98  }
99 
100  _pOwner->getManagedConsole().OLn("");
101  return retVal;
102 } // setPixelComponentOrder()

224 TPRREbool PRREImage::PRREImageImpl::cannibalize(PRREImage& victim)
225 {
226  // Note that we cannot easily copy victim's Impl since copy ctor and assignment operators are empty!
227  // Thus the following assignment cannot be used: *pImpl = *(victim.pImpl)
228  // We wouldn't even want to use copy ctor or assignment here since we do not want any array copy.
229  // That is why this function is written, to "steal" the arrays of the victim so victim will be "empty".
230  // In C++11 this could be done with a move ctor.
231 
232  _pOwner->SetName(victim.getName()); // copy the Managed part
233  _pOwner->SetFilename(victim.getFilename()); // copy the FiledManaged part
234 
235  nBits = victim.pImpl->nBits;
236  nWidth = victim.pImpl->nWidth;
237  nHeight = victim.pImpl->nHeight;
238  clrCompOrderOrig = victim.pImpl->clrCompOrderOrig;
239  clrCompOrder = victim.pImpl->clrCompOrder;
240  bUpsideDown = victim.pImpl->bUpsideDown;
241  bChanged = victim.pImpl->bChanged;
242  nSizePixels = victim.pImpl->nSizePixels;
243  pPixels = victim.pImpl->pPixels;
244 
245  victim.pImpl->nBits = 0;
246  victim.pImpl->nWidth = 0;
247  victim.pImpl->nHeight = 0;
248  victim.pImpl->bChanged = true;
249  victim.pImpl->nSizePixels = 0;
250  victim.pImpl->pPixels = PGENULL;
251 
252  return true;
253 
254 }

375 TPRREbool PRREImage::PRREImageImpl::swapColors(TPRRE_PIXEL_COMPONENT_ORDER from, TPRRE_PIXEL_COMPONENT_ORDER to, TPRREbyte swapcount)
376 {
377  _pOwner->getManagedConsole().OLn("PRREImage::swapColors(%PXCO, %PXCO, %d)", from, to, swapcount);
378  if ( swapcount < 1 )
379  {
380  _pOwner->getManagedConsole().OLn("");
381  return true;
382  }
383  else if ( swapcount == 2 )
384  {
385  to = getIntermediatePixelCompOrder(from, to);
386  _pOwner->getManagedConsole().OLn(" New target order is %PXCO.", to);
387  }
388 
389  TPRREbyte dstR, dstG, dstB, dstA;
390  if ( !setColorComponentsIndices(dstR, dstG, dstB, dstA, to) )
391  return false;
392 
393  TPRREbyte srcR, srcG, srcB, srcA;
394  if ( !setColorComponentsIndices(srcR, srcG, srcB, srcA, from) )
395  return false;
396 
397  _pOwner->getManagedConsole().OLn(" dstR, srcR == %d, %d", dstR, srcR);
398  _pOwner->getManagedConsole().OLn(" dstG, srcG == %d, %d", dstG, srcG);
399  _pOwner->getManagedConsole().OLn(" dstB, srcB == %d, %d", dstB, srcB);
400 
401  clrCompOrder = to;
402 
403  TPRREbool done = false;
404  TPRREubyte tmp = 0;
405 
406  TPRREbyte srcIdx, dstIdx;
407  for (TPRREbyte i = 0; i <= 2 && !done; i++)
408  {
409  switch (i)
410  {
411  case 0: srcIdx = srcR;
412  dstIdx = dstR;
413  break;
414  case 1: srcIdx = srcG;
415  dstIdx = dstG;
416  break;
417  default: srcIdx = srcB;
418  dstIdx = dstB;
419  }
420  if ( dstIdx-srcIdx != 0 )
421  {
422  done = true;
423  for (TPRREuint x = 0; x < nWidth; x++)
424  {
425  for (TPRREuint y = 0; y < nHeight; y++)
426  {
427  tmp = pPixels[getIndexToPixel(x,y)+srcIdx];
428  pPixels[getIndexToPixel(x,y)+srcIdx] = pPixels[getIndexToPixel(x,y)+dstIdx];
429  pPixels[getIndexToPixel(x,y)+dstIdx] = tmp;
430  } // for y
431  } // for x
432  }
433  } // for i
434 
435  _pOwner->getManagedConsole().OLn("");
436  return true;
437 } // swapColors()

542 TPRREbool PRREImage::PRREImageImpl::readBMP8pixels(const HANDLE f, const RGBQUAD* palette)
543 {
544  if ( (nWidth % 4) != 0 )
545  {
546  _pOwner->getManagedConsole().EOLn("ERROR: 8 bpp image nWidth doesnt divisible by 4 (%d mod 4 == %d)!",
547  nWidth, nWidth % 4);
548  return false;
549  }
550 
551  // we already have the palette, but still need to grab the indices into this palette for each pixel
552  unsigned char* const pPaletteIndexArray = readPaletteIndices(f);
553  if ( !pPaletteIndexArray )
554  return false;

556  TPRREuint k = 0;
557  for (TPRREuint y = 0; y < nHeight; ++y)
558  for (TPRREuint x = 0; x < nWidth; ++x)
559  {
560  for (TPRREuint c = 0; c < 3; ++c)
561  pPixels[(y*nWidth+x)*3+c] = ((const TPRREubyte*) palette)[pPaletteIndexArray[k]*4+2-c];
562  k++;
563  }
564 
565  delete[] pPaletteIndexArray;
566  clrCompOrder = PRRE_RGB;
567  clrCompOrderOrig = clrCompOrder;
