123 TPRREuint PR00FsReducedRenderingEngineImpl::initialize(
124  TPRRE_RENDERER rndr,
125  TPRREuint width, TPRREuint height,
126  TPRRE_DISPLAY_MODES dmode, TPRREuint freq,
127  TPRREint cdepth, TPRREint zdepth, TPRREint stencil,
128  TPRREint samples,
129  HWND window)
130 {
131  TPRREint st = stencil;
132 
133  getConsole().OLn("PRRE::initialize(%d, %d, %d, dmode, %d, %d, %d, %d, %d)",
134  rndr, width, height, freq, cdepth, zdepth, st, samples);
135  if ( isInitialized() )
136  {
137  getConsole().SOLn(" > already initialized!");
138  return 1;
139  }
140 
141  getConsole().OIOLn( PRRE_NAME );
142  getConsole().OLn( PRRE_VERSION );
143 
144  switch (rndr)
145  {
146  case PRRE_RENDERER_SW:
147  pRenderer = &PRRERendererSWincremental::createAndGet(wnd, screen, hwInfo);
148  break;
149  case PRRE_RENDERER_HW_FP:
150  pRenderer = &PRRERendererHWfixedPipe::createAndGet(wnd, screen, hwInfo);
151  break;
152  default:
153  getConsole().EOLnOO("ERROR: invalid renderer specified!");
154  return 1;
155  }
156 
157  if ( pRenderer->initialize(width, height, dmode, freq, cdepth, zdepth, stencil, samples, window) != 0 )
158  {
159  getConsole().EOLnOO("ERROR: failed to initialize renderer!");
160  return 1;
161  }
162 
163  // OpenGL is fully operational at this point so managers can be instantiated
164  // Note in case of SW rendering there is no OpenGL, but managers are able to detect that condition so we are safe here.
165  try
166  {
167  pImageMgr = new PRREImageManager();
168 
169  pTextureMgr = new PRRETextureManager();
170  if ( !pTextureMgr->isInitialized() )
171  {
172  getConsole().EOLn("ERROR: failed to initialize PRRETextureManager!");
173  shutdown();
174  getConsole().OO();
175  return 1;
176  }
177 
178  pMaterialMgr = new PRREMaterialManager();
179  if ( !pMaterialMgr->isInitialized() )
180  {
181  getConsole().EOLn("ERROR: failed to initialize PRREMaterialManager!");
182  shutdown();
183  getConsole().OO();
184  return 1;
185  }
186 
187  pMesh3DMgr = new PRREMesh3DManager(*pMaterialMgr);
188  if ( !pMesh3DMgr->isInitialized() )
189  {
190  getConsole().EOLn("ERROR: failed to initialize PRREMesh3DManager!");
191  shutdown();
192  getConsole().OO();
193  return 1;
194  }
195 
196  pObject3DMgr = new PRREObject3DManager(*pTextureMgr, *pMaterialMgr);
197  if ( !pObject3DMgr->isInitialized() )
198  {
199  getConsole().EOLn("ERROR: failed to initialize PRREObject3DManager!");
200  shutdown();
201  getConsole().OO();
202  return 1;
203  }
204 
205  pCamera = new PRRECamera();
206  // initialize camera viewport to window client area
207  // we can trust window clientrect size even if this is an external window
208  pCamera->SetViewport(0, 0, wnd.getClientWidth(), wnd.getClientHeight());
209  pCamera->SetAspectRatio(wnd.getClientWidth() / (TPRREfloat) wnd.getClientHeight());
210  } // try
211  catch (const std::bad_alloc&)
212  {
213  getConsole().EOLn("ERROR: failed to instantiate a manager!");
214  shutdown();
215  getConsole().OO();
216  return 1;
217  }
218 
219  uiMgr.Initialize( wnd.getWndDC() );
220 
221  pRenderer->SetManagers(pObject3DMgr, &uiMgr, pCamera);
222 
223  // PRRE is fully operational at this point
224  getConsole().OLn("");
225  getConsole().SOLn("> PRRE initialized");
226  getConsole().OLnOO("");
227 
228  return 0;
229

417 void PR00FsReducedRenderingEngineImpl::PrintTexturesUnusedByMaterials() const
418 {
419  if ( !isInitialized() )
420  return;
421 
422  const PRRETextureManager& texMgr = getTextureManager();
423  const PRREMaterialManager& matMgr = getMaterialManager();
424 
425  getConsole().OLn("List of Textures Unused by Materials");
426  getConsole().OLn("====================================");
427 
428  TPRREuint nUnusedTextures = 0;
429 
430  for (TPRREint texi = 0; texi < texMgr.getCount(); texi++)
431  {
432  PRRETexture* tex = (PRRETexture*) texMgr.getAttachedAt(texi);
433  if ( !tex )
434  continue;
435 
436  bool textureUsedByMaterial = false;
437  for (TPRREint mati = 0; !textureUsedByMaterial && (mati < matMgr.getCount()); mati++)
438  {
439  PRREMaterial* mat = (PRREMaterial*) matMgr.getAttachedAt(mati);
440  if ( !mat)
441  continue;
442 
443  for (TPRREuint layeri = 0; !textureUsedByMaterial && (layeri < matMgr.getMaximumLayerCount()); layeri++)
444  {
445  if ( mat->getTexture(layeri) == tex )
446  textureUsedByMaterial = true;
447  }
448  } // for mati
449  if ( !textureUsedByMaterial )
450  {
451  ++nUnusedTextures;
452  getConsole().OLn("Internal num.: %d, file name: %s", tex->getInternalNum(), tex->getFilename().c_str());
453  }
454  } // for texi
455 
456  getConsole().OLn("%d Textures Found to be Unused by Checking %d Materials.", nUnusedTextures, matMgr.getCount());
457  getConsole().OLn("");
458 } // PrintUnusedTextures()

601 void PR00FsReducedRenderingEngineImpl::CopyScreenToTexture(PRRETexture& tex)
602 {
603  // This function should be rather part of PRREScreen(), however it is easier to implement here.
604  // One challenge of implementing this in PRREScreen() is that that class needs to know:
605  // - HWInfo to check if multitexturing is supported, although singleton instance is easily get with getHardwareInfo();
606  // - current camera because that is not singleton and cannot be queried by that class, this is the real challenge.
607  // Camera is needed to get viewport size. It is not ok to check current resolution because:
608  // - it does not equal to expected image size when engine runs in windowed mode;
609  // - in future we may want to have multiple cameras with their own viewports, and CopyToTexture() should copy only 1 viewport content.
610 
611  if ( !isInitialized() )
612  return;
613 
614  // we don't expect texture dimension to be rectangular (w != h), but
615  // its width must be same as our viewport's width
616  if ( tex.getWidth() != getCamera().getViewport().size.width )
617  return;
618 
619  if ( tex.getHeight() == 0 )
620  return;
621 
622  if ( tex.getInternalNum() == 0 )
623  return;
624 
625  if ( getHardwareInfo().getVideo().isMultiTexturingSupported() )
626  {
627  glActiveTextureARB(GL_TEXTURE1_ARB);
628  glDisable(GL_TEXTURE_2D);
629  glActiveTextureARB(GL_TEXTURE0_ARB);
630  }
631 
632  glEnable(GL_TEXTURE_2D);
633  glBindTexture(GL_TEXTURE_2D, tex.getInternalNum());
634  glCopyTexImage2D(GL_TEXTURE_2D, 0, GL_RGB, 0, 0, tex.getWidth(), tex.getHeight(), 0);
635 }
