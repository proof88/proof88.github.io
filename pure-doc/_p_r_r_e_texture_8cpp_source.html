257 TPRREbool PRRETexture::PRRETextureImpl::setAnisoFilteringMode(TPRRE_ANISO_TEX_FILTERING filtering)
258 {
259  _pOwner->getManagedConsole().OIOLn("PRRETexture::setAnisoFilteringMode(%fX)", PRRETextureManager::getFloatFromPRREanisoTexFilteringName(filtering));
260 
261  if ( _pOwner->getManager() == PGENULL )
262  return false;
263 
264  if ( !PRREhwInfo::get().getVideo().isAcceleratorDetected() )
265  {
266  _pOwner->getManagedConsole().EOLnOO("WARNING: anisotropic filtering is not supported without acceleration!");
267  return false;
268  }
269 
270  bool result = false;
271  const TPRREfloat newFilteringF = ((PRRETextureManager*)_pOwner->getManager())->getFloatFromPRREanisoTexFilteringName(filtering);
272  const TPRREfloat availMaxFilteringF = ((PRRETextureManager*)_pOwner->getManager())->getFloatFromPRREanisoTexFilteringName(
273  ((PRRETextureManager*)_pOwner->getManager())->getMaximumAnisoFiltering() );
274  // in Release mode, although getMaximumAnisoFiltering() returns ANISO_16X, availMaxFilteringF becomes 0.
275  // This can be fixed by adding a console printout to getFloatFromPRREanisoTexFilteringName(), so probably this is some
276  // compiler optimization bug which doesnt happen in Debug mode. Would be good to play with the Release mode compiler options
277  // to see exactly which option causes this. This is weird because getFloatFromPRREanisoTexFilteringName() can never ever be 0.
278  // _pOwner->getManagedConsole().OLn( "availMaxFilteringF: %f", availMaxFilteringF);
279  if ( availMaxFilteringF > 1.0f )
280  {
281  if ( (nInternalNum == 0) || (!pglBindTexture(GL_TEXTURE_2D, nInternalNum)) )
282  {
283  _pOwner->getManagedConsole().EOLnOO("ERROR: nInternalNum is 0 or pglBindTexture() failed!");
284  return false;
285  }
286  if ( availMaxFilteringF >= newFilteringF )
287  {
288  if ( pglTexParameterf(GL_TEXTURE_2D, GL_TEXTURE_MAX_ANISOTROPY_EXT, (GLfloat) newFilteringF) )
289  {
290  filtAniso = filtering;
291  result = true;
292  }
293  }
294  else
295  {
296  _pOwner->getManagedConsole().EOLn("WARNING: selected mode not supported, setting max supported: %fX!", availMaxFilteringF);
297  if ( pglTexParameterf(GL_TEXTURE_2D, GL_TEXTURE_MAX_ANISOTROPY_EXT, (GLfloat) availMaxFilteringF) )
298  {
299  filtAniso = PRRETextureManager::getPRREanisoTexFilteringNameFromFloat(availMaxFilteringF);
300  result = true;
301  }
302  }
303  }
304  else
305  {
306  _pOwner->getManagedConsole().EOLn("ERROR: anisotropic filtering is not supported");
307  }
308  _pOwner->getManagedConsole().OO();
309  return result;
310 } // setAnisoFilteringMode()

388 TPRREbool PRRETexture::PRRETextureImpl::uploadPixels()
389 {
390  _pOwner->getManagedConsole().OIOLn("PRRETexture::uploadPixels()");
391 
392  if ( _pOwner->getPixels() == PGENULL )
393  {
394  _pOwner->getManagedConsole().EOLnOO("ERROR: getPixels() returned NULL!");
395  return false;
396  }
397 
398  if ( _pOwner->getManager() == PGENULL )
399  {
400  _pOwner->getManagedConsole().EOLnOO("ERROR: getManager() returned NULL!");
401  return false;
402  }
403 
404  if ( strcmp(typeid(*_pOwner->getManager()).raw_name(), typeid(PRRETextureManager&).raw_name()) )
405  {
406  _pOwner->getManagedConsole().EOLnOO("ERROR: getManager() type is not PRRETextureManager!");
407  return false;
408  }
409 
410  if ( PRREhwInfo::get().getVideo().isAcceleratorDetected() )
411  {
412  if (!pglBindTexture(GL_TEXTURE_2D, nInternalNum))
413  {
414  return false;
415  }
416  }
417 
418  const TPRRE_PIXEL_COMPONENT_ORDER oldco = _pOwner->getPixelComponentOrder();
419  const GLenum glTexFormat = getTransformedSourceFormat(oldco); // might invoke setPixelComponentOrder()!
420  if ( glTexFormat == GL_INVALID_ENUM )
421  {
422  _pOwner->getManagedConsole().EOLnOO("ERROR: getTransformedSourceFormat() failed!");
423  // try reset original pixel component order
424  if ( !_pOwner->setPixelComponentOrder( oldco ) )
425  {
426  _pOwner->getManagedConsole().EOLnOO("ERROR: recovery setPixelComponentOrder(oldco) failed!");
427  }
428  return false;
429  }
430 
431  const GLint internalfmt = getTargetInternalFormat();
432 
433  if ( !actualUploadProc(internalfmt, glTexFormat) )
434  {
435  _pOwner->getManagedConsole().EOLnOO("ERROR: actualUploadProc() failed!");
436  return false;
437  }
438 
439  DescribeTexFormatAndSize(internalfmt);
440 
441  if ( _pOwner->getPixelComponentOrder() != oldco )
442  {
443  if ( !_pOwner->setPixelComponentOrder( oldco ) )
444  {
445  _pOwner->getManagedConsole().EOLnOO("ERROR: finished creating new texture, but setPixelComponentOrder(oldco) failed!");
446  // intentionally let function still return true
447  }
448  }
449 
450  _pOwner->getManagedConsole().OO();
451 
452  return true;
453 } // uploadPixels()

522 GLenum PRRETexture::PRRETextureImpl::getTransformedSourceFormat(TPRRE_PIXEL_COMPONENT_ORDER oldco)
523 {
524  const PRRETextureManager& parent = *((PRRETextureManager*) _pOwner->getManager());
525  const PRREhwInfo& hwInfo = PRREhwInfo::get();
526 
527  if ( oldco == PRRE_BGR )
528  {
529  if ( hwInfo.getVideo().isNativeDIBFormatSupported() && parent.isNativeDIBFormatSupportEnabled() )
530  return GL_BGR_EXT;
531  else
532  {
533  if ( !_pOwner->setPixelComponentOrder(PRRE_RGB) )
534  return GL_INVALID_ENUM;
535  return GL_RGB;
536  }
537  }
538  else if ( oldco == PRRE_BGRA )
539  {
540  if ( hwInfo.getVideo().isNativeDIBFormatSupported() && parent.isNativeDIBFormatSupportEnabled() )
541  return GL_BGRA_EXT;
542  else
543  {
544  if ( !_pOwner->setPixelComponentOrder(PRRE_RGBA) )
545  return GL_INVALID_ENUM;
546  return GL_RGBA;
547  }
548  }
549  else if ( _pOwner->getPixelComponentOrder() % 2 == 0 )
550  {
551  if ( oldco != PRRE_RGB )
552  {
553  if ( !_pOwner->setPixelComponentOrder(PRRE_RGB) )
554  return GL_INVALID_ENUM;
555  }
556  return GL_RGB;
557  }
558  else
559  {
560  if ( oldco != PRRE_RGBA )
561  {
562  if ( !_pOwner->setPixelComponentOrder(PRRE_RGBA) )
563  return GL_INVALID_ENUM;
564  }
565  return GL_RGBA;
566  }
567 } // getTransformedSourceFormat()

576 GLint PRRETexture::PRRETextureImpl::getTargetInternalFormat()
577 {
578  const PRRETextureManager& parent = *((PRRETextureManager*) _pOwner->getManager());
579  const PRREhwInfo& hwInfo = PRREhwInfo::get();
580 
581  if ( hwInfo.getVideo().isTextureCompressionSupported() )
582  {
583  switch ( parent.getDefaultCompressionMode() )
584  {
585  case PRRE_TC_NONE: return _pOwner->getPixelComponentOrder() % 2 == 0 ? 3 : 4;
586  case PRRE_TC_S3TC_RGB_DXT1 : return GL_COMPRESSED_RGB_S3TC_DXT1_EXT;
587  case PRRE_TC_S3TC_RGBA_DXT1: return GL_COMPRESSED_RGBA_S3TC_DXT1_EXT;
588  case PRRE_TC_S3TC_RGBA_DXT3: return GL_COMPRESSED_RGBA_S3TC_DXT3_EXT;
589  case PRRE_TC_S3TC_RGBA_DXT5: return GL_COMPRESSED_RGBA_S3TC_DXT5_EXT;
590  case PRRE_TC_3DFX_RGB_FXT1 : return GL_COMPRESSED_RGB_FXT1_3DFX;
591  case PRRE_TC_3DFX_RGBA_FXT1: return GL_COMPRESSED_RGBA_FXT1_3DFX;
592  /* PRRE_TC_AUTO causes the driver to select any of the above */
593  default: /* for PRRE_TC_AUTO */ return _pOwner->getPixelComponentOrder() % 2 == 0 ? GL_COMPRESSED_RGB_ARB : GL_COMPRESSED_RGBA_ARB;
594  }
595  }
596  else
597  return _pOwner->getPixelComponentOrder() % 2 == 0 ? 3 : 4;
598 } // getTargetInternalFormat()

613 TPRREbool PRRETexture::PRRETextureImpl::actualUploadProc(GLint internalfmt, GLenum glTexFormat)
614 {
615  const PRRETextureManager& parent = *((PRRETextureManager*) _pOwner->getManager());
616  const PRREhwInfo& hwInfo = PRREhwInfo::get();
617 
618  if ( !hwInfo.getVideo().isAcceleratorDetected() )
619  {
620  _pOwner->getManagedConsole().OLn("no need to upload texture, non-accelerated rendering is active");
621  return true;
622  }
623 
624  if ( nMIPmapCount > 1 )
625  {
626  _pOwner->getManagedConsole().OLn("mipmapping on");
627  _pOwner->getManagedConsole().O("auto-generating mipmaps with ");
628  if ( hwInfo.getVideo().isHardwareMipMapGenerationSupported() && parent.isHardwareMipMapGenerationEnabled() )
629  {
630  _pOwner->getManagedConsole().OLn("HW-support.");
631  if (!pglTexParameteri(GL_TEXTURE_2D, GL_GENERATE_MIPMAP_SGIS, GL_TRUE) )
632  {
633  return false;
634  }
635 
636  // WARNING: up to OpenGL 2.x, border is allowed to be 0 or 1. However from GL 3.0 it must be 0.
637  // WARNING: if GL version does not support NPOT sizes, width must be 2n+2(border) for some integer n.
638  // WARNING: if GL version does not support NPOT sizes, height must be 2m+2(border) for some integer m.
639  // TODO: So in the future when I add support for NPOT textures, I need to consider border value when specifying width and height.
640  return pglTexImage2D(GL_TEXTURE_2D, 0, internalfmt, _pOwner->getWidth(), _pOwner->getHeight(), bBorder ? (GLint) 1 : (GLint) 0, glTexFormat, GL_UNSIGNED_BYTE, _pOwner->getPixels());
641  }
642  else
643  {
644  _pOwner->getManagedConsole().OLn("SW-only.");
645  const GLint errCode = gluBuild2DMipmaps(GL_TEXTURE_2D, internalfmt, _pOwner->getWidth(), _pOwner->getHeight(), glTexFormat, GL_UNSIGNED_BYTE, _pOwner->getPixels());
646  if ( errCode != GL_NO_ERROR )
647  {
648  _pOwner->getManagedConsole().EOLn("PRRETextureImpl::actualUploadProc() ERROR: %d, %s", errCode, gluErrorString(errCode));
649  return false;
650  }
651  return true;
652  }
653  }
654  else
655  {
656  _pOwner->getManagedConsole().OLn("mipmapping off");
657 
658  // WARNING: up to OpenGL 2.x, border is allowed to be 0 or 1. However from GL 3.0 it must be 0.
659  // WARNING: if GL version does not support NPOT sizes, width must be 2n+2(border) for some integer n.
660  // WARNING: if GL version does not support NPOT sizes, height must be 2m+2(border) for some integer m.
661  // TODO: So in the future when I add support for NPOT textures, I need to consider border value when specifying width and height.
662  return pglTexImage2D(GL_TEXTURE_2D, 0, internalfmt, _pOwner->getWidth(), _pOwner->getHeight(), bBorder ? (GLint) 1 : (GLint) 0, glTexFormat, GL_UNSIGNED_BYTE, _pOwner->getPixels());
663  }
664 } // actualUploadProc()

672 void PRRETexture::PRRETextureImpl::DescribeTexFormatAndSize(GLint internalfmt)
673 {
674  nSize = _pOwner->getWidth() * _pOwner->getHeight() * 4;
675  texFormat = PRRE_TF_RGBA;
676 
677  if ( !PRREhwInfo::get().getVideo().isAcceleratorDetected() )
678  {
679  return;
680  }
681 
682  if ( (internalfmt != 3) && (internalfmt != 4) )
683  {
684  GLint tmpret = 0;
685  glGetTexLevelParameteriv(GL_TEXTURE_2D, 0, GL_TEXTURE_COMPRESSED, &tmpret);
686  if ( tmpret == GL_TRUE )
687  {
688  glGetTexLevelParameteriv(GL_TEXTURE_2D, 0, GL_TEXTURE_INTERNAL_FORMAT, &tmpret);
689  _pOwner->getManagedConsole().O("final internal texFormat is ");
690  switch ( tmpret )
691  {
692  case GL_RGBA:
693  _pOwner->getManagedConsole().OLn("GL_RGBA");
694  break;
695  case GL_COMPRESSED_RGB_S3TC_DXT1_EXT:
696  texFormat = PRRE_TF_DXT1;
697  _pOwner->getManagedConsole().OLn("GL_COMPRESSED_RGB_S3TC_DXT1_EXT");
698  break;
699  case GL_COMPRESSED_RGBA_S3TC_DXT1_EXT:
700  texFormat = PRRE_TF_DXT1A;
701  _pOwner->getManagedConsole().OLn("GL_COMPRESSED_RGBA_S3TC_DXT1_EXT");
702  break;
703  case GL_COMPRESSED_RGBA_S3TC_DXT3_EXT:
704  texFormat = PRRE_TF_DXT3A;
705  _pOwner->getManagedConsole().OLn("GL_COMPRESSED_RGBA_S3TC_DXT3_EXT");
706  break;
707  case GL_COMPRESSED_RGBA_S3TC_DXT5_EXT:
708  texFormat = PRRE_TF_DXT5A;
709  _pOwner->getManagedConsole().OLn("GL_COMPRESSED_RGBA_S3TC_DXT5_EXT");
710  break;
711  case GL_COMPRESSED_RGB_FXT1_3DFX:
712  texFormat = PRRE_TF_FXT1;
713  _pOwner->getManagedConsole().OLn("GL_COMPRESSED_RGB_FXT1_3DFX");
714  break;
715  case GL_COMPRESSED_RGBA_FXT1_3DFX:
716  texFormat = PRRE_TF_FXT1A;
717  _pOwner->getManagedConsole().OLn("GL_COMPRESSED_RGBA_FXT1_3DFX");
718  break;
719  /* In 14.12 AMD Catalyst Omega Software, requesting auto-compression at glTexImage2D() returns general compression here instead of any if the specific values above. */
720  case GL_COMPRESSED_RGB_ARB:
721  texFormat = PRRE_TF_UNSURE;
722  _pOwner->getManagedConsole().OLn("GL_COMPRESSED_RGB_ARB");
723  break;
724  case GL_COMPRESSED_RGBA_ARB:
725  texFormat = PRRE_TF_UNSURE;
726  _pOwner->getManagedConsole().OLn("GL_COMPRESSED_RGBA_ARB");
727  break;
728  default:
729  _pOwner->getManagedConsole().EOLn("ERROR: glGetTexLevelParameteriv() returned %d for GL_TEXTURE_INTERNAL_FORMAT", tmpret);
730  break;
731  }
732  glGetTexLevelParameteriv(GL_TEXTURE_2D, 0, GL_TEXTURE_COMPRESSED_IMAGE_SIZE, &tmpret);
733  nSize = tmpret;
734  }
735  }
736  if ( nMIPmapCount > 1 )
737  {
738  nSize = PRRETextureManager::getMIPmappedTexSize(nSize, _pOwner->getWidth(), _pOwner->getHeight());
739  }
740 } // DescribeTexFormatAndSize()
