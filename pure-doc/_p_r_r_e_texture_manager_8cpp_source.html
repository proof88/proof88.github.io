400 TPRREbool PRRETextureManager::PRRETextureManagerImpl::generateAndUploadTexture(PRRETexture& texture)
401 {
402  if ( (getDefaultMinFilteringMode() == PRRE_ISO_NEAREST) || (getDefaultMinFilteringMode() == PRRE_ISO_LINEAR) )
403  texture.pImpl->nMIPmapCount = 1;
404  else
405  texture.pImpl->nMIPmapCount = PRRETextureManager::getMIPmapCount( texture.getWidth(), texture.getHeight() );
406 
407  if ( pHWInfo.getVideo().isAcceleratorDetected() )
408  {
409  if ( !pglGenTextures(1, &(texture.pImpl->nInternalNum)) )
410  {
411  return false;
412  }
413 
414  if ( !pglBindTexture(GL_TEXTURE_2D, texture.pImpl->nInternalNum) )
415  {
416  return false;
417  }
418 
419  if ( !pglTexEnvi(GL_TEXTURE_ENV, GL_TEXTURE_ENV_MODE, GL_MODULATE) )
420  {
421  return false;
422  }
423  }
424 
425  bool b = true;
426  b &= texture.setMagFilteringMode( getDefaultMagFilteringMode() );
427  b &= texture.setMinFilteringMode( getDefaultMinFilteringMode() );
428  b &= texture.setAnisoFilteringMode( getDefaultAnisoFilteringMode() );
429  b &= texture.setTextureWrappingMode( getDefaultTextureWrappingModeS(), getDefaultTextureWrappingModeT() );
430  b &= texture.setBorder( getDefaultBorder() );
431  b &= texture.setBorderColor( getDefaultBorderColor() );
432  if ( !b )
433  {
434  _pOwner->getConsole().EOLn("ERROR: failed to set an attribute to the texture!");
435  return false;
436  }
437 
438  if ( !texture.uploadPixels() )
439  {
440  _pOwner->getConsole().EOLn("ERROR: uploadPixels() failed!");
441  return false;
442  }
443 
444  return true;
445 }

787 PRRETexture* PRRETextureManager::createTextureFromImage(const PRREImage& img)
788 {
789  getConsole().OLnOI("PRRETextureManager::createTextureFromImage(...)");
790 
791  if ( !isInitialized() )
792  {
793  getConsole().OO();
794  return PGENULL;
795  }
796 
797  PRRETexture* texture = isLazyInstancingEnabled() ? (PRRETexture*) getByFilename( img.getFilename().c_str() ) : PGENULL;
798 
799  if ( texture != PGENULL )
800  { // we already have a texture with this filename
801  getConsole().SOLn("> Found loaded texture, returning.");
802  getConsole().OLnOO("");
803  return texture;
804  }
805 
806  getConsole().OLn("texture == PGENULL");
807  try
808  {
809  texture = new PRRETexture(img);
810  }
811  catch (const std::bad_alloc&)
812  {
813  getConsole().EOLn("ERROR: Failed to instantiate new PRRETexture!");
814  getConsole().OLnOO("");
815  return PGENULL;
816  }
817 
818  Attach(*texture);
819 
820  if ( !pImpl->generateAndUploadTexture(*texture) )
821  {
822  getConsole().EOLn("ERROR: generateAndUploadTexture() failed!");
823  getConsole().OLnOO("");
824  DeleteAttachedInstance(*texture);
825  return PGENULL;
826  }
827 
828  if ( !isPixelPreservingEnabled() )
829  texture->FlushResources();
830 
831  texture->SetName("Texture " + std::to_string(pImpl->nRunningCounter++));
832 
833  getConsole().OLn("Texture nWidth and nHeight: %d %d", texture->getWidth(), texture->getHeight());
834  getConsole().OLn("final nSize is %d Byte(s).", texture->pImpl->nSize);
835  getConsole().OLn("internalnum = %d", texture->pImpl->nInternalNum);
836  getConsole().OLn("");
837  getConsole().SOLn("> Texture created, name: %s!", texture->getName().c_str());
838  getConsole().OOOLn("");
839  return texture;
840 } // createTextureFromImage()

851 PRRETexture* PRRETextureManager::createFromFile(const char* filename)
852 {
853  if ( !isInitialized() )
854  {
855  return PGENULL;
856  }
857 
858  getConsole().OLnOI("PRRETextureManager::createFromFile(\"%s\")", filename);
859 
860  if ( filename == PGENULL )
861  {
862  getConsole().EOLnOO("ERROR: input was NULL, returning PGENULL");
863  getConsole().OLn("");
864  return PGENULL;
865  }
866 
867  PRRETexture* texture = isLazyInstancingEnabled() ? (PRRETexture*) getByFilename( filename ) : PGENULL;
868 
869  if ( texture != PGENULL )
870  { // we already have a texture with this filename
871  getConsole().SOLnOO("> Found loaded texture, returning: %s", texture->getName().c_str());
872  getConsole().OLn("");
873  return texture;
874  }
875 
876  getConsole().OI();
877  PRREImage* tmpImg = PRREImageManager::createFromFile(filename);
878  getConsole().OO();
879  if ( tmpImg == PGENULL )
880  {
881  getConsole().EOLnOO("ERROR: failed to create Image from file!");
882  getConsole().OLn("");
883  return PGENULL;
884  }
885 
886  try
887  {
888  texture = new PRRETexture(*tmpImg);
889  PRREImage* textureAsImage = (PRREImage*)texture;
890  if ( !textureAsImage->pImpl->cannibalize(*tmpImg) )
891  {
892  const std::string sErrMsg = "cannibalize() failed!";
893  throw std::runtime_error(sErrMsg);
894  }
895  }
896  catch (const std::exception& e)
897  {
898  getConsole().EOLn("ERROR: Failed to create or fill new PRRETexture: %s!", e.what());
899  DeleteAttachedInstance(*tmpImg);
900  getConsole().OLnOO("");
901  return PGENULL;
902  }
903  DeleteAttachedInstance(*tmpImg);
904 
905  Attach(*texture);
906 
907  if ( !pImpl->generateAndUploadTexture(*texture) )
908  {
909  getConsole().EOLn("ERROR: generateAndUploadTexture() failed!");
910  getConsole().OLnOO("");
911  DeleteAttachedInstance(*texture);
912  return PGENULL;
913  }
914 
915  if ( !isPixelPreservingEnabled() )
916  texture->FlushResources();
917 
918  texture->SetName("Texture " + std::to_string(pImpl->nRunningCounter++));
919 
920  getConsole().OLn("Texture nWidth and nHeight: %d %d", texture->getWidth(), texture->getHeight());
921  getConsole().OLn("final nSize is %d Byte(s).", texture->pImpl->nSize);
922  getConsole().OLn("internalnum = %d", texture->pImpl->nInternalNum);
923  getConsole().OLn("");
924  getConsole().SOLn("> Texture created, name: %s!", texture->getName().c_str());
925  getConsole().OOOLn("");
926  return texture;
927 } // createFromFile()
