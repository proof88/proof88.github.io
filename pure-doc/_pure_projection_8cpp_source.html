36bool PureProjection::project3dTo2d(
37 TPureFloat fWorldX, TPureFloat fWorldY, TPureFloat fWorldZ,
38 TPureFloat fCamPosX, TPureFloat fCamPosY, TPureFloat fCamPosZ,
39 TPureFloat fCamTargetX, TPureFloat fCamTargetY, TPureFloat fCamTargetZ,
40 TPureFloat fCamUpX, TPureFloat fCamUpY, TPureFloat fCamUpZ,
41 TPureFloat fCamFovY, TPureFloat fCamAspect,
42 TPureFloat fNearPlane, TPureFloat fFarPlane,
43 TPureInt nViewportX, TPureInt nViewportY,
44 TPureUInt nViewportW, TPureUInt nViewportH,
45 PureVector& vecProjected)
46{
47 // TODO: basically this is copy-paste of PureRendererSWincrementalImpl::VertexProcessing() and
48 // PureRendererSWincrementalImpl::PrimitiveAssembly(), so it would be better if those functions
49 // would use this. For that this function needs to be split into 2.
50
51 /*
52 Projection: to convert 3D world-space coordinates into 2D window-space coordinates, 4 steps are needed:
53
54 1. View Transform:
55 World Space -> Eye Space
56 2. Projection Transform:
57 Eye Space -> Clip Space
58 3. Perspective Divide:
59 Clip Space -> Normalized Device Space
60 4. Viewport Transformation:
61 Normalized Device Space -> Window/Screen Space
62
63 This is basically what my software renderer is doing in PureRendererSWincrementalImpl::GeometricStage().
64 These steps are shown with example coordinates in:
65 https://github.com/proof88/PGE/blob/master/PGE/PURE/docpages/internal/PR00FPSvsPRRE-Transformations.xlsx .
66 And also explained in https://proof88.github.io/pure-doc/architecture.html .
67 Obviously we don't do this with PureRendererHWincrementalImpl since in that case these transformations are happening in the GPU.
68 */
69
70 if ((fCamFovY <= 0.f) || (fCamAspect <= 0.f) ||
71 (fNearPlane <= 0.f) || (fNearPlane >= fFarPlane) ||
72 (nViewportW == 0) || (nViewportH == 0))
73 {
74 return false;
75 }
76
77 vecProjected.Set(fWorldX, fWorldY, fWorldZ);
78 vecProjected.Set(3, 1.f); // W, just in case someone somehow messed W up
79
80 /* 1. World -> View/Eye Space */
81 PureTransformMatrix mViewTr;
82 mViewTr.SetLookAt(
83 PureVector(fCamPosX, fCamPosY, fCamPosZ),
84 PureVector(fCamTargetX, fCamTargetY, fCamTargetZ),
85 PureVector(fCamUpX, fCamUpY, fCamUpZ));
86 vecProjected = mViewTr * vecProjected;
87
88 /* 2. View/Eye -> Clip Space */
89 /* The pyramidal frustum in eye space becomes a cube, therefore, the clipping plane equations in clip space are: X + Wc = 0, Y + Wc = 0, Z + Wc = 0, etc. */
90 PureTransformMatrix mProjTr;
91 mProjTr.SetFrustumByFovY(fCamFovY, fCamAspect, fNearPlane, fFarPlane);
92 vecProjected = mProjTr * vecProjected;
93
94 /* We can now easly decide if the given 3D coordinate is within the view frustum or not, since we are in clip space, where the frustum is actually a cube. */
95 // Note that I'm unsure if we should return false here or continue projecting, since I also think if the user gives X-Y coords that are
96 // too far horizontally or vertically from the camera, we can still project them to 2D even though they fall outside of the viewport/screen area.
97 // Maybe we should add a flag argument also that controls this.
98 if ((vecProjected.getX() < -vecProjected.getW()) || (vecProjected.getX() > vecProjected.getW())
99 ||
100 (vecProjected.getY() < -vecProjected.getW()) || (vecProjected.getY() > vecProjected.getW())
101 ||
102 (vecProjected.getZ() < -vecProjected.getW()) || (vecProjected.getZ() > vecProjected.getW()))
103
104 {
105 // out of frustum
106 return false;
107 }
108
109 /* 3. Clip -> Normalized Device Space */
110 if (vecProjected.getW() == 0.f)
111 {
112 return false;
113 }
114 vecProjected /= vecProjected.getW();
115
116 /* 4. Normalized Device -> Window/Screen Space */
117 const float DEPTH_RANGE_MIN = 0.0f;
118 const float DEPTH_RANGE_MAX = 1.0f;
119
120 vecProjected.SetX( std::roundf(nViewportW / 2.f * vecProjected.getX() + nViewportX + nViewportW / 2.f) );
121 vecProjected.SetY( std::roundf(nViewportH / 2.f * vecProjected.getY() + nViewportY + nViewportH / 2.f) );
122 vecProjected.SetZ( (DEPTH_RANGE_MAX - DEPTH_RANGE_MIN) / 2.f * vecProjected.getZ() + (DEPTH_RANGE_MAX + DEPTH_RANGE_MIN) / 2.f );
123
124 return true;
125} // project3dTo2d()

139bool PureProjection::project2dTo3d(
140 TPureUInt nScreenX, TPureUInt nScreenY, TPureFloat nScreenDepth,
141 TPureFloat fCamPosX, TPureFloat fCamPosY, TPureFloat fCamPosZ,
142 TPureFloat fCamTargetX, TPureFloat fCamTargetY, TPureFloat fCamTargetZ,
143 TPureFloat fCamUpX, TPureFloat fCamUpY, TPureFloat fCamUpZ,
144 TPureFloat fCamFovY, TPureFloat fCamAspect,
145 TPureFloat fNearPlane, TPureFloat fFarPlane,
146 TPureInt nViewportX, TPureInt nViewportY,
147 TPureUInt nViewportW, TPureUInt nViewportH,
148 PureVector& vecUnprojected)
149{
150 // In this function basically I do everything in reverse order compared to project3dTo2d(),
151 // however if there is any problem, check the formula of gluUnProject():
152 // https://registry.khronos.org/OpenGL-Refpages/gl2.1/xhtml/gluUnProject4.xml
153 //
154 // Can also check: https://gamedev.stackexchange.com/questions/45445/what-method-replaces-gl-select-for-box-selection
155
156 /*
157 Unprojection: to convert 2D window-space coordinates to 3D world-space coordinates, the reverse order steps are needed:
158
159 1. Viewport Transformation:
160 Window/Screen Space -> Normalized Device Space
161 2. Perspective Divide:
162 Normalized Device Space -> Clip Space
163 3. Projection Transform:
164 Clip Space -> Eye Space
165 4. View Transform:
166 Eye Space -> World Space
167 */
168
169 const float DEPTH_RANGE_MIN = 0.0f;
170 const float DEPTH_RANGE_MAX = 1.0f;
171
172 if ((nScreenDepth < DEPTH_RANGE_MIN) ||
173 (nScreenDepth > DEPTH_RANGE_MAX) ||
174 (fCamFovY <= 0.f) || (fCamAspect <= 0.f) ||
175 (fNearPlane <= 0.f) || (fNearPlane >= fFarPlane) ||
176 (nViewportW == 0) || (nViewportH == 0))
177 {
178 return false;
179 }
180
181 /* 1. Window/Screen -> Normalized Device Space */
182 vecUnprojected.Set(
183 (nScreenX - nViewportX - nViewportW / 2.f) / (nViewportW / 2.f),
184 (nScreenY - nViewportY - nViewportH / 2.f) / (nViewportH / 2.f),
185 (nScreenDepth - (DEPTH_RANGE_MAX + DEPTH_RANGE_MIN) / 2.f) / ((DEPTH_RANGE_MAX - DEPTH_RANGE_MIN) / 2.f));
186 vecUnprojected.Set(3, 1.f); // W, just in case someone somehow messed W up
187
188 /* 2. Normalized Device -> Clip Space */
189 // TODO: are we already in Clip space now?
190
191 /* We can now easly decide if the given 2D coordinate is within the view frustum or not, since we are in clip space, where the frustum is actually a cube. */
192 // Note that I'm unsure if we should return false here or continue unprojecting, since I also think if the user gives X-Y coords that are
193 // off-screen coords, we can still unproject them to 3D even though they fall outside of the frustum.
194 // Maybe we should add a flag argument also that controls this.
195 if ((vecUnprojected.getX() < -1.f) || (vecUnprojected.getX() > 1.f)
196 ||
197 (vecUnprojected.getY() < -1.f) || (vecUnprojected.getY() > 1.f)
198 ||
199 (vecUnprojected.getZ() < -1.f) || (vecUnprojected.getZ() > 1.f))
200
201 {
202 // out of frustum
203 return false;
204 }
205
206 /* 3-4. Clip -> View/Eye -> World Space */
207 // I'm doing these 2 steps together so I need to do only 1 matrix invert() instead of 2
208 PureTransformMatrix mProjTr;
209 mProjTr.SetFrustumByFovY(fCamFovY, fCamAspect, fNearPlane, fFarPlane);
210 PureTransformMatrix mViewTr;
211 mViewTr.SetLookAt(
212 PureVector(fCamPosX, fCamPosY, fCamPosZ),
213 PureVector(fCamTargetX, fCamTargetY, fCamTargetZ),
214 PureVector(fCamUpX, fCamUpY, fCamUpZ));
215 mProjTr *= mViewTr; // here we combine Projection with View Matrix
216
217 mProjTr.invert();
218 vecUnprojected = mProjTr * vecUnprojected;
219
220 vecUnprojected = vecUnprojected / vecUnprojected.getW();
221
222 return true;
223} // project2dTo3d()
