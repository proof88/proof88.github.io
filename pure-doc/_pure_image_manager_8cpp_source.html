79PureImage* PureImageManager::PureImageManagerImpl::loadBMP(const char* filename)
80{
81 BITMAPFILEHEADER file_header;
82 BITMAPINFOHEADER info_header;
83 DWORD bitmaplength;
84 RGBQUAD* palette = NULL;
85 DWORD palettesize = 0;
86 DWORD bytesread;
87 PureImage* pNewImage = NULL;
88
89 HANDLE bitmapfile = CreateFileA(filename, GENERIC_READ, FILE_SHARE_READ, NULL, OPEN_EXISTING, 0, 0);
90
91 _pOwner->getConsole().OLnOI("PureImageManager::loadBMP(\"%s\")", filename);
92 if ( bitmapfile == INVALID_HANDLE_VALUE )
93 return loadBMPfail(bitmapfile, pNewImage, palette, "ERROR: bitmapfile == INVALID_HANDLE_VALUE, returning PGENULL!");
94
95 ReadFile(bitmapfile, &file_header, sizeof(BITMAPFILEHEADER), &bytesread, NULL);
96 _pOwner->getConsole().OLn("Read %d bytes of total %d bytes BITMAPFILEHEADER", bytesread, sizeof(BITMAPFILEHEADER));
97 ReadFile(bitmapfile, &info_header, sizeof(BITMAPINFOHEADER), &bytesread, NULL);
98 _pOwner->getConsole().OLn("Read %d bytes of total %d bytes BITMAPINFOHEADER", bytesread, sizeof(BITMAPINFOHEADER));
99
100 if ( info_header.biCompression != BI_RGB )
101 return loadBMPfail(bitmapfile, pNewImage, palette, "ERROR: info_header.biCompression != BI_RGB, returning PGENULL!");
102
103 _pOwner->getConsole().OLn("W x H x Bpp: %d x %d x %d", info_header.biWidth, info_header.biHeight, info_header.biBitCount);
104
105 // Once I managed to save a bmp in Photoshop where biHeight was negative, so we need to make sure everything we have is positive
106 info_header.biWidth = abs(info_header.biWidth);
107 info_header.biHeight = abs(info_header.biHeight);
108 info_header.biBitCount = (WORD) abs((int) info_header.biBitCount);
109
110 if ( (bitmaplength = info_header.biWidth * info_header.biHeight * (info_header.biBitCount < 32 ? 3 : 4)) == 0 )
111 return loadBMPfail(bitmapfile, pNewImage, palette, "ERROR: bitmaplength == 0, returning returning PGENULL!");
112
113 try
114 {
115 if ( info_header.biClrUsed > 0 )
116 {
117 palettesize = info_header.biClrUsed * sizeof(RGBQUAD);
118 palette = new RGBQUAD[palettesize];
119 ReadFile(bitmapfile, palette, palettesize, &bytesread, NULL);
120 _pOwner->getConsole().OLn("Read %d bytes of total %d bytes of RGBQUAD array for palette", bytesread, palettesize);
121 if ( bytesread != palettesize )
122 {
123 return loadBMPfail(bitmapfile, pNewImage, palette, "ERROR: bytesread != palettesize, returning PGENULL!");
124 }
125 }
126
127 pNewImage = new PureImage();
128 // PURE_RGB is just a fake value at this point
129 if ( !pNewImage->pImpl->initMembers(info_header.biBitCount, info_header.biWidth, info_header.biHeight,
130 PURE_RGB, PURE_RGB, true, false, PGENULL, bitmaplength) )
131 return loadBMPfail(bitmapfile, pNewImage, palette, "ERROR: initMembers() failed, returning PGENULL!");
132
133 _pOwner->getConsole().OLn("offset: %d", file_header.bfOffBits);
134
135 if ( !pNewImage->pImpl->readBMPpixels(bitmapfile, palette, info_header.biBitCount) )
136 return loadBMPfail(bitmapfile, pNewImage, palette, "ERROR: readBMPpixels() failed!");
137
138 CloseHandle(bitmapfile);
139 delete[] palette;
140
141 _pOwner->getConsole().SOLnOO("> bitmap successfully loaded into memory, using %d bytes (%d kbytes)",
142 sizeof(pNewImage)+pNewImage->getPixelsSize(), (sizeof(*pNewImage)+pNewImage->getPixelsSize())/1024);
143 } // try
144 catch (const std::bad_alloc&)
145 {
146 return loadBMPfail(bitmapfile, pNewImage, palette, "ERROR: failed to allocated palette!");
147 }
148
149 return pNewImage;
150
151} // loadBMP()

227TPureUInt PureImageManager::getColorConversionSwapCount(TPIXCOMPORD from, TPIXCOMPORD to)
228{
229 if ( (from % 2) == 1 )
230 from = (TPIXCOMPORD)(from-1);
231 if ( (to % 2) == 1 )
232 to = (TPIXCOMPORD)(to-1);
233 if ( from == to )
234 return 0;
235
236 switch (from)
237 {
238 case PURE_RGB: return (to == PURE_GBR) || (to == PURE_BRG) ? 2 : 1;
239 case PURE_RBG: return (to == PURE_GRB) || (to == PURE_BGR) ? 2 : 1;
240 case PURE_GRB: return (to == PURE_RBG) || (to == PURE_BGR) ? 2 : 1;
241 case PURE_BGR: return (to == PURE_RBG) || (to == PURE_GRB) ? 2 : 1;
242 case PURE_GBR: return (to == PURE_RGB) || (to == PURE_BRG) ? 2 : 1;
243 default /* BRG*/: return (to == PURE_RGB) || (to == PURE_GBR) ? 2 : 1;
244 }
245} // getColorConversionSwapCount()

252TPIXCOMPORD PureImageManager::getMirroredPixelComponentOrder(TPIXCOMPORD corder)
253{
254 switch (corder)
255 {
256 case PURE_RGB : return PURE_BGR;
257 case PURE_RBG : return PURE_GBR;
258 case PURE_GBR : return PURE_RBG;
259 case PURE_GRB : return PURE_BRG;
260 case PURE_BGR : return PURE_RGB;
261 case PURE_BRG : return PURE_GRB;
262 case PURE_RGBA: return PURE_BGRA;
263 case PURE_RBGA: return PURE_GBRA;
264 case PURE_GBRA: return PURE_RBGA;
265 case PURE_GRBA: return PURE_BRGA;
266 case PURE_BGRA: return PURE_RGBA;
267 case PURE_BRGA: return PURE_GRBA;

269 }
270} // getMirroredPixelComponentOrder()

365PureImage* PureImageManager::createFromFile(const char* filename)
366{
367 getConsole().OLnOI("PureImageManager::createFromFile(\"%s\")", filename);
368
369 if ( filename == NULL )
370 return pImpl->createFromFileFail("ERROR: NULLPOINTER!");
371
372 if ( !PFL::fileExists(filename) )
373 return pImpl->createFromFileFail("ERROR: file doesn't exist!");
374
375 string sFileExt = PFL::getExtension(filename);
376 if ( sFileExt == "" )
377 return pImpl->createFromFileFail("ERROR: no file extension!");
378
379#pragma warning(disable:4244) /* int-char conversion in std::transform */
380 transform(sFileExt.begin(), sFileExt.end(), sFileExt.begin(), ::toupper);
381#pragma warning(default:4244)
382 getConsole().OLn("ext: .%s, ", sFileExt.c_str());
383 PureImage* pNewImage;
384
385 if ( sFileExt == "BMP" )
386 {
387 getConsole().OLn("calling loadBMP()... ");
388 pNewImage = pImpl->loadBMP(filename);
389 }
390 else return pImpl->createFromFileFail("ERROR: unsupported extension!");
391
392 if ( pNewImage == PGENULL )
393 return pImpl->createFromFileFail("ERROR: loadXXX() returned PGENULL!");
394
395 pNewImage->SetName("Image " + std::to_string(pImpl->nRunningCounter++));
396 pNewImage->SetFilename( filename );
397 Attach( *pNewImage );
398
399 getConsole().SOLnOO("> Image loaded, name: %s!", pNewImage->getName().c_str());
400 getConsole().OLn("");
401
402 return pNewImage;
403} // createFromFile()

415PureImage* PureImageManager::createBlank(TPureUInt width, TPureUInt height, TPureUInt bpp)
416{
417 if ( (width == 0) || (height == 0) || (bpp != 24) )
418 return PGENULL;
419
420 PureImage* pNewImage = PGENULL;
421 try
422 {
423 pNewImage = new PureImage();
424
425 // ppxls = PGENULL means it won't try to copy anything to the allocated buffer so the content of the buffer will be undefined.
426 if ( !pNewImage->pImpl->initMembers(bpp, width, height, PURE_RGB, PURE_RGB, true, false, PGENULL, width * height * bpp/8) )
427 {
428 delete pNewImage;
429 getConsole().EOLn("ERROR: Failed initMembers(%d, %d, %d, ...)", bpp, width, height);
430 getConsole().OLn("");
431 return PGENULL;
432 }
433
434 Attach( *pNewImage );
435
436 pNewImage->SetName("Image " + std::to_string(pImpl->nRunningCounter++));
437
438 getConsole().SOLnOO("> Blank Image created, name: %s!", pNewImage->getName().c_str());
439 getConsole().OLn("");
440 }
441 catch (const std::bad_alloc&)
442 {
443 getConsole().EOLn("ERROR: failed to instantiate PureImage!");
444 getConsole().OLn("");
445 }
446
447 return pNewImage;
448}
