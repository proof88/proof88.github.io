73PureOctree::ChildIndex PureOctree::calculateIndex(const PureVector& pos) const
74{
75 ChildIndex iChild = 0; // TopLeftFront
76
77 if ( pos.getX() >= vPos.getX() )
78 {
79 iChild |= RIGHT;
80 }
81 if ( pos.getY() < vPos.getY() )
82 {
83 iChild |= BOTTOM;
84 }
85 if ( pos.getZ() >= vPos.getZ() )
86 {
87 iChild |= BACK;
88 }
89
90 return iChild;
91} // calculateIndex()

105PureOctree* PureOctree::insertObject(const PureObject3D& obj)
106{
107 if ( (obj.getPosVec().getX() < vPos.getX() - fSize/2.f) || (obj.getPosVec().getX() > vPos.getX() + fSize/2.f) )
108 return PGENULL;
109
110 if ( (obj.getPosVec().getY() < vPos.getY() - fSize/2.f) || (obj.getPosVec().getY() > vPos.getY() + fSize/2.f) )
111 return PGENULL;
112
113 if ( (obj.getPosVec().getZ() < vPos.getZ() - fSize/2.f) || (obj.getPosVec().getZ() > vPos.getZ() + fSize/2.f) )
114 return PGENULL;
115
116 if ( getNodeType() == LeafEmpty )
117 { // simply store the object at this level
118 vObjects.insert(&obj);
119 return this;
120 }
121
122 if ( getNodeType() == Parent )
123 { // pass the object to next depth level
124 const ChildIndex iChild = calculateIndex(obj.getPosVec());
125 return vChildren[iChild]->insertObject(obj);
126 }
127
128 // getNodeType() == LeafContainer
129 if ( nCurrentDepth == nMaxDepth )
130 {
131 // cannot subdivide more, so just simply store the object at this last level
132 vObjects.insert(&obj);
133 return this;
134 }
135 else
136 {
137 // subdivide this node now, depth level increases
138 if ( !subdivide() )
139 {
140 return PGENULL;
141 }
142 for (TPureUInt i = 0; i < vChildren.size(); i++)
143 {
144 vChildren[i]->parent = this;
145 }
146 vChildren[TOP | LEFT | FRONT]->vPos.Set(vPos.getX() - fSize/4.f, vPos.getY() + fSize/4.f, vPos.getZ() - fSize/4.f);
147 vChildren[BOTTOM | LEFT | FRONT]->vPos.Set(vPos.getX() - fSize/4.f, vPos.getY() - fSize/4.f, vPos.getZ() - fSize/4.f);
148 vChildren[TOP | RIGHT | FRONT]->vPos.Set(vPos.getX() + fSize/4.f, vPos.getY() + fSize/4.f, vPos.getZ() - fSize/4.f);
149 vChildren[BOTTOM | RIGHT | FRONT]->vPos.Set(vPos.getX() + fSize/4.f, vPos.getY() - fSize/4.f, vPos.getZ() - fSize/4.f);
150 vChildren[TOP | LEFT | BACK ]->vPos.Set(vPos.getX() - fSize/4.f, vPos.getY() + fSize/4.f, vPos.getZ() + fSize/4.f);
151 vChildren[BOTTOM | LEFT | BACK ]->vPos.Set(vPos.getX() - fSize/4.f, vPos.getY() - fSize/4.f, vPos.getZ() + fSize/4.f);
152 vChildren[TOP | RIGHT | BACK ]->vPos.Set(vPos.getX() + fSize/4.f, vPos.getY() + fSize/4.f, vPos.getZ() + fSize/4.f);
153 vChildren[BOTTOM | RIGHT | BACK ]->vPos.Set(vPos.getX() + fSize/4.f, vPos.getY() - fSize/4.f, vPos.getZ() + fSize/4.f);
154
155 // add the already inserted object to one of the new child nodes, remove it from this node
156 const PureObject3D* objAlreadyInTree = *(vObjects.begin());
157 vObjects.clear();
158 const ChildIndex iChildForObjAlreadyInTree = calculateIndex(objAlreadyInTree->getPosVec());
159 vChildren[iChildForObjAlreadyInTree]->insertObject(*objAlreadyInTree);
160
161 // add the new object also to one of the new child nodes
162 const ChildIndex iChild = calculateIndex(obj.getPosVec());
163 return vChildren[iChild]->insertObject(obj);
164 }
165} // insertObject()

177const PureOctree* PureOctree::findObject(const PureObject3D& obj) const
178{
179 if ( getNodeType() == LeafEmpty )
180 {
181 return PGENULL;
182 }
183 else if ( getNodeType() == Parent )
184 {
185 // request the suitable children on next depth level to try find the object
186 const ChildIndex iChild = calculateIndex(obj.getPosVec());
187 return vChildren[iChild]->findObject(obj);
188 }
189
190 // getNodeType() == LeafContainer
191 if ( vObjects.find(&obj) != vObjects.end() )
192 {
193 return this;
194 }
195 else
196 {
197 return PGENULL;
198 }
199} // findObject()

330TPureBool PureOctree::subdivide()
331{
332 try
333 {
334 for (TPureUInt i = 0; i < 8; i++)
335 {
336 PureOctree* const pChildNode = new PureOctree(PureVector(), fSize/2.f, nMaxDepth, nCurrentDepth + 1);
337 vChildren.push_back(pChildNode);
338 }
339 }
340 catch (const std::exception&)
341 {
342 for (TPureUInt i = 0; i < 8; i++)
343 {
344 if ( vChildren[i] != PGENULL )
345 {
346 delete vChildren[i];
347 }
348 vChildren.resize(0);
349 }
350 return false;
351 }
352 return true;
353}

356void PureOctree::DeleteChildren()
