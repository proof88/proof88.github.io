137TPureBool PureGLsnippets::setVertexIndex(void* arr, TPureUInt index, TPureUInt value, GLenum iType)
138{
139 switch (iType)
140 {
141 case GL_UNSIGNED_BYTE :
142 if ( value > UCHAR_MAX )
143 {
144 CConsole::getConsoleInstance(sLoggerModuleName).EOLn("setVertexIndex() ERROR: storing %d as BYTE (MAX=%d)!", value, UCHAR_MAX);
145 return false;
146 }
147 ((TPureUByte*)arr)[index] = (TPureUByte) value;
148 break;
149 case GL_UNSIGNED_SHORT:
150 if ( value > USHRT_MAX )
151 {
152 CConsole::getConsoleInstance(sLoggerModuleName).EOLn("setVertexIndex() ERROR: storing %d as SHORT (MAX=%d)!", value, USHRT_MAX);
153 return false;
154 }
155 ((TPureUShort*)arr)[index] = (TPureUShort) value;
156 break;
157 default: // GL_UNSIGNED_INT
158 ((TPureUInt*)arr)[index] = value;
159 break;
160 }
161 return true;
162} // setVertexIndex()

169void PureGLsnippets::SetGLBoundingBoxRendering(TPureBool state)
170{
171 if ( state )
172 {
173 // obj->Draw() must not set these states when the given renderPass is PURE_RPASS_SYNC_OCCLUSION_START
174 glColorMask(GL_FALSE, GL_FALSE, GL_FALSE, GL_FALSE);
175 glEnable(GL_DEPTH_TEST);
176 glDepthMask(GL_FALSE);
177 glDisable(GL_CULL_FACE);
178 glDisable(GL_LIGHTING);
179 glPolygonMode(GL_FRONT_AND_BACK, GL_FILL);
180 glDisable(GL_ALPHA_TEST);
181 glLoadTexturesAndSetBlendState(PGENULL, false, false);
182 }
183 else
184 {
185 glColorMask(GL_TRUE, GL_TRUE, GL_TRUE, GL_TRUE);
186 glDepthMask(GL_TRUE);
187 glFlush(); // this to make sure any initiated GL query is actually starts to be processed by GPU so later we dont have to wait that much for results
188 }
189} // SetGLBoundingBoxRendering()

283void PureGLsnippets::glLoadTextureIntoTMU(const PureTexture* tex, TPureUInt iTMU, TPureBool bSticked)
284{
285 /* Make sure we don't call GL functions when no accelerator is present */
286 if ( !PureHwInfo::get().getVideo().isAcceleratorDetected() )
287 return;
288
289 // currently we support only 2 textured layers ...
290 if ( iTMU > 1 )
291 {
292 CConsole::getConsoleInstance(sLoggerModuleName).EOLn("ERROR: glLoadTextureIntoTMU(tex, %d)", iTMU);
293 assert( false );
294 }
295
296 if ( PureHwInfo::get().getVideo().isMultiTexturingSupported() )
297 {
298 glActiveTextureARB(GL_TEXTURE0_ARB + iTMU);
299 if ( tex != PGENULL )
300 {
301 glEnable(GL_TEXTURE_2D);
302 if ( tex->getInternalNum() != iLastTex[iTMU] )
303 {
304 glBindTexture(GL_TEXTURE_2D, tex->getInternalNum());
305 iLastTex[iTMU] = tex->getInternalNum();
306 }
307 glTexEnvf(GL_TEXTURE_ENV, GL_TEXTURE_ENV_MODE, GL_MODULATE);
308 }
309 else
310 glDisable(GL_TEXTURE_2D);
311 }
312 else
313 {
314 if ( tex != PGENULL )
315 {
316 glEnable(GL_TEXTURE_2D);
317 if ( tex->getInternalNum() != iLastTex[iTMU] )
318 {
319 glBindTexture(GL_TEXTURE_2D, tex->getInternalNum());
320 iLastTex[iTMU] = tex->getInternalNum();
321 }
322 if ( bSticked )
323 glTexEnvf(GL_TEXTURE_ENV, GL_TEXTURE_ENV_MODE, GL_REPLACE);
324 else
325 glTexEnvf(GL_TEXTURE_ENV, GL_TEXTURE_ENV_MODE, GL_MODULATE);
326 }
327 else
328 glDisable(GL_TEXTURE_2D);
329 }
330} // glLoadTextureIntoTMU()

344void PureGLsnippets::glLoadTexturesAndSetBlendState(const PureMaterial* mat, TPureBool bObjLevel1Sticked, TPureBool bObjLevel1Blended)
345{
346 if ( PGENULL == mat )
347 {
348 glDisable(GL_BLEND);
349 glLoadTextureIntoTMU( PGENULL, 0, bObjLevel1Sticked );
350 if ( PureHwInfo::get().getVideo().isMultiTexturingSupported() )
351 {
352 glLoadTextureIntoTMU( PGENULL, 1, bObjLevel1Sticked );
353 }
354 return;
355 }
356
357 if ( PureHwInfo::get().getVideo().isMultiTexturingSupported() )
358 {
359 if ( mat->isMultiTextured() )
360 {
361 // enable blending of 2nd layer
362 glEnable(GL_BLEND);
363 glBlendFunc(getGLBlendFromPureBlend(mat->getSourceBlendFunc(1)), getGLBlendFromPureBlend(mat->getDestinationBlendFunc(1)));
364 glLoadTextureIntoTMU( mat->getTexture(0), 0, bObjLevel1Sticked );
365 glLoadTextureIntoTMU( mat->getTexture(1), 1, bObjLevel1Sticked );
366 }
367 else
368 {
369 // disable blending only if base layer is not blended and level 1 object3d is also not blended ...
370 if ( !bObjLevel1Blended && !PureMaterial::isBlendFuncReallyBlending(mat->getSourceBlendFunc(), mat->getDestinationBlendFunc()) )
371 {
372 glDisable(GL_BLEND);
373 }
374 glLoadTextureIntoTMU( mat->getTexture(0), 0, bObjLevel1Sticked );
375 glLoadTextureIntoTMU( PGENULL, 1, bObjLevel1Sticked );
376 }
377 }
378 else
379 {
380 // disable blending only if base layer is not blended and level 1 object3d is also not blended ...
381 if ( !bObjLevel1Blended && !PureMaterial::isBlendFuncReallyBlending(mat->getSourceBlendFunc(), mat->getDestinationBlendFunc()) )
382 {
383 glDisable(GL_BLEND);
384 }
385 glLoadTextureIntoTMU( mat->getTexture(), 0, bObjLevel1Sticked );
386 }
387} // glLoadTexturesAndSetBlendState()
