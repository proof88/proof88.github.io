198PureMaterial* PureMaterialManager::createMaterial()
199{
200 if ( !pImpl->isInitialized() )
201 {
202 return PGENULL;
203 }
204
205 getConsole().OLnOI("PureMaterialManager::createMaterial()");
206
207 PureMaterial* mat = PGENULL;
208 try
209 {
210 mat = new PureMaterial(pImpl->nMaxLayers);
211 Attach( *mat );
212
213 mat->SetName("Material " + std::to_string(pImpl->nRunningCounter++));
214
215 getConsole().SOLnOO("> Material created successfully!");
216 }
217 catch (const std::bad_alloc&)
218 {
219 getConsole().EOLnOO("ERROR: Failed to instantiate new PureMaterial, name: %s!", mat->getName().c_str());
220 }
221 getConsole().OLn("");
222
223 return mat;
224
225} // createMaterial()
