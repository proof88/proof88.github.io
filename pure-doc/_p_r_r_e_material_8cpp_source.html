67 TPRREbool PRREMaterial::PRREMaterialImpl::allocateArrays(TPRREuint nColorCount, TPRREuint nTexcoordCount, TPRREuint nIndexCount, TPRREuint nIndexSize)
68 {
69  if ( nIndexSize == 0 )
70  {
71  _pOwner->getManagedConsole().EOLn("PRREMaterial()::allocateArrays(%d, %d, %d, %d) fail due to nIndexSize!", nColorCount, nTexcoordCount, nIndexCount, nIndexSize);
72  return false;
73  }
74 
75  if ( getTexcoordsCount(0) > 0 )
76  {
77  _pOwner->getManagedConsole().SOLn("PRREMaterial()::allocateArrays(%d, %d, %d, %d) skip, already allocated!", nColorCount, nTexcoordCount, nIndexCount, nIndexSize);
78  return true;
79  }
80 
81  nIndices = nIndexCount;
82  nIndexSize = nIndexSize;
83 
84  try
85  {
86  for (TPRREuint i = 0; i < ((PRREMaterialManager*)_pOwner->getManager())->getMaximumLayerCount(); i++)
87  {
88  layers[i].nColors_h = nColorCount;
89  layers[i].pColors = new TRGBAFLOAT[nColorCount];
90  layers[i].nTexcoords_h = nTexcoordCount;
91  layers[i].pTexcoords = new TUVW[nTexcoordCount];
92  }
93  return true;
94  }
95  catch (const std::bad_alloc&)
96  {
97  _pOwner->getManagedConsole().EOLn("PRREMaterial()::allocateArrays(%d, %d, %d, %d) fail due to out of memory!", nColorCount, nTexcoordCount, nIndexCount, nIndexSize);
98  for (TPRREuint i = 0; i < ((PRREMaterialManager*)_pOwner->getManager())->getMaximumLayerCount(); i++)
99  {
100  layers[i].nColors_h = 0;
101  delete[] layers[i].pColors;
102  layers[i].pColors = PGENULL;
103  layers[i].nTexcoords_h = 0;
104  delete[] layers[i].pTexcoords;
105  layers[i].pTexcoords = PGENULL;
106  }
107  return false;
108  }
109 }

290 TPRREbool PRREMaterial::PRREMaterialImpl::setSourceBlendFunc(TPRRE_BLENDFACTOR value, TPRREuint level)
291 {
292  if ( !_pOwner->getUtiliser() || !_pOwner->getUtiliser()->getManager() )
293  {
294  // this way we permit the user setting blending again when the utiliser Object3D is not attached to its manager,
295  // so it can be assured that in case of reattach, it is still non-blended, without even checking blending state!
296  _pOwner->getManagedConsole().EOLn("ERROR: %s either no utiliser or utiliser has no parent!", __FUNCTION__);
297  return false;
298  }
299 
300  if ( (value != PRRE_SRC_COLOR) && (value != PRRE_ONE_MINUS_SRC_COLOR) )
301  {
302  if ( level < ((PRREMaterialManager*)_pOwner->getManager())->getMaximumLayerCount() )
303  {
304  layers[level].blendFactorSource = value;
305 
306  // remember: Object3D is also a manager, so it could also implement its own HandleManagedPropertyChanged(),
307  // to handle changes in the material of its subobjects! Since Object3D doesnt implement such,
308  // the PRREManager's base HandleManagedPropertyChanged() is invoked when subobject's material is changed here!
309  _pOwner->getUtiliser()->getManager()->HandleManagedPropertyChanged( *(_pOwner->getUtiliser()) );
310 
311  return true;
312  }
313  }
314  return false;
315 } // setSourceBlendFunc()

399 TPRREbool PRREMaterial::PRREMaterialImpl::copyFromMaterial(PRREMaterial& srcMat, TPRREuint dstLevel, TPRREuint srcLevel)
400 {
401  if ( (dstLevel < ((PRREMaterialManager*)_pOwner->getManager())->getMaximumLayerCount()) &&
402  (srcLevel < ((PRREMaterialManager*)_pOwner->getManager())->getMaximumLayerCount()) )
403  {
404  // we also want to make sure that we have the same memory layout of texcoords and indices
405  // so we can safely copy data into our preallocated arrays
406  if ( (layers[dstLevel].nTexcoords_h == srcMat.pImpl->layers[srcLevel].nTexcoords_h) &&
407  (layers[dstLevel].nColors_h == srcMat.pImpl->layers[srcLevel].nColors_h) &&
408  (nIndices == srcMat.getIndicesCount()) &&
409  (nIndexSize == srcMat.pImpl->nIndexSize) )
410  {
411  if ( srcMat.getTexture(srcLevel) != NULL )
412  {
413  if ( !setTexture( srcMat.getTexture(srcLevel), dstLevel ) )
414  {
415  _pOwner->getManagedConsole().EOLn("ERROR: %s setTexture() failed for dstLevel %d!", __FUNCTION__, dstLevel);
416  return false;
417  }
418  }
419 
420  memcpy( layers[dstLevel].pTexcoords, srcMat.pImpl->layers[srcLevel].pTexcoords, sizeof(TUVW) * layers[dstLevel].nTexcoords_h );
421  memcpy( layers[dstLevel].pColors, srcMat.pImpl->layers[srcLevel].pColors, sizeof(TRGBAFLOAT) * layers[dstLevel].nColors_h );
422 
423  return true;
424  }
425  else
426  ; // TODO: set lasterr
427  }
428  else
429  ; // TODO: set lasterr
430 
431  return false;
432 }

448 PRREMaterial::PRREMaterialImpl::PRREMaterialImpl(
449  PRREMaterial* owner, TPRREuint nLayers)
450 {
451  _pOwner = owner;
452  _pOwner->getManagedConsole().OLnOI("PRREMaterial() ...");
453 
454  // sanity check
455  if ( nLayers < 1 )
456  nLayers = 1;
457 
458  for (TPRREuint i = 0; i < nLayers; i++)
459  {
460  layers.push_back( TPRRE_MATERIAL_LAYER() );
461  layers[ layers.size()-1 ].nColors_h = 0;
462  layers[ layers.size()-1 ].pColors = PGENULL;
463  layers[ layers.size()-1 ].nTexcoords_h = 0;
464  layers[ layers.size()-1 ].pTexcoords = PGENULL;
465  layers[ layers.size()-1 ].tex = PGENULL;
466  layers[ layers.size()-1 ].texEnvMode = PRRE_TEX_FUNC_MODULATE;
467  layers[ layers.size()-1 ].fTransparancy = 1.0f;
468  layers[ layers.size()-1 ].clrTexEnv.Set(255, 255, 255, 255);
469  layers[ layers.size()-1 ].blendFactorSource = PRRE_ONE;
470  layers[ layers.size()-1 ].blendFactorDestination = PRRE_ZERO;
471  }
472 
473  nIndices = 0;
474  fShininessBackFace = fShininessFrontFace = 1.0f;
475 
476  // we dont store nLayers because from now layers.size() is fix until this instance is deleted
477 
478  _pOwner->getManagedConsole().SOLnOO("Done!");
479 } // PRREObject3D()
