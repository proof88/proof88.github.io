67 TPRREbool PRREMaterial::PRREMaterialImpl::allocateArrays(TPRREuint nColorCount, TPRREuint nTexcoordCount, TPRREuint nIndexCount, TPRREuint nIndexSize)
68 {
69  if ( nIndexSize == 0 )
70  {
71  _pOwner->getManagedConsole().EOLn("PRREMaterial()::allocateArrays(%d, %d, %d, %d) fail due to nIndexSize!", nColorCount, nTexcoordCount, nIndexCount, nIndexSize);
72  return false;
73  }
74 
75  if ( getTexcoordsCount(0) > 0 )
76  {
77  _pOwner->getManagedConsole().SOLn("PRREMaterial()::allocateArrays(%d, %d, %d, %d) skip, already allocated!", nColorCount, nTexcoordCount, nIndexCount, nIndexSize);
78  return true;
79  }
80 
81  nIndices = nIndexCount;
82  nIndexSize = nIndexSize;
83 
84  try
85  {
86  for (TPRREuint i = 0; i < ((PRREMaterialManager*)_pOwner->getManager())->getMaximumLayerCount(); i++)
87  {
88  layers[i].nColors_h = nColorCount;
89  layers[i].pColors = new TRGBAFLOAT[nColorCount];
90  layers[i].nTexcoords_h = nTexcoordCount;
91  layers[i].pTexcoords = new TUVW[nTexcoordCount];
92  }
93  return true;
94  }
95  catch (const std::bad_alloc&)
96  {
97  _pOwner->getManagedConsole().EOLn("PRREMaterial()::allocateArrays(%d, %d, %d, %d) fail due to out of memory!", nColorCount, nTexcoordCount, nIndexCount, nIndexSize);
98  for (TPRREuint i = 0; i < ((PRREMaterialManager*)_pOwner->getManager())->getMaximumLayerCount(); i++)
99  {
100  layers[i].nColors_h = 0;
101  delete[] layers[i].pColors;
102  layers[i].pColors = PGENULL;
103  layers[i].nTexcoords_h = 0;
104  delete[] layers[i].pTexcoords;
105  layers[i].pTexcoords = PGENULL;
106  }
107  return false;
108  }
109 }

371 TPRREbool PRREMaterial::PRREMaterialImpl::copyFromMaterial(PRREMaterial& srcMat, TPRREuint dstLevel, TPRREuint srcLevel)
372 {
373  if ( (dstLevel < ((PRREMaterialManager*)_pOwner->getManager())->getMaximumLayerCount()) &&
374  (srcLevel < ((PRREMaterialManager*)_pOwner->getManager())->getMaximumLayerCount()) )
375  {
376  // we also want to make sure that we have the same memory layout of texcoords and indices
377  // so we can safely copy data into our preallocated arrays
378  if ( (layers[dstLevel].nTexcoords_h == srcMat.pImpl->layers[srcLevel].nTexcoords_h) &&
379  (layers[dstLevel].nColors_h == srcMat.pImpl->layers[srcLevel].nColors_h) &&
380  (nIndices == srcMat.getIndicesCount()) &&
381  (nIndexSize == srcMat.pImpl->nIndexSize) )
382  {
383  if ( srcMat.getTexture(srcLevel) != NULL )
384  {
385  if ( !setTexture( srcMat.getTexture(srcLevel), dstLevel ) )
386  {
387  _pOwner->getManagedConsole().EOLn("ERROR: %s setTexture() failed for dstLevel %d!", __FUNCTION__, dstLevel);
388  return false;
389  }
390  }
391 
392  memcpy( layers[dstLevel].pTexcoords, srcMat.pImpl->layers[srcLevel].pTexcoords, sizeof(TUVW) * layers[dstLevel].nTexcoords_h );
393  memcpy( layers[dstLevel].pColors, srcMat.pImpl->layers[srcLevel].pColors, sizeof(TRGBAFLOAT) * layers[dstLevel].nColors_h );
394 
395  return true;
396  }
397  else
398  ; // TODO: set lasterr
399  }
400  else
401  ; // TODO: set lasterr
402 
403  return false;
404 }

420 PRREMaterial::PRREMaterialImpl::PRREMaterialImpl(
421  PRREMaterial* owner, TPRREuint nLayers)
422 {
423  _pOwner = owner;
424  _pOwner->getManagedConsole().OLnOI("PRREMaterial() ...");
425 
426  // sanity check
427  if ( nLayers < 1 )
428  nLayers = 1;
429 
430  for (TPRREuint i = 0; i < nLayers; i++)
431  {
432  layers.push_back( TPRRE_MATERIAL_LAYER() );
433  layers[ layers.size()-1 ].nColors_h = 0;
434  layers[ layers.size()-1 ].pColors = PGENULL;
435  layers[ layers.size()-1 ].nTexcoords_h = 0;
436  layers[ layers.size()-1 ].pTexcoords = PGENULL;
437  layers[ layers.size()-1 ].tex = PGENULL;
438  layers[ layers.size()-1 ].texEnvMode = PRRE_TEX_FUNC_MODULATE;
439  layers[ layers.size()-1 ].fTransparancy = 1.0f;
440  layers[ layers.size()-1 ].clrTexEnv.Set(255, 255, 255, 255);
441  layers[ layers.size()-1 ].blendFactorSource = PRRE_ONE;
442  layers[ layers.size()-1 ].blendFactorDestination = PRRE_ZERO;
443  }
444 
445  nIndices = 0;
446  fShininessBackFace = fShininessFrontFace = 1.0f;
447 
448  // we dont store nLayers because from now layers.size() is fix until this instance is deleted
449 
450  _pOwner->getManagedConsole().SOLnOO("Done!");
451 } // PRREObject3D()
