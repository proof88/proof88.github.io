74 PRREOctree::ChildIndex PRREOctree::calculateIndex(const PRREVector& pos) const
75 {
76  ChildIndex iChild = 0; // TopLeftFront
77 
78  if ( pos.getX() >= vPos.getX() )
79  {
80  iChild |= RIGHT;
81  }
82  if ( pos.getY() < vPos.getY() )
83  {
84  iChild |= BOTTOM;
85  }
86  if ( pos.getZ() >= vPos.getZ() )
87  {
88  iChild |= BACK;
89  }
90 
91  return iChild;
92 } // calculateIndex()

106 PRREOctree* PRREOctree::insertObject(const PRREObject3D& obj)
107 {
108  if ( (obj.getPosVec().getX() < vPos.getX() - fSize/2.f) || (obj.getPosVec().getX() > vPos.getX() + fSize/2.f) )
109  return PGENULL;
110 
111  if ( (obj.getPosVec().getY() < vPos.getY() - fSize/2.f) || (obj.getPosVec().getY() > vPos.getY() + fSize/2.f) )
112  return PGENULL;
113 
114  if ( (obj.getPosVec().getZ() < vPos.getZ() - fSize/2.f) || (obj.getPosVec().getZ() > vPos.getZ() + fSize/2.f) )
115  return PGENULL;
116 
117  if ( getNodeType() == LeafEmpty )
118  { // simply store the object at this level
119  vObjects.insert(&obj);
120  return this;
121  }
122 
123  if ( getNodeType() == Parent )
124  { // pass the object to next depth level
125  const ChildIndex iChild = calculateIndex(obj.getPosVec());
126  return vChildren[iChild]->insertObject(obj);
127  }
128 
129  // getNodeType() == LeafContainer
130  if ( nCurrentDepth == nMaxDepth )
131  {
132  // cannot subdivide more, so just simply store the object at this last level
133  vObjects.insert(&obj);
134  return this;
135  }
136  else
137  {
138  // subdivide this node now, depth level increases
139  if ( !subdivide() )
140  {
141  return PGENULL;
142  }
143  for (TPRREuint i = 0; i < vChildren.size(); i++)
144  {
145  vChildren[i]->parent = this;
146  }
147  vChildren[TOP | LEFT | FRONT]->vPos.Set(vPos.getX() - fSize/4.f, vPos.getY() + fSize/4.f, vPos.getZ() - fSize/4.f);
148  vChildren[BOTTOM | LEFT | FRONT]->vPos.Set(vPos.getX() - fSize/4.f, vPos.getY() - fSize/4.f, vPos.getZ() - fSize/4.f);
149  vChildren[TOP | RIGHT | FRONT]->vPos.Set(vPos.getX() + fSize/4.f, vPos.getY() + fSize/4.f, vPos.getZ() - fSize/4.f);
150  vChildren[BOTTOM | RIGHT | FRONT]->vPos.Set(vPos.getX() + fSize/4.f, vPos.getY() - fSize/4.f, vPos.getZ() - fSize/4.f);
151  vChildren[TOP | LEFT | BACK ]->vPos.Set(vPos.getX() - fSize/4.f, vPos.getY() + fSize/4.f, vPos.getZ() + fSize/4.f);
152  vChildren[BOTTOM | LEFT | BACK ]->vPos.Set(vPos.getX() - fSize/4.f, vPos.getY() - fSize/4.f, vPos.getZ() + fSize/4.f);
153  vChildren[TOP | RIGHT | BACK ]->vPos.Set(vPos.getX() + fSize/4.f, vPos.getY() + fSize/4.f, vPos.getZ() + fSize/4.f);
154  vChildren[BOTTOM | RIGHT | BACK ]->vPos.Set(vPos.getX() + fSize/4.f, vPos.getY() - fSize/4.f, vPos.getZ() + fSize/4.f);
155 
156  // add the already inserted object to one of the new child nodes, remove it from this node
157  const PRREObject3D* objAlreadyInTree = *(vObjects.begin());
158  vObjects.clear();
159  const ChildIndex iChildForObjAlreadyInTree = calculateIndex(objAlreadyInTree->getPosVec());
160  vChildren[iChildForObjAlreadyInTree]->insertObject(*objAlreadyInTree);
161 
162  // add the new object also to one of the new child nodes
163  const ChildIndex iChild = calculateIndex(obj.getPosVec());
164  return vChildren[iChild]->insertObject(obj);
165  }
166 } // insertObject()

178 const PRREOctree* PRREOctree::findObject(const PRREObject3D& obj) const
179 {
180  if ( getNodeType() == LeafEmpty )
181  {
182  return PGENULL;
183  }
184  else if ( getNodeType() == Parent )
185  {
186  // request the suitable children on next depth level to try find the object
187  const ChildIndex iChild = calculateIndex(obj.getPosVec());
188  return vChildren[iChild]->findObject(obj);
189  }
190 
191  // getNodeType() == LeafContainer
192  if ( vObjects.find(&obj) != vObjects.end() )
193  {
194  return this;
195  }
196  else
197  {
198  return PGENULL;
199  }
200 } // findObject()

331 TPRREbool PRREOctree::subdivide()
332 {
333  try
334  {
335  for (TPRREuint i = 0; i < 8; i++)
336  {
337  PRREOctree* const pChildNode = new PRREOctree(PRREVector(), fSize/2.f, nMaxDepth, nCurrentDepth + 1);
338  vChildren.push_back(pChildNode);
339  }
340  }
341  catch (const std::exception&)
342  {
343  for (TPRREuint i = 0; i < 8; i++)
344  {
345  if ( vChildren[i] != PGENULL )
346  {
347  delete vChildren[i];
348  }
349  vChildren.resize(0);
350  }
351  return false;
352  }
353  return true;
354 }
