68TPureBool PureImage::PureImageImpl::setPixelComponentOrder(TPURE_PIXEL_COMPONENT_ORDER cord)
69{
70 _pOwner->getManagedConsole().OLn("PureImage::setPixelComponentOrder(%PXCO)", cord);
71
72 if ( pPixels == PGENULL )
73 return false;
74
75 if ( cord == clrCompOrder )
76 {
77 _pOwner->getManagedConsole().OLn(" WARNING: nothing to do, equal component orders!");
78 _pOwner->getManagedConsole().OLn(" clrCompOrder == %PXCO", clrCompOrder);
79 return true;
80 }
81
82 bool retVal = true;
83 if ( (abs(cord-clrCompOrder) > 1) ||
84 ( ((cord-clrCompOrder == 1) && (cord % 2 == 0)) ||
85 ((cord-clrCompOrder == -1) && (cord % 2 == 1)) ) )
86 {
87 bChanged = true;
88 TPureUInt swapcount = PureImageManager::getColorConversionSwapCount(cord, clrCompOrder);
89 _pOwner->getManagedConsole().OLn(" swapcount = %d", swapcount);
90 while ( swapcount > 0 )
91 {
92 _pOwner->getManagedConsole().OI();
93 retVal &= swapColors(clrCompOrder, cord, (TPureByte) swapcount);
94 swapcount--;
95 _pOwner->getManagedConsole().OO();
96 }
97 }
98
99 _pOwner->getManagedConsole().OLn("");
100 return retVal;
101} // setPixelComponentOrder()

104PureColor PureImage::PureImageImpl::getPixel(TPureUInt x, TPureUInt y)
105{
106 PureColor ret;
107
108 if ( pPixels == PGENULL )
109 return ret;
110
111 if ( (x >= nWidth) || (y >= nHeight) )
112 return ret;
113
114 TPureUInt index = getIndexToPixel(x,y);
115 TPureByte rIndex, gIndex, bIndex, aIndex;
116 if ( !setColorComponentsIndices(rIndex, gIndex, bIndex, aIndex, clrCompOrder) )
117 return ret;
118
119 ret.Set(pPixels[index+rIndex], pPixels[index+gIndex], pPixels[index+bIndex], aIndex == -1 ? 255 : pPixels[index+aIndex]);
120
121 return ret;
122} // getPixel()

223TPureBool PureImage::PureImageImpl::cannibalize(PureImage& victim)
224{
225 // Note that we cannot easily copy victim's Impl since copy ctor and assignment operators are empty!
226 // Thus the following assignment cannot be used: *pImpl = *(victim.pImpl)
227 // We wouldn't even want to use copy ctor or assignment here since we do not want any array copy.
228 // That is why this function is written, to "steal" the arrays of the victim so victim will be "empty".
229 // In C++11 this could be done with a move ctor.
230
231 _pOwner->SetName(victim.getName()); // copy the Managed part
232 _pOwner->SetFilename(victim.getFilename()); // copy the FiledManaged part
233
234 nBits = victim.pImpl->nBits;
235 nWidth = victim.pImpl->nWidth;
236 nHeight = victim.pImpl->nHeight;
237 clrCompOrderOrig = victim.pImpl->clrCompOrderOrig;
238 clrCompOrder = victim.pImpl->clrCompOrder;
239 bUpsideDown = victim.pImpl->bUpsideDown;
240 bChanged = victim.pImpl->bChanged;
241 nSizePixels = victim.pImpl->nSizePixels;
242 pPixels = victim.pImpl->pPixels;
243
244 victim.pImpl->nBits = 0;
245 victim.pImpl->nWidth = 0;
246 victim.pImpl->nHeight = 0;
247 victim.pImpl->bChanged = true;
248 victim.pImpl->nSizePixels = 0;
249 victim.pImpl->pPixels = PGENULL;
250
251 return true;
252
253}

270TPureBool PureImage::PureImageImpl::setColorComponentsIndices(TPureByte& r, TPureByte& g, TPureByte& b, TPureByte& a, const TPIXCOMPORD pxcord)
271{
272 a = (pxcord % 2) == 1 ? 3 : -1;
273
274 switch (pxcord)
275 {
276 case PURE_RGBA: case PURE_RGB: r = 0; g = 1; b = 2; break;
277 case PURE_RBGA: case PURE_RBG: r = 0; g = 2; b = 1; break;
278 case PURE_BGRA: case PURE_BGR: r = 2; g = 1; b = 0; break;
279 case PURE_BRGA: case PURE_BRG: r = 1; g = 2; b = 0; break;
280 case PURE_GBRA: case PURE_GBR: r = 2; g = 0; b = 1; break;
281 case PURE_GRBA: case PURE_GRB: r = 1; g = 0; b = 2; break;
282 default: return false;
283 }
284 return true;
285} // setColorComponentsIndices()

315 return PURE_RGBA;

374TPureBool PureImage::PureImageImpl::swapColors(TPURE_PIXEL_COMPONENT_ORDER from, TPURE_PIXEL_COMPONENT_ORDER to, TPureByte swapcount)
375{
376 _pOwner->getManagedConsole().OLn("PureImage::swapColors(%PXCO, %PXCO, %d)", from, to, swapcount);
377 if ( swapcount < 1 )
378 {
379 _pOwner->getManagedConsole().OLn("");
380 return true;
381 }
382 else if ( swapcount == 2 )
383 {
384 to = getIntermediatePixelCompOrder(from, to);
385 _pOwner->getManagedConsole().OLn(" New target order is %PXCO.", to);
386 }
387
388 TPureByte dstR, dstG, dstB, dstA;
389 if ( !setColorComponentsIndices(dstR, dstG, dstB, dstA, to) )
390 return false;
391
392 TPureByte srcR, srcG, srcB, srcA;
393 if ( !setColorComponentsIndices(srcR, srcG, srcB, srcA, from) )
394 return false;
395
396 _pOwner->getManagedConsole().OLn(" dstR, srcR == %d, %d", dstR, srcR);
397 _pOwner->getManagedConsole().OLn(" dstG, srcG == %d, %d", dstG, srcG);
398 _pOwner->getManagedConsole().OLn(" dstB, srcB == %d, %d", dstB, srcB);
399
400 clrCompOrder = to;
401
402 TPureBool done = false;
403 TPureUByte tmp = 0;
404
405 TPureByte srcIdx, dstIdx;
406 for (TPureByte i = 0; i <= 2 && !done; i++)
407 {
408 switch (i)
409 {
410 case 0: srcIdx = srcR;
411 dstIdx = dstR;
412 break;
413 case 1: srcIdx = srcG;
414 dstIdx = dstG;
415 break;
416 default: srcIdx = srcB;
417 dstIdx = dstB;
418 }
419 if ( dstIdx-srcIdx != 0 )
420 {
421 done = true;
422 for (TPureUInt x = 0; x < nWidth; x++)
423 {
424 for (TPureUInt y = 0; y < nHeight; y++)
425 {
426 tmp = pPixels[getIndexToPixel(x,y)+srcIdx];
427 pPixels[getIndexToPixel(x,y)+srcIdx] = pPixels[getIndexToPixel(x,y)+dstIdx];
428 pPixels[getIndexToPixel(x,y)+dstIdx] = tmp;
429 } // for y
430 } // for x
431 }
432 } // for i
433
434 _pOwner->getManagedConsole().OLn("");
435 return true;
436} // swapColors()

512unsigned char* PureImage::PureImageImpl::readPaletteIndices(const HANDLE f)
513{
514 const DWORD nPaletteIndexArraySize = nWidth * nHeight * sizeof(unsigned char) / (8 / nBits);
515 unsigned char* pPaletteIndexArray = NULL;
516 try
517 {
518 pPaletteIndexArray = new unsigned char[nPaletteIndexArraySize];
519 DWORD bytesread;
520 ReadFile(f, pPaletteIndexArray, nPaletteIndexArraySize, &bytesread, NULL);
521 if ( bytesread != nPaletteIndexArraySize )
522 {
523 _pOwner->getManagedConsole().EOLn("ERROR: bytesread != nPaletteIndexArraySize: %d != %d", bytesread, nPaletteIndexArraySize);
524 delete[] pPaletteIndexArray;
525 return NULL;
526 }
527 }
528 catch (const std::bad_alloc&)
529 {
530 _pOwner->getManagedConsole().EOLn(" ERROR: failed to allocate pPaletteIndexArray!");
531 }
532
533 return pPaletteIndexArray;
534}

541TPureBool PureImage::PureImageImpl::readBMP8pixels(const HANDLE f, const RGBQUAD* palette)
542{
543 if ( (nWidth % 4) != 0 )
544 {
545 _pOwner->getManagedConsole().EOLn("ERROR: 8 bpp image nWidth doesnt divisible by 4 (%d mod 4 == %d)!",
546 nWidth, nWidth % 4);
547 return false;
548 }
549
550 // we already have the palette, but still need to grab the indices into this palette for each pixel
551 unsigned char* const pPaletteIndexArray = readPaletteIndices(f);
552 if ( !pPaletteIndexArray )
553 return false;
554
555 TPureUInt k = 0;
556 for (TPureUInt y = 0; y < nHeight; ++y)
557 for (TPureUInt x = 0; x < nWidth; ++x)
558 {
559 for (TPureUInt c = 0; c < 3; ++c)
560 pPixels[(y*nWidth+x)*3+c] = ((const TPureUByte*) palette)[pPaletteIndexArray[k]*4+2-c];
561 k++;
562 }
563
564 delete[] pPaletteIndexArray;
565 clrCompOrder = PURE_RGB;
566 clrCompOrderOrig = clrCompOrder;
567 nBits = 24;
568 return true;
569} // readBMP8pixels()

576TPureBool PureImage::PureImageImpl::readBMP4pixels(HANDLE f, const RGBQUAD* palette)
577{
578 if ( (nWidth % 8) != 0 )
579 {
580 _pOwner->getManagedConsole().EOLn("ERROR: 4 bpp image nWidth doesnt divisible by 8 (%d mod 8 == %d)!",
581 nWidth, nWidth % 8);
582 return false;
583 }
584
585 // we already have the palette, but still need to grab the indices into this palette for each pixel
586 unsigned char* const pPaletteIndexArray = readPaletteIndices(f);
587 if ( !pPaletteIndexArray )
588 return false;
589
590 TPureUInt k = 0;
591 for (TPureUInt y = 0; y < nHeight; ++y)
592 {
593 for (TPureUInt x = 0; x < nWidth; x += 2)
594 {
595 for (int c = 0; c < 3; ++c)
596 pPixels[(y*nWidth+x)*3+c] = ((const TPureUByte*) palette)[pPaletteIndexArray[k]/16*4+2-c];
597
598 for (int c = 0; c < 3; ++c)
599 pPixels[(y*nWidth+(x+1))*3+c] = ((const TPureUByte*) palette)[pPaletteIndexArray[k]%16*4+2-c];
600
601 k++;
602 }
603 }
604
605 delete[] pPaletteIndexArray;
606 clrCompOrder = PURE_RGB;
607 clrCompOrderOrig = clrCompOrder;
608 nBits = 24;
609 return true;
610} // readBMP4pixels()
