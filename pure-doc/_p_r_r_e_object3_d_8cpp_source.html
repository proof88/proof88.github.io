364 void PRREObject3D::PRREObject3DImpl::Draw(bool bLighting)
365 {
366  // caller renderer is expected to check for GL errors, so we don't check them here
367 
368  if ( !bVisible )
369  return;
370 
371  if ( !PRREhwInfo::get().getVideo().isAcceleratorDetected() )
372  {
373  DrawSW();
374  return;
375  }
376 
377  // see if we are parent or referring to another object i.e. we are cloned object
378  if ( _pOwner->isLevel1() || getReferredObject() )
379  {
380  // Currently 3D objects are 2-level entities:
381  // first level (parent) has no geometry, owns subobjects, sets basic things,
382  // while second level (subobjects) own geometry, inherit basic things set by parent.
383 
384  // So transformations and other basic things are set by parent objects having subobjects.
385  // Or by cloned objects which refer to another original objects but still have their own position, angle, etc.
386 
387  ApplyTransformations();
388  PrepareGLbeforeDraw(bLighting);
389 
390  // take care of attached objects (subobjects)
391  // note that either we have our own subobjects, OR we are just a cloned object and we need to draw original object's subobjects
392  PRREObject3D* pWhichParent = getReferredObject() ? getReferredObject() : _pOwner;
393  pWhichParent->pImpl->bParentInitiatedOperation = true;
394  for (TPRREint i = 0; i < pWhichParent->getCount(); i++)
395  ((PRREObject3D*)pWhichParent->getAttachedAt(i))->Draw(bLighting);
396  pWhichParent->pImpl->bParentInitiatedOperation = false;
397  return;
398  }
399 
400  // if we reach this point then either our parent is drawing us as its subobject, or a cloned object is drawing us on behalf of our parent
401 
402  // subobject must ignore draw if its Draw() was not called by its parent level-1 object but someone else from outside ...
403  if ( !((PRREObject3D*)_pOwner->getManager())->pImpl->bParentInitiatedOperation )
404  {
405  _pOwner->getManagedConsole().EOLn("Draw() of subobject called outside of its level-1 parent object, ignoring draw!");
406  return;
407  }
408 
409  if ( PRREhwInfo::get().getVideo().isMultiTexturingSupported() )
410  {
411  if ( _pOwner->getMaterial().isMultiTextured() )
412  {
413  // enable blending of 2nd layer
414  glEnable(GL_BLEND);
415  glBlendFunc(getGLblendFromPRREblend(_pOwner->getMaterial().getSourceBlendFunc(1)), getGLblendFromPRREblend(_pOwner->getMaterial().getDestinationBlendFunc(1)));
416  LoadTextureIntoTMU( _pOwner->getMaterial().getTexture(0), 0 );
417  LoadTextureIntoTMU( _pOwner->getMaterial().getTexture(1), 1 );
418  }
419  else
420  {
421  // disable blending only if base layer is not blended ...
422  if ( ! PRREObject3DManager::isBlendFuncBlends(_pOwner->getMaterial().getSourceBlendFunc(), _pOwner->getMaterial().getDestinationBlendFunc()) )
423  glDisable(GL_BLEND);
424  LoadTextureIntoTMU( _pOwner->getMaterial().getTexture(0), 0 );
425  LoadTextureIntoTMU( PGENULL, 1 );
426  }
427  }
428  else
429  {
430  // disable blending only if base layer is not blended ...
431  if ( ! PRREObject3DManager::isBlendFuncBlends(_pOwner->getMaterial().getSourceBlendFunc(), _pOwner->getMaterial().getDestinationBlendFunc()) )
432  glDisable(GL_BLEND);
433  LoadTextureIntoTMU( _pOwner->getMaterial().getTexture() );
434  }
435 
436  /* currently not supporting any vendor-specific mode */
437  if ( BITF_READ(_pOwner->getVertexTransferMode(),PRRE_VT_VENDOR_BITS,3) != 0 )
438  return;
439 
440  if ( pFbBuffer == NULL )
441  {
442  /* We create this buffer only once at first run. */
443  /* We intentionally switch to feedback mode only once since now it is for debugging purposes only, for breakpoint, etc.
444  but in the future when HW transform and SW rasterization can be combined, this will be controlled by public API */
445 
446  /* num of values can be stored in this buffer */
447  /* we need to store not only the transformed values but also some extra values for grammar */
448  /* in this case, the items placed in this array are polygon items (GL_POLYGON_TOKEN) which look like this:
449  GL_POLYGON_TOKEN 3 (value value value value (value value value value) (value value value value))
450  So for a standard cube where each side has 2 triangles, it is 12 triangles in total, meaning
451  12 polygons. GL_POLYGON_TOKEN itself is 1 value, 3 itself is also 1 value, and then 3*12 values for 3 vertices.
452  So the total num of values is: numOfTriangles * (2+3*12)
453  Example with some real data is in later comment with debugbuffer[]. */
454 
455 
456  nFbBuffer_h = GLsizei(ceil((_pOwner->getVerticesCount(false) / 3.0f))) * (2+3*12);
457  try
458  {
459  /* TODO: probably in future we should rather use pVerticesTransf here as well since we already have it for that purpose, right? ;) */
460  pFbBuffer = new GLfloat[nFbBuffer_h];
461 
462  /* unfortunately only the most detailed option GL_4D_COLOR_TEXTURE will give us the w-coord of vertices so we need to use that */
463  if ( pglFeedbackBuffer(nFbBuffer_h, GL_4D_COLOR_TEXTURE, pFbBuffer) )
464  {
465  // If you want all vertices to be transformed and catched in feedback mode then dont forget to disable culling and depth testing (maybe only 1 is needed to be disabled).
466  //glDisable(GL_DEPTH_TEST);
467  glDisable(GL_CULL_FACE); /* otherwise only the front facing side of cube would be written to feedback buffer */
468 
469  glRenderMode(GL_FEEDBACK);
470  }
471  }
472  catch (const std::bad_alloc&)
473  {
474  _pOwner->getManagedConsole().EOLn("ERROR: PRREObject3D::PRREObject3DImpl::Draw() failed to allocate pFbBuffer!");
475  }
476  }
477 
478  ((PRREVertexTransfer*)_pOwner)->pImpl->TransferVertices();
479 
480  /* The number of values (not vertices) transferred to the feedback buffer. */
481  const GLint nFbBufferWritten_h = glRenderMode(GL_RENDER);
482 
483  GLfloat debugbuffer[500];
484 
485  if ( nFbBufferWritten_h > 0 )
486  {
487  // we are happy
488  for (int i = 0; i < min(500,nFbBufferWritten_h); i++)
489  debugbuffer[i] = pFbBuffer[i];
490  //Sleep(1); // suitable for placing breakpoint
491  /*
492  debugbuffer[0] = GL_POLYGON_TOKEN;
493  debugbuffer[1] = 3;
494  debugbuffer[ 2.. 5] = x y z w (xyz are in window coordinates while w is clip coordinate)
495  debugbuffer[ 6.. 9] = r g b a
496  debugbuffer[10..13] = texture coords, probably x y u w
497  debugbuffer[14..17] = x y z w (xyz are in window coordinates while w is clip coordinate)
498  debugbuffer[18..21] = r g b a
499  debugbuffer[22..25] = texture coords, probably x y u w
500  debugbuffer[26..29] = x y z w (xyz are in window coordinates while w is clip coordinate)
501  debugbuffer[30..33] = r g b a
502  debugbuffer[34..37] = texture coords, probably x y u w
503  debugbuffer[38] = GL_POLYGON_TOKEN;
504  debugbuffer[39] = 3;
505 
506  example: gamedata\\models\\cube.obj
507  assuming that nearplane is 0.1f, farplane is 100.0f, depthrange is [0, 1], fovy is 80
508  box
509  i=2 (3. vertex) debugbuffer[26]
510  model space -> world space -> view/eye space -> clip space -> normalized device coords -> screen coords
511  sw: [-0.5, -0.5, -0.5] -> [-0.5, -0.5, 2.5] -> [-0.5, -0.5, 2.5] -> [-0.446, -0.595, 2.348, 2.5] -> [-0.178, -0.238, 0.939, 2.5] -> [328, 228, 0.961, 2.5]
512  hw: [328, 228, 0.961, 2.5]
513 
514  i=6 (7. vertex) debugbuffer[?]
515  model space -> world space -> view/eye space -> clip space -> normalized device coords -> screen coords
516  sw: [ 0.5, 0.5, 0.5] -> [ 0.5, 0.5, 3.5] -> [ 0.5, 0.5, 3.5] -> [ 0.446, 0.595, 3.368, 3.5] -> [ 0.127, 0.170, 0.962, 3.5] -> [451, 351, 0.972, 3.5]
517  hw: [451, 351, 0.972, 3.5]
518 
519  ...
520  */
521  }
522  else
523  {
524  // we were already in GL_RENDER (0) mode, OR nothing was returned, OR we were in FEEDBACK but buffer was not enough to hold all values (<0) ...
525  }
526 } // Draw()

592 void PRREObject3D::PRREObject3DImpl::LoadTextureIntoTMU(const PRRETexture* tex, TPRREuint iTMU) const
593 {
594  /* Make sure we don't call GL functions when no accelerator is present */
595  if ( !PRREhwInfo::get().getVideo().isAcceleratorDetected() )
596  return;
597 
598  // TODO: the PR00FPS-style lasttex method is not present here currently ... later it should be implemented.
599  // A general TMU manager should record which TMU holds which texture, so that can avoid unnecessary glBindTexture() calls.
600 
601  // currently we support only 2 textured layers ...
602  if ( iTMU > 1 )
603  {
604  _pOwner->getManagedConsole().EOLn("ERROR: LoadTextureIntoTMU(tex, %d)", iTMU);
605  _ASSERT( false );
606  }
607 
608  if ( PRREhwInfo::get().getVideo().isMultiTexturingSupported() )
609  {
610  glActiveTextureARB(GL_TEXTURE0_ARB + iTMU);
611  if ( tex != PGENULL )
612  {
613  glEnable(GL_TEXTURE_2D);
614  glBindTexture(GL_TEXTURE_2D, tex->getInternalNum());
615  glTexEnvf(GL_TEXTURE_ENV, GL_TEXTURE_ENV_MODE, GL_MODULATE);
616  }
617  else
618  glDisable(GL_TEXTURE_2D);
619  }
620  else
621  {
622  if ( tex != PGENULL )
623  {
624  glEnable(GL_TEXTURE_2D);

640 void PRREObject3D::PRREObject3DImpl::ApplyTransformations() const
641 {
642  /* Make sure we don't call GL functions when no accelerator is present */
643  if ( !PRREhwInfo::get().getVideo().isAcceleratorDetected() )
644  return;
645 
646  glTranslatef(_pOwner->getPosVec().getX(), _pOwner->getPosVec().getY(), _pOwner->getPosVec().getZ());
647 
648  switch ( getRotationOrder() )
649  {
650  case PRRE_XYZ:
651  glRotatef(getAngleVec().getX(), 1,0,0);
652  glRotatef(getAngleVec().getY(), 0,1,0);
653  glRotatef(getAngleVec().getZ(), 0,0,1);
654  break;
655  case PRRE_XZY:
656  glRotatef(getAngleVec().getX(), 1,0,0);
657  glRotatef(getAngleVec().getZ(), 0,0,1);
658  glRotatef(getAngleVec().getY(), 0,1,0);
659  break;
660  case PRRE_YXZ:
661  glRotatef(getAngleVec().getY(), 0,1,0);
662  glRotatef(getAngleVec().getX(), 1,0,0);
663  glRotatef(getAngleVec().getZ(), 0,0,1);
664  break;
665  case PRRE_YZX:
666  glRotatef(getAngleVec().getY(), 0,1,0);
667  glRotatef(getAngleVec().getZ(), 0,0,1);
668  glRotatef(getAngleVec().getX(), 1,0,0);
669  break;
670  case PRRE_ZXY:
671  glRotatef(getAngleVec().getZ(), 0,0,1);
672  glRotatef(getAngleVec().getX(), 1,0,0);
673  glRotatef(getAngleVec().getY(), 0,1,0);
674  break;
675  case PRRE_ZYX:
676  glRotatef(getAngleVec().getZ(), 0,0,1);
677  glRotatef(getAngleVec().getY(), 0,1,0);
678  glRotatef(getAngleVec().getX(), 1,0,0);
679  break;
680  } // switch
681 
682  glScalef(getScaling().getX(), getScaling().getY(), getScaling().getZ());
683  /*
684  Well, mirror-transforming an object (scaling by negative value) makes its faces flip:
685  https://www.gamedev.net/forums/topic/640616-negative-scaling-flips-face-winding-affects-backface-culling/
686  I wanted to X-mirror objects loaded by tmcsgfxlib-to-prre wrapper for legacy projects, because I noticed they appeared mirrored compared to proofps engine behavior.
687  So I noticed when setting negative scale on X axis, faces also flipped, so objects became inside-out.
688  This is expected since vertices' winding order also flips because we still feed triangle vertices in same order even though triangles become mirrored.
689  But I don't treat this as a legacy-project-only issue, since even a new project with the new engine could come up for any reason with negative scaling for any mirroring reason.
690  So, this is rather a feature: we need to detect negative scaling factor and change front face winding order on-the-fly.
691  */
692  if ( getScaling().getX() * getScaling().getY() * getScaling().getZ() <= (0.0f-PFL::E) )
693  glFrontFace(GL_CW);
694  else
695  glFrontFace(GL_CCW);
696 }
