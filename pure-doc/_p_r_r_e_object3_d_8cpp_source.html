32  Vertex Culling (Object-Space)
33  *****************************
34 
35  GL_EXT_cull_vertex: https://www.khronos.org/registry/OpenGL/extensions/EXT/EXT_cull_vertex.txt
36  - 1996
37  - "Culling a polygon by examining its vertexes in object space can be more efficient than screen space polygon culling since
38  the transformation to screen space (which may include a division by w) can be avoided for culled vertexes. Also, vertex culling
39  can be computed before vertexes are assembled into primitives. This is a useful property when drawing meshes with shared vertexes,
40  since a vertex can be culled once, and the resulting state can be used for all primitives which share the vertex."
41  - This extension was most probably introduced for optimisation before HW T&L spread. Lack of HW T&L was present in the pre-Geforce era,
42  i.e. TNT, etc.
43  Desktop: was mostly supported by early integrated Intel VGA.
44  Mobile: no support at all!
45  - Verdict: don't use it if neither 3dfx nor other major vendors supported it before TnL!

88  // Remember: we cannot use isLevel1() or isLevel2() here because isLevel2() might return true for
89  // a level-1 object that was originally a cloned object but not referring to any object anymore after
90  // the original object has been deleted!
91  // We might check for actual vertex data because level-1 objects never have vertex data, but I think
92  // in the future even level-2 objects might also lose vertex data in their client memory if it is
93  // already uploaded to host memory.
94  // So, this dtor must execute operations keeping mind that the same code is executed for both level-1

97  // objects, all referrers are also get deleted automatically.

101  pRefersto->pImpl->referrers.erase(_pOwner);
102  }
103  else if ( referrers.size() > 0 )
104  {
105  // referrers now become zombie objects: they do not refer but they dont have any subobjects either!
106  // user's responsibility to get rid of them if original object is removed!
107  for (auto& referrer : referrers)
108  referrer->pImpl->pRefersto = PGENULL;
109  }
110 
111  SetOcclusionTested(false);
112 
113  // these are way unnecessary things if we are a level-2 object, but as explained above, we have to live
114  // with it - at least these are all harmless code even when executed for a level-2.
115  std::set< std::deque<PRREObject3D*>* > mapsEraseObjectFrom;
116  // would be much nicer with Cpp11 initializer list
117  mapsEraseObjectFrom.insert(&occluders);
118  mapsEraseObjectFrom.insert(&occludees_opaque);
119  mapsEraseObjectFrom.insert(&occludees_blended);
120  mapsEraseObjectFrom.insert(&occludees_2d_opaque);
121  mapsEraseObjectFrom.insert(&occludees_2d_blended);
122 
123  for (auto pMapEraseFrom : mapsEraseObjectFrom )

126  while ( occ_it != pMapEraseFrom->end() )
127  {
128  pMapEraseFrom->erase(occ_it);
129  occ_it = std::find(pMapEraseFrom->begin(), pMapEraseFrom->end(), _pOwner);
130  }
131  } // end for pMapEraseFrom
132 
133  _pOwner->getManagedConsole().OI();
134  _pOwner->DeleteAll();
135  _pOwner->getManagedConsole().OO();
136 
137  _pOwner->getManagedConsole().SOLnOO("Done!");
138 } // ~PRREObject3DImpl()

614 void PRREObject3D::PRREObject3DImpl::SetOcclusionTested(TPRREbool state)
615 {
616  // It can happen at rare cases e.g. at engine shutdown that isLevel1() returns false due to we are cloned object NOT having the referred object anymore,
617  // because it has just been deleted prior to deleting us (SetOcclusionTested(false) is being invoked by our dtor).
618  // This case should be somehow identified to NOT log it since that is misleading to log this call as we were really a level-2 object!
619  // I could not find a way to detect at this point that isLevel1() returns false just because our referred object has been already deleted.
620  // So, instead of just checking isLevel1() right at the beginning of the function:
621  // - check for isLevel1() only when state is true, because that path we definitely don't want to be executed for level-2 objects, it must be logged as error;
622  // - and it is definitely deleting objects case, since dtors set state to false, not true!
623  // - do not check for isLevel1() when given state is false because it that code path doesn't have any effect when invoked for a real level-2 object.
624 
625  if ( !PRREhwInfo::get().getVideo().isOcclusionQuerySupported() )
626  return;
627 
628  if ( state )
629  {
630  if ( PGENULL == _pOwner->getManager() )
631  {
632  throw std::runtime_error("no manager!");
633  }
634 
635  if ( !_pOwner->isLevel1() )

1039 void PRREObject3D::PRREObject3DImpl::Draw_ApplyTransformations() const
1040 {
1041  assert(_pOwner->isLevel1());
1042 
1043  /* Make sure we don't call GL functions when no accelerator is present */
1044  if ( !PRREhwInfo::get().getVideo().isAcceleratorDetected() )
1045  return;
1046 
1047  glTranslatef(_pOwner->getPosVec().getX(), _pOwner->getPosVec().getY(), _pOwner->getPosVec().getZ());

1053  glRotatef(getAngleVec().getY(), 0,1,0);
1054  glRotatef(getAngleVec().getZ(), 0,0,1);

1058  glRotatef(getAngleVec().getZ(), 0,0,1);
1059  glRotatef(getAngleVec().getY(), 0,1,0);
1060  break;
1061  case PRRE_YXZ:
1062  glRotatef(getAngleVec().getY(), 0,1,0);
1063  glRotatef(getAngleVec().getX(), 1,0,0);
1064  glRotatef(getAngleVec().getZ(), 0,0,1);
1065  break;
1066  case PRRE_YZX:
1067  glRotatef(getAngleVec().getY(), 0,1,0);
1068  glRotatef(getAngleVec().getZ(), 0,0,1);
1069  glRotatef(getAngleVec().getX(), 1,0,0);
1070  break;
1071  case PRRE_ZXY:
1072  glRotatef(getAngleVec().getZ(), 0,0,1);
1073  glRotatef(getAngleVec().getX(), 1,0,0);
1074  glRotatef(getAngleVec().getY(), 0,1,0);
1075  break;
1076  case PRRE_ZYX:
1077  glRotatef(getAngleVec().getZ(), 0,0,1);
1078  glRotatef(getAngleVec().getY(), 0,1,0);

1080  break;
1081  } // switch
1082

1084  /*
1085  Well, mirror-transforming an object (scaling by negative value) makes its faces flip:
1086  https://www.gamedev.net/forums/topic/640616-negative-scaling-flips-face-winding-affects-backface-culling/
1087  I wanted to X-mirror objects loaded by tmcsgfxlib-to-prre wrapper for legacy projects, because I noticed they appeared mirrored compared to proofps engine behavior.

1089  This is expected since vertices' winding order also flips because we still feed triangle vertices in same order even though triangles become mirrored.
1090  But I don't treat this as a legacy-project-only issue, since even a new project with the new engine could come up for any reason with negative scaling for any mirroring reason.
1091  So, this is rather a feature: we need to detect negative scaling factor and change front face winding order on-the-fly.
1092  */
1093  if ( getScaling().getX() * getScaling().getY() * getScaling().getZ() <= (0.0f-PFL::E) )
1094  glFrontFace(GL_CW);
1095  else
1096  glFrontFace(GL_CCW);
1097 } // Draw_ApplyTransformations()

1365 TPRREbool PRREObject3D::PRREObject3DImpl::draw_OcclusionQuery_Finish(TPRREbool bASyncQuery, TPRREbool bRenderIfQueryPending)
1366 {
1367  assert(_pOwner->isLevel1());
1368 
1369  if ( nOcclusionQuery == 0 )
1370  {
1371  return false; // let it be rendered
1372  }
1373 
1374  if ( !bOcclusionQueryStarted )
1375  {
1376  // we always end up here with legacy renderer which never starts query for any objects but
1377  // still invokes this function (i.e. Object3D::Draw() invokes it)
1378 
1379  if ( bASyncQuery )
1380  {
1381  if ( bOccluded && (OQ_MAX_FRAMES_WO_START_QUERY_WHEN_OCCLUDED == 0) )
1382  { // if query is not started, but it should had been started if constant value says it should not wait for another frame,
1383  // then this is a programmer error condition
1384  if ( !isOccluder() )
1385  { // abort only if this is really an occludee: because renderer won't start occlusion query for occluders beforehand!
1386  assert(false); // this just cannot happen
1387  }
1388  }
1389  return bOccluded; // let it be rendered if last result says not occluded
1390  }
1391  else
1392  {
1393  // it is ok to render with either legacy renderer or when occlusion query is in sync
1394  // since in latter case bOcclusionQueryStarted is false only when it is not tested for occlusion
1395  return false; // let it be rendered
1396  }
1397  }
1398 
1399  // this should be done rather with chrono after we have C++11
1400  PFL::timeval begin, end;
1401  PFL::gettimeofday(&begin, 0);
1402 
1403  GLint queryResultAvailable;
1404  do {
1405  glGetQueryObjectivARB(nOcclusionQuery, GL_QUERY_RESULT_AVAILABLE_ARB, &queryResultAvailable);
1406  } while (!queryResultAvailable && !bASyncQuery);
1407 
1408  // update sync statistics related to wait time
1409  if ( !bASyncQuery )
1410  {
1411  PFL::gettimeofday(&end, 0);
1412  if ( PFL::updateForMaxDuration(timeLongestWaitForSyncQueryFinish, begin, end) )
1413  {
1414  PFL::updateForMaxDuration(PRREObject3D::PRREObject3DImpl::stats[PRREObject3D::PRREObject3DImpl::stats.size()-1].timeLongestGlobalWaitForSyncQueryFinish, begin, end);
1415  }
1416  } // sync stats update
1417 
1418  if ( queryResultAvailable == GL_FALSE )
1419  {
1420  // this is async query
1421  nFramesWaitedForOcclusionTestResult++;
1422  return bRenderIfQueryPending ? false : bOccluded;
1423  }
1424 
1425  bOcclusionQueryStarted = false;
1426 
1427  if ( PRREhwInfo::get().getVideo().isBooleanOcclusionQuerySupported() )
1428  {
1429  GLint sampleBoolean;
1430  glGetQueryObjectivARB(nOcclusionQuery, GL_QUERY_RESULT_ARB, &sampleBoolean);
1431  bOccluded = ( sampleBoolean == GL_FALSE );
1432  }
1433  else
1434  {
1435  GLuint sampleCount;
1436  glGetQueryObjectuivARB(nOcclusionQuery, GL_QUERY_RESULT_ARB, &sampleCount);
1437  bOccluded = ( sampleCount == 0 );
1438  }
1439 
1440  // update async statistics
1441  if ( bASyncQuery )
1442  {
1443  if ( nFramesWaitedForOcclusionTestResult < nFramesWaitedForOcclusionTestResultMin )
1444  {
1445  nFramesWaitedForOcclusionTestResultMin = nFramesWaitedForOcclusionTestResult;
1446  if ( nFramesWaitedForOcclusionTestResultMin < PRREObject3D::PRREObject3DImpl::stats[PRREObject3D::PRREObject3DImpl::stats.size()-1].nFramesWaitedForOcclusionTestResultGlobalMin )
1447  {
1448  PRREObject3D::PRREObject3DImpl::stats[PRREObject3D::PRREObject3DImpl::stats.size()-1].nFramesWaitedForOcclusionTestResultGlobalMin = nFramesWaitedForOcclusionTestResultMin;
1449  }
1450  }
1451  if ( nFramesWaitedForOcclusionTestResult > nFramesWaitedForOcclusionTestResultMax )
1452  {
1453  nFramesWaitedForOcclusionTestResultMax = nFramesWaitedForOcclusionTestResult;
1454  if ( nFramesWaitedForOcclusionTestResultMax > PRREObject3D::PRREObject3DImpl::stats[PRREObject3D::PRREObject3DImpl::stats.size()-1].nFramesWaitedForOcclusionTestResultGlobalMax )
1455  {
1456  PRREObject3D::PRREObject3DImpl::stats[PRREObject3D::PRREObject3DImpl::stats.size()-1].nFramesWaitedForOcclusionTestResultGlobalMax = nFramesWaitedForOcclusionTestResultMax;
1457  }
1458  }
1459  nFramesWaitedForOcclusionTestResult = 0;
1460  nFramesWithoutOcclusionTest = 0;
1461  }
1462 
1463  return bOccluded;
1464 } // draw_OcclusionQuery_Finish()
