235void PureTransformMatrix::SetFrustum(float left, float right, float bottom, float top, float n, float f)
236{
237 /* to understand this implementation, you can check my matrix calculations done on paper here:
238 https://proof88.github.io/pure-doc/architecture.html */
239
240 if ( (n <= 0.0f) || (n >= f) )
241 return;
242
243 if ( (right <= left) || (top <= bottom) )
244 return;
245
246 SetZero();
247
248 /* 1st row */
249 mat[0][0] = 2*n / (right-left);
250 mat[0][2] = (right+left) / (right-left);
251
252 /* 2nd row */
253 mat[1][1] = 2*n / (top-bottom);
254 mat[1][2] = (top+bottom) / (top-bottom);
255
256 /* 3rd row */
257 mat[2][2] = (f+n) / (f-n);
258 mat[2][3] = -2*f*n / (f-n);
259
260 /* 4th row */
261 mat[3][2] = 1.0f;
262 mat[3][3] = 0.0f; /* because SetZero had set it as 1 */
263}

278void PureTransformMatrix::SetFrustumByFovX(float fovx, float aspect, float zNear, float zFar)
279{
280 /* to understand this implementation, you can check my matrix calculations done on paper here:
281 https://proof88.github.io/pure-doc/architecture.html */
282
283 if ( (fovx <= 0.0f) || (aspect <= 0.0f) || (zNear <= 0.0f) || (zNear >= zFar) )
284 return;
285
286 fovx = fovx * PFL::PI/180;
287
288 const float left = -(zNear * tan(fovx / 2)); /* left needs to be negative for sure */
289 const float right = -left;
290
291 const float top = right / aspect;
292 const float bottom = -top;
293
294 SetFrustum(left, right, bottom, top, zNear, zFar);
295}

309void PureTransformMatrix::SetFrustumByFovY(float fovy, float aspect, float zNear, float zFar)
310{
311 /* to understand this implementation, you can check my matrix calculations done on paper here:
312 https://proof88.github.io/pure-doc/architecture.html */
313
314 if ( (fovy <= 0.0f) || (aspect <= 0.0f) || (zNear <= 0.0f) || (zNear >= zFar) )
315 return;
316
317 fovy = fovy * PFL::PI/180;
318
319 const float top = zNear * tan(fovy / 2);
320 const float bottom = -top;
321
322 const float right = top * aspect;
323 const float left = -right;
324
325 SetFrustum(left, right, bottom, top, zNear, zFar);

337void PureTransformMatrix::SetLookAt(
338 const PureVector& vPos,
339 const PureVector& vTarget,
340 const PureVector& vUp)
341{
342 /*
343
344 My implementation doesn't equal to any of these:
345 https://www.scratchapixel.com/lessons/mathematics-physics-for-computer-graphics/lookat-function
346 https://www.khronos.org/opengl/wiki/GluLookAt_code
347 https://github.com/g-truc/glm/blob/master/glm/gtc/matrix_transform.inl#L756
348
349 And above implementations differ from each other a bit. But still good material to read and learn.
350 */
351
352 PureVector vForwardFromTargetAndPos = vTarget - vPos;
353 vForwardFromTargetAndPos.Normalize();
354
355 PureVector vAltUp = vUp;
356
357 // this is to detect when our calculated forward vector is parallel with Y-axis and that would make cross product zero right vector ...
358 // todo: switch to quaternions
359 if ( (abs(vForwardFromTargetAndPos.getX()) < PFL::E) && (abs(vForwardFromTargetAndPos.getZ()) < PFL::E) )
360 {
361 if ( vForwardFromTargetAndPos.getY() >= 0.0f )
362 vAltUp.Set(0.0f, 0.0f, 1.0f);
363 else
364 vAltUp.Set(0.0f, 0.0f, -1.0f);
365 }
366
367 PureVector vRightFromForwardAndUp = vAltUp ^ vForwardFromTargetAndPos;
368 vRightFromForwardAndUp.Normalize();
369
370 PureVector vUpFromRightAndForward = vForwardFromTargetAndPos ^ vRightFromForwardAndUp;
371 vUpFromRightAndForward.Normalize();
372
373 mat[0][0] = vRightFromForwardAndUp.getX();
374 mat[1][0] = vRightFromForwardAndUp.getY();
375 mat[2][0] = vRightFromForwardAndUp.getZ();
376 mat[3][0] = 0.0f;
377
378 mat[0][1] = vUpFromRightAndForward.getX();
379 mat[1][1] = vUpFromRightAndForward.getY();
380 mat[2][1] = vUpFromRightAndForward.getZ();
381 mat[3][1] = 0.0f;
382
383 mat[0][2] = vForwardFromTargetAndPos.getX();
384 mat[1][2] = vForwardFromTargetAndPos.getY();
385 mat[2][2] = vForwardFromTargetAndPos.getZ();
386 mat[3][2] = 0.0f;
387
388 /*
389 Normally we should use negative vPos because this is camera's coords so we should translate in reverse direction,
390 but we call invert() at the end anyway so let's stick to positive vPos now and let invert make the translation to reverse direction.
391 */
392 mat[0][3] = vPos.getX();
393 mat[1][3] = vPos.getY();
394 mat[2][3] = vPos.getZ();
395 mat[3][3] = 1.0f;
396
397 /*
398 First I didn't understand why inverting the matrix was needed here. Originally I didn't even invert.
399 But it didn't work well ... position of camera was fine as -vPos but the angles given by Right, Forward and Up vectors calculated above were bad!
400 Then I realized during debugging they were giving inverse angles! When target was right to the camera, the camera was rotating to left side, etc.
401 So what I had to do was to invert the matrix. This gave good angles, I have verified matrix content during debugging.
402 But the position of the camera was bad this time... because I had set the last column of the matrix to -vPos because I knew that I needed
403 to transform the camera in negative direction. But since invert makes translation to inverse direction, no need to specify negative vPos anymore!
404 */
405 invert();
406}
