441void PureManager::PureManagerImpl::DeleteAttachedInstance(PureManaged& m)
442{
443 _pOwner->getConsole().OLn("PureManager::DeleteAttachedInstance(...)");
444
445 bool found = false;
446 for (TPureInt i = 0; (i < nManagedAllocated) && !found; i++)

466void PureManager::PureManagerImpl::DeleteAll()
467{
468 _pOwner->getConsole().OLn("PureManager::DeleteAll()");
469 if ( pManageds == PGENULL )
470 return;
471
472 for (TPureInt i = 0; i < nManagedAllocated; i++)
473 {
474 if ( pManageds[i] != PGENULL )
475 delete pManageds[i];
476 }
477 delete[] pManageds;
478 pManageds = PGENULL;
479 nManagedAllocated = 0;
480 if ( nManagedCount != 0 )
481 { // should be 0 because manageds' dtor always calls DetachFrom()

483 }

554TPureInt PureManager::PureManagerImpl::preAlloc(TPureInt count)
555{
556 _pOwner->getConsole().OLn("PureManager::preAlloc(%d) ...", count);
557 if ( count > 0 )
558 {
559 TPureInt prev_h = nManagedAllocated;
560 nManagedAllocated += count;
561 PureManaged** pNewStorage = PGENULL;
562 try
563 {
564 pNewStorage = new PureManaged*[nManagedAllocated];
565 }
566 catch (const std::bad_alloc&)
567 {
568 _pOwner->getConsole().EOLnOO("ERROR: failed to alloc new storage!");
569 return -1;
570 }
571 memcpy(pNewStorage, pManageds, prev_h*sizeof(PureManaged*));
572 delete[] pManageds;
573 pManageds = pNewStorage;
574 for (TPureInt i = prev_h; i < nManagedAllocated; i++)
575 pManageds[i] = PGENULL;
576 return prev_h;
577 }
578
579 _pOwner->getConsole().EOLn("PureManager::preAlloc(%d) ERROR!", count);
580 return -1;
581} // preAlloc()

808void PureManager::WriteList() const
809{
810 getConsole().OLn("PureManager::WriteList()");
811 getConsole().OLn("========================");
812 const TPureUInt nSysMemTotal = getUsedSystemMemory();
813
814 getConsole().OIOLn("total slots == %d, # of manageds == %d", getSize(), getCount());
815 for (TPureInt i = 0; i < getSize(); i++)
816 {
817 const PureManaged* const pMngd = getAttachedAt(i);
818 if ( pMngd != PGENULL )
819 {
820 getConsole().OLn("pManageds[%d], sys mem: %d Bytes:", i, pMngd->getUsedSystemMemory());
821 WriteListCallback( *pMngd );
822 }
823 else
824 getConsole().OLn("pManageds[%d] == PGENULL (free slot)", i);
825 }
826 getConsole().OLnOO("> total sys mem = %d Bytes <= %d kBytes <= %d MBytes",
827 nSysMemTotal, (int)ceil(nSysMemTotal/1024.0f), (int)ceil(nSysMemTotal/1024.0f/1024.0f));
828 getConsole().OLn("");
829} // WriteList()
