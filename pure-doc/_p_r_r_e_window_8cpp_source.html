216 LRESULT __stdcall PRREdefaultWindowProc(HWND hWindow, UINT wndMsg, WPARAM wParam, LPARAM lParam)
217 {
218  PRREWindowImpl* const wnd = (PRREWindowImpl*) GetWindowLongA(hWindow, GWL_USERDATA);
219 
220  if ( wnd == NULL )
221  return DefWindowProc(hWindow, wndMsg, wParam, lParam);
222 
223  switch ( wndMsg )
224  {
225  case WM_ACTIVATE : return wnd->onActivate((LOWORD(wParam) == WA_ACTIVE) || (LOWORD(wParam) == WA_CLICKACTIVE));
226  case WM_SYSCOMMAND : return wnd->onSysCommand(wParam, lParam);
227  case WM_CLOSE : return wnd->onClose();
228  case WM_DESTROY : return wnd->onDestroy();
229  case WM_INPUT : return wnd->onInput( (HRAWINPUT) lParam);
230  case WM_KEYDOWN : return wnd->onKeyDown(wParam);
231  case WM_KEYUP : return wnd->onKeyUp(wParam);
232  case WM_MOUSEMOVE : return wnd->onMouseMove(GET_X_LPARAM(lParam), GET_Y_LPARAM(lParam));
233  case WM_MOUSEWHEEL : return wnd->onMouseWheel( HIWORD(wParam) );
234  // Note: Quake3::win_wndproc.c states that Win32 may accumulate multiple events into wParam ... I havent experienced this yet though ...
235  case WM_LBUTTONDOWN: return wnd->onMouseDown();
236  case WM_LBUTTONUP : return wnd->onMouseUp();
237  case WM_MBUTTONDOWN: return wnd->onMouseDown();
238  case WM_MBUTTONUP : return wnd->onMouseUp();
239  case WM_RBUTTONDOWN: return wnd->onMouseDown();
240  case WM_RBUTTONUP : return wnd->onMouseUp();
241  case WM_REDRAW_REQUIRED: return wnd->onRedrawRequired();
242  case WM_WINDOWPOSCHANGED: return wnd->onPosChanged( (LPWINDOWPOS) lParam );
243  case WM_G3T_PING: return wnd->onG3tCommand(wndMsg, wParam, lParam);
244  default: return DefWindowProc(hWindow, wndMsg, wParam, lParam);
245  } // switch()

299 TPRREbool PRREWindowImpl::initialize(TPRREuint width, TPRREuint height, TPRREbool bFullscreen, const char* caption, HWND window)
300 {
301  if ( hWindow != NULL )
302  return false;
303 
304  if ( width == 0 || height == 0 )
305  return false;
306 
307  getConsole().OIOLn("PRREWindow(%d, %d, %b, \"%s\", con) ...", width, height, bFullscreen, caption);
308  getConsole().OI();
309 
310  SetPropertiesBeforeInitialize();
311  ResetDefaults();
312 
313  if ( window == NULL )
314  {
315  if ( !setupWindowClass(clWindow) )
316  {
317  getConsole().OO();
318  getConsole().OO();
319  return false;
320  }
321 
322  if ( !actualCreateWindow(width, height, bFullscreen, caption, clWindow) )
323  {
324  getConsole().OO();
325  getConsole().OO();
326  UnregisterClass(clWindow.lpszClassName, clWindow.hInstance);
327  SetPropertiesBeforeInitialize();
328  return false;
329  }
330  }
331  else
332  {
333  getConsole().OLn("Skipped creating new window as an already existing window has been specified!");
334  hWindow = window;
335  bExternalOwner = true;
336  }
337 
338  GetClientRect(hWindow, &rectWindowClient);
339  getConsole().OLn("GetClientRect(): %d x %d",
340  rectWindowClient.right - rectWindowClient.left,
341  rectWindowClient.bottom - rectWindowClient.top);
342 
343  dwStyle = GetWindowLong(hWindow, GWL_STYLE);
344  dwStyleEx = GetWindowLong(hWindow, GWL_EXSTYLE);
345 
346  if ( !bExternalOwner )
347  {
348  SetWindowLong(hWindow, GWL_USERDATA, (LONG) this);
349  SetCaption(caption);
350  }
351 
352  ProcessMessages(); // let the window keep up the pace
353 
354  // triggering an onPosChanged() event to retrieve additional geometric details
355  tagWINDOWPOS tmpStuff;
356  tmpStuff.cx = rectWindowFull.right - rectWindowFull.left;
357  tmpStuff.cy = rectWindowFull.bottom - rectWindowFull.top;
358  tmpStuff.x = rectWindowFull.left;
359  tmpStuff.y = rectWindowFull.top;
360  onPosChanged(&tmpStuff);
361 
362  getConsole().OOSOLn("> done!");
363  getConsole().OO();
364 
365  return true;
366 } // Initialize()
