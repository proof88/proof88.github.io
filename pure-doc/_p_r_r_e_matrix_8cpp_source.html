188 TPRREfloat PRREMatrix::getDeterminant() const
189 {
190  /*
191  https://hu.wikipedia.org/wiki/Determin%C3%A1ns#4%C3%974-es
192  https://en.wikipedia.org/wiki/Determinant
193  https://en.wikipedia.org/wiki/Laplace_expansion
194  https://hu.wikipedia.org/wiki/Kifejt%C3%A9si_t%C3%A9tel
195  http://www.mathwords.com/c/cofactor_matrix.htm
196  */
197 
198  const TPRREfloat det2233 = mat[2][2]*mat[3][3] - mat[2][3]*mat[3][2];
199  const TPRREfloat det2133 = mat[2][1]*mat[3][3] - mat[2][3]*mat[3][1];
200  const TPRREfloat det2033 = mat[2][0]*mat[3][3] - mat[2][3]*mat[3][0];
201  const TPRREfloat det2031 = mat[2][0]*mat[3][1] - mat[2][1]*mat[3][0];
202  const TPRREfloat det2132 = mat[2][1]*mat[3][2] - mat[2][2]*mat[3][1];
203  const TPRREfloat det2032 = mat[2][0]*mat[3][2] - mat[2][2]*mat[3][0];
204 
205  const TPRREfloat det00 = mat[1][1]*det2233 - mat[1][2]*det2133 + mat[1][3]*det2132;
206  const TPRREfloat det01 = mat[1][0]*det2233 - mat[1][2]*det2033 + mat[1][3]*det2032;
207  const TPRREfloat det02 = mat[1][0]*det2133 - mat[1][1]*det2033 + mat[1][3]*det2031;
208  const TPRREfloat det03 = mat[1][0]*det2132 - mat[1][1]*det2032 + mat[1][2]*det2031;
209 
210  return mat[0][0]*det00 - mat[0][1]*det01 + mat[0][2]*det02 - mat[0][3]*det03;
211 }

528 void PRREMatrix::TransposeArray(TPRREfloat* arr, TPRREbyte rows, TPRREbyte cols)
529 {
530  // although we are in 4x4 matrix class, this is a generalized transpose function for matrix with arbitrary rows and cols
531  const TPRREbyte maxdim = max(rows, cols);
532  const TPRREbyte mindim = min(rows, cols);
533 
534  // no matter if rows or cols is bigger, we always let cols alias j run until max dimension since j is always bigger than i
535  // j is always bigger than i to avoid extra swaps those silly transpose functions found on the internet are doing ...
536  for (int i = 0; i < mindim; i++)
537  {
538  for (int j = i+1; j < maxdim; j++)
539  {
540  TPRREfloat tmp = arr[i*rows+j];
541  arr[i*rows+j] = arr[j*rows+i];
542  arr[j*rows+i] = tmp;
543  }
544  }
545 
546  // obviously rows and cols values are needed to be swapped after transpose, but we don't need to do it in this 4x4 case
547 }

556 TPRREbool PRREMatrix::invertArray(TPRREfloat* arr, TPRREbyte rows, TPRREbyte cols)
557 {
558  // this invert implementation is just for rows = cols = 4
559  if ( (rows != cols) || (rows != 4) )
560  return false;
561 
562  /*
563  https://en.wikipedia.org/wiki/Adjugate_matrix
564  https://hu.wikipedia.org/wiki/Adjung%C3%A1lt_(m%C3%A1trixinvert%C3%A1l%C3%A1s)
565  https://www.euclideanspace.com/maths/algebra/matrix/functions/inverse/index.htm
566  https://www.euclideanspace.com/maths/algebra/matrix/functions/inverse/threeD/index.htm
567  https://www.euclideanspace.com/maths/algebra/matrix/functions/inverse/fourD/index.htm
568  */
569 
570  const TPRREfloat det2233 = arr[ind(2,2)]*arr[ind(3,3)] - arr[ind(2,3)]*arr[ind(3,2)];
571  const TPRREfloat det2133 = arr[ind(2,1)]*arr[ind(3,3)] - arr[ind(2,3)]*arr[ind(3,1)];
572  const TPRREfloat det2033 = arr[ind(2,0)]*arr[ind(3,3)] - arr[ind(2,3)]*arr[ind(3,0)];
573  const TPRREfloat det2031 = arr[ind(2,0)]*arr[ind(3,1)] - arr[ind(2,1)]*arr[ind(3,0)];
574  const TPRREfloat det2132 = arr[ind(2,1)]*arr[ind(3,2)] - arr[ind(2,2)]*arr[ind(3,1)];
575  const TPRREfloat det2032 = arr[ind(2,0)]*arr[ind(3,2)] - arr[ind(2,2)]*arr[ind(3,0)];
576 
577  // minors for row 0
578  const TPRREfloat det00 = arr[ind(1,1)]*det2233 - arr[ind(1,2)]*det2133 + arr[ind(1,3)]*det2132;
579  const TPRREfloat det01 = arr[ind(1,0)]*det2233 - arr[ind(1,2)]*det2033 + arr[ind(1,3)]*det2032;
580  const TPRREfloat det02 = arr[ind(1,0)]*det2133 - arr[ind(1,1)]*det2033 + arr[ind(1,3)]*det2031;
581  const TPRREfloat det03 = arr[ind(1,0)]*det2132 - arr[ind(1,1)]*det2032 + arr[ind(1,2)]*det2031;
582 
583  // above lines are from getDeterminant()
584  const TPRREfloat det = arr[ind(0,0)]*det00 - arr[ind(0,1)]*det01 + arr[ind(0,2)]*det02 - arr[ind(0,3)]*det03;
585 
586  // check if matrix is singular
587  if ( abs(det) < PFL::E )
588  return false; // cannot divive by det
589 
590  // following lines continue calculating minors for next rows exactly same way we calculated above for row0 ...
591 
592  // minors for row 1
593  const TPRREfloat det10 = arr[ind(0,1)]*det2233 - arr[ind(0,2)]*det2133 + arr[ind(0,3)]*det2132;
594  const TPRREfloat det11 = arr[ind(0,0)]*det2233 - arr[ind(0,2)]*det2033 + arr[ind(0,3)]*det2032;
595  const TPRREfloat det12 = arr[ind(0,0)]*det2133 - arr[ind(0,1)]*det2033 + arr[ind(0,3)]*det2031;
596  const TPRREfloat det13 = arr[ind(0,0)]*det2132 - arr[ind(0,1)]*det2032 + arr[ind(0,2)]*det2031;
597 
598  const TPRREfloat det1233 = arr[ind(1,2)]*arr[ind(3,3)] - arr[ind(1,3)]*arr[ind(3,2)];
599  const TPRREfloat det1133 = arr[ind(1,1)]*arr[ind(3,3)] - arr[ind(1,3)]*arr[ind(3,1)];
600  const TPRREfloat det1033 = arr[ind(1,0)]*arr[ind(3,3)] - arr[ind(1,3)]*arr[ind(3,0)];
601  const TPRREfloat det1132 = arr[ind(1,1)]*arr[ind(3,2)] - arr[ind(1,2)]*arr[ind(3,1)];
602  const TPRREfloat det1032 = arr[ind(1,0)]*arr[ind(3,2)] - arr[ind(1,2)]*arr[ind(3,0)];
603  const TPRREfloat det1031 = arr[ind(1,0)]*arr[ind(3,1)] - arr[ind(1,1)]*arr[ind(3,0)];
604 
605  // minors for row 2
606  const TPRREfloat det20 = arr[ind(0,1)]*det1233 - arr[ind(0,2)]*det1133 + arr[ind(0,3)]*det1132;
607  const TPRREfloat det21 = arr[ind(0,0)]*det1233 - arr[ind(0,2)]*det1033 + arr[ind(0,3)]*det1032;
608  const TPRREfloat det22 = arr[ind(0,0)]*det1133 - arr[ind(0,1)]*det1033 + arr[ind(0,3)]*det1031;
609  const TPRREfloat det23 = arr[ind(0,0)]*det1132 - arr[ind(0,1)]*det1032 + arr[ind(0,2)]*det1031;
610 
611  const TPRREfloat det1223 = arr[ind(1,2)]*arr[ind(2,3)] - arr[ind(1,3)]*arr[ind(2,2)];
612  const TPRREfloat det1123 = arr[ind(1,1)]*arr[ind(2,3)] - arr[ind(1,3)]*arr[ind(2,1)];
613  const TPRREfloat det1122 = arr[ind(1,1)]*arr[ind(2,2)] - arr[ind(1,2)]*arr[ind(2,1)];
614  const TPRREfloat det1023 = arr[ind(1,0)]*arr[ind(2,3)] - arr[ind(1,3)]*arr[ind(2,0)];
615  const TPRREfloat det1022 = arr[ind(1,0)]*arr[ind(2,2)] - arr[ind(1,2)]*arr[ind(2,0)];
616  const TPRREfloat det1021 = arr[ind(1,0)]*arr[ind(2,1)] - arr[ind(1,1)]*arr[ind(2,0)];
617 
618  // minors for row 3
619  const TPRREfloat det30 = arr[ind(0,1)]*det1223 - arr[ind(0,2)]*det1123 + arr[ind(0,3)]*det1122;
620  const TPRREfloat det31 = arr[ind(0,0)]*det1223 - arr[ind(0,2)]*det1023 + arr[ind(0,3)]*det1022;
621  const TPRREfloat det32 = arr[ind(0,0)]*det1123 - arr[ind(0,1)]*det1023 + arr[ind(0,3)]*det1021;
622  const TPRREfloat det33 = arr[ind(0,0)]*det1122 - arr[ind(0,1)]*det1022 + arr[ind(0,2)]*det1021;
623 
624  // now we know all minors, we need to build the adjugate matrix and simply divide by det
625  // adjugate matrix is the transposed form of the cofactor matrix
626  // cofactor matrix consists of signed minors
627  // we build it now right into the original array
628 
629  arr[ind(0,0)] = det00 / det; arr[ind(0,1)] = -det10 / det; arr[ind(0,2)] = det20 / det; arr[ind(0,3)] = -det30 / det;
630  arr[ind(1,0)] = -det01 / det; arr[ind(1,1)] = det11 / det; arr[ind(1,2)] = -det21 / det; arr[ind(1,3)] = det31 / det;
631  arr[ind(2,0)] = det02 / det; arr[ind(2,1)] = -det12 / det; arr[ind(2,2)] = det22 / det; arr[ind(2,3)] = -det32 / det;
632  arr[ind(3,0)] = -det03 / det; arr[ind(3,1)] = det13 / det; arr[ind(3,2)] = -det23 / det; arr[ind(3,3)] = det33 / det;
633 
634  return true;
635 }
