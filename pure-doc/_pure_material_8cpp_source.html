47PureMaterial::PureMaterialImpl::~PureMaterialImpl()
48{
49 _pOwner->getManagedConsole().OLnOI("~PureMaterial() ...");
50
51 for (TPureUInt i = 0; i < layers.size(); i++)
52 {
53 layers[i].nColors_h = layers[i].nTexcoords_h = 0;
54 delete[] layers[i].pColors;
55 delete[] layers[i].pTexcoords;
56 layers[i].pColors = PGENULL;
57 layers[i].pTexcoords = PGENULL;
58 }
59 nIndices = 0;
60 m_nIndexSize = 0;
61
62 _pOwner->getManagedConsole().SOLnOO("Done!");
63} // ~PureTexture()

66TPureBool PureMaterial::PureMaterialImpl::allocateArrays(TPureUInt nColorCount, TPureUInt nTexcoordCount, TPureUInt nIndexCount, TPureUInt nIndexSize)
67{
68 if ( nIndexSize == 0 )
69 {
70 _pOwner->getManagedConsole().EOLn("PureMaterial()::allocateArrays(%d, %d, %d, %d) fail due to nIndexSize!", nColorCount, nTexcoordCount, nIndexCount, nIndexSize);
71 return false;
72 }
73
74 if ( getTexcoordsCount(0) > 0 )
75 {
76 _pOwner->getManagedConsole().SOLn("PureMaterial()::allocateArrays(%d, %d, %d, %d) skip, already allocated!", nColorCount, nTexcoordCount, nIndexCount, nIndexSize);
77 return true;
78 }
79
80 nIndices = nIndexCount;
81 m_nIndexSize = nIndexSize;
82
83 try
84 {
85 for (TPureUInt i = 0; i < ((PureMaterialManager*)_pOwner->getManager())->getMaximumLayerCount(); i++)
86 {
87 layers[i].nColors_h = nColorCount;
88 layers[i].pColors = new TRGBAFLOAT[nColorCount];
89 layers[i].nTexcoords_h = nTexcoordCount;
90 layers[i].pTexcoords = new TUVW[nTexcoordCount];
91 }
92 return true;
93 }
94 catch (const std::bad_alloc&)
95 {
96 _pOwner->getManagedConsole().EOLn("PureMaterial()::allocateArrays(%d, %d, %d, %d) fail due to out of memory!", nColorCount, nTexcoordCount, nIndexCount, nIndexSize);
97 for (TPureUInt i = 0; i < ((PureMaterialManager*)_pOwner->getManager())->getMaximumLayerCount(); i++)
98 {
99 layers[i].nColors_h = 0;
100 delete[] layers[i].pColors;
101 layers[i].pColors = PGENULL;
102 layers[i].nTexcoords_h = 0;
103 delete[] layers[i].pTexcoords;
104 layers[i].pTexcoords = PGENULL;
105 }
106 return false;
107 }
108}

301TPureBool PureMaterial::PureMaterialImpl::setSourceBlendFunc(TPURE_BLENDFACTOR value, TPureUInt level)
302{
303 if ( !_pOwner->getUtiliser() || !_pOwner->getUtiliser()->getManager() )
304 {
305 // this way we permit the user setting blending again when the utiliser Object3D is not attached to its manager,
306 // so it can be assured that in case of reattach, it is still non-blended, without even checking blending state!
307 _pOwner->getManagedConsole().EOLn("ERROR: %s either no utiliser or utiliser has no parent!", __FUNCTION__);
308 return false;
309 }
310
311 if ( (value != PURE_SRC_COLOR) && (value != PURE_ONE_MINUS_SRC_COLOR) )
312 {
313 if ( level < ((PureMaterialManager*)_pOwner->getManager())->getMaximumLayerCount() )
314 {
315 layers[level].blendFactorSource = value;
316
317 // remember: Object3D is also a manager, so it could also implement its own HandleManagedPropertyChanged(),
318 // to handle changes in the material of its subobjects! Since Object3D doesnt implement such,
319 // the PureManager's base HandleManagedPropertyChanged() is invoked when subobject's material is changed here!
320 _pOwner->getUtiliser()->getManager()->HandleManagedPropertyChanged( *(_pOwner->getUtiliser()) );
321
322 return true;
323 }
324 }
325 return false;
326} // setSourceBlendFunc()

376TPURE_BLENDMODE PureMaterial::PureMaterialImpl::getBlendMode(TPureUInt level) const
377{
378 if ( level < ((PureMaterialManager*)_pOwner->getManager())->getMaximumLayerCount() )
379 {
380 if ( (layers[level].blendFactorSource == PURE_ONE) && (layers[level].blendFactorDestination == PURE_ZERO) )
381 return PURE_BM_NONE;
382 else if ( (layers[level].blendFactorSource == PURE_SRC_ALPHA) && (layers[level].blendFactorDestination == PURE_ONE_MINUS_SRC_ALPHA) )
383 return PURE_BM_STANDARD_TRANSPARENCY;
384 else
385 return PURE_BM_MANUAL;
386 }
387 else
388 {
389 // TODO: set lasterr
390 return PURE_BM_NONE;
391 }

410TPureBool PureMaterial::PureMaterialImpl::copyFromMaterial(PureMaterial& srcMat, TPureUInt dstLevel, TPureUInt srcLevel)
411{
412 if ( (dstLevel < ((PureMaterialManager*)_pOwner->getManager())->getMaximumLayerCount()) &&
413 (srcLevel < ((PureMaterialManager*)_pOwner->getManager())->getMaximumLayerCount()) )
414 {
415 // we also want to make sure that we have the same memory layout of texcoords and indices
416 // so we can safely copy data into our preallocated arrays
417 if ( (layers[dstLevel].nTexcoords_h == srcMat.pImpl->layers[srcLevel].nTexcoords_h) &&
418 (layers[dstLevel].nColors_h == srcMat.pImpl->layers[srcLevel].nColors_h) &&
419 (nIndices == srcMat.getIndicesCount()) &&
420 (m_nIndexSize == srcMat.pImpl->m_nIndexSize) )
421 {
422 if ( srcMat.getTexture(srcLevel) != NULL )
423 {
424 if ( !setTexture( srcMat.getTexture(srcLevel), dstLevel ) )
425 {
426 _pOwner->getManagedConsole().EOLn("ERROR: %s setTexture() failed for dstLevel %d!", __FUNCTION__, dstLevel);
427 return false;
428 }
429 }
430
431 memcpy( layers[dstLevel].pTexcoords, srcMat.pImpl->layers[srcLevel].pTexcoords, sizeof(TUVW) * layers[dstLevel].nTexcoords_h );
432 memcpy( layers[dstLevel].pColors, srcMat.pImpl->layers[srcLevel].pColors, sizeof(TRGBAFLOAT) * layers[dstLevel].nColors_h );
433
434 return true;
435 }
436 else
437 ; // TODO: set lasterr
438 }
439 else
440 ; // TODO: set lasterr
441
442 return false;
443}

459PureMaterial::PureMaterialImpl::PureMaterialImpl(
460 PureMaterial* owner, TPureUInt nLayers)
461{
462 _pOwner = owner;
463 _pOwner->getManagedConsole().OLnOI("PureMaterial() ...");
464
465 // sanity check
466 if ( nLayers < 1 )
467 nLayers = 1;
468
469 for (TPureUInt i = 0; i < nLayers; i++)
470 {
471 layers.push_back( TPURE_MATERIAL_LAYER() );
472 layers[ layers.size()-1 ].nColors_h = 0;
473 layers[ layers.size()-1 ].pColors = PGENULL;
474 layers[ layers.size()-1 ].nTexcoords_h = 0;
475 layers[ layers.size()-1 ].pTexcoords = PGENULL;
476 layers[ layers.size()-1 ].tex = PGENULL;
477 layers[ layers.size()-1 ].texEnvMode = PURE_TEX_FUNC_MODULATE;
478 layers[ layers.size()-1 ].fTransparancy = 1.0f;
479 layers[ layers.size()-1 ].clrTexEnv.Set(255, 255, 255, 255);
480 layers[ layers.size()-1 ].blendFactorSource = PURE_ONE;
481 layers[ layers.size()-1 ].blendFactorDestination = PURE_ZERO;
482 }
483
484 nIndices = 0;
485 fShininessBackFace = fShininessFrontFace = 1.0f;
486 bDecalOffset = false;
487
488 // we dont store nLayers because from now layers.size() is fix until this instance is deleted
489
490 _pOwner->getManagedConsole().SOLnOO("Done!");
491} // PureObject3D()
