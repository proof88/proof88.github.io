125TPureBool PureMatrix::isIdentity() const
126{
127 for (int i = 0; i < 4; i++)
128 {
129 for (int j = 0; j < 4; j++)
130 {
131 if ( i == j )
132 {
133 if ( (mat[i][i] <= (1.0f-PFL::E)) || (mat[i][i] >= (1.0f+PFL::E)) )
134 return false;
135 }
136 else
137 {
138 if ( abs(mat[i][j]) >= PFL::E )
139 return false;
140 }
141 }
142 }
143 return true;
144} // isIdentity()

187TPureFloat PureMatrix::getDeterminant() const
188{
189 /*
190 https://hu.wikipedia.org/wiki/Determin%C3%A1ns#4%C3%974-es
191 https://en.wikipedia.org/wiki/Determinant
192 https://en.wikipedia.org/wiki/Laplace_expansion
193 https://hu.wikipedia.org/wiki/Kifejt%C3%A9si_t%C3%A9tel
194 http://www.mathwords.com/c/cofactor_matrix.htm
195 */
196
197 const TPureFloat det2233 = mat[2][2]*mat[3][3] - mat[2][3]*mat[3][2];
198 const TPureFloat det2133 = mat[2][1]*mat[3][3] - mat[2][3]*mat[3][1];
199 const TPureFloat det2033 = mat[2][0]*mat[3][3] - mat[2][3]*mat[3][0];
200 const TPureFloat det2031 = mat[2][0]*mat[3][1] - mat[2][1]*mat[3][0];
201 const TPureFloat det2132 = mat[2][1]*mat[3][2] - mat[2][2]*mat[3][1];
202 const TPureFloat det2032 = mat[2][0]*mat[3][2] - mat[2][2]*mat[3][0];
203
204 const TPureFloat det00 = mat[1][1]*det2233 - mat[1][2]*det2133 + mat[1][3]*det2132;
205 const TPureFloat det01 = mat[1][0]*det2233 - mat[1][2]*det2033 + mat[1][3]*det2032;
206 const TPureFloat det02 = mat[1][0]*det2133 - mat[1][1]*det2033 + mat[1][3]*det2031;
207 const TPureFloat det03 = mat[1][0]*det2132 - mat[1][1]*det2032 + mat[1][2]*det2031;
208
209 return mat[0][0]*det00 - mat[0][1]*det01 + mat[0][2]*det02 - mat[0][3]*det03;
210}

527void PureMatrix::TransposeArray(TPureFloat* arr, TPureByte rows, TPureByte cols)
528{
529 // although we are in 4x4 matrix class, this is a generalized transpose function for matrix with arbitrary rows and cols
530 const TPureByte maxdim = std::max(rows, cols);
531 const TPureByte mindim = std::min(rows, cols);
532
533 // no matter if rows or cols is bigger, we always let cols alias j run until max dimension since j is always bigger than i
534 // j is always bigger than i to avoid extra swaps those silly transpose functions found on the internet are doing ...
535 for (int i = 0; i < mindim; i++)
536 {
537 for (int j = i+1; j < maxdim; j++)
538 {
539 TPureFloat tmp = arr[i*rows+j];
540 arr[i*rows+j] = arr[j*rows+i];
541 arr[j*rows+i] = tmp;
542 }
543 }
544
545 // obviously rows and cols values are needed to be swapped after transpose, but we don't need to do it in this 4x4 case
546}

555TPureBool PureMatrix::invertArray(TPureFloat* arr, TPureByte rows, TPureByte cols)
556{
557 // this invert implementation is just for rows = cols = 4
558 if ( (rows != cols) || (rows != 4) )
559 return false;
560
561 /*
562 https://en.wikipedia.org/wiki/Adjugate_matrix
563 https://hu.wikipedia.org/wiki/Adjung%C3%A1lt_(m%C3%A1trixinvert%C3%A1l%C3%A1s)
564 https://www.euclideanspace.com/maths/algebra/matrix/functions/inverse/index.htm
565 https://www.euclideanspace.com/maths/algebra/matrix/functions/inverse/threeD/index.htm
566 https://www.euclideanspace.com/maths/algebra/matrix/functions/inverse/fourD/index.htm
567 */
568
569 const TPureFloat det2233 = arr[ind(2,2)]*arr[ind(3,3)] - arr[ind(2,3)]*arr[ind(3,2)];
570 const TPureFloat det2133 = arr[ind(2,1)]*arr[ind(3,3)] - arr[ind(2,3)]*arr[ind(3,1)];
571 const TPureFloat det2033 = arr[ind(2,0)]*arr[ind(3,3)] - arr[ind(2,3)]*arr[ind(3,0)];
572 const TPureFloat det2031 = arr[ind(2,0)]*arr[ind(3,1)] - arr[ind(2,1)]*arr[ind(3,0)];
573 const TPureFloat det2132 = arr[ind(2,1)]*arr[ind(3,2)] - arr[ind(2,2)]*arr[ind(3,1)];
574 const TPureFloat det2032 = arr[ind(2,0)]*arr[ind(3,2)] - arr[ind(2,2)]*arr[ind(3,0)];
575
576 // minors for row 0
577 const TPureFloat det00 = arr[ind(1,1)]*det2233 - arr[ind(1,2)]*det2133 + arr[ind(1,3)]*det2132;
578 const TPureFloat det01 = arr[ind(1,0)]*det2233 - arr[ind(1,2)]*det2033 + arr[ind(1,3)]*det2032;
579 const TPureFloat det02 = arr[ind(1,0)]*det2133 - arr[ind(1,1)]*det2033 + arr[ind(1,3)]*det2031;
580 const TPureFloat det03 = arr[ind(1,0)]*det2132 - arr[ind(1,1)]*det2032 + arr[ind(1,2)]*det2031;
581
582 // above lines are from getDeterminant()
583 const TPureFloat det = arr[ind(0,0)]*det00 - arr[ind(0,1)]*det01 + arr[ind(0,2)]*det02 - arr[ind(0,3)]*det03;
584
585 // check if matrix is singular
586 if ( abs(det) < PFL::E )
587 return false; // cannot divive by det
588
589 // following lines continue calculating minors for next rows exactly same way we calculated above for row0 ...
590
591 // minors for row 1
592 const TPureFloat det10 = arr[ind(0,1)]*det2233 - arr[ind(0,2)]*det2133 + arr[ind(0,3)]*det2132;
593 const TPureFloat det11 = arr[ind(0,0)]*det2233 - arr[ind(0,2)]*det2033 + arr[ind(0,3)]*det2032;
594 const TPureFloat det12 = arr[ind(0,0)]*det2133 - arr[ind(0,1)]*det2033 + arr[ind(0,3)]*det2031;
595 const TPureFloat det13 = arr[ind(0,0)]*det2132 - arr[ind(0,1)]*det2032 + arr[ind(0,2)]*det2031;
596
597 const TPureFloat det1233 = arr[ind(1,2)]*arr[ind(3,3)] - arr[ind(1,3)]*arr[ind(3,2)];
598 const TPureFloat det1133 = arr[ind(1,1)]*arr[ind(3,3)] - arr[ind(1,3)]*arr[ind(3,1)];
599 const TPureFloat det1033 = arr[ind(1,0)]*arr[ind(3,3)] - arr[ind(1,3)]*arr[ind(3,0)];
600 const TPureFloat det1132 = arr[ind(1,1)]*arr[ind(3,2)] - arr[ind(1,2)]*arr[ind(3,1)];
601 const TPureFloat det1032 = arr[ind(1,0)]*arr[ind(3,2)] - arr[ind(1,2)]*arr[ind(3,0)];
602 const TPureFloat det1031 = arr[ind(1,0)]*arr[ind(3,1)] - arr[ind(1,1)]*arr[ind(3,0)];
603
604 // minors for row 2
605 const TPureFloat det20 = arr[ind(0,1)]*det1233 - arr[ind(0,2)]*det1133 + arr[ind(0,3)]*det1132;
606 const TPureFloat det21 = arr[ind(0,0)]*det1233 - arr[ind(0,2)]*det1033 + arr[ind(0,3)]*det1032;
607 const TPureFloat det22 = arr[ind(0,0)]*det1133 - arr[ind(0,1)]*det1033 + arr[ind(0,3)]*det1031;
608 const TPureFloat det23 = arr[ind(0,0)]*det1132 - arr[ind(0,1)]*det1032 + arr[ind(0,2)]*det1031;
609
610 const TPureFloat det1223 = arr[ind(1,2)]*arr[ind(2,3)] - arr[ind(1,3)]*arr[ind(2,2)];
611 const TPureFloat det1123 = arr[ind(1,1)]*arr[ind(2,3)] - arr[ind(1,3)]*arr[ind(2,1)];
612 const TPureFloat det1122 = arr[ind(1,1)]*arr[ind(2,2)] - arr[ind(1,2)]*arr[ind(2,1)];
613 const TPureFloat det1023 = arr[ind(1,0)]*arr[ind(2,3)] - arr[ind(1,3)]*arr[ind(2,0)];
614 const TPureFloat det1022 = arr[ind(1,0)]*arr[ind(2,2)] - arr[ind(1,2)]*arr[ind(2,0)];
615 const TPureFloat det1021 = arr[ind(1,0)]*arr[ind(2,1)] - arr[ind(1,1)]*arr[ind(2,0)];
616
617 // minors for row 3
618 const TPureFloat det30 = arr[ind(0,1)]*det1223 - arr[ind(0,2)]*det1123 + arr[ind(0,3)]*det1122;
619 const TPureFloat det31 = arr[ind(0,0)]*det1223 - arr[ind(0,2)]*det1023 + arr[ind(0,3)]*det1022;
620 const TPureFloat det32 = arr[ind(0,0)]*det1123 - arr[ind(0,1)]*det1023 + arr[ind(0,3)]*det1021;
621 const TPureFloat det33 = arr[ind(0,0)]*det1122 - arr[ind(0,1)]*det1022 + arr[ind(0,2)]*det1021;
622
623 // now we know all minors, we need to build the adjugate matrix and simply divide by det;
624 // adjugate matrix is the transposed form of the cofactor matrix;
625 // cofactor matrix consists of signed minors;
626 // we build it now right into the original array
627
628 arr[ind(0,0)] = det00 / det; arr[ind(0,1)] = -det10 / det; arr[ind(0,2)] = det20 / det; arr[ind(0,3)] = -det30 / det;
629 arr[ind(1,0)] = -det01 / det; arr[ind(1,1)] = det11 / det; arr[ind(1,2)] = -det21 / det; arr[ind(1,3)] = det31 / det;
630 arr[ind(2,0)] = det02 / det; arr[ind(2,1)] = -det12 / det; arr[ind(2,2)] = det22 / det; arr[ind(2,3)] = -det32 / det;
631 arr[ind(3,0)] = -det03 / det; arr[ind(3,1)] = det13 / det; arr[ind(3,2)] = -det23 / det; arr[ind(3,3)] = det33 / det;
632
633 return true;
634}
