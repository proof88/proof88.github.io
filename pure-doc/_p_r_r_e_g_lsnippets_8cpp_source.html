170 void PRREGLsnippets::SetGLBoundingBoxRendering(TPRREbool state)
171 {
172  if ( state )
173  {
174  // obj->Draw() must not set these states when the given renderPass is PRRE_RPASS_SYNC_OCCLUSION_START
175  glColorMask(GL_FALSE, GL_FALSE, GL_FALSE, GL_FALSE);
176  glEnable(GL_DEPTH_TEST);
177  glDepthMask(GL_FALSE);
178  glDisable(GL_CULL_FACE);
179  glDisable(GL_LIGHTING);
180  glPolygonMode(GL_FRONT_AND_BACK, GL_FILL);
181  glDisable(GL_ALPHA_TEST);
182  glLoadTexturesAndSetBlendState(PGENULL, false, false);
183  }
184  else
185  {
186  glColorMask(GL_TRUE, GL_TRUE, GL_TRUE, GL_TRUE);
187  glDepthMask(GL_TRUE);
188  glFlush(); // this to make sure any initiated GL query is actually starts to be processed by GPU so later we dont have to wait that much for results
189  }

284 void PRREGLsnippets::glLoadTextureIntoTMU(const PRRETexture* tex, TPRREuint iTMU, TPRREbool bSticked)
285 {
286  /* Make sure we don't call GL functions when no accelerator is present */
287  if ( !PRREhwInfo::get().getVideo().isAcceleratorDetected() )
288  return;
289 
290  // currently we support only 2 textured layers ...
291  if ( iTMU > 1 )
292  {
293  CConsole::getConsoleInstance(sLoggerModuleName).EOLn("ERROR: glLoadTextureIntoTMU(tex, %d)", iTMU);
294  assert( false );
295  }
296 
297  if ( PRREhwInfo::get().getVideo().isMultiTexturingSupported() )
298  {
299  glActiveTextureARB(GL_TEXTURE0_ARB + iTMU);
300  if ( tex != PGENULL )
301  {
302  glEnable(GL_TEXTURE_2D);
303  if ( tex->getInternalNum() != iLastTex[iTMU] )
304  {
305  glBindTexture(GL_TEXTURE_2D, tex->getInternalNum());
306  iLastTex[iTMU] = tex->getInternalNum();
307  }
308  glTexEnvf(GL_TEXTURE_ENV, GL_TEXTURE_ENV_MODE, GL_MODULATE);
309  }
310  else
311  glDisable(GL_TEXTURE_2D);
312  }
313  else
314  {
315  if ( tex != PGENULL )
316  {
317  glEnable(GL_TEXTURE_2D);
318  if ( tex->getInternalNum() != iLastTex[iTMU] )
319  {
320  glBindTexture(GL_TEXTURE_2D, tex->getInternalNum());
321  iLastTex[iTMU] = tex->getInternalNum();
322  }
323  if ( bSticked )
324  glTexEnvf(GL_TEXTURE_ENV, GL_TEXTURE_ENV_MODE, GL_REPLACE);
325  else
326  glTexEnvf(GL_TEXTURE_ENV, GL_TEXTURE_ENV_MODE, GL_MODULATE);
327  }
328  else
329  glDisable(GL_TEXTURE_2D);
330  }
331 } // glLoadTextureIntoTMU()

345 void PRREGLsnippets::glLoadTexturesAndSetBlendState(const PRREMaterial* mat, TPRREbool bObjLevel1Sticked, TPRREbool bObjLevel1Blended)
346 {
347  if ( PGENULL == mat )
348  {
349  glDisable(GL_BLEND);
350  glLoadTextureIntoTMU( PGENULL, 0, bObjLevel1Sticked );
351  if ( PRREhwInfo::get().getVideo().isMultiTexturingSupported() )
352  {
353  glLoadTextureIntoTMU( PGENULL, 1, bObjLevel1Sticked );
354  }
355  return;
356  }
357 
358  if ( PRREhwInfo::get().getVideo().isMultiTexturingSupported() )
359  {
360  if ( mat->isMultiTextured() )
361  {
362  // enable blending of 2nd layer
363  glEnable(GL_BLEND);
364  glBlendFunc(getGLBlendFromPRREBlend(mat->getSourceBlendFunc(1)), getGLBlendFromPRREBlend(mat->getDestinationBlendFunc(1)));
365  glLoadTextureIntoTMU( mat->getTexture(0), 0, bObjLevel1Sticked );
366  glLoadTextureIntoTMU( mat->getTexture(1), 1, bObjLevel1Sticked );
367  }
368  else
369  {
370  // disable blending only if base layer is not blended and level 1 object3d is also not blended ...
371  if ( !bObjLevel1Blended && !PRREMaterial::isBlendFuncReallyBlending(mat->getSourceBlendFunc(), mat->getDestinationBlendFunc()) )
372  {
373  glDisable(GL_BLEND);
374  }
375  glLoadTextureIntoTMU( mat->getTexture(0), 0, bObjLevel1Sticked );
376  glLoadTextureIntoTMU( PGENULL, 1, bObjLevel1Sticked );
377  }
378  }
379  else
380  {
381  // disable blending only if base layer is not blended and level 1 object3d is also not blended ...
382  if ( !bObjLevel1Blended && !PRREMaterial::isBlendFuncReallyBlending(mat->getSourceBlendFunc(), mat->getDestinationBlendFunc()) )
383  {
384  glDisable(GL_BLEND);
385  }
386  glLoadTextureIntoTMU( mat->getTexture(), 0, bObjLevel1Sticked );
387  }
388 } // glLoadTexturesAndSetBlendState()
