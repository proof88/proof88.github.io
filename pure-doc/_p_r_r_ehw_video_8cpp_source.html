1018  const int nLineLeftMargin = 2; // each line starts at this position
1019  const int nLineMax = 80; // assuming we have this number of chars available per line
1020  const int nLineEndMargin = 2; // need to leave this number of characters empty at end of line
1021  const int nLeftColMax = nLineMax/2 - nLineLeftMargin - nLineEndMargin; // left column have this width
1022 
1023  // temporarily reset console indentation level so we have enough space
1024  getConsole().SetIndent(nLineLeftMargin);
1025 
1026  // we print extensions in 2 columns
1027  do
1028  {
1029  string sub;
1030  ssFeatures >> sub;
1031 
1032  // sub is empty when no more extension left in ssFeatures
1033  if ( sub.empty() )
1034  continue;
1035 
1036  if ( i % 2 == 0 )
1037  {
1038  // left column in table
1039  getConsole().O("%s", sub.c_str());
1040  numspaces = nLeftColMax - sub.length();
1041  }
1042  else
1043  {
1044  // right column in table
1045  if ( (sub.length() <= (nLeftColMax+2)) && (numspaces > 0) )
1046  {
1047  tmpSpacesString = "";
1048  for (int j = 0; j < numspaces; j++)
1049  tmpSpacesString += ' ';
1050  getConsole().O(tmpSpacesString.c_str());
1051  }
1052  else
1053  {
1054  // too long extension needs to be written to next line (left column) to avoid line wrap
1055  // we fell here due to one of 2 reasons:
1056  // - sub.length() > 40 means our current string originally targeted to right column is too long
1057  // - numspaces <= 0 means the previous string already printed to left column is too long, did not leave enough space for us
1058  // either reason we need to jump to next line
1059  getConsole().OLn("");
1060  }
1061  getConsole().OLn("%s", sub.c_str());
1062  }
1063  i++;
1064  } while (ssFeatures);
1065 
1066  getConsole().SetIndent(nIndentOriginal);
1067 
1068  if ( i % 2 == 1 )
1069  getConsole().OLn("");
1070  getConsole().OLn("");

1072  getConsole().OLn("");
