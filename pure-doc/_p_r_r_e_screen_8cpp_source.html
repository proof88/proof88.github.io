220 TPRREbool PRREScreenImpl::applyDisplaySettings(HDC dc, TPRRE_SCREEN_PF pixelFormat)
221 {
222  getConsole().OLn("PRREScreen::applyDisplaySettings()");
223 
224  if ( bDisplaySettingsApplied )
225  {
226  getConsole().OLn(" Already applied!");
227  if ( pixelFormat != PRRE_SCREEN_PF_NONE )
228  FindAdvancedPixelFormat(dc);
229 
230  return false;
231  }
232 
233  if ( (nBitsColor == 0) || (nBitsDepth == 0) || (nResX == 0) || (nResY == 0) )
234  {
235  getConsole().EOLn(" ERROR: rr == %d, clr == %d, z == %d, %d x %d",
236  nRefreshRate, nBitsColor, nBitsDepth, nResX, nResY);
237  return false;
238  }
239 
240  bool canproceed = true;
241 
242  if ( bFullScreen )
243  {
244  if ( nRefreshRate == 0 )
245  { // maximum Hz
246  nRefreshRate = findMaximumRefreshRate(nResX, nResY, nBitsColor);
247  getConsole().OLn(" selected refresh rate = %d Hz", nRefreshRate);
248  }
249  canproceed = switchToFullscreenMode(nResX, nResY, nBitsColor, nRefreshRate);
250  } // bFullScreen
251 
252  if ( canproceed )
253  {
254  getConsole().SOn(); getConsole().OLn(" can proceed"); getConsole().SOff();
255  ReadCurrentDisplaySettings(nResX, nResY, nBitsColor, nRefreshRate);
256  getConsole().OLn(" Current display settings: %dx%dx%d @ %d Hz",
257  nResX, nResY, nBitsColor, nRefreshRate);
258  FindSimplePixelFormat(dc);
259  switch (pixelFormat)
260  {
261  case PRRE_SCREEN_PF_SIMPLE:
262  bDisplaySettingsApplied = applyPixelFormat(dc, iSimplePixelFormat, &pfd);
263  break;
264  case PRRE_SCREEN_PF_ADVANCED:
265  bDisplaySettingsApplied = applyPixelFormat(dc, iAdvancedPixelFormat, &pfd);
266  break;
267  case PRRE_SCREEN_PF_NONE:
268  // TODO: create DIB in RAM to render to as in CSWRenderer ctor!
269  bDisplaySettingsApplied = true;
270  break;
271  default:
272  bDisplaySettingsApplied = false;
273  getConsole().EOLn("ERROR: invalid pixelFormat specified!");
274  }
275 
276  bFSAA_ready = (iAdvancedPixelFormat == iAppliedPixelFormat);
277 
278  return bDisplaySettingsApplied;
279  } // canproceed
280 
281  return false;
282 
283 } // applyDisplaySettings()

700 void PRREScreenImpl::FindSimplePixelFormat(HDC dc)
701 {
702  getConsole().OLn(" FindSimplePixelFormat() ...");
703  if ( iSimplePixelFormat > 0 )
704  {
705  getConsole().OLn(" already found!");
706  return;
707  }
708 
709  memset(&pfd, 0, sizeof(PIXELFORMATDESCRIPTOR));
710  pfd.nSize = sizeof(PIXELFORMATDESCRIPTOR);
711  pfd.nVersion = (WORD) 1;
712  pfd.dwFlags = PFD_DRAW_TO_WINDOW | PFD_SUPPORT_OPENGL | PFD_DOUBLEBUFFER | PFD_GENERIC_ACCELERATED;
713  pfd.iPixelType = PFD_TYPE_RGBA;
714  pfd.iLayerType = PFD_MAIN_PLANE;
715  pfd.cColorBits = (BYTE) nBitsColor;
716  pfd.cDepthBits = (BYTE) nBitsDepth;
717  pfd.cStencilBits = (BYTE) nBitsStencil;
718 
719  int iformat = ChoosePixelFormat(dc, &pfd);
720  while ( iformat == 0 && (pfd.cDepthBits > 16) )
721  {
722  getConsole().EOLn(" WARNING: %d-bit Z-Buffer failed", pfd.cDepthBits);
723  pfd.cDepthBits -= 8;
724  iformat = ChoosePixelFormat(dc, &pfd);
725  }
726 
727  if ( iformat == 0 )
728  {
729  getConsole().EOLn(" ERROR: not found pixel format!");
730  return;
731  }
732 
733  nBitsDepth = (TPRREuint) pfd.cDepthBits;
734  iSimplePixelFormat = iformat;
735  getConsole().SOLn(" > Simple pixel format collected!");
736 } // FindSimplePixelFormat()
