163TPureBool PureScreenImpl::applyDisplaySettings(HDC dc, TPURE_SCREEN_PF pixelFormat)
164{
165 getConsole().OLn("PureScreen::applyDisplaySettings()");
166
167 if ( bDisplaySettingsApplied )
168 {
169 getConsole().OLn(" Already applied!");
170 if (pixelFormat != PURE_SCREEN_PF_NONE)
171 {
172 FindAdvancedPixelFormat(dc);
173 }
174
175 return false;
176 }
177
178 if ( (nBitsColor == 0) || (nBitsDepth == 0) || (nResX == 0) || (nResY == 0) )
179 {
180 getConsole().EOLn(" ERROR: rr == %d, clr == %d, z == %d, %d x %d",
181 nRefreshRate, nBitsColor, nBitsDepth, nResX, nResY);
182 return false;
183 }
184
185 bool canproceed = true;
186
187 if ( bFullScreen )
188 {
189 if ( nRefreshRate == 0 )
190 { // maximum Hz
191 nRefreshRate = findMaximumRefreshRate(nResX, nResY, nBitsColor);
192 getConsole().OLn(" selected refresh rate = %d Hz", nRefreshRate);
193 }
194 canproceed = switchToFullscreenMode(nResX, nResY, nBitsColor, nRefreshRate);
195 } // bFullScreen
196
197 if ( canproceed )
198 {
199 getConsole().SOn(); getConsole().OLn(" can proceed"); getConsole().SOff();
200 ReadCurrentDisplaySettings(nResX, nResY, nBitsColor, nRefreshRate);
201 getConsole().OLn(" Current display settings: %dx%dx%d @ %d Hz",
202 nResX, nResY, nBitsColor, nRefreshRate);
203 FindSimplePixelFormat(dc);
204 switch (pixelFormat)
205 {
206 case PURE_SCREEN_PF_SIMPLE:
207 bDisplaySettingsApplied = applyPixelFormat(dc, iSimplePixelFormat, &m_pfd);
208 break;
209 case PURE_SCREEN_PF_ADVANCED:
210 bDisplaySettingsApplied = applyPixelFormat(dc, m_iAdvancedPixelFormat, &m_pfd);
211 break;
212 case PURE_SCREEN_PF_NONE:
213 // TODO: create DIB in RAM to render to as in CSWRenderer ctor!
214 bDisplaySettingsApplied = true;
215 break;
216 default:
217 bDisplaySettingsApplied = false;
218 getConsole().EOLn("ERROR: invalid pixelFormat specified!");
219 }
220
221 bFSAA_ready = (m_iAdvancedPixelFormat == m_iAppliedPixelFormat);
222
223 return bDisplaySettingsApplied;
224 } // canproceed
225
226 return false;
227
228} // applyDisplaySettings()

231void PureScreenImpl::ResetDisplaySettings()
232{
233 getConsole().OLn("PureScreen::ResetDisplaySettings()");
234
235 if ( !bDisplaySettingsApplied || !bFullScreen )
236 {
237 getConsole().OLn(" nothing to reset.");

252 getConsole().EOLn("ChangeDisplaySettings(NULL, 0) failed: %d", cdsret);

276void PureScreenImpl::SetResolution(TPureUInt w, TPureUInt h)
277{

279 return;
280
281 if ( (w == 0) && (h == 0) )
282 {

287 {

289 nResY = dm.dmPelsHeight;
290 }
291 else
292 {
293 nResX = 800;
294 nResY = 600;
295 }
296 }
297 else

299 nResX = w;
300 nResY = h;

508void PureScreenImpl::FindSimplePixelFormat(HDC dc)
509{
510 getConsole().OLn(" FindSimplePixelFormat() ...");
511 if ( iSimplePixelFormat > 0 )
512 {
513 getConsole().OLn(" already found!");
514 return;
515 }
516
517 memset(&m_pfd, 0, sizeof(PIXELFORMATDESCRIPTOR));
518 m_pfd.nSize = sizeof(PIXELFORMATDESCRIPTOR);
519 m_pfd.nVersion = (WORD) 1;
520 m_pfd.dwFlags = PFD_DRAW_TO_WINDOW | PFD_SUPPORT_OPENGL | PFD_DOUBLEBUFFER | PFD_GENERIC_ACCELERATED;
521 m_pfd.iPixelType = PFD_TYPE_RGBA;
522 m_pfd.iLayerType = PFD_MAIN_PLANE;
523 m_pfd.cColorBits = (BYTE) nBitsColor;
524 m_pfd.cDepthBits = (BYTE) nBitsDepth;
525 m_pfd.cStencilBits = (BYTE) nBitsStencil;
526
527 int iformat = ChoosePixelFormat(dc, &m_pfd);
528 while ( iformat == 0 && (m_pfd.cDepthBits > 16) )
529 {
530 getConsole().EOLn(" WARNING: %d-bit Z-Buffer failed", m_pfd.cDepthBits);
531 m_pfd.cDepthBits -= 8;
532 iformat = ChoosePixelFormat(dc, &m_pfd);
533 }
534
535 if ( iformat == 0 )
536 {
537 getConsole().EOLn(" ERROR: not found pixel format!");
538 return;
539 }
540
541 nBitsDepth = (TPureUInt) m_pfd.cDepthBits;
542 iSimplePixelFormat = iformat;
543 getConsole().SOLn(" > Simple pixel format collected!");
544} // FindSimplePixelFormat()

553void PureScreenImpl::FindAdvancedPixelFormat(HDC dc)
554{
555 getConsole().OLn(" FindAdvancedPixelFormat() ...");
556 if ( m_iAdvancedPixelFormat > 0 )
557 {
558 getConsole().OLn(" already found!");
559 return;
560 }
561
562 if ( wglChoosePixelFormatARB == PGENULL )
563 {
564 getConsole().EOLn(" ERROR: wglChoosePixelFormatARB() is NULL!");
565 return;
566 }
567
568 int pixelformat = 0;
569 UINT numformats = 0;
570 float fAttributes[] = {0,0};
571 int iAttributes[] =
572 {
573 WGL_DRAW_TO_WINDOW_ARB, GL_TRUE,
574 WGL_SUPPORT_OPENGL_ARB, GL_TRUE,
575 WGL_ACCELERATION_ARB, WGL_FULL_ACCELERATION_ARB,
576 WGL_COLOR_BITS_ARB, nBitsColor,
577 WGL_ALPHA_BITS_ARB, 8,
578 WGL_DEPTH_BITS_ARB, nBitsDepth,
579 WGL_STENCIL_BITS_ARB, nBitsStencil,
580 WGL_DOUBLE_BUFFER_ARB, GL_TRUE,
581 WGL_SAMPLE_BUFFERS_ARB, GL_TRUE,
582 WGL_SAMPLES_ARB, (int) iFSAA_req,
583 0, 0
584 };
585 int valid = wglChoosePixelFormatARB(dc, iAttributes, fAttributes, (UINT) 1, &pixelformat, &numformats);
586
587 if ( (valid == FALSE) || (numformats < 1) || (pixelformat < 1) )
588 {
589 getConsole().EOLn(" ERROR: wglChoosePixelFormatARB() failed!");
590 return;
591 }
592 m_iAdvancedPixelFormat = pixelformat;
593 getConsole().SOLn(" > Advanced pixel format collected!");
594} // FindAdvancedPixelFormat()

600void PureScreenImpl::ReadAdvancedPixelFormatInfo(HDC dc, int iAppliedPixelFormat)
601{
602 if ( wglGetPixelFormatAttribivARB == PGENULL )
603 return;
604
605 const UINT nParams = 13;
606 int iAttributes[nParams] =
607 {
608 WGL_ACCELERATION_ARB,
609 WGL_SWAP_LAYER_BUFFERS_ARB,
610 WGL_SWAP_METHOD_ARB,
611 WGL_DOUBLE_BUFFER_ARB,
612 WGL_COLOR_BITS_ARB,
613 WGL_RED_BITS_ARB,
614 WGL_GREEN_BITS_ARB,
615 WGL_BLUE_BITS_ARB,
616 WGL_ALPHA_BITS_ARB,
617 WGL_DEPTH_BITS_ARB,
618 WGL_STENCIL_BITS_ARB,
619 WGL_SAMPLE_BUFFERS_ARB,
620 WGL_SAMPLES_ARB
621 };
622 int nValues[nParams];
623
624 if ( !wglGetPixelFormatAttribivARB(dc, iAppliedPixelFormat, 0, nParams, iAttributes, nValues) )
625 {
626 getConsole().EOLn(" WARNING: wglGetPixelFormatAttribivARB() not null but failed!");
627 return;
628 }
629
630 getConsole().OLn(" clr: %d R%dG%dB%dA%d Z%d S%d", nValues[4], nValues[5], nValues[6], nValues[7],
631 nValues[8], nValues[9], nValues[10]);
632 switch ( nValues[0] )
633 {
634 case WGL_FULL_ACCELERATION_ARB: getConsole().OLn(" Accelerated by ICD."); break;
635 case WGL_GENERIC_ACCELERATION_ARB: getConsole().EOLn(" Accelerated by MCD."); break;
636 case WGL_NO_ACCELERATION_ARB: getConsole().EOLn(" Software rasterization."); break;
637 } // switch
638
639 /*if ( nValues[1] )
640 getConsole().OLn(" BackBuffer is present.");
641 else
642 getConsole().EOLn(" WARNING: no backbuffer!");
643 */
644
645 switch ( nValues[2] )
646 {
647 case WGL_SWAP_EXCHANGE_ARB: getConsole().OLn(" Swap mode: exchange."); break;
648 case WGL_SWAP_COPY_ARB: getConsole().OLn(" Swap mode: copy."); break;
649 case WGL_SWAP_UNDEFINED_ARB: getConsole().OLn(" Swap mode: copy w/ undefined."); break;
650 } // switch
651
652 if ( nValues[3] )
653 getConsole().OLn(" DoubleBuffering.");
654 else
655 getConsole().EOLn(" WARNING: no DoubleBuffering!");
656
657 getConsole().OLn(" Number of sample buffers: %d.", nValues[11]);
658 iFSAA_state = nValues[12];
659 getConsole().OLn(" FSAA is ready @ %dX mode.", iFSAA_state);
660} // ReadAdvancedPixelFormatInfo()

669TPureBool PureScreenImpl::applyPixelFormat(HDC dc, int iPixelFormat, PIXELFORMATDESCRIPTOR* pfd)
670{
671 if ( SetPixelFormat(dc, iPixelFormat, pfd) )
672 {
673 getConsole().SOLn(" SetPixelFormat() passed");
674 m_iAppliedPixelFormat = GetPixelFormat(dc);
675 if (m_iAppliedPixelFormat != iPixelFormat)
676 {
677 getConsole().EOLn(" WARNING: GetPixelFormat() did not return the expected pixel format!");
678 }
679 DescribePixelFormat(dc, m_iAppliedPixelFormat, sizeof(PIXELFORMATDESCRIPTOR), pfd);
680 getConsole().OLn(" clr: %d R%dG%dB%dA%d Z%d S%d ver: %d",
681 pfd->cColorBits, pfd->cRedBits, pfd->cGreenBits, pfd->cBlueBits,
682 pfd->cAlphaBits, pfd->cDepthBits, pfd->cStencilBits, pfd->nVersion);
683 ReadAdvancedPixelFormatInfo(dc, m_iAppliedPixelFormat);
684 return true;
685 } // SetPixelFormat()
686
687 getConsole().EOLn(" ERROR: SetPixelFormat() failed!");
688 return false;
689} // applyPixelFormat()

715TPureUInt PureScreenImpl::findMaximumRefreshRate(TPureUInt nWidth, TPureUInt nHeight, TPureInt nCDepth)
716{
717 TPureUInt nFoundMaxRefreshRate = 0;
718 DEVMODE dm;
719 memset(&dm, 0, sizeof(DEVMODE));
720 dm.dmSize = sizeof(DEVMODE);
721 for (DWORD i = 0; EnumDisplaySettings(NULL, i, &dm); i++)
722 {
723 if ( (dm.dmPelsWidth == (DWORD) nWidth) && (dm.dmPelsHeight == (DWORD) nHeight) && (dm.dmBitsPerPel == (DWORD) nCDepth) )
724 {
725 if ( dm.dmDisplayFrequency > (DWORD) nFoundMaxRefreshRate )
726 nFoundMaxRefreshRate = (TPureUInt) dm.dmDisplayFrequency;
727 }
728 }
729 return nFoundMaxRefreshRate;
