130TPureUInt PR00FsUltimateRenderingEngineImpl::initialize(
131 TPURE_RENDERER rndr,
132 TPureUInt width, TPureUInt height,
133 TPURE_DISPLAY_MODES dmode, TPureUInt freq,
134 TPureInt cdepth, TPureInt zdepth, TPureInt stencil,
135 TPureInt samples,
136 HWND window)
137{
138 TPureInt st = stencil;
139
140 getConsole().OLn("Pure::initialize(%d, %d, %d, dmode, %d, %d, %d, %d, %d)",
141 rndr, width, height, freq, cdepth, zdepth, st, samples);
142 if ( isInitialized() )
143 {
144 getConsole().SOLn(" > already initialized!");
145 return 1;
146 }
147
148 getConsole().OIOLn( PURE_NAME );
149 getConsole().OLn( PURE_VERSION );
150
151 switch (rndr)
152 {
153 case PURE_RENDERER_SW:
154 pRenderer = &PureRendererSWincremental::createAndGet(m_cfgProfiles, wnd, screen, hwInfo);
155 break;
156 case PURE_RENDERER_HW_FP:
157 pRenderer = &PureRendererHWfixedPipe::createAndGet(m_cfgProfiles, wnd, screen, hwInfo);
158 break;
159 default:
160 getConsole().EOLnOO("ERROR: invalid renderer specified!");
161 return 1;
162 }
163
164 if ( pRenderer->initialize(width, height, dmode, freq, cdepth, zdepth, stencil, samples, window) != 0 )
165 {
166 getConsole().EOLnOO("ERROR: failed to initialize renderer!");
167 return 1;
168 }
169
170 // OpenGL is fully operational at this point so managers can be instantiated
171 // Note in case of SW rendering there is no OpenGL, but managers are able to detect that condition so we are safe here.
172 try
173 {
174 // I know this error handling is a joke!
175 // the try is added to catch the exceptions might be thrown by new,
176 // but since the ctors are not throwing YET, I need to check isInitialized() after each other.
177 // obviously I will add exception throwing to ctors and then this code will be much cleaner.
178
179 pImageMgr = new PureImageManager();
180
181 pTextureMgr = new PureTextureManager();
182 if ( !pTextureMgr->isInitialized() )
183 {
184 getConsole().EOLn("ERROR: failed to initialize PureTextureManager!");
185 shutdown();
186 getConsole().OO();
187 return 1;
188 }
189
190 pMaterialMgr = new PureMaterialManager();
191 if ( !pMaterialMgr->isInitialized() )
192 {
193 getConsole().EOLn("ERROR: failed to initialize PureMaterialManager!");
194 shutdown();
195 getConsole().OO();
196 return 1;
197 }
198
199 pMesh3DMgr = new PureMesh3DManager(*pMaterialMgr);
200 if ( !pMesh3DMgr->isInitialized() )
201 {
202 getConsole().EOLn("ERROR: failed to initialize PureMesh3DManager!");
203 shutdown();
204 getConsole().OO();
205 return 1;
206 }
207
208 pObject3DMgr = new PureObject3DManager(*pTextureMgr, *pMaterialMgr);
209 if ( !pObject3DMgr->isInitialized() )
210 {
211 getConsole().EOLn("ERROR: failed to initialize PureObject3DManager!");
212 shutdown();
213 getConsole().OO();
214 return 1;
215 }
216
217 pCamera = new PureCamera();
218 // initialize camera viewport to window client area
219 // we can trust window clientrect size even if this is an external window
220 pCamera->SetViewport(0, 0, wnd.getClientWidth(), wnd.getClientHeight());
221 pCamera->SetAspectRatio(wnd.getClientWidth() / (TPureFloat) wnd.getClientHeight());
222 } // try
223 catch (const std::bad_alloc&)
224 {
225 getConsole().EOLn("ERROR: failed to instantiate a manager!");
226 shutdown();
227 getConsole().OO();
228 return 1;
229 }
230
231 uiMgr.Initialize( wnd.getWndHandle(), wnd.getWndDC());
232
233 pRenderer->SetManagers(pObject3DMgr, &uiMgr, pCamera);
234
235 // PURE is fully operational at this point
236 getConsole().OLn("");
237 getConsole().SOLn("> PURE initialized");
238 getConsole().OLnOO("");
239
240 return 0;
241
242} // initialize()

253TPureBool PR00FsUltimateRenderingEngineImpl::shutdown()
254{
255 getConsole().OLn("PR00FsUltimateRenderingEngine::shutdown()");
256 getConsole().OLn("");
257
258 if ( getAutoWriteStatsAtShutdown() )
259 {
260 getConsole().SetLoggingState("4LLM0DUL3S", true);
261 getConsole().OIOLn("Dumping states ...");
262 getConsole().OLn("");
263
264 WriteList();
265 getConsole().OLnOO("");
266
267 getConsole().SetLoggingState("4LLM0DUL3S", false);
268 }
269
270 getConsole().OIOLn("Shutting down subsystems ...");
271 getConsole().OLn("");
272 getConsole().OI();
273
274 uiMgr.Deinitialize();
275
276 // managed objects like objects, textures, etc are implicitly deleted by calling dtor of their managers below
277 delete pCamera;
278 pCamera = PGENULL;
279
280 delete pObject3DMgr;
281 pObject3DMgr = PGENULL;
282
283 delete pMesh3DMgr;
284 pMesh3DMgr = PGENULL;
285
286 delete pMaterialMgr;
287 pMaterialMgr = PGENULL;
288
289 delete pTextureMgr;
290 pTextureMgr = PGENULL;
291
292 delete pImageMgr;
293 pImageMgr = PGENULL;
294
295 bool success = true;
296
297 // since renderer owns GL context, it must be shut down last, so managers still have the chance above to properly
298 // delete any GL resources
299 if ( pRenderer )
300 {
301 if (!( success = pRenderer->shutdown() ))
302 getConsole().EOLnOO("ERROR: failed to shut down renderer!");
303 pRenderer = PGENULL;
304 }
305
306 // theoretically window, screen, and hwinfo are already deinitialized by renderer, however we need to take care of proper housekeeping
307 wnd.Deinitialize();
308 screen.ResetDisplaySettings();
309 hwInfo.Deinitialize();
310
311 getConsole().OO();
312
313 getConsole().OLnOO("END OF SHUTDOWN!");
314
315 return success;
316} // shutdown()

445void PR00FsUltimateRenderingEngineImpl::PrintTexturesUnusedByMaterials() const
446{
447 if ( !isInitialized() )
448 return;
449
450 const PureTextureManager& texMgr = getTextureManager();
451 const PureMaterialManager& matMgr = getMaterialManager();
452
453 getConsole().OLn("List of Textures Unused by Materials");
454 getConsole().OLn("====================================");
455
456 TPureUInt nUnusedTextures = 0;
457
458 for (TPureInt texi = 0; texi < texMgr.getCount(); texi++)
459 {
460 PureTexture* tex = (PureTexture*) texMgr.getAttachedAt(texi);
461 if ( !tex )
462 continue;
463
464 bool textureUsedByMaterial = false;
465 for (TPureInt mati = 0; !textureUsedByMaterial && (mati < matMgr.getCount()); mati++)
466 {
467 PureMaterial* mat = (PureMaterial*) matMgr.getAttachedAt(mati);
468 if ( !mat)
469 continue;
470
471 for (TPureUInt layeri = 0; !textureUsedByMaterial && (layeri < matMgr.getMaximumLayerCount()); layeri++)
472 {
473 if ( mat->getTexture(layeri) == tex )
474 textureUsedByMaterial = true;
475 }
476 } // for mati
477 if ( !textureUsedByMaterial )
478 {
479 ++nUnusedTextures;
480 getConsole().OLn("Internal num.: %d, file name: %s", tex->getInternalNum(), tex->getFilename().c_str());
481 }
482 } // for texi
483
484 getConsole().OLn("%d Textures Found to be Unused by Checking %d Materials.", nUnusedTextures, matMgr.getCount());
485 getConsole().OLn("");
486} // PrintUnusedTextures()

504 PureMaterial* mat = (PureMaterial*) matMgr.getAttachedAt(mati);

699void PR00FsUltimateRenderingEngineImpl::CopyScreenToTexture(PureTexture& tex)
700{
701 // This function should be rather part of PureScreen(), however it is easier to implement here.
702 // One challenge of implementing this in PureScreen() is that that class needs to know:
703 // - HWInfo to check if multitexturing is supported, although singleton instance is easily get with getHardwareInfo();
704 // - current camera because that is not singleton and cannot be queried by that class, this is the real challenge.
705 // Camera is needed to get viewport size. It is not ok to check current resolution because:
706 // - it does not equal to expected image size when engine runs in windowed mode;
707 // - in future we may want to have multiple cameras with their own viewports, and CopyToTexture() should copy only 1 viewport content.
708
709 if ( !isInitialized() )
710 return;
711
712 // we don't expect texture dimension to be rectangular (w != h), but
713 // its width must be same as our viewport's width
714 if ( tex.getWidth() != getCamera().getViewport().size.width )
715 return;
716
717 if ( tex.getHeight() == 0 )
718 return;
719
720 if ( tex.getInternalNum() == 0 )
721 return;
722
723 if ( getHardwareInfo().getVideo().isMultiTexturingSupported() )
724 {
725 glActiveTextureARB(GL_TEXTURE1_ARB);
726 glDisable(GL_TEXTURE_2D);
727 glActiveTextureARB(GL_TEXTURE0_ARB);
728 }
729
730 glEnable(GL_TEXTURE_2D);
731 glBindTexture(GL_TEXTURE_2D, tex.getInternalNum());
732 glCopyTexImage2D(GL_TEXTURE_2D, 0, GL_RGB, 0, 0, tex.getWidth(), tex.getHeight(), 0);
733}
