102 void DiscoverDrawRangeElementsAvailability()

112 getConsole().EOLn("DrawRangeElements is disabled, upgrade graphics card driver!");
113 bSuppDrawRangeElements = false;

117
118 glGetIntegerv(GL_MAX_ELEMENTS_INDICES_EXT, &(tmpRet[1]));

128 getConsole().SOLn("DrawRangeElements supported!");
129 getConsole().OLn(" Max recommended vertex array vertices: %d", tmpRet[0]);
130 getConsole().OLn(" Max recommended vertex array indices: %d", tmpRet[1]);

133 getConsole().EOLn("DrawRangeElements not supported!");
