121 void PRREhwInfoImpl::Initialize(
122  HGLRC rc, HDC wnd_dc,
123  TPRREuint nResX, TPRREuint nResY,
124  TPRREint nColorBits, TPRREint nDepthBits,
125  TPRREint nStencilBits, TPRREint nFSAAlevel )
126 {
127 
128  getConsole().OIOLn("PRREhwInfo::Initialize() ...");
129  getConsole().OI();
130  if ( bInitialized )
131  {
132  getConsole().OLnOO("Already initialized!");
133  getConsole().OO();
134  return;
135  }
136 
137  if ( rc == NULL )
138  {
139  getConsole().OLn("Warning: rc is NULL!");
140  }
141 
142  PreInitialize();
143  hwCentralProcessor.Initialize();
144  hwMemory.Initialize();
145  hwVideo.Initialize(rc, wnd_dc, nResX, nResY, nColorBits, nDepthBits, nStencilBits, nFSAAlevel);
146  hwAudio.Initialize();
147  bInitialized = hwCentralProcessor.isInitialized() &&
148  hwMemory.isInitialized() &&
149  hwVideo.isInitialized() &&
150  hwAudio.isInitialized();
151 
152  if ( bInitialized )
153  {
154  getConsole().OOSOLn("done!");
155  getConsole().OO();
156  }
157  else
158  {
159  getConsole().OOEOLn("Failed to initialize 1 or more hardware class(es)!");
160  getConsole().OO();
161  WriteStats();
162  return;
163  }
164 
165 } // Initialize(...)
