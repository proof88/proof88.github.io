117 glGetQueryivARB(GL_SAMPLES_PASSED_ARB, GL_QUERY_COUNTER_BITS_ARB, &(tmpRet[0]));

120 getConsole().EOLn("Occlusion query ext. is exported, but query cntr bits is %d.", tmpRet[0]);
