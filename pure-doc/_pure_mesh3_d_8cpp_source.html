267void PureMesh3D::PureMesh3DImpl::RecalculateSize()
268{
269 if ( isLevel1() )
270 {
271 // we are parent, ask our submeshes to determine their sizes and object-local positions
272 for (TPureInt i = 0; i < _pOwner->getCount(); i++)
273 ((PureMesh3D*)_pOwner->getAttachedAt(i))->RecalculateSize();
274
275 // now we can determine our own size as parent based on min and max subobject positions and size
276 TXYZ minvals, maxvals;
277 const PureMesh3D* subMesh = (PureMesh3D*)_pOwner->getAttachedAt(0);
278 minvals.x = subMesh->getPosVec().getX() - subMesh->getSizeVec().getX() / 2.0f;
279 minvals.y = subMesh->getPosVec().getY() - subMesh->getSizeVec().getY() / 2.0f;
280 minvals.z = subMesh->getPosVec().getZ() - subMesh->getSizeVec().getZ() / 2.0f;
281 maxvals.x = subMesh->getPosVec().getX() + subMesh->getSizeVec().getX() / 2.0f;
282 maxvals.y = subMesh->getPosVec().getY() + subMesh->getSizeVec().getY() / 2.0f;
283 maxvals.z = subMesh->getPosVec().getZ() + subMesh->getSizeVec().getZ() / 2.0f;
284
285 for (TPureInt i = 1; i < _pOwner->getCount(); i++)
286 {
287 subMesh = (PureMesh3D*)_pOwner->getAttachedAt(i);
288 if ( minvals.x > subMesh->getPosVec().getX() - subMesh->getSizeVec().getX() / 2.0f )
289 minvals.x = subMesh->getPosVec().getX() - subMesh->getSizeVec().getX() / 2.0f;
290 if ( maxvals.x < subMesh->getPosVec().getX() + subMesh->getSizeVec().getX() / 2.0f )
291 maxvals.x = subMesh->getPosVec().getX() + subMesh->getSizeVec().getX() / 2.0f;
292 if ( minvals.y > subMesh->getPosVec().getY() - subMesh->getSizeVec().getY() / 2.0f )
293 minvals.y = subMesh->getPosVec().getY() - subMesh->getSizeVec().getY() / 2.0f;
294 if ( maxvals.y < subMesh->getPosVec().getY() + subMesh->getSizeVec().getY() / 2.0f )
295 maxvals.y = subMesh->getPosVec().getY() + subMesh->getSizeVec().getY() / 2.0f;
296 if ( minvals.z > subMesh->getPosVec().getZ() - subMesh->getSizeVec().getZ() / 2.0f )
297 minvals.z = subMesh->getPosVec().getZ() - subMesh->getSizeVec().getZ() / 2.0f;
298 if ( maxvals.z < subMesh->getPosVec().getZ() + subMesh->getSizeVec().getZ() / 2.0f )
299 maxvals.z = subMesh->getPosVec().getZ() + subMesh->getSizeVec().getZ() / 2.0f;
300 }
301
302 vSize.SetX( abs( minvals.x - maxvals.x ) );
303 vSize.SetY( abs( minvals.y - maxvals.y ) );
304 vSize.SetZ( abs( minvals.z - maxvals.z ) );
305
306 // since we are parent we dont need to modify our world-space position at all based on our subobjects
307
308 vRelPos.SetX( minvals.x + vSize.getX() / 2.0f );
309 vRelPos.SetY( minvals.y + vSize.getY() / 2.0f );
310 vRelPos.SetZ( minvals.z + vSize.getZ() / 2.0f );
311
312 return;
313 }
314
315 // as level-2 in theory we have geometry but check it for sure
316 if ( (nVertices_h == 0) || (pVertices == PGENULL) )
317 {
318 _pOwner->getManagedConsole().EOLn("PureMesh3D::PureMesh3DImpl::RecalculateSize() in level-2 but vertex data is not present!");
319 return;
320 }
321
322 // we are subobject, so we need to calculate our own size and object-local position based on min and max vertex positions
323
324 TXYZ minvals, maxvals;
325 minvals.x = pVertices[0].x;
326 minvals.y = pVertices[0].y;
327 minvals.z = pVertices[0].z;
328 maxvals.x = minvals.x;
329 maxvals.y = minvals.y;
330 maxvals.z = minvals.z;
331
332 for (TPureUInt i = 1; i < nVertices_h; i++)
333 {
334 if ( minvals.x > pVertices[i].x ) minvals.x = pVertices[i].x;
335 if ( maxvals.x < pVertices[i].x ) maxvals.x = pVertices[i].x;
336 if ( minvals.y > pVertices[i].y ) minvals.y = pVertices[i].y;
337 if ( maxvals.y < pVertices[i].y ) maxvals.y = pVertices[i].y;
338 if ( minvals.z > pVertices[i].z ) minvals.z = pVertices[i].z;
339 if ( maxvals.z < pVertices[i].z ) maxvals.z = pVertices[i].z;
340 }
341
342 vSize.SetX( abs( minvals.x - maxvals.x ) );
343 vSize.SetY( abs( minvals.y - maxvals.y ) );
344 vSize.SetZ( abs( minvals.z - maxvals.z ) );
345
346 vPos.SetX( minvals.x + vSize.getX() / 2.0f );
347 vPos.SetY( minvals.y + vSize.getY() / 2.0f );
348 vPos.SetZ( minvals.z + vSize.getZ() / 2.0f );
349
350 // vRelPos is not updated for level-2, as default [0,0,0] is always valid since we set position above based on minvals and size!
351
352} // RecalculateSize()

392TPureBool PureMesh3D::PureMesh3DImpl::cannibalize(PureMesh3D& victim)
393{
394 // Note that we cannot easily copy victim's Impl since copy ctor and assignment operators are empty!
395 // Thus the following assignment cannot be used: *(this->pImpl) = *(victim.pImpl)
396 // We wouldn't even want to use copy ctor or assignment here since we do not want any array copy.
397 // That is why this function is written, to "steal" the arrays of the victim so victim will be "empty".
398 // In C++11 this could be done with a move ctor.
399
400 // Note: this function could be recursive and be called for level-1 mesh only, it would get the job
401 // done for all the submeshes too. But now it was easier to implement this silly way so it needs to be
402 // invoked manually for all submeshes too.
403
404 // we should not have any submesh at this point, and no geometry yet
405 if ( (_pOwner->getCount() != 0) || (pVertices != PGENULL) || (nVertices_h > 0) )
406 {
407 _pOwner->getManagedConsole().EOLn(
408 "PureMesh3D::cannibalize() ERROR: cannibalizer has non-zero managed count (%d) or has geometry (%d)!", _pOwner->getCount(), nVertices_h);
409 return false;
410 }
411
412 // at this point we do not need to deal with the Manager or FiledManager parts
413
414 _pOwner->SetName(victim.getName()); // copy the Managed part
415 _pOwner->SetFilename(victim.getFilename()); // copy the FiledManaged part
416
417 primitiveFormat = victim.pImpl->primitiveFormat;
418
419 pVertices = victim.pImpl->pVertices;
420 pNormals = victim.pImpl->pNormals;
421 pVertexIndices = victim.pImpl->pVertexIndices;
422 nMinVertexIndex = victim.pImpl->nMinVertexIndex;
423 nMaxVertexIndex = victim.pImpl->nMaxVertexIndex;
424 nIndicesType = victim.pImpl->nIndicesType;
425
426 nVertices_h = victim.pImpl->nVertices_h;
427 nVertexIndices_h = victim.pImpl->nVertexIndices_h;
428 nFaces_h = victim.pImpl->nFaces_h;
429
430 vPos = victim.pImpl->vPos;
431 vRelPos = victim.pImpl->vRelPos;
432 vSize = victim.pImpl->vSize;
433
434 pMaterial = victim.pImpl->pMaterial;
435 pMaterial->SetUtiliser(this->_pOwner);
436
437 victim.pImpl->pVertices = PGENULL;
438 victim.pImpl->pNormals = PGENULL;
439 victim.pImpl->pVertexIndices = PGENULL;
440 victim.pImpl->pMaterial = PGENULL;
441 victim.pImpl->nMinVertexIndex = UINT_MAX;
442 victim.pImpl->nMaxVertexIndex = 0;
443 victim.pImpl->nIndicesType = GL_UNSIGNED_BYTE;
444 victim.pImpl->nVertices_h = 0;
445 victim.pImpl->nVertexIndices_h = 0;
446 victim.pImpl->nFaces_h = 0;
447
448 return true;
449} // cannibalize()

880void PureMesh3D::RecalculateSize()
