57PureBoundingVolumeHierarchy* PureBoundingVolumeHierarchy::insertObject(const PureObject3D& obj)
58{
59 // Note 1:
60 // most of the time dynamic_cast should be used for downcasting, however in this specific case,
61 // we can guarantee that the returned Octree node is actually a derived PureBoundingVolumeHierarchy node,
62 // since we are in PureBoundingVolumeHierarchy method. So we can use static_cast for faster cast.
63
64 // Note 2: although we call ancestor's insertObject(), due to polymorphism it will still call our
65 // derived Subdivide() method, so children will be PureBoundingVolumeHierarchy instances!
66 PureBoundingVolumeHierarchy* node = static_cast<PureBoundingVolumeHierarchy*>(PureOctree::insertObject(obj));
67 if ( PGENULL == node )
68 {
69 return PGENULL;
70 }
71
72 const PureAxisAlignedBoundingBox obj_bbox(obj.getPosVec(), obj.getScaledSizeVec());
73 node->aabb.ExtendBy(obj_bbox);
74
75 PureBoundingVolumeHierarchy* prevChild = node;
76 PureBoundingVolumeHierarchy* p = static_cast<PureBoundingVolumeHierarchy*>(node->parent);
77 while (p != PGENULL)
78 {
79 p->aabb.ExtendBy(prevChild->aabb);
80 prevChild = p;
81 p = static_cast<PureBoundingVolumeHierarchy*>(p->parent);
82 }
83
84 return node;
85} // insertObject()

143TPureBool PureBoundingVolumeHierarchy::subdivide()
144{
145 assert(vChildren.size() == 0);
146 try
147 {
148 for (TPureUInt i = 0; i < 8; i++)
149 {
150 PureBoundingVolumeHierarchy* const pChildNode = new PureBoundingVolumeHierarchy(PureVector(), getSize()/2.f, getMaxDepthLevel(), getDepthLevel() + 1);
151 vChildren.push_back(pChildNode);
152 }
153 }
154 catch (const std::exception&)
155 {
156 for (TPureUInt i = 0; i < 8; i++)
157 {
158 if ( vChildren[i] != PGENULL )
159 {
160 delete vChildren[i];
161 }
162 vChildren.resize(0);
163 }
164 return false;
165 }
166 return true;
167}
