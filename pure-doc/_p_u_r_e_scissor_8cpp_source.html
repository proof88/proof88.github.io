22  1001 | 1000 | 1010

65 bool PUREScissor::DoScissor(int& x0, int& y0, int& x1, int& y1, int xmin, int xmax, int ymin, int ymax)
66 {
67 
68  // Cohen-Sutherland.
69  // Might be studied against parametric line clipping ( Cyrus & Beck (1978), Liang & Barsky (1984) ) from performance perspective.
70 
71  /*
72  // unsure why this swap is present in my old code, theoretically this is not needed at all
73  if ( x0 > x1 )
74  {
75  Swap(x0, x1);
76  Swap(y0, y1);
77  }
78  */
79 
80  CS_RGNCODE c0 = ComputeCode(x0, y0, xmin, xmax, ymin, ymax);
81  CS_RGNCODE c1 = ComputeCode(x1, y1, xmin, xmax, ymin, ymax);
82 
83  while ( !( canTrivialAccept(c0, c1) || canTrivialReject(c0, c1) ) )
84  {
85  // either P0 or P1 is outside
86  if ( c0 == 0 )
87  {
88  Swap(x0, x1);
89  Swap(y0, y1);
90  Swap(c0, c1);
91  }
92 
93  // clip the line against a clip region
94  if ( (c0 & CS_LEFT) == CS_LEFT ) IntersectLeft(x0, y0, x1, y1, xmin);
95  else if ( (c0 & CS_RIGHT) == CS_RIGHT ) IntersectRight(x0, y0, x1, y1, xmax);
96  else if ( (c0 & CS_BOTTOM) == CS_BOTTOM ) IntersectBottom(x0, y0, x1, y1, ymin);
97  else if ( (c0 & CS_TOP) == CS_TOP ) IntersectTop(x0, y0, x1, y1, ymax);
98 
99  c0 = ComputeCode(x0, y0, xmin, xmax, ymin, ymax);
100  }
101 
102  return canTrivialAccept(c0, c1);
103 } // DoScissor()

123 PUREScissor::CS_RGNCODE PUREScissor::ComputeCode(int x, int y, int xmin, int xmax, int ymin, int ymax)
124 {
125  PUREScissor::CS_RGNCODE code = CS_INSIDE;
126 
127  if (x < xmin) // to the left of clip window
128  code |= CS_LEFT;
129  else if (x > xmax) // to the right of clip window
130  code |= CS_RIGHT;
131 
132  if (y < ymin) // below the clip window
133  code |= CS_BOTTOM;
134  else if (y > ymax) // above the clip window
135  code |= CS_TOP;
136 
137  return code;
138 }
