18/*
19 Region codes for Cohen-Sutherland line clipping algorithm.
20 | |
21 1001 | 1000 | 1010
22 _____|______|_____
23 | |
24 0001 | 0000 | 0010
25 _____|______|_____
26 | |
27 0101 | 0100 | 0110
28 | |
29
30 The sequence for reading the codes' bits is TBRL (Top, Bottom, Right, Left).
31 Example: 1010 means Top Right region.
32*/
33
34const PUREScissor::CS_RGNCODE PUREScissor::CS_INSIDE = 0; // 0000

64bool PUREScissor::DoScissor(int& x0, int& y0, int& x1, int& y1, int xmin, int xmax, int ymin, int ymax)
65{
66
67 // Cohen-Sutherland.
68 // Might be studied against parametric line clipping ( Cyrus & Beck (1978), Liang & Barsky (1984) ) from performance perspective.
69
70 /*
71 // unsure why this swap is present in my old code, theoretically this is not needed at all
72 if ( x0 > x1 )
73 {
74 Swap(x0, x1);
75 Swap(y0, y1);
76 }
77 */
78
79 CS_RGNCODE c0 = ComputeCode(x0, y0, xmin, xmax, ymin, ymax);
80 CS_RGNCODE c1 = ComputeCode(x1, y1, xmin, xmax, ymin, ymax);
81
82 while ( !( canTrivialAccept(c0, c1) || canTrivialReject(c0, c1) ) )
83 {
84 // either P0 or P1 is outside
85 if ( c0 == 0 )
86 {
87 Swap(x0, x1);
88 Swap(y0, y1);
89 Swap(c0, c1);
90 }
91
92 // clip the line against a clip region
93 if ( (c0 & CS_LEFT) == CS_LEFT ) IntersectLeft(x0, y0, x1, y1, xmin);
94 else if ( (c0 & CS_RIGHT) == CS_RIGHT ) IntersectRight(x0, y0, x1, y1, xmax);
95 else if ( (c0 & CS_BOTTOM) == CS_BOTTOM ) IntersectBottom(x0, y0, x1, y1, ymin);
96 else if ( (c0 & CS_TOP) == CS_TOP ) IntersectTop(x0, y0, x1, y1, ymax);
97
98 c0 = ComputeCode(x0, y0, xmin, xmax, ymin, ymax);
99 }
100
101 return canTrivialAccept(c0, c1);
102} // DoScissor()

122PUREScissor::CS_RGNCODE PUREScissor::ComputeCode(int x, int y, int xmin, int xmax, int ymin, int ymax)
123{
124 PUREScissor::CS_RGNCODE code = CS_INSIDE;
125
126 if (x < xmin) // to the left of clip window
127 code |= CS_LEFT;
128 else if (x > xmax) // to the right of clip window
129 code |= CS_RIGHT;
130
131 if (y < ymin) // below the clip window
132 code |= CS_BOTTOM;
133 else if (y > ymax) // above the clip window
134 code |= CS_TOP;
135
136 return code;
137}

150void PUREScissor::IntersectLeft(int& x0, int& y0, int& x1, int& y1, int xmin)
151{
152 float m = (y1-y0) / (float)(x1-x0);
153 y0 = (int) (m * (xmin-x0) + y0);
154 x0 = xmin;
155}
