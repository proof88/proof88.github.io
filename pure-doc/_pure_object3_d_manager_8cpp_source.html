253void PureObject3DManager::Attach(PureManaged& m)
254{
255 if ( m.getManager() != PGENULL )
256 return;
257
258 PureMesh3DManager::Attach(m);
259 try
260 {
261 PureObject3D& obj = dynamic_cast<PureObject3D&>(m);
262
263 // it will be an occludee for sure, since:
264 // - either this is a brand new object attached first time to its manager,
265 // - or Detach() had previously removed it from occludees_* and occluders lists, and without manager it cannot be set as occluder so it stayed unchanged for sure!
266 obj.SetOccluder(false);
267 }
268 catch (const std::exception&)
269 {
270 getConsole().OLn("WARNING: PureObject3DManager::Attach() param is NOT Object3D ref!");
271 }
272} // Attach()

280void PureObject3DManager::Detach(PureManaged& m)
281{
282 if ( m.getManager() == PGENULL )
283 return;
284
285 try
286 {
287 PureObject3D& obj = dynamic_cast<PureObject3D&>(m);
288 obj.SetOcclusionTested(false);
289 obj.getMaterial(false).setBlendMode(PURE_BM_NONE); // removes from occludees_blended and occludees_2d_blended container
290 obj.SetOccluder(false); // removes from occluders container
291
292 auto occ_it = std::find(PureObject3D::PureObject3DImpl::occludees_opaque.begin(), PureObject3D::PureObject3DImpl::occludees_opaque.end(), &obj);
293 if ( occ_it != PureObject3D::PureObject3DImpl::occludees_opaque.end() )
294 {
295 PureObject3D::PureObject3DImpl::occludees_opaque.erase(occ_it);
296 }
297 else
298 {
299 occ_it = std::find(PureObject3D::PureObject3DImpl::occludees_2d_opaque.begin(), PureObject3D::PureObject3DImpl::occludees_2d_opaque.end(), &obj);
300 if ( occ_it != PureObject3D::PureObject3DImpl::occludees_2d_opaque.end() )
301 {
302 PureObject3D::PureObject3DImpl::occludees_2d_opaque.erase(occ_it);
303 }
304 }
305 }
306 catch (const std::exception&)
307 {
308 // no need to log
309 //getConsole().OLn("WARNING: PureObject3DManager::Detach() param is NOT Object3D ref!");
310 }
311
312 PureMesh3DManager::Detach(m);
313} // Detach()

485PureObject3D* PureObject3DManager::createPlane(
486 TPureFloat a, TPureFloat b,
487 TPURE_VERTEX_MODIFYING_HABIT vmod,
488 TPURE_VERTEX_REFERENCING_MODE vref,
489 TPureBool bForceUseClientMemory)
490{
491 if ( !pImpl->isInitialized() )
492 {
493 return PGENULL;
494 }
495
496 getConsole().OLnOI("PureObject3DManager::createPlane(%f, %f)", a, b);
497
498 PureObject3D* obj = PGENULL;
499 PureObject3D* subobj = PGENULL;
500 try
501 {
502 obj = new PureObject3D(pImpl->materialMgr, vmod, vref, bForceUseClientMemory);
503 // do not attach obj to object3dmanager yet, we need its material created first!
504
505 subobj = new PureObject3D(pImpl->materialMgr, vmod, vref, bForceUseClientMemory);
506 obj->Attach( *subobj );
507
508 PureMesh3DManager* const me = this;
509 if ( !(me->pImpl->convertToPlane(*obj, a, b)) )
510 {
511 throw std::runtime_error("convertToPlane() failed!");
512 }
513
514 // now we have material, it can be attached to object3dmanager!
515 Attach( *obj );
516
517 subobj->pImpl->pVerticesTransf = new TPURE_TRANSFORMED_VERTEX[subobj->getVerticesCount()];
518 }
519 catch (const std::exception& e)
520 {
521 delete subobj; // will invoke obj->Detach()
522 delete obj; // would delete subobj, except if it is not yet attached, so we explicitly delete subobj above
523 getConsole().EOLnOO("ERROR: PureObject3DManager::createPlane() failed to instantiate or fill PureObject3D: %s!", e.what());
524 getConsole().OLn("");
525 return PGENULL;
526 }
527
528 // although PPP has already selected the vtransmode, we set it again
529 // to actually allocate the needed resources for the geometry
530 getConsole().OI();
531 if ( !obj->setVertexTransferMode( obj->getVertexTransferMode() ) )
532 {
533 getConsole().OO();
534 DeleteAttachedInstance( *obj );
535 getConsole().EOLnOO("ERROR: setVertexTransferMode() failed, returning NULL now!");
536 getConsole().OLn("");
537 return PGENULL;
538 }
539 getConsole().OO();
540
541 obj->SetName("Object3D " + std::to_string(pImpl->nRunningCounter++));
542 obj->SetOccluder(false);
543
544 getConsole().SOLnOO("> Plane created successfully, name: %s!", obj->getName().c_str());
545 getConsole().OLn("");
546
547 return obj;
548
549} // createPlane()

688PureObject3D* PureObject3DManager::createFromFile(
689 const char* filename,
690 TPURE_VERTEX_MODIFYING_HABIT vmod,
691 TPURE_VERTEX_REFERENCING_MODE vref,
692 TPureBool bForceUseClientMemory )
693{
694 if ( !pImpl->isInitialized() )
695 {
696 return PGENULL;
697 }
698
699 getConsole().OLnOI("PureObject3DManager::createFromFile(\"%s\")", filename);
700
701 if ( filename == NULL )
702 {
703 getConsole().EOLnOO("ERROR: input was NULL, returning PGENULL");
704 getConsole().OLn("");
705 return PGENULL;
706 }
707
708 string sFileExt = PFL::getExtension(filename);
709 if ( sFileExt == "" )
710 {
711 getConsole().EOLnOO("ERROR: no file extension!");
712 getConsole().OLn("");
713 return PGENULL;
714 }
715 if ( !PFL::fileExists(filename) )
716 {
717 getConsole().EOLnOO("ERROR: file doesn't exist!");
718 getConsole().OLn("");
719 return PGENULL;
720 }
721
722 PureObject3D* obj = PGENULL;
723 PureObject3D* subobject = PGENULL;
724 PureMesh3D* tmpMesh = PGENULL;
725 #pragma warning(disable:4244) /* int-char conversion in std::transform */
726 transform(sFileExt.begin(), sFileExt.end(), sFileExt.begin(), ::toupper);
727 #pragma warning(default:4244)
728 getConsole().OLn("ext: .%s", sFileExt.c_str());
729 if ( sFileExt == "OBJ" )
730 {
731 try
732 {
733 getConsole().OI();
734 tmpMesh = PureMesh3DManager::createFromFile(filename);
735 if ( !tmpMesh )
736 {
737 getConsole().OO();
738 getConsole().EOLnOO("ERROR: failed to create tmpMesh!");
739 getConsole().OLn("");
740 return PGENULL;
741 }
742 obj = new PureObject3D(pImpl->materialMgr, vmod, vref, bForceUseClientMemory);
743 if ( !((PureMesh3D*)obj)->pImpl->cannibalize(*tmpMesh) )
744 {
745 const std::string sErrMsg = "cannibalize() failed for level-1 obj!";
746 throw std::runtime_error(sErrMsg);
747 }
748 obj->SetName("Object3D " + std::to_string(pImpl->nRunningCounter++));
749
750 TPureUInt nVerticesTotalTmpMesh = 0;
751 for (TPureInt i = 0; i < tmpMesh->getCount(); i++)
752 {
753 subobject = new PureObject3D(pImpl->materialMgr, vmod, vref, bForceUseClientMemory);
754 obj->Attach( *subobject );
755 if ( !((PureMesh3D*)subobject)->pImpl->cannibalize(*(PureMesh3D*)(tmpMesh->getAttachedAt(i))) )
756 {
757 const std::string sErrMsg = "cannibalize() failed for level-2 subobj " + std::to_string(i) + "!";
758 throw std::runtime_error(sErrMsg);
759 }
760 subobject->pImpl->pVerticesTransf = new TPURE_TRANSFORMED_VERTEX[subobject->getVerticesCount()];
761 nVerticesTotalTmpMesh += subobject->getVertexIndicesCount();
762
763 // Legacy tmcsgfxlib behavior: auto-load textures for subobjects where pipe character is present in name
764 const std::string::size_type nPipePos = subobject->getName().find('|');
765 if ( nPipePos != std::string::npos )
766 {
767 const std::string sTexName = subobject->getName().substr(nPipePos+1);
768 getConsole().OLn("Legacy: loading texture implicitly by submodel name: %s", sTexName.c_str());
769 PureTexture* const tex = pImpl->textureMgr.createFromFile( PFL::getDirectory(filename).append(sTexName).c_str() );
770 if ( tex )
771 {
772 if ( !subobject->getMaterial().setTexture(tex) )
773 {
774 getConsole().EOLn("ERROR: failed to set texture for subobject! Continuing ...");
775 }
776 }
777 else
778 {
779 getConsole().EOLn("ERROR: PureObject3DManager::createFromFile() failed to load texture: %s! Continuing ...", sTexName.c_str());
780 }
781 // since we have loaded texture from submodelname, we can get rid of the texture filename part of it
782 subobject->SetName( subobject->getName().substr(0, nPipePos) );
783 }
784 }
785 this->DeleteAttachedInstance(*tmpMesh);
786 tmpMesh = PGENULL;
787
788 obj->SetFilename(filename);
789 getConsole().OO();
790 Attach( *obj );
791
792 obj->SetOccluder(false);
793 if ( PureHwInfo::get().getVideo().isOcclusionQuerySupported() && pImpl->isEligibleForOcclusionQuery(nVerticesTotalTmpMesh) )
794 {
795 obj->SetOcclusionTested(true);
796 }
797
798 // although PPP has already selected the vtransmode, we set it again
799 // to actually allocate the needed resources for the geometry
800 getConsole().OI();
801 if ( !obj->setVertexTransferMode( obj->getVertexTransferMode() ) )
802 {
803 getConsole().OO();
804 DeleteAttachedInstance( *obj );
805 getConsole().EOLnOO("ERROR: setVertexTransferMode() failed, returning NULL now!");
806 getConsole().OLn("");
807 return PGENULL;
808 }
809 getConsole().OO();
810 } // try
811 catch (const std::exception& e)
812 {
813 // note: textures won't be deallocated cause we don't track what textures were loaded for this object!
814 delete tmpMesh;
815 delete subobject; // will invoke obj->Detach()
816 delete obj; // would delete subobj, except if it is not yet attached, so we explicitly delete subobj above
817 getConsole().OO();
818 getConsole().EOLnOO("ERROR: failed to instantiate: %s!", e.what());
819 getConsole().OLn("");
820 return PGENULL;
821 } // catch
822 }

959void PureObject3DManager::UpdateOccluderStates()
960{
961 //getConsole().SetLoggingState(PureObject3DManager::getLoggerModuleName(), true);
962
963 TPureFloat fAverageBiggestAreaScaled = 0.0f;
964 TPureUInt nCount = 0;
965 for (TPureInt i = 0; i < getSize(); i++)
966 {
967 const PureObject3D* const pMngd = (PureObject3D*) getAttachedAt(i);
968 if ( pMngd != PGENULL )
969 {
970 if ( !pMngd->isRenderingAllowed() )
971 continue;
972
973 const TPureBool bOpaque = !PureMaterial::isBlendFuncReallyBlending(pMngd->getMaterial(false).getSourceBlendFunc(), pMngd->getMaterial(false).getDestinationBlendFunc() );
974
975 if ( !bOpaque )
976 continue;
977
978 if ( pMngd->isStickedToScreen() )
979 continue;
980
981 if ( !pMngd->isAffectingZBuffer() )
982 continue;
983
984 if ( pMngd->isWireframed() )
985 continue;
986
987 nCount++;
988 fAverageBiggestAreaScaled += pMngd->getBiggestAreaScaled();
989 }
990 }
991 fAverageBiggestAreaScaled /= (TPureFloat)nCount;
992
993 //getConsole().OLnOI("%s() START: avg area: %f", __FUNCTION__, fAverageBiggestAreaScaled);
994 //getConsole().OLn("area, bAreaBiggerThanAvg, name, filename");
995 std::deque<PureObject3D*> occluderCandidates;
996 for (TPureInt i = 0; i < getSize(); i++)
997 {
998 PureObject3D* const pMngd = (PureObject3D*) getAttachedAt(i);
999 if ( pMngd != PGENULL )
1000 {
1001 if ( !pMngd->isRenderingAllowed() )
1002 continue;
1003
1004 const TPureBool bOpaque = !PureMaterial::isBlendFuncReallyBlending(pMngd->getMaterial(false).getSourceBlendFunc(), pMngd->getMaterial(false).getDestinationBlendFunc() );
1005
1006 if ( !bOpaque )
1007 continue;
1008
1009 if ( pMngd->isStickedToScreen() )
1010 continue;
1011
1012 if ( !pMngd->isAffectingZBuffer() )
1013 continue;
1014
1015 if ( pMngd->isWireframed() )
1016 continue;
1017
1018 const TPureBool bAreaBiggerThanAvg = (pMngd->getBiggestAreaScaled() > (fAverageBiggestAreaScaled * pImpl->fOccluderSelectionBias));
1019
1020 //getConsole().OLn("%f, %b, %s, %s", pMngd->getBiggestAreaScaled(), bAreaBiggerThanAvg, pMngd->getName().c_str(), pMngd->getFilename().c_str());
1021
1022 if ( bAreaBiggerThanAvg )
1023 {
1024 // dont let it be occluder yet at this point, just put in the candidates list, order by biggest scaled area, from bigger to smaller
1025 auto it = occluderCandidates.begin();
1026 while ( it != occluderCandidates.end() )
1027 {
1028 if ( (*it)->getBiggestAreaScaled() > pMngd->getBiggestAreaScaled() )
1029 {
1030 it++;
1031 }
1032 else
1033 {
1034 break;
1035 }
1036 }
1037 occluderCandidates.insert(it, pMngd);
1038 }
1039 else
1040 {
1041 pMngd->SetOccluder(false);
1042 }
1043 }
1044 }
1045 //getConsole().OOOLn("%s() END", __FUNCTION__);
1046
1047 // now we really set occluder state for the first nMaxOccluderCount element of the candidates list
1048 TPureUInt occCntr = 0;
1049 auto it = occluderCandidates.begin();
1050 while ( it != occluderCandidates.end() )
1051 {
1052 if ( occCntr < pImpl->nMaxOccluderCount )
1053 {
1054 //getConsole().OLn("%f, %s, %s", (*it)->getBiggestAreaScaled(), (*it)->getName().c_str(), (*it)->getFilename().c_str());
1055 (*it)->SetOccluder(true);
1056 }
1057 else
1058 {
1059 (*it)->SetOccluder(false);
1060 }
1061 occCntr++;
1062 it++;
1063 }
1064
1065 //getConsole().SetLoggingState(PureObject3DManager::getLoggerModuleName(), false);
1066} // UpdateOccluderStates()
