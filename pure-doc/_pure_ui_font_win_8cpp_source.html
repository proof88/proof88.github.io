35PureUiFontWin::PureUiFontWin(
36 const std::string& fontFaceName, int height, bool bold, bool italic, bool underline, bool strikeout,
37 HDC wnd_dc )
38{
39 sFontFaceName = fontFaceName;
40 nHeight = height;
41 nWeight = bold ? FW_BOLD : FW_NORMAL;
42 bItalic = italic ? TRUE : FALSE;
43 bUnderline = underline ? TRUE : FALSE;
44 bStrikeOut = strikeout ? TRUE : FALSE;
45
46 hDC = wnd_dc;
47
48 hFont = CreateFont(
49 nHeight, 0, 0, 0, nWeight, bItalic, bUnderline, bStrikeOut, EASTEUROPE_CHARSET,
50 OUT_TT_PRECIS, CLIP_DEFAULT_PRECIS, ANTIALIASED_QUALITY, FF_DONTCARE | DEFAULT_PITCH, sFontFaceName.c_str() );
51
52 if ( !hFont )
53 {
54 getConsole().EOLn(
55 "PureUiFontWin(%s, %d, %b, %b, %b, %b) FAILED to create font!",
56 fontFaceName.c_str(), height, bold, italic, underline, strikeout );
57 // exception should be thrown but ui system will be replaced anyway ...
58 return;
59 }
60
61
62 if ( PureHwInfo::get().getVideo().isAcceleratorDetected() ) {
63
64 // Create space for 255 characters.
65 base = glGenLists(255);
66
67 if ( base == 0 )
68 {
69 getConsole().EOLn(
70 "PureUiFontWin(%s, %d, %b, %b, %b, %b) FAILED to allocate display lists!",
71 fontFaceName.c_str(), height, bold, italic, underline, strikeout );
72 // DeleteObject(hFont); should i release the font object this way?
73 // exception should be thrown but ui system will be replaced anyway ...
74 return;
75 }
76
77 if ( NULL == SelectObject(hDC, hFont) )
78 {
79 getConsole().EOLn(
80 "PureUiFontWin(%s, %d, %b, %b, %b, %b) ERROR: SelectObject() FAILED!",
81 fontFaceName.c_str(), height, bold, italic, underline, strikeout );
82 // DeleteObject(hFont); should i release the font object this way?
83 // exception should be thrown but ui system will be replaced anyway ...
84 return;
85 }
86
87 // rumor is that wglUseFontBitmaps() fails if there is a gl error before ... so calling glGetError() before this solves that problem.
88 // display lists created here will contain 1 call only: glBitmap().
89 // also pls check what are the return values of this function!
90 if ( !wglUseFontBitmaps(hDC, 0, 255, base) )
91 {
92 getConsole().EOLn(
93 "PureUiFontWin(%s, %d, %b, %b, %b, %b) FAILED to generate bitmaps!",
94 fontFaceName.c_str(), height, bold, italic, underline, strikeout );
95 return;
96 }
97
98 }
99
100 getConsole().SOLn(
101 "PureUiFontWin(%s, %d, %b, %b, %b, %b) created font successfully, handle: %d.",
102 fontFaceName.c_str(), height, bold, italic, underline, strikeout, hFont );
103}

143PureUiFontWin::~PureUiFontWin()
144{
145 if ( hFont )
146 {
147 // should i release the font object this way?
148 if ( DeleteObject(hFont) )
149 {
150 getConsole().SOLn(
151 "~PureUiFontWin(%s, %d, %b, %b, %b, %b) deleted font successfully, handle: %d.",
152 sFontFaceName.c_str(), nHeight, nWeight, bItalic, bUnderline, bStrikeOut, hFont );
153 hFont = NULL;
154 }
155 else
156 getConsole().EOLn(
157 "~PureUiFontWin(%s, %d, %b, %b, %b, %b) FAILED to delete font, handle: %d!",
158 sFontFaceName.c_str(), nHeight, nWeight, bItalic, bUnderline, bStrikeOut, hFont );
159
160 if ( base != 0 )
161 {
162 glDeleteLists(base, 255);
163 base = 0;
164 }
165
166 }
167 else
168 getConsole().OLn(
169 "~PureUiFontWin(%s, %d, %b, %b, %b, %b) no handle.",
170 sFontFaceName.c_str(), nHeight, nWeight, bItalic, bUnderline, bStrikeOut );
171}
