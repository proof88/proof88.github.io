441PureVector& PureVector::operator^=(const PureVector& vec)
442{
443 TPURE_XYZW newpos;
444 newpos.x = pos.y*vec.pos.z - pos.z*vec.pos.y;
445 newpos.y = pos.z*vec.pos.x - pos.x*vec.pos.z;
446 newpos.z = pos.x*vec.pos.y - pos.y*vec.pos.x;
447 pos = newpos;
448 return *this;
