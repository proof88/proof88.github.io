210void Bullet::Update(const unsigned int& nFactor)
211{
212 /*
213 * In the PR00FPS Promo flash game I did this:
214 * 1.) when firing the bullet, the xMove and yMove properties of the new bullet were initialized like this:
215 * eval("_root.mc_map.mc_bullet"+_root.bulletCount).xMove = Math.cos((this.mc_player_arm1._rotation+aimShit)*Math.PI/180.0) * neg;
216 * eval("_root.mc_map.mc_bullet"+_root.bulletCount).yMove = Math.sin((this.mc_player_arm1._rotation+aimShit)*Math.PI/180.0) * neg;
217 *
218 * And initial positions were set based on the above like this:
219 * eval("_root.mc_map.mc_bullet"+_root.bulletCount)._x = this.newX + eval("_root.mc_map.mc_bullet"+_root.bulletCount).xMove*this.mc_player_arm1._width;
220 * eval("_root.mc_map.mc_bullet"+_root.bulletCount)._y = this.newY+150 + eval("_root.mc_map.mc_bullet"+_root.bulletCount).yMove*this.mc_player_arm1._width;
221 *
222 * 2.) And bulletProc updated the position of the bullets with a simple multiplication:
223 * this._x += this.xMove * _root.BULLETSPEED;
224 * this._y += this.yMove * _root.BULLETSPEED;
225 *
226 * Maybe this approach would be faster than using PUT.Move() like below.
227 * However, collision check is the most expensive thing now anyway, so I don't think about this now.
228 */
229 m_put.Move(m_speed / nFactor);
230 m_obj->getPosVec() = m_put.getPosVec();
231}

626void Weapon::UpdatePositions(const PureVector& playerPos, const PureVector& targetPos2D)
627{
628 getObject3D().getPosVec().Set( playerPos.getX(), playerPos.getY() + WpnYBiasToPlayerCenter, playerPos.getZ() );
629
630 /*
631 By default with AngleY 0 and AngleZ 0, weapon looks to <- direction.
632
633 When AngleY is 0, and:
634 - AngleZ is 0, it means weapon looks to <- direction.
635 ^
636 - AngleZ is -45, it means weapon looks to \ direction.
637
638 - AngleZ is 45, it means weapon looks to / direction.
639
640
641 When AngleY is 180, and:
642 - AngleZ is 0, it means weapon looks to -> direction.
643 ^
644 - AngleZ is -45, it means weapon looks to / direction.
645
646 - AngleZ is 45, it means weapon looks to \ direction.
647
648
649 This means that when targetX is less than posX, then angleY should be 0, otherwise it needs to be 180, and angleX always needs to be between 90 and -90.
650
651 Y
652 ^
653 |
654 | (x) (x) is the xhair, (w) is the weapon, and is angleZ of wpn we need to set.
655 | |\ Length of A and B are obviously known. Thus tan can be calculated by B/A, so in radians will be arctan(B/A).
656 | B| \
657 | |___\(w)
658 | A
659 -|------------------------->X
660 |
661
662 */
663
664 PureVector vecWpnPosProjected2D;
665 if (!m_gfx.getCamera().project3dTo2d(
666 getObject3D().getPosVec().getX(),
667 getObject3D().getPosVec().getY(),
668 getObject3D().getPosVec().getZ(),
669 vecWpnPosProjected2D))
670 {
671 // Failure is expected when the weapon is out of frustum, so this is not a real error, ignore it silently
672 //getConsole().EOLn("PRooFPSddPGE::%s(): project3dTo2d() failed!", __func__);
673 return;
674 }
675 else
676 {
677 vecWpnPosProjected2D.SetX(vecWpnPosProjected2D.getX() - m_gfx.getCamera().getViewport().size.width / 2.f);
678 vecWpnPosProjected2D.SetY(vecWpnPosProjected2D.getY() - m_gfx.getCamera().getViewport().size.height / 2.f);
679 // getConsole().EOLn("PRooFPSddPGE::%s(): wpn2d: (%f,%f,%f), target2d: (%f,%f,%f)",
680 // __func__,
681 // vecWpnPosProjected2D.getX(), vecWpnPosProjected2D.getY(), vecWpnPosProjected2D.getZ(),
682 // targetPos2D.getX(), targetPos2D.getY(), targetPos2D.getZ());
683 }
684
685 const PureVector vecNewTargetPos2D = targetPos2D - vecWpnPosProjected2D;
686 const float distYXratio = (vecNewTargetPos2D.getX() == 0.f) ? vecNewTargetPos2D.getY() : vecNewTargetPos2D.getY()/ vecNewTargetPos2D.getX();
687 if (vecNewTargetPos2D.getX() < 0.f )
688 {
689 getObject3D().getAngleVec().SetY( 0.f );
690 getObject3D().getAngleVec().SetZ( atan( distYXratio )*180.f / PFL::PI );
691 }
692 else
693 {
694 getObject3D().getAngleVec().SetY( 180.f );
695 getObject3D().getAngleVec().SetZ( -atan( distYXratio )*180.f / PFL::PI );
696 }
697}

842bool Weapon::update()
843{
844 UpdateGraphics();
845
846 if ( m_state == WPN_READY )
847 {
848 return false;
849 }
850
851 PFL::timeval timeNow;
852 PFL::gettimeofday(&timeNow, 0);
853
854 if ( m_state == WPN_SHOOTING )
855 {
856 const TPureFloat fMillisecsSinceLastShot = PFL::getTimeDiffInUs(timeNow, m_timeLastShot) / 1000.f;
857 if ( getVars()["firing_cooldown"].getAsInt() <= fMillisecsSinceLastShot )
858 {
859 m_state = WPN_READY;
860 }
861 return false;
862 }
863
864 // WPN_RELOADING
865 if ( getVars()["reload_per_mag"].getAsBool() )
866 {
867 const TPureFloat fMillisecsSinceReloadStarted = PFL::getTimeDiffInUs(timeNow, m_timeReloadStarted) / 1000.f;
868 if ( getVars()["reload_time"].getAsInt() <= fMillisecsSinceReloadStarted )
869 {
870 if ( getVars()["reload_whole_mag"].getAsBool() )
871 {
872 m_nMagBulletCount = m_nBulletsToReload;
873 }
874 else
875 {
876 m_nMagBulletCount += m_nBulletsToReload;
877 }
878 m_nUnmagBulletCount -= m_nBulletsToReload;
879 m_state = WPN_READY;
880 return true;
881 }
882 }
883 else
884 {
885 const TPureFloat fMillisecsSinceLastBulletReload = PFL::getTimeDiffInUs(timeNow, m_timeReloadStarted) / 1000.f;
886 if ( getVars()["reload_time"].getAsInt() <= fMillisecsSinceLastBulletReload )
887 {
888 m_nMagBulletCount++;
889 m_nUnmagBulletCount--;
890 m_nBulletsToReload--;
891 if ( m_nBulletsToReload == 0 )
892 {
893 m_state = WPN_READY;
894 }
895 else
896 {
897 m_timeReloadStarted = timeNow;
898 }
899 return true;
900 }
901 }
902 return false;
903}

990TPureBool Weapon::pullTrigger(bool bMoving, bool bRun, bool bDuck)
991{
992 const bool bPrevTriggerReleased = m_bTriggerReleased;
993 m_bTriggerReleased = false;
994
995 if ( (m_state != WPN_READY) && /* reloading can be stopped if it is per-bullet */
996 !( (m_state == WPN_RELOADING) && (!getVars()["reload_per_mag"].getAsBool()) ) )
997 {
998 return false;
999 }
1000
1001 if (!bPrevTriggerReleased && (m_firingMode != WPN_FM_AUTO))
1002 {
1003 // in non-automatic modes, to trigger a next shot, we require the trigger to be released before being pulled
1004 return false;
1005 }
1006
1007 if ( m_nMagBulletCount == 0 )
1008 {
1009 return false;
1010 }
1011
1012 m_state = WPN_SHOOTING;
1013 m_nMagBulletCount--;
1014
1015
1016
1017 const float fRelativeBulletAngleZ = getRandomRelativeBulletAngle(bMoving, bRun, bDuck);
1018 //getConsole().EOLn("bMoving: %b, bRun: %b, bDuck: %b, fRelativeBulletAngleZ: %f! ", bMoving, bRun, bDuck, fRelativeBulletAngleZ);
1019
1020 m_bullets.push_back(
1021 Bullet(
1022 m_gfx,
1023 m_connHandle,
1024 m_obj->getPosVec().getX(), m_obj->getPosVec().getY(), m_obj->getPosVec().getZ(),
1025 m_obj->getAngleVec().getX(), m_obj->getAngleVec().getY(), m_obj->getAngleVec().getZ() + fRelativeBulletAngleZ,
1026 getVars()["bullet_size_x"].getAsFloat(),
1027 getVars()["bullet_size_y"].getAsFloat(),
1028 getVars()["bullet_size_z"].getAsFloat(),
1029 getVars()["bullet_speed"].getAsFloat(),
1030 getVars()["bullet_gravity"].getAsFloat(),
1031 getVars()["bullet_drag"].getAsFloat(),
1032 getVars()["bullet_fragile"].getAsBool(),
1033 getVars()["damage_ap"].getAsInt(),
1034 getVars()["damage_hp"].getAsInt(),
1035 getVars()["damage_area_size"].getAsFloat(),
1036 getVars()["damage_area_pulse"].getAsFloat())
1037 );
1038
1039 PFL::gettimeofday(&m_timeLastShot, 0);
1040
1041 return true;
1042}

1199float Weapon::getMomentaryRecoilMultiplier() const
1200{
1201 const float fRecoilMax = getMaximumRecoilMultiplier();
1202 if (fRecoilMax <= 1.f)
1203 {
1204 // fRecoilMax is minimized at 1.f by ctor but it is better to to have the condition that way
1205 return 1.f;
1206 }
1207
1208 PFL::timeval timeNow;
1209 PFL::gettimeofday(&timeNow, 0);
1210 const TPureFloat fMillisecsSinceLastShot = PFL::getTimeDiffInUs(timeNow, m_timeLastShot) / 1000.f;
1211
1212 if (fMillisecsSinceLastShot < 0.f)
1213 {
1214 // has never ever fired a shot yet
1215 return 1.f;
1216 }
1217
1218 const float fRecoilCooldownMillisecs = getVars().at("recoil_cooldown").getAsFloat();
1219
1220 // ctor makes sure that recoil_cooldown is positive (bigger than firing_cooldown) if recoil_m is > 1.f, so
1221 // this assertion is implied from ctor behavior, thus we cannot divide by zero below!
1222 assert(fRecoilCooldownMillisecs > 0.f);
1223
1224 if (fRecoilCooldownMillisecs <= fMillisecsSinceLastShot)
1225 {
1226 return 1.f;
1227 }
1228
1229 return PFL::lerp(1.f, fRecoilMax, 1.f - (fMillisecsSinceLastShot / fRecoilCooldownMillisecs));
1230}
