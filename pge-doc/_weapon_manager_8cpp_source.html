376void Bullet::Update(const unsigned int& nFactor)
377{
378 /*
379 * In the PR00FPS Promo flash game I did this:
380 * 1.) when firing the bullet, the xMove and yMove properties of the new bullet were initialized like this:
381 * eval("_root.mc_map.mc_bullet"+_root.bulletCount).xMove = Math.cos((this.mc_player_arm1._rotation+aimShit)*Math.PI/180.0) * neg;
382 * eval("_root.mc_map.mc_bullet"+_root.bulletCount).yMove = Math.sin((this.mc_player_arm1._rotation+aimShit)*Math.PI/180.0) * neg;
383 *
384 * And initial positions were set based on the above like this:
385 * eval("_root.mc_map.mc_bullet"+_root.bulletCount)._x = this.newX + eval("_root.mc_map.mc_bullet"+_root.bulletCount).xMove*this.mc_player_arm1._width;
386 * eval("_root.mc_map.mc_bullet"+_root.bulletCount)._y = this.newY+150 + eval("_root.mc_map.mc_bullet"+_root.bulletCount).yMove*this.mc_player_arm1._width;
387 *
388 * 2.) And bulletProc updated the position of the bullets with a simple multiplication:
389 * this._x += this.xMove * _root.BULLETSPEED;
390 * this._y += this.yMove * _root.BULLETSPEED;
391 *
392 * Maybe this approach would be faster than using PUT.Move() like below.
393 * However, collision check is the most expensive thing now anyway, so I don't think about this now.
394 */
395 const float fMoveDistance = m_speed / nFactor;
396 m_put.Move(fMoveDistance);
397 m_obj->getPosVec() = m_put.getPosVec();
398 m_fDistTravelled += fMoveDistance;
399

935void Weapon::UpdatePositions(const PureVector& playerPos, const PureVector& targetPos2D)
936{
937 getObject3D().getPosVec().Set( playerPos.getX(), playerPos.getY() + WpnYBiasToPlayerCenter, playerPos.getZ() );
938
939 /*
940 By default with AngleY 0 and AngleZ 0, weapon looks to <- direction.
941
942 When AngleY is 0, and:
943 - AngleZ is 0, it means weapon looks to <- direction.
944 ^
945 - AngleZ is -45, it means weapon looks to \ direction.
946
947 - AngleZ is 45, it means weapon looks to / direction.
948
949
950 When AngleY is 180, and:
951 - AngleZ is 0, it means weapon looks to -> direction.
952 ^
953 - AngleZ is -45, it means weapon looks to / direction.
954
955 - AngleZ is 45, it means weapon looks to \ direction.
956
957
958 This means that when targetX is less than posX, then angleY should be 0, otherwise it needs to be 180, and angleX always needs to be between 90 and -90.
959
960 Y
961 ^
962 |
963 | (x) (x) is the xhair, (w) is the weapon, and is angleZ of wpn we need to set.
964 | |\ Length of A and B are obviously known. Thus tan can be calculated by B/A, so in radians will be arctan(B/A).
965 | B| \
966 | |___\(w)
967 | A
968 -|------------------------->X
969 |
970
971 */
972
973 PureVector vecWpnPosProjected2D;
974 if (!m_gfx.getCamera().project3dTo2d(
975 getObject3D().getPosVec().getX(),
976 getObject3D().getPosVec().getY(),
977 getObject3D().getPosVec().getZ(),
978 vecWpnPosProjected2D))
979 {
980 // Failure is expected when the weapon is out of frustum, so this is not a real error, ignore it silently
981 //getConsole().EOLn("PRooFPSddPGE::%s(): project3dTo2d() failed!", __func__);
982 return;
983 }
984 else
985 {
986 vecWpnPosProjected2D.SetX(vecWpnPosProjected2D.getX() - m_gfx.getCamera().getViewport().size.width / 2.f);
987 vecWpnPosProjected2D.SetY(vecWpnPosProjected2D.getY() - m_gfx.getCamera().getViewport().size.height / 2.f);
988 // getConsole().EOLn("PRooFPSddPGE::%s(): wpn2d: (%f,%f,%f), target2d: (%f,%f,%f)",
989 // __func__,
990 // vecWpnPosProjected2D.getX(), vecWpnPosProjected2D.getY(), vecWpnPosProjected2D.getZ(),
991 // targetPos2D.getX(), targetPos2D.getY(), targetPos2D.getZ());
992 }
993
994 const PureVector vecNewTargetPos2D = targetPos2D - vecWpnPosProjected2D;
995 const float distYXratio = (vecNewTargetPos2D.getX() == 0.f) ? vecNewTargetPos2D.getY() : vecNewTargetPos2D.getY()/ vecNewTargetPos2D.getX();
996 if (vecNewTargetPos2D.getX() < 0.f )
997 {
998 getObject3D().getAngleVec().SetY( 0.f );
999 getObject3D().getAngleVec().SetZ( atan( distYXratio )*180.f / PFL::PI );
1000 }
1001 else
1002 {
1003 getObject3D().getAngleVec().SetY( 180.f );
1004 getObject3D().getAngleVec().SetZ( -atan( distYXratio )*180.f / PFL::PI );
1005 }

1177 if ( getVars()["reload_time"].getAsInt() <= fMillisecsSinceReloadStarted )
1178 {

1180 {
1181 m_nMagBulletCount = m_nBulletsToReload;

1186 }
1187 m_nUnmagBulletCount -= m_nBulletsToReload;
1188 m_state = WPN_READY;
1189 return true;
1190 }
1191 }
1192 else

1198 m_nUnmagBulletCount--;

1202 m_state = WPN_READY;
1203 }
1204 else
1205 {
1206 m_timeReloadStarted = timeNow;

1209 }

1284

1301TPureBool Weapon::pullTrigger(bool bMoving, bool bRun, bool bDuck)
1302{
1303 const bool bPrevTriggerReleased = m_bTriggerReleased;
1304 m_bTriggerReleased = false;
1305
1306 if ( (m_state != WPN_READY) && /* reloading can be stopped if it is per-bullet */
1307 !( (m_state == WPN_RELOADING) && (!getVars()["reload_per_mag"].getAsBool()) ) )
1308 {
1309 return false;
1310 }
1311
1312 if (!bPrevTriggerReleased && (m_firingMode != WPN_FM_AUTO))
1313 {
1314 // in non-automatic modes, to trigger a next shot, we require the trigger to be released before being pulled
1315 return false;
1316 }
1317
1318 if ( m_nMagBulletCount == 0 )
1319 {
1320 return false;
1321 }
1322
1323 m_state = WPN_SHOOTING;
1324
1325 if (m_type != Type::Melee)
1326 {
1327 m_nMagBulletCount--;
1328 }
1329
1330 const float fRelativeBulletAngleZ = getRandomRelativeBulletAngle(bMoving, bRun, bDuck);
1331 //getConsole().EOLn("bMoving: %b, bRun: %b, bDuck: %b, fRelativeBulletAngleZ: %f! ", bMoving, bRun, bDuck, fRelativeBulletAngleZ);
1332
1333 // here create() invokes PooledBullet::init(), should invoke the server version!
1334 if (!m_bullets.create(
1335 m_id,
1336 m_gfx,
1337 m_connHandle,
1338 m_obj->getPosVec().getX(), m_obj->getPosVec().getY(), m_obj->getPosVec().getZ(),
1339 m_obj->getAngleVec().getX(), m_obj->getAngleVec().getY(), m_obj->getAngleVec().getZ() + fRelativeBulletAngleZ,
1340 getVars()["bullet_visible"].getAsBool(),
1341 getVars()["bullet_size_x"].getAsFloat(),
1342 getVars()["bullet_size_y"].getAsFloat(),
1343 getVars()["bullet_size_z"].getAsFloat(),
1344 getVars()["bullet_speed"].getAsFloat(),
1345 getVars()["bullet_gravity"].getAsFloat(),
1346 getVars()["bullet_drag"].getAsFloat(),
1347 getVars()["bullet_fragile"].getAsBool(),
1348 getVars()["bullet_distance_max"].getAsFloat(),
1349 (getVars()["bullet_particle"].getAsString() == "smoke" ?
1350 Bullet::ParticleType::Smoke :
1351 Bullet::ParticleType::None),
1352 getVars()["damage_ap"].getAsInt(),
1353 getVars()["damage_hp"].getAsInt(),
1354 getVars()["damage_area_size"].getAsFloat(),
1355 (getVars()["damage_area_effect"].getAsString() == "linear" ?
1356 Bullet::DamageAreaEffect::Linear :
1357 Bullet::DamageAreaEffect::Constant),
1358 getVars()["damage_area_pulse"].getAsFloat()))
1359 {
1360 getConsole().EOLn("Weapon::pullTrigger(): pool did not create bullet!");
1361 return false;

1542{

1545 {
