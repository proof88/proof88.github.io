209 langTable = new string*[n](); // () makes all ptrs NULL

310int PGEcfgProfiles::addProfile(const char* sUser, const char* sNick)
311{
312 if ((sUser == NULL) || (sNick == NULL))
313 {
314 return -2;
315 }
316
317 // this is ridiculous, PFL still doesnt have std::string-compatible strClr() !!!
318 char cLine[200]{};
319 strncpy_s(cLine, sizeof(cLine), sNick, sizeof(cLine)); // last param should be something else but here I have no better choice ...
320 if ((strlen(sNick) == 0) || (PFL::strClr(cLine) != strlen(sNick)))
321 {
322 return -2;
323 }
324
325 getConsole().OLnOI("PGEcfgProfiles::addProfile(%s, %s)", sUser, sNick);
326
327 if (getPathToProfiles().empty())
328 {
329 getConsole().EOLnOO("ERROR: profiles path unknown!");
330 return -3;
331 }
332
333 const std::string sFileToCreate = getPathToProfiles() + sUser + "\\" + sUser + ".cfg";
334 const std::string sDirToCreate = getPathToProfiles() + sUser;
335
336 const std::string** const sProfileList = getProfilesList();
337 bool bSameUserFound = false;
338 for (int i = 0; i < getProfilesCount(); i++)
339 {
340 bSameUserFound = ( *(sProfileList[i]) == sUser );
341 if ( bSameUserFound )
342 break;
343 }
344
345 if ( bSameUserFound )
346 {
347 getConsole().EOLnOO("ERROR: same user found!");
348 return -1;
349 }
350
351 if ( _mkdir( sDirToCreate.c_str() ) != 0 )
352 {
353 getConsole().EOLnOO("ERROR: couldn't create dir: %s", sDirToCreate.c_str());
354 return -2;
355 }
356
357 ofstream f_cfg( sFileToCreate );
358 if ( f_cfg.fail() )
359 {
360 getConsole().EOLnOO("ERROR: couldn't create file: %s", sFileToCreate.c_str());
361 return -2;
362 }
363 // write optional header, otherwise switching to this profile may fail if validateOnLoad() fails
364 if (!validateOnSave(f_cfg))
365 {
366 getConsole().EOLnOO("ERROR: validateOnSave() failed for file: %s", sFileToCreate.c_str());
367 return -2;
368 }
369 // write nickname as it is mandatory for valid cfg
370 f_cfg << PGE_SYS_CFG_PLAYER_NAME_CVAR << " = " << sNick << std::endl;
371 f_cfg.close();
372
373 LoadProfilesList();
374
375 const std::string** const sNewProfileList = getProfilesList();
376 for (int i = 0; i < getProfilesCount(); i++)
377 {
378 if ( *(sNewProfileList[i]) == sUser )
379 {
380 getConsole().SOLnOO("> done!");
381 return i;
382 }
383 }
384
385 // if we have reached this point, we couldn't find the added user, so return error
386 getConsole().EOLnOO("ERROR: couldn't find the added profile!");
387 return -2;
388} // addProfile()

405int PGEcfgProfiles::deleteProfile(int nIndex)
406{
407 getConsole().OLnOI("PGEcfgProfiles::deleteProfile(%d)", nIndex);
408
409 if ( (nIndex < 0) || (nIndex >= getProfilesCount()) )
410 {
411 getConsole().EOLnOO("ERROR: invalid index!");
412 return -1;
413 }
414
415 if ( nIndex == getProfileIndex() )
416 {
417 getConsole().OLn("deleting current profile ...");
418 setProfile( -1 );
419 }
420 const string sUserToDelete = *getProfilesList()[nIndex];
421 const string sFileToDelete = getPathToProfiles() + sUserToDelete + "\\" + sUserToDelete + ".cfg";
422 const string sDirToDelete = getPathToProfiles() + sUserToDelete;
423
424 if ( remove( sFileToDelete.c_str() ) != 0 )

427 return -2;

452void PGEcfgProfiles::ProcessCommandLine(const char* szCmdLine)
453{
454 const std::string sCmdLine(szCmdLine);
455
456 // I wanted this function to be less complex, using ancestor class PGEcfgFile's
457 // lineIsValueAssignment() and lineHandleAssignment() functions, but they contain
458 // some extra logic not fitting here ... anyway, maybe in the future some refactor
459 // could make them usable here too.
460
461 size_t nDoubleDashPos = sCmdLine.find("--", 0);
462 size_t nAssignmentPos = sCmdLine.find('=', nDoubleDashPos);
463 size_t nNextSpacePos = sCmdLine.find(' ', nAssignmentPos);
464
465 while ((nDoubleDashPos != std::string::npos) && (nAssignmentPos != std::string::npos))
466 {
467 if ((nAssignmentPos > (nDoubleDashPos + 2)) &&
468 ((nNextSpacePos == std::string::npos) || (nNextSpacePos > (nAssignmentPos + 1))))
469 { // --a=2
470 const std::string sAssignmentExpression =
471 sCmdLine.substr(
472 nDoubleDashPos + 2,
473 ((nNextSpacePos == std::string::npos) ? nNextSpacePos : nNextSpacePos - (nDoubleDashPos + 2))
474 );
475
476 if (sAssignmentExpression.find(' ') == std::string::npos)
477 {
478 const std::string sVar = sCmdLine.substr(nDoubleDashPos + 2, nAssignmentPos - (nDoubleDashPos+2));
479 const std::string sValue = sCmdLine.substr(
480 nAssignmentPos + 1,
481 (nNextSpacePos == std::string::npos) ? nNextSpacePos : (nNextSpacePos - nAssignmentPos - 1)
482 );
483 // this looks to be a valid assignment
484 getVars()[sVar] = sValue.c_str();
485 m_commandLineVars[sVar] = sValue.c_str();
486 }
487 }
488
489 nDoubleDashPos = sCmdLine.find("--", nAssignmentPos);
490 nAssignmentPos = sCmdLine.find('=', nDoubleDashPos);
491 nNextSpacePos = sCmdLine.find(' ', nAssignmentPos);
492 }
493}

534 if ( nIndex >= getProfilesCount() )
535 {

570 getConsole().SOLnOO("> done!");

586bool PGEcfgProfiles::setProfile(const std::string& sProfileName)
587{
588 getConsole().OLnOI("PGEcfgProfiles::SetProfile(%s)", sProfileName.c_str());
589 if (sProfileName.empty())
590 {
591 // request erasing out profile data
592 const bool b = setProfile(-1);
593 getConsole().OO();

595 }
596
597 if (getProfilesCount() == 0)
598 {
599 getConsole().EOLnOO("ERROR: there is no any profile!");

601 }
602
603 for (int i = 0; i < nProfilesCount; i++)
604 {
605 if (*sFoundProfiles[i] == sProfileName)
606 {
607 const bool b = setProfile(i);
608 // sanity check: setProfile() should set requested profile name too
609 if (b)
610 {
611 if (getProfileName() != sProfileName)
612 {
613 getConsole().EOLnOO("ERROR: %s != %s!", getProfileName().c_str(), sProfileName.c_str());
614 // either readConfiguration() in SetProfile(index) failed or there is something else, so erase profile data
615 setProfile(-1);
616 return false;
617 }
618 else
619 {
620 getConsole().SOLnOO("> done!");
621 return true;
622 }
623 }
624 getConsole().EOLnOO("ERROR: found profile with index %d but could not set it!", i);
625 return false;
626 }
627 }
628
629 getConsole().EOLnOO("ERROR: there is no such profile!");
630 return false;

840void PGEcfgProfiles::LoadProfilesList()
841{
842 getConsole().OLnOI("PGEcfgProfiles::LoadProfilesList() ...");
843 // save current user
844 string sOriginalCurrentProfileUser;
845 if ( getProfileIndex() != -1 )
846 sOriginalCurrentProfileUser = *( getProfilesList()[getProfileIndex()] );
847
848 nProfilesCount = 0;
849 char pOrigCurrDirr[1024];
850 char pCurrDirr[1024];
851 GetCurrentDirectory(1024, pOrigCurrDirr);
852 GetCurrentDirectory(1024, pCurrDirr);
853 if ( !SetCurrentDirectory(sPathToProfiles.c_str()) && (GetLastError() != ERROR_SUCCESS) )
854 {
855 getConsole().EOLnOO("ERROR: couldn't set current dir to: %s", sPathToProfiles.c_str());
856 return;
857 }
858
859 SetLastError(ERROR_SUCCESS); // need this because sometimes a last error was stuck and we couldn't enter the loop below
860
861 WIN32_FIND_DATA dirdata;
862 HANDLE hDirSearch = FindFirstFile("*.*", &dirdata);
863 if ( hDirSearch == INVALID_HANDLE_VALUE )
864 {
865 getConsole().EOLnOO("ERROR: hDirSearch == INVALID_HANDLE_VALUE");
866 return;
867 }
868
869 sFoundProfiles = PGENULL;
870 sFoundProfilePlayerNames = PGENULL;
871 while ( (GetLastError() != ERROR_NO_MORE_FILES) && (hDirSearch != INVALID_HANDLE_VALUE) )
872 {
873 string sCfgFileName;
874 if ( (strcmp(dirdata.cFileName,".") != 0) && (strcmp(dirdata.cFileName,"..") != 0) )
875 {
876 if ( (dirdata.dwFileAttributes & FILE_ATTRIBUTE_DIRECTORY) != 0 )
877 {
878 sCfgFileName = dirdata.cFileName;
879 sCfgFileName += ".cfg";
880 SetCurrentDirectory(dirdata.cFileName);
881 WIN32_FIND_DATA filedata;
882 HANDLE hFileSearch = FindFirstFile(sCfgFileName.c_str(), &filedata);
883 if ( (GetLastError() != ERROR_NO_MORE_FILES) && (hFileSearch != INVALID_HANDLE_VALUE) )
884 {
885 string sPlayerName;
886 // note: temporarily disabled file size checking ... decide later if it's needed
887 /*if ( (filedata.nFileSizeLow >= PGE_CFG_SETTINGSFILE_MIN_SIZE) &&
888 (filedata.nFileSizeLow <= PGE_CFG_SETTINGSFILE_MAX_SIZE) )*/
889 { // checking file size, just in case ...
890 if ( getPlayerNameFromFile(sCfgFileName.c_str(), sPlayerName) )
891 {
892 nProfilesCount++;
893 std::string** sNewFoundProfiles = PGENULL;
894 try
895 {
896 if ( sFoundProfiles )
897 { // mimicing realloc, this to be changed to vector or something!
898 sNewFoundProfiles = new string*[nProfilesCount]();
899 memcpy(sNewFoundProfiles, sFoundProfiles, sizeof(string*)*nProfilesCount);
900 delete[] sFoundProfiles;
901 sFoundProfiles = sNewFoundProfiles;
902 }
903 else
904 {
905 sFoundProfiles = new string*[nProfilesCount]();
906 }
907 sFoundProfiles[nProfilesCount-1] = new string(dirdata.cFileName);
908 if ( sFoundProfilePlayerNames )
909 { // mimicing realloc, this to be changed to vector or something!
910 std::string** sNewFoundProfilePlayerNames = new string*[nProfilesCount]();
911 memcpy(sNewFoundProfilePlayerNames, sFoundProfilePlayerNames, sizeof(string*)*nProfilesCount);
912 delete[] sFoundProfilePlayerNames;
913 sFoundProfilePlayerNames = sNewFoundProfilePlayerNames;
