112 bool pge_network::PgePacket::initPktPgeMsgUserConnected(
113 PgePacket& pkt,
114 const PgeNetworkConnectionHandle& connHandleServerSide,
115 bool bCurrentClient,
116 const std::string& sIpAddress)
117 {
118 if (sIpAddress.length() >= sizeof(PgePacket::getMessageAsUserConnected(pkt).m_szIpAddress))
119 {
120 return false;
121 }
122 pge_network::PgePacket::initPktBasic(pkt, MsgUserConnectedServerSelf::id, connHandleServerSide, pge_network::PgePacket::AutoFill::ZERO);
123
124 MsgUserConnectedServerSelf& msgUserConnected = PgePacket::getMessageAsUserConnected(pkt);
125 msgUserConnected.m_bCurrentClient = bCurrentClient;
126
127 /*
128 I explicitly zero the whole pkt area above because I want clean pkt.
129 However, the debug versions of some security-enhanced CRT functions fill the buffer passed to them with
130 a special character (0xFE):
131 https://learn.microsoft.com/en-us/cpp/c-runtime-library/reference/crtsetdebugfillthreshold?view=msvc-170
132
133 This can turn it off: _CrtSetDebugFillThreshold( 0 );
134 And I'm not turning this back on because I don't want to use this debug feature.
135 */
136 _CrtSetDebugFillThreshold(0);
137 return (strncpy_s(
138 msgUserConnected.m_szIpAddress,
139 sizeof(msgUserConnected.m_szIpAddress),
140 sIpAddress.c_str(),
141 sIpAddress.length()) == 0);
142 }

214 bool PgePacket::addPktMsgApp(
215 pge_network::PgePacket& pkt,
216 const pge_network::MsgApp& msgApp)
217 {
218 if (pge_network::PgePacket::getPacketId(pkt) != MsgApp::id)
219 {
220 // this packet is initialized for non-MsgApp messages
221 return false;
222 }
223
224 // we receive msgApp structure from outside, dont take the validity of its content granted!
225 if (MsgApp::getMsgAppDataActualSizeBytes(msgApp) > MsgApp::nMaxMessageLengthBytes)
226 {
227 // given nMsgAppDataSize indicates too big app msg data
228 return false;
229 }
230
231 /* use uint32_t instead of uint8_t or MsgApp::TMsgSize here to avoid numeric overflow */
232 const uint32_t nActualTotalAppMsgSize = offsetof(pge_network::MsgApp, m_cMsgData) + MsgApp::getMsgAppDataActualSizeBytes(msgApp);
233 assert(nActualTotalAppMsgSize <= std::numeric_limits<MsgApp::TMsgSize>::max());
234 if (nActualTotalAppMsgSize > std::numeric_limits<MsgApp::TMsgSize>::max())
235 {
236 // given nMsgAppDataSize indicates too big app msg data
237 return false;
238 }
239
240 if (pge_network::PgePacket::getMessageAppArea(pkt).m_nActualMessagesAreaLength + nActualTotalAppMsgSize > pge_network::MsgAppArea::nMaxMessagesAreaLengthBytes)
241 {
242 // not enough space in this packet to store the given app msg
243 return false;
244 }
245
246 if (pge_network::PgePacket::getMessageAppArea(pkt).m_nMessageCount == std::numeric_limits<uint8_t>::max())
247 {
248 // already reached max app msg count
249 return false;
250 }
251
252 // TODO: find location for next app msg (separate function)
253 // TODO: store the given app msg at the proper location
254
255 memcpy(pge_network::PgePacket::getMessageAppArea(pkt).m_cData, &msgApp, nActualTotalAppMsgSize);
256 pge_network::PgePacket::getMessageAppArea(pkt).m_nMessageCount++;
257 pge_network::PgePacket::getMessageAppArea(pkt).m_nActualMessagesAreaLength += static_cast<MsgAppArea::TAreaLength>(nActualTotalAppMsgSize);
258
259 return true;
260 }

349 bool MsgApp::fillMsgApp(
350 pge_network::MsgApp& myAppMsg,
351 const pge_network::MsgApp::TMsgId& msgAppMsgId,
352 const TByte* msgAppData,
353 MsgApp::TMsgSize nMsgAppDataSize)
354 {

356 {
357 // given data cannot fit into target struct

360
361 pge_network::MsgApp::getMsgAppMsgId(myAppMsg) = msgAppMsgId;
362 memcpy_s(
363 pge_network::MsgApp::getMsgAppData(myAppMsg),
364 pge_network::MsgApp::nMaxMessageLengthBytes,
365 static_cast<const void*>(msgAppData),
366 nMsgAppDataSize);
367 pge_network::MsgApp::getMsgAppDataActualSizeBytes(myAppMsg) = nMsgAppDataSize;
368
369 return true;
