44bool PgeGnsWrapper::init()
45{
46 if (isInitialized())
47 {
48 CConsole::getConsoleInstance("PgeGnsWrapper").EOLn("ERROR: GameNetworkingSockets is already initialized!");
49 return false;
50 }
51
52 SteamDatagramErrMsg errMsg;
53
54 if (GameNetworkingSockets_Init(nullptr, errMsg))
55 {
56 CConsole::getConsoleInstance("PgeGnsWrapper").SOLn("Initialized GameNetworkingSockets %s!", GAMENETWORKINGSOCKETS_VER_STR);
57 }
58 else
59 {
60 CConsole::getConsoleInstance("PgeGnsWrapper").EOLn("Failed to initialize GameNetworkingSockets %s: %s!", GAMENETWORKINGSOCKETS_VER_STR, errMsg);
61 return false;
62 }
63
64 SteamNetworkingUtils()->SetDebugOutputFunction(k_ESteamNetworkingSocketsDebugOutputType_Msg, NetworkDbg);
65
66 // Select instance to use. For now we'll always use the default.
67 // But we could use SteamGameServerNetworkingSockets() on Steam.
68 m_pInterface = SteamNetworkingSockets();
69
70 // now main engine loop can invoke pollIncomingMessages() and pollConnectionStateChanges()
71
72 return true;
73} // init()

132bool PgeGnsWrapper::pollIncomingMessages()
133{
134 static const int nIncomingMsgArraySize = 10;
135 ISteamNetworkingMessage* pIncomingGnsMsg[nIncomingMsgArraySize];
136
137 if (!isInitialized())
138 {
139 CConsole::getConsoleInstance("PgeGnsWrapper").EOLn("%s not initialized!", __func__);
140 return false;
141 }
142
143 // numGnsMsgs is actually the number of received PgePackets, however GNS talks about SteamNetworkingMessage,
144 // but PGE puts a PgePacket into such message, and a PgePacket can contain multiple MsgApps, so pls don't
145 // mix the MsgApps with SteamNetworkingMessages, they are not the same messages.
146 const int numGnsMsgs = receiveMessages(pIncomingGnsMsg, nIncomingMsgArraySize);
147 if (numGnsMsgs == 0)
148 {
149 return false;
150 }
151
152 if (numGnsMsgs < 0)
153 {
154 // This case falling here would be normal when GNS instance is not connected, actually sometimes
155 // we do disconnect on purpose in different cases other than exiting the application.
156 // However, we still log this as error and expect the caller NOT invoke us when they know we are not connected.
157 CConsole::getConsoleInstance("PgeGnsWrapper").EOLn("%s: Error checking for messages!", __func__);
158 return false;
159 }
160
161 if (!pIncomingGnsMsg)
162 {
163 CConsole::getConsoleInstance("PgeGnsWrapper").EOLn("%s: pIncomingGnsMsg null!", __func__);
164 return false;
165 }
166
167 for (int i = 0; i < numGnsMsgs; i++)
168 {
169 if (!validateSteamNetworkingMessage(pIncomingGnsMsg[i]->m_conn))
170 {
171 continue;
172 }
173
174 // PgePacket is a fixed-size memory area, during sending we truncate it to actual size, which is this size:
175 const int nActualPktSize = (pIncomingGnsMsg[i])->m_cbSize;
176 pge_network::PgePacket pkt;
177 assert(nActualPktSize <= sizeof(pkt));
178 assert(nActualPktSize > 0);
179
180 memcpy(&pkt, (pIncomingGnsMsg[i])->m_pData, nActualPktSize);
181 updateIncomingPgePacket(pkt, pIncomingGnsMsg[i]->m_conn);
182
183 // We don't need this anymore.
184 // Note that we could even push pIncomingGnsMsg to a queue, and process it later, and
185 // release it later, that could be a good speed optimization.
186 (pIncomingGnsMsg[i])->Release();
187
188 const pge_network::PgePacket& pktAsConst = pkt; // from now on we use this const version
189
190 if (pge_network::PgePacket::getPacketId(pktAsConst) == pge_network::PgePktId::Application)
191 {
192 const uint8_t nMessageCount = pge_network::PgePacket::getMessageAppCount(pktAsConst);
193 if (nMessageCount == 0)
194 {
195 CConsole::getConsoleInstance("PgeGnsWrapper").EOLn("%s: app message pkt with msg count 0 from connection %u!",
196 __func__, pge_network::PgePacket::getServerSideConnectionHandle(pktAsConst));
197 assert(false);
198 continue;
199 }
200
201 // for now we support only 1 app msg / pkt
202 assert(nMessageCount == 1);
203
204 const pge_network::MsgApp* pMsgApp = pge_network::PgePacket::getMsgAppFromPkt(pktAsConst);
205 assert(pMsgApp); // never null since it points into pkt
206
207 for (uint8_t iAppMsg = 0; iAppMsg < nMessageCount; iAppMsg++)
208 {
209 // TODO: nooo I need proper iteration for every msgId!
210 const pge_network::MsgApp::TMsgId& msgAppId = pge_network::MsgApp::getMsgAppMsgId(*pMsgApp);
211 if (m_allowListedAppMessages.end() == m_allowListedAppMessages.find(msgAppId))
212 {
213 CConsole::getConsoleInstance("PgeGnsWrapper").EOLn("%s: non-allowlisted app message received: %u from connection %u!",
214 __func__, msgAppId, pge_network::PgePacket::getServerSideConnectionHandle(pktAsConst));
215 assert(false);
216 continue;
217 }
218 ++m_nRxMsgCount[msgAppId];
219
220 // we could also check if nMsgSize is non-zero, however we shouldnt: app is allowed to define zero-size AppMsg, it is
221 // not our business here to judge.
222 pMsgApp += pge_network::MsgApp::getMsgAppTotalActualSizeBytes(*pMsgApp);
223 }
224 }
225 else
226 {
227 if (m_allowListedPgeMessages.end() == m_allowListedPgeMessages.find(pge_network::PgePacket::getPacketId(pktAsConst)))
228 {
229 CConsole::getConsoleInstance("PgeGnsWrapper").EOLn("%s: non-allowlisted pge message received: %u from connection %u!",
230 __func__, pge_network::PgePacket::getPacketId(pktAsConst), pge_network::PgePacket::getServerSideConnectionHandle(pktAsConst));
231 assert(false);
232 continue;
233 }
234
235 if (pgeMessageIsHandledAtGnsLevel(pktAsConst))
236 {
237 CConsole::getConsoleInstance("PgeGnsWrapper").OLn("%s: PGE message %u handled at GNS level without reaching PGE level, from connection %u!",
238 __func__, pge_network::PgePacket::getPacketId(pktAsConst), pge_network::PgePacket::getServerSideConnectionHandle(pktAsConst));
239 continue;
240 }
241 }
242
243 // TODO: m_nRxByteCount should be incremented earlier since there are cases in the loop where we continue to next pkt, or we should
244 // update the definition of m_nRxByteCount to counting only those pkts which are actually pushed into PGE queue!
245 m_nRxByteCount += nActualPktSize;
246 // TODO: would be nice if even the size of pushed packets would be the actual to copy less memory
247 // or at least use emplace_back()
248 m_queuePackets.push_back(pktAsConst);
249 } // for i
250
251 if (m_nRxPktCount == 0)
252 {
253 m_time1stRxPkt = std::chrono::steady_clock::now();
254 }
255 m_nRxPktCount += numGnsMsgs;
256
257 return true;
258}

445void PgeGnsWrapper::logDetailedConnectionStatus(const HSteamNetConnection& connHandle) const
446{
447 CConsole::getConsoleInstance("PgeGnsWrapper").OLn("Detailed Connection Status for connHandle %u: ", connHandle);
448
449 if (connHandle == k_HSteamNetConnection_Invalid)
450 {
451 CConsole::getConsoleInstance("PgeGnsWrapper").OLn(" Returning immediately, this is invalid handle (for server it is itself)!");
452 return;
453 }
454
455 // we need this line-by-line logging otherwise no paragraph breaks will be inserted into the html at each newline;
456 // note that this could be a feature of the logger lib in the future.
457 std::stringstream ssDetailedConnStatus(getDetailedConnectionStatus(connHandle));
458 while (ssDetailedConnStatus)
459 {
460 std::string sLine;
461 std::getline(ssDetailedConnStatus, sLine);
462 CConsole::getConsoleInstance("PgeGnsWrapper").OLn("%s", sLine.c_str());
463 }
464 CConsole::getConsoleInstance("PgeGnsWrapper").OLn("");
465}
