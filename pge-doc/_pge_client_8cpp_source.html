275void PgeClientImpl::WriteList() const
276{
277 getConsole().OLnOI("PgeClient::WriteList() start");
278 if (isInitialized())
279 {
280 getConsole().OLn("Role: Client");
281 if (getConnectionHandle() == pge_network::ServerConnHandle)
282 {
283 getConsole().EOLn("Connection Handle: Invalid");
284 }
285 else
286 {
287 getConsole().OLn("Connection Handle: %u", static_cast<uint32_t>(getConnectionHandle()));
288 getConsole().OLn("Server Address: %s", getServerAddress());
289 }
290 }
291 else
292 {
293 getConsole().OLn("PgeClient is NOT initialized!");
294 }
295
296 getConsole().OOOLn("PgeClient::WriteList() end");
297} // WriteList()
