45bool PgeGsnWrapper::init()
46{
47 if (isInitialized())
48 {
49 CConsole::getConsoleInstance("PgeGsnWrapper").EOLn("ERROR: GameNetworkingSockets is already initialized!");
50 return false;
51 }
52
53 SteamDatagramErrMsg errMsg;
54
55 if (GameNetworkingSockets_Init(nullptr, errMsg))
56 {
57 CConsole::getConsoleInstance("PgeGsnWrapper").SOLn("Initialized GameNetworkingSockets %s!", GAMENETWORKINGSOCKETS_VER_STR);
58 }
59 else
60 {
61 CConsole::getConsoleInstance("PgeGsnWrapper").EOLn("Failed to initialize GameNetworkingSockets %s: %s!", GAMENETWORKINGSOCKETS_VER_STR, errMsg);
62 return false;
63 }
64
65 SteamNetworkingUtils()->SetDebugOutputFunction(k_ESteamNetworkingSocketsDebugOutputType_Msg, NetworkDbg);
66
67 // Select instance to use. For now we'll always use the default.
68 // But we could use SteamGameServerNetworkingSockets() on Steam.
69 m_pInterface = SteamNetworkingSockets();
70
71 // now main engine loop can invoke pollIncomingMessages() and pollConnectionStateChanges()
72
73 return true;
74} // init()

133bool PgeGsnWrapper::pollIncomingMessages()
134{
135 static const int nIncomingMsgArraySize = 10; // TODO: make this configurable from outside
136 ISteamNetworkingMessage* pIncomingMsg[nIncomingMsgArraySize];
137
138 if (!isInitialized())
139 {
140 CConsole::getConsoleInstance("PgeGsnWrapper").EOLn("%s not initialized!", __func__);
141 return false;
142 }
143
144 const int numMsgs = receiveMessages(pIncomingMsg, nIncomingMsgArraySize);
145 if (numMsgs == 0)
146 {
147 return false;
148 }
149
150 if (numMsgs < 0)
151 {
152 CConsole::getConsoleInstance("PgeGsnWrapper").EOLn("%s: Error checking for messages!", __func__);
153 return false;
154 }
155
156 if (!pIncomingMsg)
157 {
158 CConsole::getConsoleInstance("PgeGsnWrapper").EOLn("%s: pIncomingMsg null!", __func__);
159 return false;
160 }
161
162 for (int i = 0; i < numMsgs; i++)
163 {
164 if (!validateSteamNetworkingMessage(pIncomingMsg[i]->m_conn))
165 {
166 continue;
167 }
168
169 pge_network::PgePacket pkt;
170 assert((pIncomingMsg[i])->m_cbSize == sizeof(pkt));
171 memcpy(&pkt, (pIncomingMsg[i])->m_pData, (pIncomingMsg[i])->m_cbSize);
172 updateIncomingPgePacket(pkt, pIncomingMsg[i]->m_conn);
173
174 // We don't need this anymore.
175 // Note that we could even push pIncomingMsg to a queue, and process it later, and
176 // release it later, that could be a good speed optimization.
177 (pIncomingMsg[i])->Release();
178
179 if (pkt.pktId == pge_network::PgePktId::APP)
180 {
181 if (m_allowListedAppMessages.end() == m_allowListedAppMessages.find(pkt.msg.app.msgId))
182 {
183 CConsole::getConsoleInstance("PgeGsnWrapper").EOLn("%s: non-allowlisted app message received: %u from connection %u!",
184 __func__, pkt.msg.app.msgId, pkt.m_connHandleServerSide);
185 assert(false);
186 continue;
187 }
188 ++m_nRxMsgCount[pkt.msg.app.msgId];
189 }
190 else
191 {
192 if (m_allowListedPgeMessages.end() == m_allowListedPgeMessages.find(pkt.pktId))
193 {
194 CConsole::getConsoleInstance("PgeGsnWrapper").EOLn("%s: non-allowlisted pge message received: %u from connection %u!",
195 __func__, pkt.pktId, pkt.m_connHandleServerSide);
196 assert(false);
197 continue;
198 }
199 }
200
201 m_nRxByteCount += sizeof(pkt);
202 m_queuePackets.push_back(pkt);
203 } // for i
204
205 if (m_nRxPktCount == 0)
206 {
207 m_time1stRxPkt = std::chrono::steady_clock::now();
208 }
209 m_nRxPktCount += numMsgs;
210
211 return true;
212}

260uint32_t PgeGsnWrapper::getRxPacketPerSecondCount() const
261{
262 const auto nSecsSince1stRxPkt =
263 std::chrono::duration_cast<std::chrono::seconds>(std::chrono::steady_clock::now() - m_time1stRxPkt).count();
264
265 return static_cast<uint32_t>(nSecsSince1stRxPkt != 0 ? (getRxPacketCount() / nSecsSince1stRxPkt) : 0);
266}
