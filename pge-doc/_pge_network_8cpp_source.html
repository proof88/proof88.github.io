81bool PgeNetworkImpl::initialize()
82{
83 if (m_cfgProfiles.getVars()[CVAR_NET_SERVER].getAsString().empty())
84 {
85 m_bServer = (IDYES == MessageBox(0, "Want to be a Server?", ":)", MB_YESNO | MB_ICONQUESTION | MB_SETFOREGROUND));
86 }
87 else
88 {
89 m_bServer = m_cfgProfiles.getVars()[CVAR_NET_SERVER].getAsBool();
90 CConsole::getConsoleInstance("PgeGnsWrapper").OLn("s_bServer from config: %b", m_bServer);
91 }
92
93 if (m_bServer)
94 {
95 m_pServerClient = &m_pServer;
96 }
97 else
98 {
99 m_pServerClient = &m_pClient;
100 }
101
102 if (!m_pServerClient->initialize())
103 {
104 getConsole().EOLn("ServerClient has FAILED to initialize!");
105 return false;
106 }
107
108 return true;
109} // initialize()

189void PgeNetworkImpl::Update()
