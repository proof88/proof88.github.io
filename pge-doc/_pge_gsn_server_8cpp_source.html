54bool PgeGsnServer::startListening()
55{
56 if (isInitialized())
57 {
58 CConsole::getConsoleInstance("PgeGsnServer").OLn("%s() ERROR: Server is already listening on port %d", __func__, m_nPort);
59 return false;
60 }
61
62 SteamNetworkingIPAddr serverLocalAddr;
63 serverLocalAddr.Clear();
64 serverLocalAddr.m_port = m_nPort;
65 SteamNetworkingConfigValue_t opt;
66 opt.SetPtr(k_ESteamNetworkingConfig_Callback_ConnectionStatusChanged, (void*)steamNetConnectionStatusChangedCallback);
67
68 // GameNetworkingSockets - 1.4.0\tests\test_connection.cpp :: Test_Connection() contains example how to initiate connection to ourselves!
69 m_hListenSock = m_pInterface->CreateListenSocketIP(serverLocalAddr, 1, &opt);
70 if (m_hListenSock == k_HSteamListenSocket_Invalid)
71 {
72 CConsole::getConsoleInstance("PgeGsnServer").EOLn("%s() Failed to listen on port %d!", __func__, m_nPort);
73 return false;
74 }
75
76 m_hPollGroup = m_pInterface->CreatePollGroup();
77 if (m_hPollGroup == k_HSteamNetPollGroup_Invalid)
78 {
79 CConsole::getConsoleInstance("PgeGsnServer").EOLn("%s() Failed to create poll group!", __func__);
80 destroy();
81 return false;
82 }
83 CConsole::getConsoleInstance("PgeGsnServer").OLn("%s() Server listening on port %d", __func__, m_nPort);
84
85 // Add ourselves to the client list, using std::map wacky syntax
86 // k_HSteamNetConnection_Invalid will mean the server itself
87 m_mapClients[k_HSteamNetConnection_Invalid];
88 memset(m_mapClients[k_HSteamNetConnection_Invalid].m_szAddr, sizeof(m_mapClients[k_HSteamNetConnection_Invalid].m_szAddr), 0);
89
90 // here we create a client connect pkt that will be injected to our queue so app level will process it and create
91 // player object or whatever they want for the server itself, as it was a real client
92 pge_network::PgePacket pkt;
93 pge_network::initPktPgeMsgUserConnected(
94 pkt,
95 static_cast<pge_network::PgeNetworkConnectionHandle>(k_HSteamNetConnection_Invalid),
96 true,
97 "" /* we dont know our own IP address, later the 1st connecting client will tell us anyway */);
98
99 // we push this packet to our pkt queue, this is how we "send" message to ourselves so server game loop can process it
100 m_queuePackets.push_back(pkt);
101
102 // TODOOO: network layer needs to set user name! SetClientNick(k_HSteamNetConnection_Invalid, pkt.msg.userConnected.sUserName);
103
104 return true;
105}

107bool PgeGsnServer::stopListening()
108{
109 if (!isInitialized())
110 {
111 CConsole::getConsoleInstance("PgeGsnServer").OLn("%s() Server was not listening, no need to stop.", __func__);
112 return true;
113 }
114
115 // Close all the connections
116 CConsole::getConsoleInstance("PgeGsnServer").OLn("Server closing connections...");
117 for (const auto& it : m_mapClients)
118 {
119 // Send them one more goodbye message. Note that we also have the
120 // connection close reason as a place to send final data. However,
121 // that's usually best left for more diagnostic/debug text not actual
122 // protocol strings.
123 //SendStringToClient(it.first, "Server is shutting down. Goodbye.");
124
125 // Close the connection. We use "linger mode" to ask SteamNetworkingSockets
126 // to flush this out and close gracefully.
127 m_pInterface->CloseConnection(it.first, k_ESteamNetConnectionEnd_App_Generic, "PgeGsnServer Server Graceful shutdown", true);
128 }
129 m_mapClients.clear();
130
131 if (m_hListenSock != k_HSteamListenSocket_Invalid)
132 {
133 m_pInterface->CloseListenSocket(m_hListenSock);
134 m_hListenSock = k_HSteamListenSocket_Invalid;
135 }
136
137 if (m_hPollGroup != k_HSteamNetPollGroup_Invalid)
138 {
139 m_pInterface->DestroyPollGroup(m_hPollGroup);
140 m_hPollGroup = k_HSteamNetPollGroup_Invalid;
141 }
142
143 return true;
144}

146void PgeGsnServer::sendToClient(const HSteamNetConnection& conn, const pge_network::PgePacket& pkt)
147{
148 static_assert(k_HSteamNetConnection_Invalid == 0U, "on upper layers we use connHandle 0 to identify server, so here k_HSteamNetConnection_Invalid must be 0");
149 if (conn == k_HSteamNetConnection_Invalid)
150 {
151 // silent ignore, in the future maybe we will simply inject a message to ourselves' message queue
152 // that is how server can handle itself as a client similar to any other client
153 return;
154 }
155 m_pInterface->SendMessageToConnection(conn, &pkt, (uint32)sizeof(pkt), k_nSteamNetworkingSend_Reliable, nullptr);
156 if (m_nTxPktCount == 1)
157 {
158 m_time1stTxPkt = std::chrono::steady_clock::now();
159 }
160 m_nTxPktCount++;
161 if (pkt.pktId == pge_network::PgePktId::APP)
162 {
163 ++m_nTxMsgCount[pkt.msg.app.msgId];
164 }
165 m_nTxByteCount += sizeof(pkt);
166}

168void PgeGsnServer::sendToAllClientsExcept(const pge_network::PgePacket& pkt, const HSteamNetConnection& except)
169{
170 static_assert(k_HSteamNetConnection_Invalid == 0U, "on upper layers we use connHandle 0 to identify server, so here k_HSteamNetConnection_Invalid must be 0");
171 for (auto& client : m_mapClients)
172 {
173 if (client.first == k_HSteamNetConnection_Invalid)
174 {
175 // silent ignore, in the future maybe we will simply inject a message to ourselves' message queue
176 // that is how server can handle itself as a client similar to any other client
177 continue;
178 }
179 if (client.first != except)
180 {
181 sendToClient(client.first, pkt);
182 }
183 }

270void PgeGsnServer::onSteamNetConnectionStatusChanged(SteamNetConnectionStatusChangedCallback_t* pInfo)
271{
272 // This function is also invoked on main thread when I call PgeGsnServer.PollConnectionStateChanges() from PGE::runGame()
273 // so no need to utilize mutexes around here.
274 // And the other function PollIncomingMessages() is also invoked by PGE::runGame().
275 // So it is safe to do operations on m_queuePackets.
276 char szTemp[1024];
277
278 // What's the state of the connection?
279 switch (pInfo->m_info.m_eState)
280 {
281 case k_ESteamNetworkingConnectionState_None:
282 // NOTE: We will get callbacks here when we destroy connections. You can ignore these.
283 break;
284
285 case k_ESteamNetworkingConnectionState_ClosedByPeer:
286 case k_ESteamNetworkingConnectionState_ProblemDetectedLocally:
287 {
288 // Ignore if they were not previously connected. (If they disconnected
289 // before we accepted the connection.)
290 if (pInfo->m_eOldState == k_ESteamNetworkingConnectionState_Connected)
291 {
292 // Locate the client. Note that it should have been found, because this
293 // is the only codepath where we remove clients (except on shutdown),
294 // and connection change callbacks are dispatched in queue order.
295 auto itClient = m_mapClients.find(pInfo->m_hConn);
296 assert(itClient != m_mapClients.end());
297
298 // Select appropriate log messages
299 const char* pszDebugLogAction;
300 if (pInfo->m_info.m_eState == k_ESteamNetworkingConnectionState_ProblemDetectedLocally)
301 {
302 pszDebugLogAction = "problem detected locally";
303 sprintf(szTemp, "%s disappeared (%s)", itClient->second.m_sCustomName.c_str(), pInfo->m_info.m_szEndDebug);
304 }
305 else
306 {
307 // Note that here we could check the reason code to see if
308 // it was a "usual" connection or an "unusual" one.
309 pszDebugLogAction = "closed by peer";
310 sprintf(szTemp, "%s closed connection", itClient->second.m_sCustomName.c_str());
311 }
312
313 // Spew something to our own log. Note that because we put their nick
314 // as the connection description, it will show up, along with their
315 // transport-specific data (e.g. their IP address)
316 CConsole::getConsoleInstance("PgeGsnServer").OLn("%s: SERVER Connection %s %s, reason %d: %s\n",
317 __func__,
318 pInfo->m_info.m_szConnectionDescription,
319 pszDebugLogAction,
320 pInfo->m_info.m_eEndReason,
321 pInfo->m_info.m_szEndDebug
322 );
323
324 pge_network::PgePacket pkt;
325 memset(&pkt, 0, sizeof(pkt));
326 pkt.pktId = pge_network::MsgUserDisconnected::id;
327 pkt.m_connHandleServerSide = static_cast<pge_network::PgeNetworkConnectionHandle>(pInfo->m_hConn);
328
329 m_mapClients.erase(itClient); // dont try to send anything to the disconnected client :)
330
331 // we push this packet to our pkt queue, this is how we "send" message to ourselves so server game loop can process it
332 m_queuePackets.push_back(pkt);
333 sendToAllClientsExcept(pkt);
334 }
335 else
336 {
337 assert(pInfo->m_eOldState == k_ESteamNetworkingConnectionState_Connecting);
338 }
339
340 // Clean up the connection. This is important!
341 // The connection is "closed" in the network sense, but
342 // it has not been destroyed. We must close it on our end, too
343 // to finish up. The reason information do not matter in this case,
344 // and we cannot linger because it's already closed on the other end,
345 // so we just pass 0's.
346 m_pInterface->CloseConnection(pInfo->m_hConn, 0, nullptr, false);
347 break;
348 }
349
350 case k_ESteamNetworkingConnectionState_Connecting:
351 {
352 // This must be a new connection
353 assert(m_mapClients.find(pInfo->m_hConn) == m_mapClients.end());
354
355 CConsole::getConsoleInstance("PgeGsnServer").OLn("%s: SERVER Connection request from %s", __func__, pInfo->m_info.m_szConnectionDescription);
356
357 // A client is attempting to connect
358 // Try to accept the connection.
359 if (m_pInterface->AcceptConnection(pInfo->m_hConn) != k_EResultOK)
360 {
361 // This could fail. If the remote host tried to connect, but then
362 // disconnected, the connection may already be half closed. Just
363 // destroy whatever we have on our side.
364 m_pInterface->CloseConnection(pInfo->m_hConn, 0, nullptr, false);
365 CConsole::getConsoleInstance("PgeGsnServer").EOLn("%s: SERVER Can't accept connection. (It was already closed?)", __func__);
366 break;
367 }
368
369 // Assign the poll group
370 if (!m_pInterface->SetConnectionPollGroup(pInfo->m_hConn, m_hPollGroup))
371 {
372 m_pInterface->CloseConnection(pInfo->m_hConn, 0, nullptr, false);
373 CConsole::getConsoleInstance("PgeGsnServer").EOLn("%s: SERVER Failed to set poll group!", __func__);
374 break;
375 }
376
377 // Add them to the client list, using std::map wacky syntax
378 m_mapClients[pInfo->m_hConn];
379 pInfo->m_info.m_addrRemote.ToString(m_mapClients[pInfo->m_hConn].m_szAddr, sizeof(m_mapClients[pInfo->m_hConn].m_szAddr), true);
380 CConsole::getConsoleInstance("PgeGsnServer").OLn("%s: SERVER A client is connecting from %s ...", __func__, m_mapClients[pInfo->m_hConn].m_szAddr);
381
382 pge_network::PgePacket pkt;
383 pge_network::initPktPgeMsgUserConnected(
384 pkt,
385 static_cast<pge_network::PgeNetworkConnectionHandle>(pInfo->m_hConn),
386 false,
387 m_mapClients[pInfo->m_hConn].m_szAddr);
388
389 // we push this packet to our pkt queue, this is how we "send" message to ourselves so server game loop can process it
390 m_queuePackets.push_back(pkt);
391
392 // TODOOO: network layer needs to set user name! SetClientNick(pInfo->m_hConn, pkt.msg.userConnected.sUserName);
393
394 break;
395 }
396
397 case k_ESteamNetworkingConnectionState_Connected:
398 {
399 // We will get a callback immediately after accepting the connection.
400 // Since we are the server, we can ignore this, it's not news to us.
401
402 const auto itClient = m_mapClients.find(pInfo->m_hConn);
403 if (itClient == m_mapClients.end())
404 {
405 CConsole::getConsoleInstance("PgeGsnServer").EOLn("%s: SERVER Cannot happen: a client (%u) has reached state Connected but not present in clients map!",
406 __func__, pInfo->m_hConn);
407 assert(false);
408 }
409 else
410 {
411 CConsole::getConsoleInstance("PgeGsnServer").OLn("%s: SERVER Client with connHandle %u has reached state Connected!", __func__, pInfo->m_hConn);
412 }
413 break;
414 }
415
416 default:
417 // Silences -Wswitch
418 break;
419 }
420}
