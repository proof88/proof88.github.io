58bool PgeGnsClient::connectToServer(
59 const std::string& sServerAddress,
60 const std::string& sAppVersion)
61{
62 if (isConnected())
63 {
64 CConsole::getConsoleInstance("PgeGnsClient").EOLn("%s ERROR: already connected to %s!", __func__, m_szAddr);
65 return false;
66 }
67
68 if (!isInitialized())
69 {
70 CConsole::getConsoleInstance("PgeGnsClient").EOLn("%s() ERROR: not initialized!", __func__);
71 return true;
72 }
73
74 // currently there is no separate initialization logic here, only connecting to server, however
75 // we still have a logical separation of isInitialized() and isConnected()
76
77 m_addrServer.Clear();
78 if (!m_addrServer.ParseString(sServerAddress.c_str()))
79 {
80 CConsole::getConsoleInstance("PgeGnsClient").EOLn("%s() Failed to parse address: %s!", __func__, sServerAddress.c_str());
81 return false;
82 }
83
84 if (m_addrServer.m_port == 0)
85 {
86 m_addrServer.m_port = DEFAULT_SERVER_PORT;
87 CConsole::getConsoleInstance("PgeGnsClient").OLn("%s() Using default port: %d", __func__, m_addrServer.m_port);
88 }
89 else
90 {
91 CConsole::getConsoleInstance("PgeGnsClient").OLn("%s() Using custom port: %d", __func__, m_addrServer.m_port);
92 }
93
94 // Parse back to string just to make sure we are logging and connecting to proper address
95 m_addrServer.ToString(m_szAddr, sizeof(m_szAddr), true);
96 CConsole::getConsoleInstance("PgeGnsClient").OLn("%s Connecting to server at %s", __func__, m_szAddr);
97
98 m_sAppVersion = sAppVersion;
99 if (m_sAppVersion.empty())
100 {
101 CConsole::getConsoleInstance("PgeGnsClient").OLn("%s() no client app version is specified!", __func__);
102 }
103 else
104 {
105 CConsole::getConsoleInstance("PgeGnsClient").OLn("%s() client app version is specified as: %s!", __func__, m_sAppVersion.c_str());
106 }
107
108 SteamNetworkingConfigValue_t opt;
109 opt.SetPtr(k_ESteamNetworkingConfig_Callback_ConnectionStatusChanged, (void*)steamNetConnectionStatusChangedCallback);
110
111 // From now steamNetConnectionStatusChangedCallback() will be called back for connection status updates and that is where m_sAppVersion is also sent to server
112 m_hConnection = m_pInterface->ConnectByIPAddress(m_addrServer, 1, &opt);
113 if (m_hConnection == k_HSteamNetConnection_Invalid)
114 {
115 CConsole::getConsoleInstance("PgeGnsClient").EOLn("Client Failed to create connection");
116 return false;
117 }
118
119 //SteamNetConnectionInfo_t connInfo;
120 //m_pInterface->GetConnectionInfo(m_hConnection, &connInfo);
121
122 return true;
123}

125bool PgeGnsClient::disconnectClient(const std::string& sExtraDebugText)
126{
127 if (!isConnected())
128 {
129 CConsole::getConsoleInstance("PgeGnsClient").OLn("%s not connected.", __func__);
130 return true;
131 }
132
133 const std::string sExtraDebugTextToSend =
134 sExtraDebugText.empty() ?
135 "PgeGnsClient Client Graceful disconnectClient()" :
136 "PgeGnsClient Client Graceful disconnectClient(), reason: " + sExtraDebugText;
137
138 CConsole::getConsoleInstance("PgeGnsClient").OLn(
139 "Client closing connection ... Reason: %s",
140 sExtraDebugText.empty() ? "" : sExtraDebugText.c_str());
141
142 // here we create a server disconnect pkt that will be injected to our queue so app level can process it
143 // for the server itself, as it was a real server disconnecting
144 pge_network::PgePacket pkt;
145 pge_network::PgePacket::initPktPgeMsgUserDisconnected(
146 pkt,
147 static_cast<pge_network::PgeNetworkConnectionHandle>(k_HSteamNetConnection_Invalid));
148 // we push this packet to our pkt queue, this is how we "send" message to ourselves so client game loop can process it
149 m_queuePackets.push_back(pkt);
150
151 // we cannot inject similar pkt about ourselves because we dont have our server-side connection handle (that would be stored in
152 // m_hConnectionServerSide - somehow I never implemented it)
153
154 m_pInterface->CloseConnection(m_hConnection, k_ESteamNetConnectionEnd_App_Generic, sExtraDebugTextToSend.c_str(), true);
155 m_hConnection = k_HSteamNetConnection_Invalid;
156 m_hConnectionServerSide = k_HSteamNetConnection_Invalid;
157 m_addrServer.Clear();
158 memset(m_szAddr, sizeof(m_szAddr), 0);
159 memset(&m_connRtStatus, 0, sizeof(m_connRtStatus));
160
161 return true;
162}

184void PgeGnsClient::sendToServer(const pge_network::PgePacket& pkt)
185{
186 if (!isConnected())
187 {
188 CConsole::getConsoleInstance("PgeGnsClient").EOLn("%s not connected!", __func__);
189 return;
190 }
191
192 uint32_t nActualPktSize;
193 if (pge_network::PgePacket::getPacketId(pkt) == pge_network::PgePktId::Application)
194 {
195 // We need the real used memory size so we can truncate the sent pkt to that.
196 // We need to consider member padding and the actual message sizes to have a correct value.
197 // 'cData' is our point of view since from there we need to calculate.
198 const pge_network::MsgApp* pMsgApp = pge_network::PgePacket::getMsgAppFromPkt(pkt);
199 assert(pMsgApp); // never null since it points into pkt
200 const pge_network::TByte* pMsgAppInByteSteps = reinterpret_cast<const pge_network::TByte*>(pMsgApp); // we can step this ptr in Bytes
201
202 const uint8_t nMessageCount = pge_network::PgePacket::getMessageAppCount(pkt);
203 assert(nMessageCount == 1); // for now only 1 msg/pkt
204 for (uint8_t i = 0; i < nMessageCount; i++)
205 {
206 pMsgApp = reinterpret_cast<const pge_network::MsgApp*>(pMsgAppInByteSteps);
207 // moving pMsgAppInByteSteps by the _actual_ size of the current MsgApp struct (considering the actual app message size there)
208 pMsgAppInByteSteps += pge_network::MsgApp::getMsgAppTotalActualSizeBytes(*pMsgApp);
209 }
210 // now pMsgAppInByteSteps points to RIGHT AFTER the 1st byte of the last application message
211 const pge_network::TByte* pPkt = (const pge_network::TByte*)(&pkt);
212 nActualPktSize = pMsgAppInByteSteps - pPkt;
213 }
214 else
215 {
216 nActualPktSize = static_cast<uint32_t>(sizeof(pkt));
217 }
218
219 m_pInterface->SendMessageToConnection(m_hConnection, &pkt, nActualPktSize, k_nSteamNetworkingSend_Reliable, nullptr);
220 if (m_nTxPktCount == 1)
221 {
222 m_time1stTxPkt = std::chrono::steady_clock::now();
223 }
224 m_nTxPktCount++;
225 if (pge_network::PgePacket::getPacketId(pkt) == pge_network::PgePktId::Application)
226 {
227 const pge_network::MsgApp* pMsgApp = pge_network::PgePacket::getMsgAppFromPkt(pkt);
228 const uint8_t nMessageCount = pge_network::PgePacket::getMessageAppCount(pkt);
229 for (uint8_t i = 0; i < nMessageCount; i++)
230 {
231 // TODO: nooo, I need to properly iterate to next appmsg as I iterate in above loop!
232 ++m_nTxMsgCount[pge_network::MsgApp::getMsgAppMsgId(*pMsgApp)];
233 }
234 }
235 m_nTxByteCount += nActualPktSize;
236}

316void PgeGnsClient::onSteamNetConnectionStatusChanged(SteamNetConnectionStatusChangedCallback_t* pInfo)
317{
318 // This function is also invoked on main thread when I call PgeGnsClient.pollConnectionStateChanges() from PGE::runGame()
319 // so no need to utilize mutexes around here.
320 // And the other function pollIncomingMessages() is also invoked by PGE::runGame().
321 // So it is safe to do operations on m_queuePackets.
322
323 assert(pInfo->m_hConn == m_hConnection || m_hConnection == k_HSteamNetConnection_Invalid);
324
325 // What's the state of the connection?
326 switch (pInfo->m_info.m_eState)
327 {
328 case k_ESteamNetworkingConnectionState_None:
329 // NOTE: We will get callbacks here when we destroy connections. You can ignore these.
330 break;
331
332 case k_ESteamNetworkingConnectionState_ClosedByPeer:
333 case k_ESteamNetworkingConnectionState_ProblemDetectedLocally:
334 {
335 // Print an appropriate message
336 if (pInfo->m_eOldState == k_ESteamNetworkingConnectionState_Connecting)
337 {
338 // Note: we could distinguish between a timeout, a rejected connection,
339 // or some other transport problem.
340 CConsole::getConsoleInstance("PgeGnsClient").EOLn("%s: CLIENT problem 1 (rejected I guess) (%s)",
341 __func__, pInfo->m_info.m_szEndDebug);
342 }
343 else if (pInfo->m_info.m_eState == k_ESteamNetworkingConnectionState_ProblemDetectedLocally)
344 {
345 CConsole::getConsoleInstance("PgeGnsClient").EOLn("%s: CLIENT problem 2 (guess our machine has lost network connection) (%s)",
346 __func__, pInfo->m_info.m_szEndDebug);
347 }
348 else
349 {
350 // NOTE: We could check the reason code for a normal disconnection
351 CConsole::getConsoleInstance("PgeGnsClient").EOLn("%s: CLIENT problem 3 (closed by peer) (%s)",
352 __func__, pInfo->m_info.m_szEndDebug);
353 }
354
355 logDetailedConnectionStatus(pInfo->m_hConn);
356
357 // Clean up the connection. This is important!
358 // The connection is "closed" in the network sense, but
359 // it has not been destroyed. We must close it on our end, too
360 // to finish up. The reason information do not matter in this case,
361 // and we cannot linger because it's already closed on the other end,
362 // so we just pass 0's.
363 m_pInterface->CloseConnection(pInfo->m_hConn, 0, nullptr, false);
364 m_hConnection = k_HSteamNetConnection_Invalid;
365
366 // TODO: it would be better to invoke disconnectClient() because if we are here because server disappeared unexpectedly, then
367 // it would be better to call disconnectClient() which also pushes a server disconnect message to application level.
368 // However, when I invoked it, then in the meantime PRooFPS-dd's onGameRunning() loop detects the connection to be lost and
369 // invokes connectToServer() which will eventually update m_hConnection with a new handle, since isConnected() is NOT reliable,
370 // as described here: https://github.com/proof88/PRooFPS-dd/issues/261 .
371 // So to properly solve this issue, https://github.com/proof88/PRooFPS-dd/issues/261 needs to be solved.
372 // And then we can invoke disconnectClient here. Until it is solved, there is no proper way to handle on client-side at application
373 // level when the server makes an ungraceful shutdown (except if we remove the auto reconnect thing from app level).
374 // I'm tracking this bug in this ticket: https://github.com/proof88/PRooFPS-dd/issues/272 .
375 //disconnectClient("Server disappeared");
376 break;
377 }
378
379 case k_ESteamNetworkingConnectionState_Connecting:
380 // We will get this callback when we start connecting.
381 // We can ignore this.
382 CConsole::getConsoleInstance("PgeGnsClient").OLn("%s: CLIENT Connecting ...", __func__);
383 break;
384
385 case k_ESteamNetworkingConnectionState_Connected:
386 {
387 CConsole::getConsoleInstance("PgeGnsClient").OLn("%s: CLIENT Connected to server OK", __func__);
388
389 // KEKEKEKEKE
390 // you jumped here from PgePacket.h MsgUserConnectedServerSelf.
391 char szAddr[SteamNetworkingIPAddr::k_cchMaxString];
392 pInfo->m_info.m_addrRemote.ToString(szAddr, sizeof(szAddr), true);
393 CConsole::getConsoleInstance("PgeGnsClient").OLn("%s: CLIENT Remote address: %s", __func__, szAddr);
394
395 CConsole::getConsoleInstance("PgeGnsClient").OLn("%s: CLIENT conn. description: %s", __func__, pInfo->m_info.m_szConnectionDescription);
396
397 // server accepted our connection, now immediately send our app version, otherwise server app won't start talking to us.
398 pge_network::PgePacket pktMsgClientAppVersion;
399 pge_network::PgePacket::initPktPgeMsgClientAppVersionFromClient(
400 pktMsgClientAppVersion,
401 m_sAppVersion);
402
403 sendToServer(pktMsgClientAppVersion);
404
405 break;
406 }
407 default:
408 // Silences -Wswitch
409 break;
410 }
