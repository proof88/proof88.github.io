54bool PgeGnsServer::startListening(const std::string& sAppVersion)
55{
56 if (isListening())
57 {
58 CConsole::getConsoleInstance("PgeGnsServer").OLn("%s() ERROR: Server is already listening on port %d", __func__, m_nPort);
59 return false;
60 }
61
62 if (!isInitialized())
63 {
64 CConsole::getConsoleInstance("PgeGnsServer").EOLn("%s() ERROR: not initialized!", __func__);
65 return true;
66 }
67
68 m_sAppVersion = sAppVersion;
69 if (m_sAppVersion.empty())
70 {
71 CConsole::getConsoleInstance("PgeGnsServer").OLn("%s() no server app version is specified!", __func__);
72 }
73 else
74 {
75 CConsole::getConsoleInstance("PgeGnsServer").OLn("%s() server app version is specified as: %s!", __func__, m_sAppVersion.c_str());
76 }
77
78 SteamNetworkingIPAddr serverLocalAddr;
79 serverLocalAddr.Clear();
80 serverLocalAddr.m_port = m_nPort;
81 SteamNetworkingConfigValue_t opt;
82 opt.SetPtr(k_ESteamNetworkingConfig_Callback_ConnectionStatusChanged, (void*)steamNetConnectionStatusChangedCallback);
83
84 // GameNetworkingSockets - 1.4.0\tests\test_connection.cpp :: Test_Connection() contains example how to initiate connection to ourselves!
85 m_hListenSock = m_pInterface->CreateListenSocketIP(serverLocalAddr, 1, &opt);
86 if (m_hListenSock == k_HSteamListenSocket_Invalid)
87 {
88 CConsole::getConsoleInstance("PgeGnsServer").EOLn("%s() Failed to listen on port %d!", __func__, m_nPort);
89 return false;
90 }
91
92 m_hPollGroup = m_pInterface->CreatePollGroup();
93 if (m_hPollGroup == k_HSteamNetPollGroup_Invalid)
94 {
95 CConsole::getConsoleInstance("PgeGnsServer").EOLn("%s() Failed to create poll group!", __func__);
96 destroy();
97 return false;
98 }
99 CConsole::getConsoleInstance("PgeGnsServer").OLn("%s() Server listening on port %d", __func__, m_nPort);
100
101 // Add ourselves to the client list, using std::map wacky syntax
102 // k_HSteamNetConnection_Invalid will mean the server itself
103 m_mapClients[k_HSteamNetConnection_Invalid];
104 memset(m_mapClients[k_HSteamNetConnection_Invalid].m_szAddr, sizeof(m_mapClients[k_HSteamNetConnection_Invalid].m_szAddr), 0);
105 memset(&m_mapClients[k_HSteamNetConnection_Invalid].m_connRtStatus, sizeof(m_mapClients[k_HSteamNetConnection_Invalid].m_connRtStatus), 0);
106
107 // here we create a client connect pkt that will be injected to our queue so app level will process it and create
108 // player object or whatever they want for the server itself, as it was a real client
109 pge_network::PgePacket pktUserConnected;
110 pge_network::PgePacket::initPktPgeMsgUserConnected(
111 pktUserConnected,
112 static_cast<pge_network::PgeNetworkConnectionHandle>(k_HSteamNetConnection_Invalid),
113 true /* bCurrentClient */,
114 "" /* we dont know our own IP address, later the 1st connecting client will tell us anyway */);
115
116 // we push this packet to our pkt queue, this is how we "send" message to ourselves so server game loop can process it
117 m_queuePackets.push_back(pktUserConnected);
118
119 // TODOOO: network layer needs to set user name! SetClientNick(k_HSteamNetConnection_Invalid, PgePacket::getMessageAsUserConnected(pkt).sUserName);
120
121 return true;
122}

124bool PgeGnsServer::stopListening(const std::string& sExtraDebugText)
125{
126 if (!isListening())
127 {
128 CConsole::getConsoleInstance("PgeGnsServer").OLn("%s() Server was not listening, no need to stop.", __func__);
129 return true;
130 }
131
132 // Close all the connections
133 CConsole::getConsoleInstance("PgeGnsServer").OLn(
134 "Server closing connections for %u client(s) (including itself) ... Reason: %s",
135 m_mapClients.size(),
136 sExtraDebugText.empty() ? "unspecified" : sExtraDebugText.c_str());
137
138 // I always forget but m_mapClients also contains the server itself with invalid handle, this should be renamed so I never forget again!
139 for (const auto& itClient : m_mapClients)
140 {
141 const HSteamNetConnection& hClientConnection = itClient.first;
142 logDetailedConnectionStatus(hClientConnection);
143
144 // inform all clients about disconnecting this client;
145 // basically this way all clients will know about all clients got disconnected.
146 pge_network::PgePacket pkt;
147 pge_network::PgePacket::initPktPgeMsgUserDisconnected(
148 pkt,
149 static_cast<pge_network::PgeNetworkConnectionHandle>(hClientConnection));
150 sendToAllClientsExcept(pkt, hClientConnection);
151
152 if (hClientConnection == k_HSteamNetConnection_Invalid) {
153 // this is us, we don't have GNS connection to ourselves, so continue with real clients
154 continue;
155 }
156
157 const std::string sExtraDebugTextToSend =
158 sExtraDebugText.empty() ?
159 "PgeGnsServer Graceful stopListening()" :
160 "PgeGnsServer Graceful stopListening(), reason: " + sExtraDebugText;
161
162 // Close the connection. We use "linger mode" to ask SteamNetworkingSockets
163 // to flush this out (pending reliable message are actually sent) and close gracefully.
164 m_pInterface->CloseConnection(hClientConnection, k_ESteamNetConnectionEnd_App_Generic, sExtraDebugTextToSend.c_str(), true);
165 }
166
167 if (m_hListenSock != k_HSteamListenSocket_Invalid)
168 {
169 m_pInterface->CloseListenSocket(m_hListenSock);
170 m_hListenSock = k_HSteamListenSocket_Invalid;
171 }
172
173 if (m_hPollGroup != k_HSteamNetPollGroup_Invalid)
174 {
175 m_pInterface->DestroyPollGroup(m_hPollGroup);
176 m_hPollGroup = k_HSteamNetPollGroup_Invalid;
177 }
178
179 // here we create a client disconnect pkt that will be injected to our queue so app level can process it
180 // for the server itself, as it was a real client disconnecting
181 pge_network::PgePacket pkt;
182 pge_network::PgePacket::initPktPgeMsgUserDisconnected(
183 pkt,
184 static_cast<pge_network::PgeNetworkConnectionHandle>(k_HSteamNetConnection_Invalid));
185
186 // we push this packet to our pkt queue, this is how we "send" message to ourselves so server game loop can process it
187 m_queuePackets.push_back(pkt);
188
189 return true;
190}

197void PgeGnsServer::sendToClient(const HSteamNetConnection& conn, const pge_network::PgePacket& pkt)
198{
199 static_assert(k_HSteamNetConnection_Invalid == 0U, "on upper layers we use connHandle 0 to identify server, so here k_HSteamNetConnection_Invalid must be 0");
200 if (conn == k_HSteamNetConnection_Invalid)
201 {
202 // silent ignore, in the future maybe we will simply inject a message to ourselves' message queue
203 // that is how server can handle itself as a client similar to any other client
204 return;
205 }
206
207 uint32_t nActualPktSize;
208 if (pge_network::PgePacket::getPacketId(pkt) == pge_network::PgePktId::Application)
209 {
210 // We need the real used memory size so we can truncate the sent pkt to that.
211 // We need to consider member padding and the actual message sizes to have a correct value.
212 // 'cData' is our point of view since from there we need to calculate.
213 const pge_network::MsgApp* pMsgApp = pge_network::PgePacket::getMsgAppFromPkt(pkt);
214 assert(pMsgApp); // never null since it points into pkt
215 const pge_network::TByte* pMsgAppInByteSteps = reinterpret_cast<const pge_network::TByte*>(pMsgApp); // we can step this ptr in Bytes
216
217 const uint8_t nMessageCount = pge_network::PgePacket::getMessageAppCount(pkt);
218 assert(nMessageCount == 1); // for now only 1 msg/pkt
219 for (uint8_t i = 0; i < nMessageCount; i++)
220 {
221 pMsgApp = reinterpret_cast<const pge_network::MsgApp*>(pMsgAppInByteSteps);
222 // moving pMsgAppInByteSteps by the _actual_ size of the current MsgApp struct (considering the actual app message size there)
223 pMsgAppInByteSteps += pge_network::MsgApp::getMsgAppTotalActualSizeBytes(*pMsgApp);
224 }
225 // now pMsgAppInByteSteps points to RIGHT AFTER the 1st byte of the last application message
226 const pge_network::TByte* pPkt = (const pge_network::TByte*)(&pkt);
227 nActualPktSize = pMsgAppInByteSteps - pPkt;
228 }
229 else
230 {
231 nActualPktSize = static_cast<uint32_t>(sizeof(pkt));
232 }
233
234 m_pInterface->SendMessageToConnection(conn, &pkt, nActualPktSize, k_nSteamNetworkingSend_Reliable, nullptr);
235 if (m_nTxPktCount == 1)
236 {
237 m_time1stTxPkt = std::chrono::steady_clock::now();
238 }
239 m_nTxPktCount++;
240 if (pge_network::PgePacket::getPacketId(pkt) == pge_network::PgePktId::Application)
241 {
242 const pge_network::MsgApp* pMsgApp = pge_network::PgePacket::getMsgAppFromPkt(pkt);
243 const uint8_t nMessageCount = pge_network::PgePacket::getMessageAppCount(pkt);
244 for (uint8_t i = 0; i < nMessageCount; i++)
245 {
246 // TODO: nooo, I need to properly iterate to next appmsg as I iterate in above loop!
247 ++m_nTxMsgCount[pge_network::MsgApp::getMsgAppMsgId(*pMsgApp)];
248 //if (pge_network::MsgApp::getMsgAppMsgId(*pMsgApp) == 1u )
249 //{
250 // CConsole::getConsoleInstance("PgeGnsServer").OLn("%s() Sending out UserSetupFromServer to %u", __func__, conn);
251 //}
252 }
253 }
254 m_nTxByteCount += nActualPktSize;
255}

278void PgeGnsServer::inject(const pge_network::PgePacket& pkt)
279{
280 m_queuePackets.push_back(pkt);
281 if (m_nInjectPktCount == 1)
282 {
283 m_time1stInjectPkt = std::chrono::steady_clock::now();
284 }
285 m_nInjectPktCount++;
286 if (pge_network::PgePacket::getPacketId(pkt) == pge_network::PgePktId::Application)
287 {
288 const pge_network::MsgApp* pMsgApp = pge_network::PgePacket::getMsgAppFromPkt(pkt);
289 const uint8_t nMessageCount = pge_network::PgePacket::getMessageAppCount(pkt);
290 assert(nMessageCount == 1); // for now only 1 msg/pkt
291 for (uint8_t i = 0; i < nMessageCount; i++)
292 {
293 // TODO: nooo, I need to properly iterate to next appmsg as I iterate in sendToClient()!
294 ++m_nInjectMsgCount[pge_network::MsgApp::getMsgAppMsgId(*pMsgApp)];
295 }
296 }
297 m_nInjectByteCount += sizeof(pkt);

368bool PgeGnsServer::pgeMessageIsHandledAtGnsLevel(const pge_network::PgePacket& pktReceivedFromClient)
369{
370 if (pge_network::PgePacket::getPacketId(pktReceivedFromClient) == pge_network::MsgClientAppVersionFromClient::id)
371 {
372 // this should be the very 1st PgePacket we receive from a just connected client
373 const auto nClientConnHandle = pge_network::PgePacket::getServerSideConnectionHandle(pktReceivedFromClient);
374 const auto itClient = m_mapClients.find(nClientConnHandle);
375 if (itClient == m_mapClients.end())
376 {
377 CConsole::getConsoleInstance("PgeGnsServer").EOLn("%s: SERVER Cannot happen: a client (%u) sent MsgClientAppVersionFromClient but not present in clients map!",
378 __func__, nClientConnHandle);
379 assert(false);
380 }
381 else
382 {
383 const auto& msgClientAppVersion = pge_network::PgePacket::getMessageAsClientAppVersionFromClient(pktReceivedFromClient);
384 if (m_sAppVersion == msgClientAppVersion.m_szAppVersion)
385 {
386 CConsole::getConsoleInstance("PgeGnsServer").OLn("%s: SERVER client (%u) app version matching (%s), admitting client!",
387 __func__, nClientConnHandle, msgClientAppVersion.m_szAppVersion);
388
389 // here we create a client connect pkt that will be injected to our queue so app level will process it and create
390 // player object or whatever they want for the client
391 pge_network::PgePacket pktUserConnected;
392 pge_network::PgePacket::initPktPgeMsgUserConnected(
393 pktUserConnected,
394 itClient->first,
395 false /* bCurrentClient */,
396 itClient->second.m_szAddr);
397
398 // we push this packet to our pkt queue, this is how we "send" message to ourselves so server game loop can process it
399 m_queuePackets.push_back(pktUserConnected);
400 }
401 else
402 {
403 CConsole::getConsoleInstance("PgeGnsServer").EOLn("%s: SERVER client (%u) app version mismatching (%s), disconnecting client!",
404 __func__, nClientConnHandle, msgClientAppVersion.m_szAppVersion);
405
406 m_mapClients.erase(itClient);
407 const std::string sDebugText = "Server app version mismatching with client app version: " + m_sAppVersion + " != " + std::string(msgClientAppVersion.m_szAppVersion);
408 m_pInterface->CloseConnection(nClientConnHandle, 0, sDebugText.c_str(), false);
409 }
410 }
411 return true;
412 }
413 return false;
414}

446void PgeGnsServer::onSteamNetConnectionStatusChanged(SteamNetConnectionStatusChangedCallback_t* pInfo)
447{
448 // This function is also invoked on main thread when I call PgeGnsServer.PollConnectionStateChanges() from PGE::runGame()
449 // so no need to utilize mutexes around here.
450 // And the other function PollIncomingMessages() is also invoked by PGE::runGame().
451 // So it is safe to do operations on m_queuePackets.
452 char szTemp[1024];
453
454 const char* pszDebugLogAction;
455 switch (pInfo->m_info.m_eState)
456 {
457 case k_ESteamNetworkingConnectionState_None:
458 // sometimes we fall here immediately with m_eOldState k_ESteamNetworkingConnectionState_Connected when closing down connection
459 // fall-through
460 case k_ESteamNetworkingConnectionState_ClosedByPeer:
461 // fall-through
462 case k_ESteamNetworkingConnectionState_ProblemDetectedLocally:
463 {
464 if ((pInfo->m_eOldState == k_ESteamNetworkingConnectionState_Connected) || (pInfo->m_eOldState == k_ESteamNetworkingConnectionState_Connecting))
465 {
466 switch (pInfo->m_info.m_eState)
467 {
468 case k_ESteamNetworkingConnectionState_None:
469 pszDebugLogAction = "disappeared immediately";
470 break;
471 case k_ESteamNetworkingConnectionState_ClosedByPeer:
472 // Note that here we could check the reason code to see if it was a "usual" or an "unusual" one.
473 pszDebugLogAction = "disconnected";
474 break;
475 default: // k_ESteamNetworkingConnectionState_ProblemDetectedLocally
476 pszDebugLogAction = "disappeared, problem detected locally";
477 } // end switch m_eState
478 const auto itClient = m_mapClients.find(pInfo->m_hConn);
479 if (itClient == m_mapClients.end())
480 {
481 sprintf(szTemp, "Not stored (maybe already deleted) client %s, reason %d: %s", pszDebugLogAction, pInfo->m_info.m_eEndReason, pInfo->m_info.m_szEndDebug);
482 }
483 else
484 {
485 sprintf(szTemp, "%s %s, reason %d: %s", itClient->second.m_sCustomName.c_str(), pszDebugLogAction, pInfo->m_info.m_eEndReason, pInfo->m_info.m_szEndDebug);
486 m_mapClients.erase(itClient); // dont try to send anything to the disconnected client :)
487
488 // App level should be notified only if we found it in m_mapClients, otherwise probably this client was never known by app level
489 pge_network::PgePacket pkt;
490 pge_network::PgePacket::initPktPgeMsgUserDisconnected(pkt, pInfo->m_hConn);
491 // we push this packet to our pkt queue, this is how we "send" message to ourselves so server game loop can process it
492 m_queuePackets.push_back(pkt);
493 sendToAllClientsExcept(pkt);
494 }
495 CConsole::getConsoleInstance("PgeGnsServer").OLn("%s: SERVER Connection %s (handle %u) %s",
496 __func__,
497 pInfo->m_info.m_szConnectionDescription,
498 pInfo->m_hConn,
499 szTemp
500 );
501 logDetailedConnectionStatus(pInfo->m_hConn);
502
503 // Clean up the connection. This is important!
504 // The connection is "closed" in the network sense, but
505 // it has not been destroyed. We must close it on our end, too
506 // to finish up. The reason information do not matter in this case,
507 // and we cannot linger because it's already closed on the other end.
508 m_pInterface->CloseConnection(pInfo->m_hConn, 0, nullptr, false);
509 }
510 else
511 {
512 CConsole::getConsoleInstance("PgeGnsServer").OLn("%s: SERVER NO ACTION TAKEN: Connection %s (handle %u)",
513 __func__,
514 pInfo->m_info.m_szConnectionDescription,
515 pInfo->m_hConn
516 );
517 }
518 break;
519 }
520
521 case k_ESteamNetworkingConnectionState_Connecting:
522 {
523 // This must be a new connection
524 assert(m_mapClients.find(pInfo->m_hConn) == m_mapClients.end());
525
526 CConsole::getConsoleInstance("PgeGnsServer").OLn("%s: SERVER Connection request from %s", __func__, pInfo->m_info.m_szConnectionDescription);
527
528 // A client is attempting to connect
529 // Try to accept the connection.
530 if (m_pInterface->AcceptConnection(pInfo->m_hConn) != k_EResultOK)
531 {
532 // This could fail. If the remote host tried to connect, but then
533 // disconnected, the connection may already be half closed. Just
534 // destroy whatever we have on our side.
535 m_pInterface->CloseConnection(pInfo->m_hConn, 0, nullptr, false);
536 CConsole::getConsoleInstance("PgeGnsServer").EOLn("%s: SERVER Can't accept connection. (It was already closed?)", __func__);
537 break;
538 }
539
540 // Assign the poll group
541 if (!m_pInterface->SetConnectionPollGroup(pInfo->m_hConn, m_hPollGroup))
542 {
543 m_pInterface->CloseConnection(pInfo->m_hConn, 0, "Server failed SetConnectionPollGroup()", false);
544 CConsole::getConsoleInstance("PgeGnsServer").EOLn("%s: SERVER Failed to set poll group!", __func__);
545 break;
546 }
547
548 // Add them to the client list, using std::map wacky syntax
549 m_mapClients[pInfo->m_hConn];
550 pInfo->m_info.m_addrRemote.ToString(m_mapClients[pInfo->m_hConn].m_szAddr, sizeof(m_mapClients[pInfo->m_hConn].m_szAddr), true);
551 CConsole::getConsoleInstance("PgeGnsServer").OLn("%s: SERVER A client is connecting from %s ...", __func__, m_mapClients[pInfo->m_hConn].m_szAddr);
552
553 // since v0.2.4, we do NOT inject MsgUserConnected, instead we expect client now to send us a MsgClientAppVersionFromClient that we handle in
554 // pgeMessageIsHandledAtGnsLevel(), and based on version match, we inject MsgUserConnected there!
555
556 break;
557 }
558
559 case k_ESteamNetworkingConnectionState_Connected:
560 {
561 // We will get a callback immediately after accepting the connection.
562 // Since we are the server, we can ignore this, it's not news to us.
563
564 const auto itClient = m_mapClients.find(pInfo->m_hConn);
565 if (itClient == m_mapClients.end())
566 {

619const PgeGnsServer::TClient* PgeGnsServer::isClientConnectionHandleValid(const HSteamNetConnection& connHandle) const
620{
621 static_assert(k_HSteamNetConnection_Invalid == 0U, "on upper layers we use connHandle 0 to identify server, so here k_HSteamNetConnection_Invalid must be 0");
622
623 if (connHandle == k_HSteamNetConnection_Invalid)
624 {
625 CConsole::getConsoleInstance("PgeGnsServer").EOLn("%s: Invalid (server) connHandle specified!", __func__);
626 return nullptr;
627 }
628
629 if (!isInitialized())
630 {
631 CConsole::getConsoleInstance("PgeGnsServer").EOLn("%s not listening!", __func__);
632 return nullptr;
633 }
634
635 const auto itClient = m_mapClients.find(connHandle);
636 if (itClient == m_mapClients.end())
637 {
638 CConsole::getConsoleInstance("PgeGnsServer").EOLn("%s: connHandle %u not found!", __func__, connHandle);
639 return nullptr;
640 }
641
642 return &itClient->second;
643}
