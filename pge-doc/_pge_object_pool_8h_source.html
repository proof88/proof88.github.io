280 void deallocate()
281 {
282 if (m_capacity == 0)
283 {
284 return;
285 }
286
287 // I have to manually call the dtors because I allocated them 1 by 1 with placement new, so "delete[] m_pool" is not enough.
288 for (size_t i = 0; i < m_capacity; i++)
289 {
290 m_pool[i].~T();
291 }
292
293 // And then actually free up the memory pool.
294 ::operator delete[](m_pool);

364 template<typename... Args>
365 T* create(Args&&... pooledObjArgs)
366 {
367 if (!m_firstAvailable)
368 {
369 // no more available
370 return nullptr;
371 }
372
373 PgePooledObject* const ptr = m_firstAvailable;
374 m_firstAvailable = ptr->next();
375 m_count++;
376
377 T* const ptrAsT = static_cast<T*>(ptr);
378 ptrAsT->init(std::forward<Args>(pooledObjArgs)...);
379
380 // derived can override onSetUsed(), so setUsed() is the last action here after everything else is set!
381 ptr->setUsed(true);
382
383 return ptrAsT;

395 void remove(PgePooledObject& obj)
396 {
397 if (&obj.getParentPool() != this)
398 {
399 throw std::runtime_error("PgeObjectPool::remove(): PgePooledObject parent pool mismatch!");
400 return;
401 }
402
403 if (!obj.used())
404 {
405 return;
406 }
407
408 obj.setNext(m_firstAvailable);
409 m_firstAvailable = &obj;
410 m_count--;
411
412 // derived can override onSetUsed(), so setUsed() is the last action here after everything else is set!
413 obj.setUsed(false);

465 template<typename... Args>
466 void reserve(const std::string& poolName, const size_t& capacity, Args&&... pooledObjArgs)
467 {
468 if (m_capacity != 0)
469 {
470 throw std::runtime_error("PgeObjectPool::reserve(): already allocated pool " + m_name + "!");
471 return;
472 }
473
474 m_name = poolName;
475 m_capacity = capacity;
476
477 // early return in case capacity is 0, this is because even with 0 capacity operator new[] would
478 // return a non-null address down there which I definitely want to avoid!
479 // https://stackoverflow.com/questions/1087042/c-new-int0-will-it-allocate-memory
480 if (m_capacity == 0)
481 {
482 getConsole().SOLn("PgeObjectPool %s with zero capacity created successfully!", m_name.c_str());
483 return;
484 }
485
486 /* I cannot simply forward arguments like this:
487 new T[capacity]{ std::forward<Args>(args)... }
488 because aggregate initialization would use the default ctor of T, which might not be available!
489 So we are forwarding arguments in the loop below. */
490 m_pool = static_cast<T*>(::operator new[](capacity * sizeof(T)));
491 m_firstAvailable = m_pool;
492
493 for (size_t i = 0; i < m_capacity; i++)
494 {
495 /* Perfect forwarding : https://devblogs.microsoft.com/oldnewthing/20230727-00/?p=108494 .
496 I use placement new here so the already allocated mem pool is used.
497 Reason explained above. */
498 new (&m_pool[i]) T(*this, std::forward<Args>(pooledObjArgs)...);
499 }
500
501 for (size_t i = 0; i < (m_capacity - 1); i++)
502 {
503 m_pool[i].setNext(&(m_pool[i + 1]));
504 }
505 // no need for last elem's setNext(nullptr) since its ctor sets it by default
506
507 m_rawArrayWrapper = blIteratorAPI::getRawArrayWrapper(m_pool, m_capacity);
508
509 getConsole().SOLn("PgeObjectPool %s with capacity of %u elems (%u Bytes) created successfully!", poolName.c_str(), capacity, capacity * sizeof(T));
510 }
