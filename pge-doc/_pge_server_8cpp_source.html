298void PgeServerImpl::WriteList() const
299{
300 getConsole().OLnOI("PgeServer::WriteList() start");
301 if (isInitialized())
302 {
303 getConsole().OLn("Role: Server");
304 // TODO: PgeGnsWrapper will obviously use PgeGnsWrapper as module name when writing to console, so it is recommended now
305 // to always turn on PgeGnsWrapper logging as well together with PgeServer
306 m_gnsServer.WriteServerClientList();
307 }
308 else
309 {
310 getConsole().OLn("PgeServer is NOT initialized!");
311 }
312
313 getConsole().OOOLn("PgeServer::WriteList() end");
314} // WriteList()
