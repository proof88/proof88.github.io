30void proofps_dd::Player::genUniqueUserName(
31 char szNewUserName[proofps_dd::MsgUserNameChangeAndBootupDone::nUserNameBufferLength],
32 const std::string& sNameFromConfig,
33 const std::map<pge_network::PgeNetworkConnectionHandle, proofps_dd::Player>& m_mapPlayers)
34{
35 std::string sNewPlayerName(sNameFromConfig);
36 if (sNewPlayerName.empty())
37 {
38 sNewPlayerName = "Player";
39 }
40 sNewPlayerName = sNewPlayerName.substr(0, proofps_dd::MsgUserNameChangeAndBootupDone::nUserNameBufferLength - 1);
41
42 // if everybody is connecting with the SAME name, the maximum number of unique player names is: 10 ^ nUniqueNumberWidthInName
43 constexpr size_t nUniqueNumberWidthInName = 3; // used only if sInitialName collides with another name
44 static_assert(nUniqueNumberWidthInName >= 2);
45 static_assert(nUniqueNumberWidthInName < 5);
46
47 bool bNameCollision = true;
48 while (bNameCollision)
49 {
50 bNameCollision = false;
51 for (const auto& player : m_mapPlayers)
52 {
53 bNameCollision = (player.second.getName() == sNewPlayerName);
54 if (bNameCollision)
55 {
56 break;
57 }
58 }
59
60 if (bNameCollision)
61 {
62 // if we are here, we are even allowed to be a bit invasive with the name ...
63 sNewPlayerName = sNewPlayerName.substr(0, proofps_dd::MsgUserNameChangeAndBootupDone::nUserNameBufferLength - 1 - nUniqueNumberWidthInName) +
64 std::to_string(static_cast<size_t>(std::pow(10, nUniqueNumberWidthInName - 1)) + (rand() % static_cast<size_t>(std::pow(10, nUniqueNumberWidthInName))));
65 sNewPlayerName = sNewPlayerName.substr(0, proofps_dd::MsgUserNameChangeAndBootupDone::nUserNameBufferLength - 1);
66 }
67 };
68
69 strncpy_s(szNewUserName, proofps_dd::MsgUserNameChangeAndBootupDone::nUserNameBufferLength, sNewPlayerName.c_str(), sNewPlayerName.length());
70}

96 // load audio sources only once whenever 1st player object is created, does not matter if that is server or current client player,
97 // because these are static instances need to be set up only once for our player only
98 if (!m_sndWpnAmmo)

100 // no need to destruct in dtor, static destructions will take care anyway when exiting

112
113 // note that due to config can be changed in settings, we need to check if sound is actually loaded and try load again if it was
114 // not loaded previously, as loadSound() loads only if audio is enabled and initialized!
115 if (m_sndWpnAmmo->getLength() == 0.f)
116 {
117 // AudioSource was not yet loaded from file, let's try it now then.
118 // If we load stuff now below, and later user turns audio off, that is also fine, no audio will be played, and if later

348void proofps_dd::Player::updateAudioVisuals(const proofps_dd::Config& config, bool bServer)
349{
350 if (getInvulnerability())
351 {
352 constexpr auto nBlinkPeriodMillisecs = 100;
353 const auto nInvulTimeElapsedMillisecs = std::chrono::duration_cast<std::chrono::milliseconds>(std::chrono::steady_clock::now() - m_timeStartedInvulnerability).count();
354 // only server instance is allowed to decide about ENDING invulnerability, client will be notified over network msg
355 const bool bEndInvul = bServer && (static_cast<unsigned int>(nInvulTimeElapsedMillisecs) >= config.getPlayerRespawnInvulnerabilityDelaySeconds() * 1000);
356 const bool bShowPlayer = bEndInvul || ((nInvulTimeElapsedMillisecs / nBlinkPeriodMillisecs) % 2) == 0;
357 setVisibilityState(bShowPlayer);
358 if (bEndInvul)
359 {
360 setInvulnerability(false);
361 }
362 }
363 else
364 {
365 if (getHealth() > 0)
366 {
367 show();
368 }
369 }
370
371 // so far this is the only 3D sound that needs continuous position update because it is relatively longer than others AND played when player position is rapidly changing,
372 // AND other players need to continuously hear from the updated position e.g. we want them experience the sound coming down from above with the falling player :)
373 if (m_audio.getAudioEngineCore().isValidVoiceHandle(m_handleFallYell))
374 {
375 const PureVector& playerPos = getPos().getNew();
376 m_audio.getAudioEngineCore().set3dSourcePosition(m_handleFallYell, playerPos.getX(), playerPos.getY(), playerPos.getZ());
377 }
378}

413bool proofps_dd::Player::isDirty() const
414{
415 bool bDirtyFound = false;
416 for (auto it = m_vecOldNewValues.begin(); (it != m_vecOldNewValues.end()) && !bDirtyFound; it++)
417 {
418 // this is the "2. value-returning visitor" from example here: https://en.cppreference.com/w/cpp/utility/variant/visit
419 bDirtyFound = std::visit([](auto&& oldNewValue) -> bool { return oldNewValue.isDirty(); }, it->second);
420 }
421 return bDirtyFound;
422}

930void proofps_dd::Player::doCrouchServer()
931{
932 if (getCrouchStateCurrent())
933 {
934 return;
935 }
936
937 // Scaling change is legal since player object will be smaller thus no unexpected collision can happen;
938 // position change is also legal since player area stays within previous standing area.
939 // We need to set Object3D scaling since that is used in physics calculations also in serverPlayerCollisionWithWalls(),
940 // but we dont need to set Object3D position because Player object has its own position vector that is used in physics.
941 // On the long run we should use colliders so physics does not depend on graphics.
942 getObject3D()->SetScaling(PureVector(1.f, Player::fObjHeightCrouchScaling, 1.f));
943 getCrouchStateCurrent() = true; // can always go to crouching immediately
944
945 const bool bPullUpLegs = isInAir();
946 if (bPullUpLegs)
947 {
948 // reposition is allowed only if being in the air: pulling up the legs, so the head supposed to stay in same position as before,
949 // however we don't reposition if being on ground because it looks bad
950 // PPPKKKGGGGGG
951 PureVector playerPos = getPos().getNew();
952 playerPos.SetY(playerPos.getY() + Player::fObjHeightStanding / 2.f - (Player::fObjHeightStanding * Player::fObjHeightCrouchScaling) / 2.f);
953 getPos().set(playerPos);
954 // since we are at the beginning of a tick, it is legal to commit the position now, as old and new positions supposed to be the same at this point
955 getPos().commit();
956 }
957}

1024void proofps_dd::Player::startSomersaultServer(bool bJumpInduced)
1025{
1026 m_bWillSomersault = false;
1027
1028 // sanity check
1029 if (isSomersaulting())
1030 {
1031 return;
1032 }
1033
1034 if (bJumpInduced)
1035 {
1036 // mid-air sanity check
1037 if ((isJumping() && isJumpingInitiatedFromCrouching()) || (isJumping() != bJumpInduced) || isJumpInitiatedByJumppad())
1038 {
1039 return;
1040 }
1041 }
1042 else
1043 {
1044 // on-ground sanity check
1045 if (isInAir() || (m_strafe == Strafe::NONE))
1046 {
1047 return;
1048 }
1049 }
1050
1051 if (!getCrouchStateCurrent())
1052 {
1053 // for somersaulting on ground, we always require manual crouch, however for mid-air somersaulting, crouching depends on server config!
1054 if (bJumpInduced && m_cfgProfiles.getVars()[szCVarSvSomersaultMidAirAutoCrouch].getAsBool())
1055 {
1056 doCrouchServer();
1057 }
1058 else
1059 {
1060 return;
1061 }
1062 }
1063
1064 // just set the initial m_fSomersaultAngleZ by setting a small value, so stepSomersaultAngleServer() will know in which direction it should continue changing angle
1065 switch (m_strafe)
1066 {
1067 case Strafe::LEFT:
1068 m_fSomersaultAngleZ = (getObject3D()->getAngleVec().getY() == 0.f) ? 0.1f : -0.1f;
1069 break;
1070 case Strafe::RIGHT:
1071 m_fSomersaultAngleZ = (getObject3D()->getAngleVec().getY() == 0.f) ? -0.1f : 0.1f;
1072 break;
1073 default /* Strafe::NONE */:
1074 m_fSomersaultAngleZ = 0.1f;
1075 }
1076 getAngleZ() = m_fSomersaultAngleZ;
1077
1078 if (bJumpInduced)
1079 {
1080 // triggering mid-air somersaulting modifies player jump force and gravity, not the impact force
1081 // TODO: this should be accessed thru Config::getSomersaultMidAirJumpForceMultiplier(), however that introduces unforeseen mass of compilation problems now!
1082 m_vecJumpForce.SetX(m_vecJumpForce.getX() * m_cfgProfiles.getVars()[szCVarSvSomersaultMidAirJumpForceMultiplier].getAsFloat());
1083 m_fGravity *= m_cfgProfiles.getVars()[szCVarSvSomersaultMidAirJumpForceMultiplier].getAsFloat();
1084 }
1085 else
1086 {
1087 // triggering on-ground somersaulting modifies impact force only, not jump force or gravity
1088 switch (m_strafe)
1089 {
1090 case Strafe::LEFT:
1091 if (m_vecImpactForce.getX() > -fSomersaultGroundImpactForceX)
1092 {
1093 m_vecImpactForce.SetX(std::max(-fSomersaultGroundImpactForceX, m_vecImpactForce.getX() - fSomersaultGroundImpactForceX));
1094 }
1095 break;
1096 case Strafe::RIGHT:
1097 if (m_vecImpactForce.getX() < fSomersaultGroundImpactForceX)
1098 {
1099 m_vecImpactForce.SetX(std::min(fSomersaultGroundImpactForceX, m_vecImpactForce.getX() + fSomersaultGroundImpactForceX));
1100 }
1101 break;
1102 default /* Strafe::NONE */:
1103 break; // no-op
1104 }
1105 }
1106}
