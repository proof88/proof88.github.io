74void proofps_dd::Minimap::draw()
75{
76 // expected to be invoked every frame
77 if (!visible())
78 {
79 return;
80 }
81
82 if (!m_pge.getConfigProfiles().getVars()[Minimap::szCvarGuiMinimapShow].getAsBool())
83 {
84 return;
85 }
86
87 if ((m_maps.width() * m_maps.height()) == 0)
88 {
89 // map not yet loaded but GUI is already rendering the MenuState::None path, i.e. we are NOT in main menu
90 return;
91 }
92
93 static constexpr auto nMinimapPosLeft = 10;
94 static constexpr auto nMinimapPosTop = 10;
95 static const float fAlpha =
96 m_pge.getConfigProfiles().getVars()[Minimap::szCvarGuiMinimapTransparent].getAsBool() ? 0.8f : 1.f;
97 static const ImVec4 clrMinimapBgVec4 = ImVec4(0.f, 0.57f, 0.f, fAlpha);
98 static const ImVec4 clrMinimapBorderVec4 = ImVec4(0.f, 0.0f, 0.f, fAlpha);
99 static const ImVec4 clrViewportRectBgVec4 = ImVec4(0.f, 0.69f, 0.f, fAlpha);
100 static const ImVec4 clrPlayerRectBgVec4 = ImVec4(1.f, 1.f, 1.f, fAlpha);
101
102 const auto minimapSize = getMinimapSizeInPixels();
103 if ((minimapSize.x <= 5) || (minimapSize.y <= 5))
104 {
105 // there is no use of drawing anything if we are on such small map
106 return;
107 }
108
109 ImDrawList* const dl = ImGui::GetWindowDrawList();
110 assert(dl);
111
112 // draw minimap background rect
113
114 static const auto clrMinimapBgU32 = ImGui::GetColorU32(clrMinimapBgVec4);
115 dl->AddRectFilled(
116 ImVec2(nMinimapPosLeft, nMinimapPosTop),
117 ImVec2(nMinimapPosLeft + minimapSize.x, nMinimapPosTop + minimapSize.y),
118 clrMinimapBgU32);
119
120 // draw inside viewport rect
121
122 /* for viewport rect, we generate world-space coords of the top-left and bottom-right corners of the PURE camera viewport, and from these coords
123 we can calculate where the 2D viewport rect should be located on the minimap */
124 PureVector vecCamViewportTopLeftInWorldSpace;
125 if (!m_pge.getPure().getCamera().project2dTo3d(
126 0,
127 static_cast<TPureUInt>(roundf(m_pge.getPure().getCamera().getViewport().size.height - 1)),
128 /* in v0.2.5 this is player's Z (-1.2f as per GAME_PLAYERS_POS_Z) mapped to depth buffer: 0.9747f,
129 I'm using it now, however in the future if camera Z position can be change, we will need a dynamic value here */
130 0.9747f,
131 vecCamViewportTopLeftInWorldSpace))
132 {
133 //getConsole().EOLn("drawMinimap::%s(): project2dTo3d(0,0,...) failed!", __func__);
134 return;
135 }
136 m_pObjDebugVpTopLeft->getPosVec() = vecCamViewportTopLeftInWorldSpace;
137
138 PureVector vecCamViewportBottomRightInWorldSpace;
139 if (!m_pge.getPure().getCamera().project2dTo3d(
140 static_cast<TPureUInt>(roundf(m_pge.getPure().getCamera().getViewport().size.width - 1)),
141 0,
142 /* in v0.2.5 this is player's Z (-1.2f as per GAME_PLAYERS_POS_Z) mapped to depth buffer: 0.9747f,
143 I'm using it now, same way as in XHair, however in the future if camera Z position can be change, we will need a dynamic value here */
144 0.9747f,
145 vecCamViewportBottomRightInWorldSpace))
146 {
147 //getConsole().EOLn("drawMinimap::%s(): project2dTo3d(0,0,...) failed!", __func__);
148 return;
149 }
150 m_pObjDebugVpBottomRight->getPosVec() = vecCamViewportBottomRightInWorldSpace;
151
152 //static int i = 0;
153 //if (i++ == 300)
154 //{
155 // i = 0; // for adding debug breakpoint here, which is hit every 5 seconds @ 60 FPS
156 //}
157
158 // although we also calculate rectangles to stay inside the minimap, it is good to use a clip rectangle just in case ...
159 ImGui::PushClipRect(
160 ImVec2(nMinimapPosLeft, nMinimapPosTop),
161 ImVec2(nMinimapPosLeft + minimapSize.x + 1, nMinimapPosTop + minimapSize.y + 1),
162 false);
163
164 const ImVec2 fViewportRectTopLeft2D(
165 nMinimapPosLeft + getMinimapXfromWorldSpaceX(vecCamViewportTopLeftInWorldSpace.getX()),
166 nMinimapPosTop + getMinimapYfromWorldSpaceY(vecCamViewportTopLeftInWorldSpace.getY()));
167 const ImVec2 fViewportRectBottomRight2D(
168 nMinimapPosLeft + getMinimapXfromWorldSpaceX(vecCamViewportBottomRightInWorldSpace.getX()),
169 nMinimapPosTop + getMinimapYfromWorldSpaceY(vecCamViewportBottomRightInWorldSpace.getY()));
170
171 static const auto clrViewportRectBgU32 = ImGui::GetColorU32(clrViewportRectBgVec4);
172 dl->AddRectFilled(fViewportRectTopLeft2D, fViewportRectBottomRight2D, clrViewportRectBgU32);
173
174 // draw players rects
175
176 static const auto clrPlayerRectBgU32 = ImGui::GetColorU32(clrPlayerRectBgVec4);
177 for (const auto& playerPair : m_mapPlayers)
178 {
179 const auto& player = playerPair.second;
180 if ((player.getHealth() <= 0) || (!player.getObject3D()->isRenderingAllowed()))
181 {
182 continue;
183 }
184
185 ImVec2 fPlayerRectTopLeft2D(
186 nMinimapPosLeft + getMinimapXfromWorldSpaceX(player.getObject3D()->getPosVec().getX()),
187 nMinimapPosTop + getMinimapYfromWorldSpaceY(player.getObject3D()->getPosVec().getY()));
188
189 fPlayerRectTopLeft2D.x = std::min(fPlayerRectTopLeft2D.x, nMinimapPosLeft + minimapSize.x - 1);
190 fPlayerRectTopLeft2D.y = std::min(fPlayerRectTopLeft2D.y, nMinimapPosTop + minimapSize.y - 1);
191
192 dl->AddRectFilled(
193 fPlayerRectTopLeft2D,
194 ImVec2(fPlayerRectTopLeft2D.x + 2, fPlayerRectTopLeft2D.y + 2),
195 clrPlayerRectBgU32);
196 }
197
198 // finally draw the border of the minimap, it is last to make sure it is overdrawing everything else on the edges of the minimap rect
199
200 // Note: no need to set dl->Flags to 0 since I'm rounding return values in getMinimap...() functions, but I'm leaving here the commented code for some time ...
201 //const auto flagsDrawListOriginal = dl->Flags;
202 //dl->Flags = 0; // in case anti-aliasing is set, we turn it off temporarily because border looks sharper and better without it
203 static const auto clrMinimapBorderU32 = ImGui::GetColorU32(clrMinimapBorderVec4);
204 dl->AddRect(
205 ImVec2(nMinimapPosLeft, nMinimapPosTop),
206 ImVec2(nMinimapPosLeft + minimapSize.x + 1, nMinimapPosTop + minimapSize.y + 1),
207 clrMinimapBorderU32,
208 0.f /* rounding */,
209 0 /* flags */,
210 1.f /* thickness */);
211 //dl->Flags = flagsDrawListOriginal;
212
213 ImGui::PopClipRect();
214}

216ImVec2 proofps_dd::Minimap::getMinimapSizeInPixels() const
217{
218 if (m_maps.width() * m_maps.height() == 0)
219 {
220 // map not yet loaded but GUI is already rendering the MenuState::None path, i.e. we are NOT in main menu
221 return ImVec2();
222 }
223
224 /*
225 * With fScreenDimensionPercentageMax = 0.2f (meaning max 20% of screen dimension can be allocated for minimap, either on horizontal or vertical axis).
226 * if ( map.width > map.height)
227 * {
228 * nWidth = screen width * fScreenDimensionPercentageMax;
229 * nHeight = nWidth * (map.height / map.width);
230 * }
231 * else
232 * {
233 * nHeight = screen height * fScreenDimensionPercentageMax;
234 * nWidth = nHeight * (map.width / map.height);
235 * }
236 *
237 * Example Scenario 1:
238 * - screen width x height (pixels): 1980 x 1080
239 * - map width x height (blocks): 50 x 70
240 * - nHeight = 1080 * 0.2f = 216;
241 * - nWidth = 216 * (50/70) = 154.
242 *
243 * Example Scenario 2:
244 * - screen width x height (pixels): 1024 x 768
245 * - map width x height (blocks): 70 x 50
246 * - nWidth = 1024 * 0.2f = 205;
247 * - nHeight = 205 * (50/70) = 146.
248 */
249
250 static constexpr auto fScreenDimensionPercentageMax = 0.2f;
251 ImVec2 retSize;
252 if (m_maps.width() > m_maps.height())
253 {
254 retSize.x = std::roundf(ImGui::GetWindowWidth() * fScreenDimensionPercentageMax);
255 retSize.y = std::roundf(retSize.x * (m_maps.height() / static_cast<float>(m_maps.width())));
256 }
257 else
258 {
259 retSize.y = std::roundf(ImGui::GetWindowHeight() * fScreenDimensionPercentageMax);
260 retSize.x = std::roundf(retSize.y * (m_maps.width() / static_cast<float>(m_maps.height())));
261 }
262
263 return retSize;
264}

273float proofps_dd::Minimap::getMinimapXfromWorldSpaceX(const float& posWorldX) const
274{
275 /* make sure the input world-space coordinate is not out of map world-space bounds, where
276 map world-space bounds on X-axis are: [0, m_maps.width() * m_maps.fMapBlockSizeWidth] */
277 const float fConstrainedPosWorldX = std::min(
278 m_maps.width() * m_maps.fMapBlockSizeWidth,
279 std::max(0.f, posWorldX));
280
281 // PURE World-space to Dear ImGui Minimap space is easy on X-axis since 0 means left, increasing X means going to the RIGHT in both spaces
282 return std::roundf((fConstrainedPosWorldX / (m_maps.width() * m_maps.fMapBlockSizeWidth)) * getMinimapSizeInPixels().x);
283}

285float proofps_dd::Minimap::getMinimapYfromWorldSpaceY(const float& posWorldY) const
286{
287 const float fLowestMapPosWorldY = -1.f * static_cast<float>(m_maps.height()) * m_maps.fMapBlockSizeHeight /* 0 or negative value */;
288 if (fLowestMapPosWorldY == 0.f)
289 {
290 // do not divide by 0, this is possible with a map where there is only 1 row of map elements!
291 return 0.f;
292 }
293
294 /* make sure the input world-space coordinate is not out of map world-space bounds, where
295 map world-space bounds on Y-axis are: [-m_maps.width() * m_maps.fMapBlockSizeWidth, 0], since
296 the 1st row of map elems have Y-pos 0, the rest have decreasing thus negative Y-pos! */
297 const float fConstrainedPosWorldY = /* 0 or negative value */ std::min(
298 0.f,
299 std::max(fLowestMapPosWorldY, posWorldY));
300
301 // PURE World-space to Dear ImGui Minimap space is difficult on Y-axis since:
302 // - PURE 2D viewport (0,0) is the CENTER, and positive Y goes UPWARDS from CENTER;
303 // - Dear ImGui viewport (0,0) is the TOP LEFT, and positive Y goes DOWNWARDS from the TOP.
304
305 // positive value since these are both negative or become 0
306 const float fPosWorldYtoLowestMapPosWorldYRatio = fConstrainedPosWorldY / fLowestMapPosWorldY;
307 assert(fPosWorldYtoLowestMapPosWorldYRatio <= 1.f);
308 assert(fPosWorldYtoLowestMapPosWorldYRatio >= 0.f);
309
310 /* note that the UPWARDS vs DOWNWARDS difference between PURE and Dear ImGui has been implicitly handled above where
311 we created positive value from the 2 negative values division*/
312 return std::roundf(fPosWorldYtoLowestMapPosWorldYRatio * getMinimapSizeInPixels().y);
313}
