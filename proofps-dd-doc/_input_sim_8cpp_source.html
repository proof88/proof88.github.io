64 // I also tried mouse_event() and wanted to avoid using relative coordinates, because "relative mouse motion is subject to the settings for mouse

82void input_sim_test::bringWindowToFront(HWND hTargetWindow) noexcept(false)
83{
84 // technique copied from: https://stackoverflow.com/questions/916259/win32-bring-a-window-to-top
85 // slightly modified: try not to accept NULL foreground window, try a bit more ...
86 HWND hTmpWnd = GetForegroundWindow();
87 for (int i = 0; (i < 3) && (hTmpWnd == NULL); i++)
88 {
89 std::this_thread::sleep_for(std::chrono::milliseconds(500));
90 hTmpWnd = GetForegroundWindow();
91 }
92 const HWND hCurWnd = hTmpWnd;
93 if (hCurWnd == NULL)
94 {
95 // Win32 SDK: "The foreground window can be NULL in certain circumstances, such as when a window is losing activation."
96 // I still want to report this as an error since we use hCurWnd as input to other functions.
97 throw std::exception(
98 std::string("ERROR: GetForegroundWindow() returned NULL!").c_str());
99 }
100
101 const DWORD dwMyID = GetCurrentThreadId();
102 if (dwMyID == 0)
103 {
104 throw std::exception(
105 std::string("ERROR: GetCurrentThreadId() returned 0!").c_str());
106 }
107
108 const DWORD dwCurID = GetWindowThreadProcessId(hCurWnd, NULL);
109 if (dwCurID == 0)
110 {
111 throw std::exception(
112 std::string("ERROR: GetWindowThreadProcessId() returned 0!").c_str());
113 }
114
115 if (0 == AttachThreadInput(dwCurID, dwMyID, TRUE))
116 {
117 throw std::exception(
118 std::string("ERROR: AttachThreadInput(..., TRUE) failed: " + std::to_string(GetLastError()) + "!").c_str());
119 }
120
121 if (0 == SetWindowPos(hTargetWindow, HWND_TOPMOST, 0, 0, 0, 0, SWP_NOSIZE | SWP_NOMOVE))
122 {
123 throw std::exception(
124 std::string("ERROR: SetWindowPos(TOPMOST) failed: " + std::to_string(GetLastError()) + "!").c_str());
125 }
126
127 if (0 == SetWindowPos(hTargetWindow, HWND_NOTOPMOST, 0, 0, 0, 0, SWP_SHOWWINDOW | SWP_NOSIZE | SWP_NOMOVE))
128 {
129 throw std::exception(
130 std::string("ERROR: SetWindowPos(NOTOPMOST) failed: " + std::to_string(GetLastError()) + "!").c_str());
131 }
132
133 if (0 == SetForegroundWindow(hTargetWindow))
134 {
135 throw std::exception(
136 std::string("ERROR: SetForegroundWindow() failed!").c_str());
137 }
138
