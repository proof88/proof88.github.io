97void proofps_dd::Maps::shutdown()
98{
99 getConsole().OLnOI("Maps::shutdown() ...");
100 if (isInitialized())
101 {
102 // There is no specific need to delete loaded sounds, and I could not find specific functions to free up loaded sound data.
103 // When initialize() is invoked again, and we load sounds again, SoLoud frees up previously allocated sound data for the
104 // same wavs, so in that case memory cannot leak.
105
106 /* Current map handling */
107 unload();
108 m_sServerMapFilenameToLoad.clear();
109
110 /* Mapcycle and Available Maps Handling */
111 m_mapcycle.shutdown();
112
113 delete m_texDecorJumpPadVertical;
114 m_texDecorJumpPadVertical = PGENULL;
115
116 delete m_texRed;
117 m_texRed = PGENULL;
118 }
119 getConsole().OOOLn("Maps::shutdown() done!");
120}

122const std::string& proofps_dd::Maps::serverDecideFirstMapAndUpdateNextMapToBeLoaded()
123{
124 // PGEcfgProfiles allows the value of a CVAR be full of spaces (it is a valid case), which means that here we should trim
125 // the SV_MAP value properly since we at this level KNOW that spaces should NOT be present in this specific CVAR.
126 {
127 // this is ridiculous, PFL still doesnt have std::string-compatible strClr() !!!
128 const std::string sOriginalMapName = m_cfgProfiles.getVars()[szCVarSvMap].getAsString();
129 char cLine[200]{};
130 strncpy_s(cLine, sizeof(cLine), sOriginalMapName.c_str(), sOriginalMapName.length());
131 PFL::strClr(cLine);
132 m_cfgProfiles.getVars()[szCVarSvMap].Set(cLine);
133 }
134
135 if (m_cfgProfiles.getVars()[szCVarSvMap].getAsString().empty())
136 {
137 m_mapcycle.mapcycleRewindToFirst();
138 m_sServerMapFilenameToLoad = m_mapcycle.mapcycleGetCurrent();
139 if (m_sServerMapFilenameToLoad.empty())
140 {
141 return m_sServerMapFilenameToLoad; // error
142 }
143 else
144 {
145 getConsole().OLn("First map by mapcycle: %s", m_sServerMapFilenameToLoad.c_str());
146 }
147 }
148 else
149 {
150 m_sServerMapFilenameToLoad = m_cfgProfiles.getVars()[szCVarSvMap].getAsString();
151 getConsole().OLn("First map by config (%s): %s", szCVarSvMap, m_sServerMapFilenameToLoad.c_str());
152
153 // we go to last map in mapcycle so that when server switches from szCVarSvMap to next map, it will be the
154 // first map in mapcycle, this is how it will properly start looping mapcycle.

156 }

170bool proofps_dd::Maps::load(const char* fname, std::function<void(int)>& cbDisplayProgressUpdate)
171{
172 getConsole().OLnOI("Maps::load(%s) ...", fname);
173
174 if (!isInitialized())
175 {
176 getConsole().EOLnOO("ERROR: map handler is not initialized!");
177 return false;
178 }
179
180 if (loaded())
181 {
182 getConsole().EOLnOO("ERROR: %s is already loaded, should call unload first!", m_sFileName.c_str());
183 return false;
184 }
185
186 cbDisplayProgressUpdate(0);
187
188 // this wont be needed after we require unload() before consecutive load()
189 proofps_dd::MapItem::resetGlobalData();
190
191 const std::string sFilenameWithRelativePath = std::string(Mapcycle::GAME_MAPS_DIR) + fname;
192 m_sFileName = PFL::getFilename(sFilenameWithRelativePath.c_str());
193 m_sRawName = PFL::changeExtension(m_sFileName.c_str(), "");
194 if (m_sRawName.empty())
195 {
196 getConsole().EOLnOO("ERROR: empty raw name!");
197 unload();
198 return false;
199 }
200
201 std::ifstream f;
202 f.open(sFilenameWithRelativePath.c_str(), std::ifstream::in);
203 if ( !f.good() )
204 {
205 getConsole().EOLnOO("ERROR: failed to open file %s!", m_sFileName.c_str());
206 unload();
207 return false;
208 }
209
210 m_sServerMapFilenameToLoad = fname;
211
212 const TPURE_ISO_TEX_FILTERING texFilterMinOriginal = m_gfx.getTextureManager().getDefaultMinFilteringMode();
213 const TPURE_ISO_TEX_FILTERING texFilterMagOriginal = m_gfx.getTextureManager().getDefaultMagFilteringMode();
214 m_gfx.getTextureManager().setDefaultIsoFilteringMode(
215 TPURE_ISO_TEX_FILTERING::PURE_ISO_LINEAR_MIPMAP_LINEAR,
216 TPURE_ISO_TEX_FILTERING::PURE_ISO_LINEAR);
217
218 bool bParseError = false;
219 bool bMapLayoutReached = false;
220 constexpr std::streamsize nBuffSize = 1024;
221 char cLine[nBuffSize];
222 TPureFloat y = 0.f; // before v0.2.5 it was 4.f but I dont know why, I'm changing to 0 so all blocks have their Y-pos <= 0.f
223 std::streampos fStreamPosMapLayoutStart, fStreamPosPrevLine;
224 // First the map layout handling will be "dry", meaning that the map won't be actually created but
225 // the number of required blocks will be counted. This is useful, because then in the next non-dry run,
226 // we preallocate exactly that array size required for that number of blocks, this way no reallocation of array will be needed.
227 // Since the maps are typed into text file by humans, we cannot expect them to save the exact number of blocks. :)
228 while ( !bParseError && !f.eof() )
229 {
230 fStreamPosPrevLine = f.tellg();
231 f.getline(cLine, nBuffSize);
232 // TODO: we should finally have a strClr() version for std::string or FINALLY UPGRADE TO NEWER CPP THAT MAYBE HAS THIS FUNCTIONALITY!!!
233 PFL::strClr( cLine );
234 const std::string sLine(cLine);
235 std::string sVar, sValue;
236 if ( lineShouldBeIgnored(sLine) )
237 {
238 continue;
239 }
240 else if ( lineIsValueAssignment(sLine, sVar, sValue, bParseError) )
241 {
242 if ( bMapLayoutReached )
243 {
244 getConsole().EOLn("ERROR: parse: assignment after map layout block: %s!", sLine.c_str());
245 bParseError = true;
246 }
247 else
248 {
249 lineHandleAssignment(sVar, sValue);
250 }
251 }
252 else if ( !bParseError )
253 {
254 if (!bMapLayoutReached)
255 {
256 getConsole().OLn("Just reached map layout ...");
257 bMapLayoutReached = true;
258 fStreamPosMapLayoutStart = fStreamPosPrevLine;
259 }
260 bParseError = !lineHandleLayout(sLine, y, true);
261 }
262 };
263
264 // non-dry-run starts, this is the slowest part of map loading, thus we invoke cbDisplayProgressUpdate in this block
265 if (!bParseError)
266 {
267 getConsole().OLn(
268 "Just finished dry run, now building up the map with width %u, height %u, m_blocks_h %d, m_foregroundBlocks_h %d ...",
269 m_width, m_height, m_blocks_h, m_foregroundBlocks_h);
270
271 // During dry-run above, m_blocks_h and m_foregroundBlocks_h had been properly set, but they will be reset and counted again now
272 // during the non-dry-run, so save m_blocks_h now to calculate loading progress
273 const int nTotalBlocks_h = m_blocks_h;
274 f.clear(); // clears error states, including eof bit
275 f.seekg(fStreamPosMapLayoutStart);
276 y = 0.f; // just reset this to same value as it was before the loop
277 while (!bParseError && !f.eof())
278 {
279 f.getline(cLine, nBuffSize);
280 // TODO: we should finally have a strClr() version for std::string or FINALLY UPGRADE TO NEWER CPP THAT MAYBE HAS THIS FUNCTIONALITY!!!
281 PFL::strClr(cLine);
282 const std::string sLine(cLine);
283 if (lineShouldBeIgnored(sLine))
284 {
285 continue;
286 }
287 bParseError = !lineHandleLayout(sLine, y, false);
288 cbDisplayProgressUpdate(nTotalBlocks_h == 0 ? 0 : static_cast<int>((m_blocks_h / static_cast<float>(nTotalBlocks_h)) * 100));
289 }
290 }
291 f.close();
292
293 // variables are loaded during dry-run, but blocks are not, thus this is the earliest time to fail if their count does not match!
294 const auto nJumppadVarsCount = getJumppadValidVarsCount();
295 if (m_jumppads.size() != nJumppadVarsCount)
296 {
297 getConsole().EOLn("ERROR: jumppads size (%u) != valid jumppad vars count (%u)", m_jumppads.size(), nJumppadVarsCount);
298 bParseError = true;
299 }
300
301 m_gfx.getTextureManager().setDefaultIsoFilteringMode(
302 texFilterMinOriginal,
303 texFilterMagOriginal);
304
305 if ( bParseError )
306 {
307 getConsole().EOLnOO("ERROR: failed to parse file!");
308 unload();
309 return false;
310 }
311
312 getConsole().OLn("Just built up the map with m_blocks_h %d, m_foregroundBlocks_h %d ...", m_blocks_h, m_foregroundBlocks_h);
313
314 m_blockPosMin = m_blocks[0]->getPosVec();
315 m_blockPosMax = m_blocks[0]->getPosVec();
316 for (int i = 0; i < m_blocks_h; i++)
317 {
318 if ( m_blocks[i] != PGENULL )
319 {
320 if (m_blocks[i]->getPosVec().getX() < m_blockPosMin.getX())
321 {
322 m_blockPosMin.SetX(m_blocks[i]->getPosVec().getX());
323 }
324 else if (m_blocks[i]->getPosVec().getX() > m_blockPosMax.getX())
325 {
326 m_blockPosMax.SetX(m_blocks[i]->getPosVec().getX());
327 }
328
329 if (m_blocks[i]->getPosVec().getY() < m_blockPosMin.getY())
330 {
331 m_blockPosMin.SetY(m_blocks[i]->getPosVec().getY());
332 }
333 else if (m_blocks[i]->getPosVec().getY() > m_blockPosMax.getY())
334 {
335 m_blockPosMax.SetY(m_blocks[i]->getPosVec().getY());
336 }
337
338 if (m_blocks[i]->getPosVec().getZ() < m_blockPosMin.getZ())
339 {
340 m_blockPosMin.SetZ(m_blocks[i]->getPosVec().getZ());
341 }
342 else if (m_blocks[i]->getPosVec().getZ() > m_blockPosMax.getZ())
343 {
344 m_blockPosMax.SetZ(m_blocks[i]->getPosVec().getZ());
345 }
346 }
347 }
348 m_blocksVertexPosMin.Set(
349 m_blockPosMin.getX() - proofps_dd::Maps::fMapBlockSizeWidth / 2.f,
350 m_blockPosMin.getY() - proofps_dd::Maps::fMapBlockSizeHeight / 2.f,
351 m_blockPosMin.getZ() - proofps_dd::Maps::fMapBlockSizeDepth / 2.f);
352 m_blocksVertexPosMax.Set(
353 m_blockPosMax.getX() + proofps_dd::Maps::fMapBlockSizeWidth / 2.f,
354 m_blockPosMax.getY() + proofps_dd::Maps::fMapBlockSizeHeight / 2.f,
355 m_blockPosMax.getZ() + proofps_dd::Maps::fMapBlockSizeDepth / 2.f);
356
357 getConsole().SOLnOO("> Map loaded with width %u and height %u!", m_width, m_height);
358 return true;
359}

361void proofps_dd::Maps::unload()
362{
363 getConsole().OLnOI("Maps::unload() ...");
364 m_sServerMapFilenameToLoad.clear();
365 m_sRawName.clear();
366 m_sFileName.clear();
367 m_Block2Texture.clear();
368 if ( m_blocks )
369 {
370 for (int i = 0; i < m_blocks_h; i++)
371 {
372 m_gfx.getObject3DManager().DeleteAttachedInstance( *(m_blocks[i]) );
373 }
374 free( m_blocks );
375 m_blocks = NULL;
376
377 free(m_foregroundBlocks);
378 m_foregroundBlocks = NULL;
379 }
380 m_blocks_h = 0;
381 m_foregroundBlocks_h = 0;
382
383 for (auto& pairChar2RefBlockObject3D : m_mapReferenceBlockObject3Ds)
384 {
385 delete pairChar2RefBlockObject3D.second; // will remove from object3dmanager too
386 }
387 m_mapReferenceBlockObject3Ds.clear();
388
389 for (auto& itemPair : m_items)
390 {
391 delete itemPair.second;
392 }
393 m_items.clear();
394 for (auto& pDecorObj : m_decorations)
395 {
396 delete pDecorObj; // will remove from object3dmanager too
397 }
398 m_decorations.clear();
399 m_jumppads.clear();
400 m_fJumppadForceFactors.clear();
401 proofps_dd::MapItem::resetGlobalData();
402
403 m_width = 0;
404 m_height = 0;
405 m_blockPosMin.SetZero();
406 m_blockPosMax.SetZero();
407 m_blocksVertexPosMin.SetZero();
408 m_blocksVertexPosMax.SetZero();
409 m_vars.clear();
410 m_nValidJumppadVarsCount = 0;
411 m_spawnpoints.clear();
412
413 getConsole().OOOLn("Maps::unload() done!");
414}

426void proofps_dd::Maps::updateVisibilitiesForRenderer()
427{
428 const PureVector& campos = m_gfx.getCamera().getPosVec();
429
430 for (int i = 0; i < m_blocks_h; i++)
431 {
432 PureObject3D* const obj = m_blocks[i];
433 if ( obj != PGENULL )
434 {
435 if ( (obj->getPosVec().getX() + obj->getSizeVec().getX()/2.0f) <= campos.getX() - 13.f )
436 {
437 obj->SetRenderingAllowed(false);
438 }
439 else
440 {
441 if ( (obj->getPosVec().getX() - obj->getSizeVec().getX()/2.0f) >= campos.getX() + 13.f )
442 {
443 obj->SetRenderingAllowed(false);
444 }
445 else
446 {
447 obj->SetRenderingAllowed(true);
448 }
449 }
450 }
451 }

567size_t proofps_dd::Maps::getJumppadValidVarsCount()
568{
569 // This is a lazy evaluator, so it will count valid vars only at first call, and in subsequent calls it will just return the
570 // previously calculated value.
571 // It keeps recalculating when number of previously calculated valid vars are not matching with the number of jump pads.
572 // This is to make sure it always returns valid value, just in case someone would call it in the middle of map loading.
573 // In that case speed is not an issue, later during actual gameplay it will be very fast due to no recalculating.
574 // And obviously 0 == 0 case is also very straightforward: it does not calculate if there is no jumppad in map layout.
575 if (m_jumppads.size() == m_nValidJumppadVarsCount)
576 {
577 return m_nValidJumppadVarsCount;
578 }
579
580 if (m_decorations.size() != m_jumppads.size())
581 {
582 getConsole().EOLn(
583 "PRooFPSddPGE::%s(): decorations count %u != jumppads count %u!",
584 __func__,
585 m_decorations.size(),
586 m_jumppads.size());
587 return 0;
588 }
589
590 m_fJumppadForceFactors.clear();
591 m_nValidJumppadVarsCount = 0;
592 size_t iJumppad = 0;
593 for (const auto& var : m_vars)
594 {
595 if (var.first.find("jumppad_") != std::string::npos)
596 {
597 if (var.first != (std::string("jumppad_") + std::to_string(iJumppad)))
598 {
599 getConsole().EOLn(
600 "PRooFPSddPGE::%s(): index error: variable name is %s but our running index is %u!",
601 __func__,
602 var.first.c_str(),
603 iJumppad);
604 continue;
605 }
606
607 if (var.second.getAsString().empty())
608 {
609 getConsole().EOLn("PRooFPSddPGE::%s(): variable %s empty!", __func__, var.first.c_str());
610 continue;
611 }
612
613 const auto itCommaChar = var.second.getAsString().find(' ');
614 if (itCommaChar == std::string::npos)
615 {
616 const float fForceY = var.second.getAsFloat();
617 if (fForceY > 0.f)
618 {
619 m_fJumppadForceFactors.push_back({0.f, fForceY});
620 m_nValidJumppadVarsCount++;
621 }
622 }
623 else
624 {
625 std::stringstream sstr(var.second.getAsString());
626 TPURE_XY fForces{};
627 sstr >> fForces.y;
628 sstr >> fForces.x;
629 if (fForces.y > 0.f)
630 {
631 m_fJumppadForceFactors.push_back(fForces);
632 m_nValidJumppadVarsCount++;
633
634 // This trick I'm doing here is without any serious scientific background and not even something I should be proud of, but
635 // this is how it works: since the jumppad mult factors are specified as Y and X force multipliers, which are input to the
636 // Physics engine for calculating the INITIAL jump force, and we here in Maps do not know how Physics is doing the calculations,
637 // we are just making an ESTIMATION about the resulting jump angle the player will experience!
638 // We are ASSUMING that a relatively big X multiplier (compared to the Y multiplier) will result in a more horizontal force,
639 // rather than a more vertical force, and vice versa.
640 // So, by treating the Y and X multipliers as components of a force vector, and by normalizing this vector, we will have a ROUGH
641 // estimation of the horizontal force compared to the vertical force, and, if we treat this horizontal force component
642 // as sine of the Z angle of the force (where -90 <= Z angle <= 90), we can make a ROUGH ESTIMATION about the Z angle of
643 // the force, and we set that angle for the decoration arrow. Which might be a wrong estimation since we dont have a clue about
644 // how really these multipliers contribute to the final movement, because at this point Physics is a blackbox to Maps.
645 // Still, I find this working more or less ok, so I leave it here.
646 // I'm treating normalized x value as sine, because after normalization, vec will have: -1 <= x <= 1, and 0 means 0.
647 PureVector vecForces(fForces.x, fForces.y, 0.f);
648 vecForces.Normalize();
649 //const float fForcesXYratio = fForces.x / fForces.y;
650 //getConsole().EOLn("PRooFPSddPGE::%s(): jumppad %u forces ratio: %f!", __func__, iJumppad, fForcesXYratio);
651 //getConsole().EOLn("PRooFPSddPGE::%s(): jumppad %u forces vec normalized: x: %f, y: %f!", __func__, iJumppad, vecForces.getX(), vecForces.getY());
652
653 assert(m_decorations[iJumppad]); // no nulls here
654
655 // I'm flipping the object, otherwise positive Z angle would turn it to the left instead right
656 m_decorations[iJumppad]->getAngleVec().SetY(180.f);
657 m_decorations[iJumppad]->SetDoubleSided(true);
658
659 m_decorations[iJumppad]->getAngleVec().SetZ(
660 PFL::radToDeg(std::asinf(vecForces.getX()))
661 );
662
663 // maybe it is also a good idea to displace this decoration horizontally based on the horizontal component:
664 // (would not need this if we could modify the center of the object before rotation i.e. pivot point)
665 m_decorations[iJumppad]->getPosVec().SetX(
666 m_decorations[iJumppad]->getPosVec().getX() + vecForces.getX()
667 );
668 }
669 }
670
671 iJumppad++;
672 }
673 }
674 return m_nValidJumppadVarsCount;
675}

678{

694void proofps_dd::Maps::update(const float& fps)
695{
696 // invoked by both server and client
697
698 static constexpr float MAPITEM_VERTICAL_ANIM_UPDATE_SPEED = 480.0f;
699
700 for (auto& itemPair : getItems())
701 {
702 if (!itemPair.second)
703 {
704 continue;
705 }
706
707 proofps_dd::MapItem& mapItem = *(itemPair.second);
708 if (mapItem.isTaken())
709 {
710 continue;
711 }
712
713 mapItem.update(MAPITEM_VERTICAL_ANIM_UPDATE_SPEED / fps);
714 }
715
716 static constexpr float DECOR_ALPHA_CHANGE_SPEED = 180.0f;
717 static constexpr float fDecorAlphaMin = 100.f;
718 static constexpr float fDecorAlphaMax = 200.f;
719 static_assert(fDecorAlphaMin >= 0, "will be casted to TPureUByte");
720 static_assert(fDecorAlphaMin <= 255, "will be casted to TPureUByte");
721 static float fDecorAlpha = 200.f;
722 static bool bDecorAlphaUp = false;
723
724 if (bDecorAlphaUp)
725 {
726 fDecorAlpha += DECOR_ALPHA_CHANGE_SPEED / fps;
727 if (fDecorAlpha > fDecorAlphaMax)
728 {
729 fDecorAlpha = fDecorAlphaMax;
730 bDecorAlphaUp = false;
731 }
732 }
733 else
734 {
735 fDecorAlpha -= DECOR_ALPHA_CHANGE_SPEED / fps;
736 if (fDecorAlpha < fDecorAlphaMin)
737 {
738 fDecorAlpha = fDecorAlphaMin;
739 bDecorAlphaUp = true;
740 }
741 }
742
743 for (auto& pDecor : m_decorations)
744 {
745 assert(pDecor); // we dont store nulls here
746 pDecor->getMaterial(false).getTextureEnvColor().SetAlpha(
747 static_cast<TPureUByte>(std::llroundl(fDecorAlpha))
748 );
749 }
750}

794bool proofps_dd::Maps::lineIsValueAssignment(const std::string& sLine, std::string& sVar, std::string& sValue, bool& bParseError)
795{
796 const std::string::size_type nAssignmentPos = sLine.find('=');
797 if ( nAssignmentPos == std::string::npos )
798 {
799 return false;
800 }
801
802 if ( (nAssignmentPos == (sLine.length() - 1)) || (nAssignmentPos == 0 ) )
803 {
804 CConsole::getConsoleInstance("Maps").EOLn("ERROR: erroneous assignment: %s!", sLine.c_str());
805 bParseError = true;
806 return false;
807 }
808
809 // sLine is already trimmed: neither leading nor trailing spaces
810
811 // get rid of trailing spaces from the variable name itself, standing before the '=' char
812 // TODO: should rather use std::string compatible PFL::strClr()
813 std::string::size_type nSpPos = sLine.find(' ');
814 if ( nSpPos != std::string::npos )
815 {
816 if ( nSpPos < nAssignmentPos )
817 {
818 sVar = sLine.substr(0, nSpPos);
819 if ( sVar.find(' ') != std::string::npos )
820 {
821 // we should not have more space before '=' char
822 CConsole::getConsoleInstance("Maps").EOLn("ERROR: erroneous assignment, failed to parse variable in line: %s!", sLine.c_str());
823 bParseError = true;
824 return false;
825 }
826 }
827 else
828 {
829 // should never reach this point based on above 2 conditions
830 CConsole::getConsoleInstance("Maps").EOLn("ERROR: erroneous assignment: %s!", sLine.c_str());
831 bParseError = true;
832 return false;
833 }
834 }
835 else
836 {
837 sVar = sLine.substr(0, nAssignmentPos);
838 }
839
840 // get rid of leading spaces from the value itself, standing after the '=' char
841 std::string::size_type i = nAssignmentPos+1;
842 while ( (i < sLine.length()) && sLine[i] == ' ' )
843 {
844 i++;
845 }
846
847 if ( i < sLine.length() )
848 {
849 sValue = sLine.substr(i);
850 }
851 else
852 {
853 CConsole::getConsoleInstance("Maps").EOLn("ERROR: erroneous assignment, failed to parse value in line: %s!", sLine.c_str());
854 bParseError = true;
855 return false;
856 }
857
858 return true;
859}

887bool proofps_dd::Maps::lineHandleLayout(const std::string& sLine, TPureFloat& y, bool bDryRun)
888{
889 if (bDryRun)
890 {
891 m_height++;
892 if (m_width < sLine.length())
893 {
894 m_width = sLine.length();
895 }
896 }
897 else
898 {
899 if (m_blocks == nullptr)
900 {
901 // first line in non-dry run, now we can allocate memory for all blocks
902 // TODO: handle memory allocation errors
903 m_blocks = (PureObject3D**)malloc(m_blocks_h * sizeof(PureObject3D*));
904 m_blocks_h = 0; // we are incrementing it again during non-dry run
905 m_foregroundBlocks = (PureObject3D**)malloc(m_foregroundBlocks_h * sizeof(PureObject3D*));
906 m_foregroundBlocks_h = 0; // we are incrementing it again during non-dry run
907 }
908 }
909
910 TPureFloat x = 0.0f;
911 TPureFloat maxx = x;
912 std::string::size_type iLinePos = 0;
913
914 // Item character specifies the item type, but not the background behind the item.
915 // So the idea is to copy the previous _neighbor_ background block to be used behind the item, but
916 // only if there is a neighbor block created previously, otherwise we should not put any
917 // background block behind the item.
918 // So iObjectBgToBeCopied is > -1 only if there is neighbor background block created previously.
919 int iObjectBgToBeCopied = -1;
920
921 // The idea with special foreground block copying the previous neighbor foreground block is similar as
922 // described above with special background blocks.
923 int iObjectFgToBeCopied = -1;
924
925 while ( iLinePos != sLine.length() )
926 {
927 const char c = sLine[iLinePos];
928 iLinePos++;
929 const bool bForeground = foregroundBlocks.find(c) != foregroundBlocks.end();
930 const bool bBackground = backgroundBlocks.find(c) != backgroundBlocks.end();
931
932 x = x + proofps_dd::Maps::fMapBlockSizeWidth;
933 if ( x > maxx )
934 {
935 maxx = x;
936 }
937
938 if ( !bForeground && !bBackground )
939 {
940 iObjectBgToBeCopied = -1;
941 iObjectFgToBeCopied = -1;
942 continue;
943 }
944
945 if (bForeground && bBackground)
946 {
947 const std::string sc(1, c); // WA for CConsole lack support of %c
948 getConsole().EOLn("%s Block defined as both foreground and background: %s!", __func__, sc.c_str());
949 assert(false);
950 return false;
951 }
952
953 // special background block handling
954 bool bCopyPreviousBgBlock = false;
955 bool bSpecialBgBlock = false;
956 switch (c)
957 {
958 case ',':
959 {
960 if (bDryRun)
961 {
962 proofps_dd::MapItem* pMapItem = new proofps_dd::MapItem(m_gfx, MapItemType::ITEM_ARMOR, PureVector(x, y, GAME_ITEMS_POS_Z));
963 m_items.insert({ pMapItem->getId(), pMapItem });
964 }
965 bSpecialBgBlock = true;
966 bCopyPreviousBgBlock = iObjectBgToBeCopied > -1;
967 break;
968 }
969 case '+':
970 {
971 if (bDryRun)
972 {
973 proofps_dd::MapItem* pMapItem = new proofps_dd::MapItem(m_gfx, MapItemType::ITEM_HEALTH, PureVector(x, y, GAME_ITEMS_POS_Z));
974 m_items.insert({ pMapItem->getId(), pMapItem });
975 }
976 bSpecialBgBlock = true;
977 bCopyPreviousBgBlock = iObjectBgToBeCopied > -1;
978 break;
979 }
980 case '2':
981 {
982 if (bDryRun)
983 {
984 proofps_dd::MapItem* pMapItem = new proofps_dd::MapItem(m_gfx, MapItemType::ITEM_WPN_PISTOL, PureVector(x, y, GAME_ITEMS_POS_Z));
985 m_items.insert({ pMapItem->getId(), pMapItem });
986 }
987 bSpecialBgBlock = true;
988 bCopyPreviousBgBlock = iObjectBgToBeCopied > -1;
989 break;
990 }
991 case '3':
992 {
993 if (bDryRun)
994 {
995 proofps_dd::MapItem* pMapItem = new proofps_dd::MapItem(m_gfx, MapItemType::ITEM_WPN_MACHINEGUN, PureVector(x, y, GAME_ITEMS_POS_Z));
996 m_items.insert({ pMapItem->getId(), pMapItem });
997 }
998 bSpecialBgBlock = true;
999 bCopyPreviousBgBlock = iObjectBgToBeCopied > -1;
1000 break;
1001 }
1002 case '4':
1003 {
1004 if (bDryRun)
1005 {
1006 proofps_dd::MapItem* pMapItem = new proofps_dd::MapItem(m_gfx, MapItemType::ITEM_WPN_BAZOOKA, PureVector(x, y, GAME_ITEMS_POS_Z));
1007 m_items.insert({ pMapItem->getId(), pMapItem });
1008 }
1009 bSpecialBgBlock = true;
1010 bCopyPreviousBgBlock = iObjectBgToBeCopied > -1;
1011 break;
1012 }
1013 case 'S':
1014 {
1015 if (!bDryRun)
1016 {
1017 // spawnpoint is background block by default
1018 const PureVector vecSpawnPointPos(x, y, GAME_PLAYERS_POS_Z);
1019 if (m_spawnpoints.empty())
1020 {
1021 m_spawnpointLeftMost = vecSpawnPointPos;
1022 m_spawnpointRightMost = vecSpawnPointPos;
1023 }
1024 else
1025 {
1026 if (x < m_spawnpointLeftMost.getX())
1027 {
1028 m_spawnpointLeftMost = vecSpawnPointPos;
1029 }
1030 if (x > m_spawnpointRightMost.getX())
1031 {
1032 m_spawnpointRightMost = vecSpawnPointPos;
1033 }
1034 }
1035 m_spawnpoints.insert(vecSpawnPointPos);
1036 }
1037 bSpecialBgBlock = true;
1038 bCopyPreviousBgBlock = iObjectBgToBeCopied > -1;
1039 break;
1040 }
1041 default: /* NOP */;
1042 break;
1043 }
1044
1045 // special foreground block handling
1046 bool bCopyPreviousFgBlock = false;
1047 bool bSpecialFgBlock = false;
1048 bool bJumppad = false;
1049 switch (c)
1050 {
1051 case '^':
1052 {
1053 if (bDryRun)
1054 {
1055 PureObject3D* const pDecorObj = m_gfx.getObject3DManager().createPlane(1.f, 1.2f);
1056 pDecorObj->getPosVec().Set(
1057 x,
1058 y + proofps_dd::Maps::fMapBlockSizeHeight + pDecorObj->getSizeVec().getY() / 2.f,
1059 GAME_DECOR_POS_Z);
1060 pDecorObj->getMaterial().setTexture(m_texDecorJumpPadVertical);
1061 pDecorObj->getMaterial(false).setBlendFuncs(PURE_SRC_ALPHA, PURE_ONE_MINUS_SRC_ALPHA);
1062 pDecorObj->getMaterial(false).getTextureEnvColor().SetAlpha(200u);
1063 m_decorations.push_back(pDecorObj);
1064 }
1065 bJumppad = true;
1066 bSpecialFgBlock = true;
1067 bCopyPreviousFgBlock = iObjectFgToBeCopied > -1;
1068 break;
1069 }
1070 default:
1071 break;
1072 }
1073
1074 if (bSpecialFgBlock && bSpecialBgBlock)
1075 {
1076 const std::string sc(1, c); // WA for CConsole lack support of %c
1077 getConsole().EOLn("%s Block defined both as special foreground and special background: %s!", __func__, sc.c_str());
1078 assert(false);
1079 return false;
1080 }
1081
1082 PureObject3D* pNewBlockObj = nullptr;
1083 if (!bSpecialBgBlock || (bSpecialBgBlock && bCopyPreviousBgBlock))
1084 {
1085 m_blocks_h++;
1086 if (!bSpecialBgBlock && bBackground)
1087 {
1088 iObjectBgToBeCopied = m_blocks_h - 1;
1089 } else if (!bSpecialFgBlock && bForeground)
1090 {
1091 iObjectFgToBeCopied = m_blocks_h - 1;
1092 }
1093 if (!bDryRun)
1094 {
1095 if (bSpecialBgBlock && bCopyPreviousBgBlock)
1096 {
1097 pNewBlockObj = m_gfx.getObject3DManager().createCloned(*(m_blocks[iObjectBgToBeCopied]->getReferredObject()));
1098 }
1099 else if (bSpecialFgBlock && bCopyPreviousFgBlock)
1100 {
1101 pNewBlockObj = m_gfx.getObject3DManager().createCloned(*(m_blocks[iObjectFgToBeCopied]->getReferredObject()));
1102 }
1103 else
1104 {
1105 const auto it = m_mapReferenceBlockObject3Ds.find(c);
1106 if (it == m_mapReferenceBlockObject3Ds.end())
1107 {
1108 m_mapReferenceBlockObject3Ds[c] = m_gfx.getObject3DManager().createBox(proofps_dd::Maps::fMapBlockSizeWidth, proofps_dd::Maps::fMapBlockSizeWidth, proofps_dd::Maps::fMapBlockSizeWidth);
1109 m_mapReferenceBlockObject3Ds[c]->Hide();
1110 PureTexture* tex = PGENULL;
1111 if (m_Block2Texture.find(c) == m_Block2Texture.end())
1112 {
1113 const std::string sc(1, c); // WA for CConsole lack support of %c
1114 getConsole().EOLn("%s No texture defined for block %s!", __func__, sc.c_str());
1115 tex = m_texRed;
1116 }
1117 else
1118 {
1119 const std::string sTexName = proofps_dd::GAME_TEXTURES_DIR + m_sRawName + "\\" + m_Block2Texture[c];
1120 tex = m_gfx.getTextureManager().createFromFile(sTexName.c_str());
1121 if (!tex)
1122 {
1123 getConsole().EOLn("%s Could not load texture %s!", __func__, sTexName.c_str());
1124 tex = m_texRed;
1125 }
1126 }
1127 if (!tex)
1128 {
1129 // should happen only if default red texture could not be loaded, but that should had been detected in initialize() tho
1130 const std::string sc(1, c); // WA for CConsole lack support of %c
1131 getConsole().EOLn("%s Not assigning any texture for block %s!", __func__, sc.c_str());
1132 }
1133 m_mapReferenceBlockObject3Ds[c]->getMaterial().setTexture(tex);
1134 }
1135 pNewBlockObj = m_gfx.getObject3DManager().createCloned(*(m_mapReferenceBlockObject3Ds[c]));
1136 }
1137 // dont need to show, can stay hidden, since main game loop invokes UpdateVisibilitiesForRenderer() anyway which shows what is needed
1138 //pNewBlockObj->Show();
1139 m_blocks[m_blocks_h - 1] = pNewBlockObj;
1140 m_blocks[m_blocks_h - 1]->SetLit(true);
1141 }
1142
1143 if (bForeground)
1144 {
1145 m_foregroundBlocks_h++;
1146 if (!bDryRun)
1147 {
1148 m_foregroundBlocks[m_foregroundBlocks_h - 1] = pNewBlockObj;
1149
1150 if (bJumppad)
1151 {
1152 m_jumppads.push_back(pNewBlockObj);
1153 }
1154 }
1155 }
1156 }
1157
1158 if (!pNewBlockObj)
1159 {
1160 continue;
1161 }
1162
1163 pNewBlockObj->getPosVec().Set(x, y, bBackground ? 0.0f : -proofps_dd::Maps::fMapBlockSizeDepth);
1164 } // while
1165 y = y - proofps_dd::Maps::fMapBlockSizeHeight;
1166 return true;
1167} // lineHandleLayout()
