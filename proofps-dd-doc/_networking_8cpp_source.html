58bool proofps_dd::Networking::reinitializeNetworking()
59{
60 bool bRet = m_pge.getNetwork().reinitialize();
61 if (bRet)
62 {
63 // We tell the names of our app messages to the network engine so it can properly log message stats with message names
64 if (m_pge.getNetwork().getServerClientInstance()->getMsgAppId2StringMap().empty())
65 {
66 // never need to clear the map since its content should not change in between server and client instance switches
67 for (const auto& msgAppId2StringPair : MapMsgAppId2String)
68 {
69 m_pge.getNetwork().getServerClientInstance()->getMsgAppId2StringMap().insert(
70 { static_cast<pge_network::MsgApp::TMsgId>(msgAppId2StringPair.msgId),
71 std::string(msgAppId2StringPair.zstring) }
72 );
73 }
74 }
75
76 // however allowlisting needs to be invoked always since different messages are allowed for server and clients
77 allowListAppMessages();
78 }
79 return bRet;
80}

96void proofps_dd::Networking::allowListAppMessages()
97{
98 m_pge.getNetwork().getClient().getAllowListedAppMessages().clear();
99 m_pge.getNetwork().getServer().getAllowListedAppMessages().clear();
100
101 // following messages are received/sent by both clients and server over network:
102 m_pge.getNetwork().getClient().getAllowListedAppMessages().insert(static_cast<pge_network::MsgApp::TMsgId>(proofps_dd::MsgUserNameChangeAndBootupDone::id));
103 m_pge.getNetwork().getServer().getAllowListedAppMessages().insert(static_cast<pge_network::MsgApp::TMsgId>(proofps_dd::MsgUserNameChangeAndBootupDone::id));
104
105 if (m_pge.getNetwork().isServer())
106 {
107 m_pge.getNetwork().getServer().getAllowListedAppMessages().insert(static_cast<pge_network::MsgApp::TMsgId>(proofps_dd::MsgUserCmdFromClient::id));
108 }
109 else
110 {
111 // MsgMapChangeFromServer is also processed by server, but it injects this pkt into its own queue when needed.
112 // MsgMapChangeFromServer MUST NOT be received by server over network!
113 // MsgMapChangeFromServer is received only by clients over network!
114 m_pge.getNetwork().getClient().getAllowListedAppMessages().insert(static_cast<pge_network::MsgApp::TMsgId>(proofps_dd::MsgMapChangeFromServer::id));
115
116 // MsgUserSetupFromServer is also processed by server, but it injects this pkt into its own queue when needed.
117 // MsgUserSetupFromServer MUST NOT be received by server over network!
118 // MsgUserSetupFromServer is received only by clients over network!
119 m_pge.getNetwork().getClient().getAllowListedAppMessages().insert(static_cast<pge_network::MsgApp::TMsgId>(proofps_dd::MsgUserSetupFromServer::id));
120
121 // MsgUserUpdateFromServer is also processed by server, but it injects this pkt into its own queue when needed.
122 // MsgUserUpdateFromServer MUST NOT be received by server over network!
123 // MsgUserUpdateFromServer is received only by clients over network!
124 m_pge.getNetwork().getClient().getAllowListedAppMessages().insert(static_cast<pge_network::MsgApp::TMsgId>(proofps_dd::MsgUserUpdateFromServer::id));
125
126 // following messages are received only by clients over network:
127 m_pge.getNetwork().getClient().getAllowListedAppMessages().insert(static_cast<pge_network::MsgApp::TMsgId>(proofps_dd::MsgGameSessionStateFromServer::id));
128 m_pge.getNetwork().getClient().getAllowListedAppMessages().insert(static_cast<pge_network::MsgApp::TMsgId>(proofps_dd::MsgServerInfoFromServer::id));
129 m_pge.getNetwork().getClient().getAllowListedAppMessages().insert(static_cast<pge_network::MsgApp::TMsgId>(proofps_dd::MsgBulletUpdateFromServer::id));
130 m_pge.getNetwork().getClient().getAllowListedAppMessages().insert(static_cast<pge_network::MsgApp::TMsgId>(proofps_dd::MsgMapItemUpdateFromServer::id));
131 m_pge.getNetwork().getClient().getAllowListedAppMessages().insert(static_cast<pge_network::MsgApp::TMsgId>(proofps_dd::MsgWpnUpdateFromServer::id));
132 m_pge.getNetwork().getClient().getAllowListedAppMessages().insert(static_cast<pge_network::MsgApp::TMsgId>(proofps_dd::MsgCurrentWpnUpdateFromServer::id));
133 m_pge.getNetwork().getClient().getAllowListedAppMessages().insert(static_cast<pge_network::MsgApp::TMsgId>(proofps_dd::MsgDeathNotificationFromServer::id));
134 m_pge.getNetwork().getClient().getAllowListedAppMessages().insert(static_cast<pge_network::MsgApp::TMsgId>(proofps_dd::MsgPlayerEventFromServer::id));
135 }
136}
