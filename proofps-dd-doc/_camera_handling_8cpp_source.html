72void proofps_dd::CameraHandling::cameraUpdatePosAndAngle(
73 const Player& player,
74 const XHair& xhair,
75 const float& fFps,
76 bool bCamFollowsXHair,
77 bool bCamTiltingAllowed,
78 bool bCamRollAllowed)
79{
80 const std::chrono::time_point<std::chrono::steady_clock> timeStart = std::chrono::steady_clock::now();
81
82 cameraSmoothShakeForceTowardsZero(fFps);
83 cameraUpdateShakeFactorXY(fFps);
84
85 auto& cam = m_pge.getPure().getCamera();
86
87 if (bCamRollAllowed && player.isSomersaulting())
88 {
89 cameraUpdatePosAndAngleWhenPlayerIsSomersaulting(cam, player);
90 }
91 else
92 {
93 cameraUpdatePosAndAngleWhenPlayerIsInNormalSituation(cam, player, xhair, fFps, bCamFollowsXHair, bCamTiltingAllowed);
94 }
95
96 /*
97 * Looks like it is enough if I pass (0,0,1) as At vector.
98 * Anyway, in the future if there is any fishy 3D audio issue related to orientation, remember these tickets:
99 * - https://github.com/jarikomppa/soloud/issues/94
100 * - https://github.com/jarikomppa/soloud/issues/105
101 */
102 m_pge.getAudio().getAudioEngineCore().set3dListenerParameters(
103 cam.getPosVec().getX(), cam.getPosVec().getY(), cam.getPosVec().getZ(),
104 0, 0, 1 /*cam.getTargetVec().getX(), cam.getTargetVec().getY(), cam.getTargetVec().getZ()*/,
105 cam.getUpVec().getX(), cam.getUpVec().getY(), cam.getUpVec().getZ());
106 m_pge.getAudio().getAudioEngineCore().update3dAudio();
107
108 m_durations.m_nCameraMovementDurationUSecs += std::chrono::duration_cast<std::chrono::microseconds>(std::chrono::steady_clock::now() - timeStart).count();
109}

120void proofps_dd::CameraHandling::cameraSmoothShakeForceTowardsZero(const float& fFps)
121{
122 const float GAME_FPS_RATE_LERP_FACTOR = (fFps - GAME_TICKRATE_MIN) / static_cast<float>(GAME_TICKRATE_MAX - GAME_TICKRATE_MIN);
123 const float GAME_IMPACT_FORCE_CHANGE = PFL::lerp(2150.f, 2160.f, GAME_FPS_RATE_LERP_FACTOR);
124 const float fCamShakeForceChangePerFrame = GAME_IMPACT_FORCE_CHANGE / 36.f / fFps; /* smaller number means longer shaking in time */
125 if (m_vecCamShakeForce.getX() > 0.f)
126 {
127 m_vecCamShakeForce.SetX(m_vecCamShakeForce.getX() - fCamShakeForceChangePerFrame);
128 if (m_vecCamShakeForce.getX() < 0.f)
129 {
130 m_vecCamShakeForce.SetX(0.f);
131 }
132 }
133 if (m_vecCamShakeForce.getY() > 0.f)
134 {
135 m_vecCamShakeForce.SetY(m_vecCamShakeForce.getY() - fCamShakeForceChangePerFrame);
136 if (m_vecCamShakeForce.getY() < 0.f)
137 {
138 m_vecCamShakeForce.SetY(0.f);
139 }
140 }
141}

143void proofps_dd::CameraHandling::cameraUpdateShakeFactorXY(const float& fFps)
144{
145 if ((m_vecCamShakeForce.getX() == 0.f) && (m_vecCamShakeForce.getY() == 0.f))
146 {
147 m_fShakeFactorX = 0.f;
148 m_fShakeFactorY = 0.f;
149 m_fShakeDegree = 0.f;
150 return;
151 }
152
153 assert(fFps > 0.f); // PRooFPSddPGE::updateFramesPerSecond() makes sure fFps is never 0
154 m_fShakeDegree += 1200 / fFps;
155 while (m_fShakeDegree >= 360.f)
156 {
157 m_fShakeDegree -= 360.f;
158 }
159 float fShakeSine = sin(m_fShakeDegree * PFL::PI / 180.f);
160
161 m_fShakeFactorX = fShakeSine * m_vecCamShakeForce.getX() / fFps;
162 m_fShakeFactorY = fShakeSine * m_vecCamShakeForce.getY() / fFps;
163}

165void proofps_dd::CameraHandling::cameraUpdatePosAndAngleWhenPlayerIsSomersaulting(PureCamera& cam, const Player& player)
166{
167 cam.getPosVec().Set(
168 player.getObject3D()->getPosVec().getX() + m_fShakeFactorX,
169 player.getObject3D()->getPosVec().getY() + m_fShakeFactorY,
170 GAME_CAM_Z);
171 cam.getTargetVec().Set(
172 player.getObject3D()->getPosVec().getX() + m_fShakeFactorX,
173 player.getObject3D()->getPosVec().getY() + m_fShakeFactorY,
174 player.getObject3D()->getPosVec().getZ()
175 );
176
177 PureVector vecNewUp(0.f, 1.f, 0.f);
178 PureTransformMatrix matRotZ;
179 matRotZ.SetRotationZ((player.getObject3D()->getAngleVec().getY() == 0.f) ? -player.getSomersaultAngle() : player.getSomersaultAngle());
180 vecNewUp *= matRotZ;
181 cam.getUpVec() = vecNewUp;
182}

184void proofps_dd::CameraHandling::cameraUpdatePosAndAngleWhenPlayerIsInNormalSituation(
185 PureCamera& cam,
186 const Player& player,
187 const XHair& xhair,
188 const float& fFps,
189 bool bCamFollowsXHair,
190 bool bCamTiltingAllowed)
191{
192 constexpr unsigned int nBlocksToKeepCameraWithinMapBoundsHorizontally = 3;
193 constexpr unsigned int nBlocksToKeepCameraWithinMapBottom = 5;
194 constexpr unsigned int nBlocksToKeepCameraWithinMapTop = 1;
195
196 const float fCamMinAllowedPosXtoKeepAwayFromMapHorizontalBounds =
197 m_maps.width() < (nBlocksToKeepCameraWithinMapBoundsHorizontally * 2 + 1) ?
198 m_maps.getBlocksVertexPosMin().getX() :
199 m_maps.getBlocksVertexPosMin().getX() + (proofps_dd::Maps::fMapBlockSizeWidth * (nBlocksToKeepCameraWithinMapBoundsHorizontally));
200 const float fCamMaxAllowedPosXtoKeepAwayFromMapHorizontalBounds =
201 m_maps.width() < (nBlocksToKeepCameraWithinMapBoundsHorizontally * 2 + 1) ?
202 m_maps.getBlocksVertexPosMax().getX() :
203 m_maps.getBlocksVertexPosMin().getX() + (proofps_dd::Maps::fMapBlockSizeWidth * (m_maps.width() - nBlocksToKeepCameraWithinMapBoundsHorizontally));
204
205 const float fCamMinAllowedPosYtoKeepAwayFromMapVerticalBounds =
206 m_maps.height() < (nBlocksToKeepCameraWithinMapBottom + 1) ?
207 m_maps.getBlocksVertexPosMin().getY() :
208 m_maps.getBlocksVertexPosMin().getY() + (proofps_dd::Maps::fMapBlockSizeHeight * (nBlocksToKeepCameraWithinMapBottom - 1));
209 const float fCamMaxAllowedPosYtoKeepAwayFromMapVerticalBounds =
210 m_maps.height() < (nBlocksToKeepCameraWithinMapTop + 1) ?
211 m_maps.getBlocksVertexPosMax().getY() :
212 m_maps.getBlocksVertexPosMin().getY() + (proofps_dd::Maps::fMapBlockSizeHeight * (m_maps.height() - nBlocksToKeepCameraWithinMapTop + 1));
213
214 // TODO: multipliers 6 and 4 are roughly okay for 1024x768 and 1080p, but later ratio should be introduced based on actual screen width and height
215 constexpr float fCamMaxXDistanceFromPlayer = Maps::fMapBlockSizeWidth * 6;
216 constexpr float fCamMaxYDistanceFromPlayer = Maps::fMapBlockSizeHeight * 4;
217 float fCamPosXTarget, fCamPosYTarget;
218 if (bCamFollowsXHair)
219 {
220 const float fPosXBetweenPlayerAndCamera =
221 (xhair.getUnprojectedCoords().getX() >= player.getObject3D()->getPosVec().getX()) ?
222 (std::min(
223 player.getObject3D()->getPosVec().getX() + player.getObject3D()->getPosVec().getX() + fCamMaxXDistanceFromPlayer,
224 player.getObject3D()->getPosVec().getX() + xhair.getUnprojectedCoords().getX()) / 2.f) :
225 (std::max(
226 player.getObject3D()->getPosVec().getX() + player.getObject3D()->getPosVec().getX() - fCamMaxXDistanceFromPlayer,
227 player.getObject3D()->getPosVec().getX() + xhair.getUnprojectedCoords().getX()) / 2.f);
228
229 const float fPosYBetweenPlayerAndCamera =
230 (xhair.getUnprojectedCoords().getY() >= player.getObject3D()->getPosVec().getY()) ?
231 (std::min(
232 player.getObject3D()->getPosVec().getY() + player.getObject3D()->getPosVec().getY() + fCamMaxYDistanceFromPlayer,
233 player.getObject3D()->getPosVec().getY() + xhair.getUnprojectedCoords().getY()) / 2.f) :
234 (std::max(
235 player.getObject3D()->getPosVec().getY() + player.getObject3D()->getPosVec().getY() - fCamMaxYDistanceFromPlayer,
236 player.getObject3D()->getPosVec().getY() + xhair.getUnprojectedCoords().getY()) / 2.f);
237
238 fCamPosXTarget = std::min(
239 fCamMaxAllowedPosXtoKeepAwayFromMapHorizontalBounds,
240 std::max(fCamMinAllowedPosXtoKeepAwayFromMapHorizontalBounds, fPosXBetweenPlayerAndCamera));
241
242 fCamPosYTarget = std::min(
243 fCamMaxAllowedPosYtoKeepAwayFromMapVerticalBounds,
244 std::max(fCamMinAllowedPosYtoKeepAwayFromMapVerticalBounds, fPosYBetweenPlayerAndCamera));
245 }
246 else
247 {
248 fCamPosXTarget = std::min(
249 fCamMaxAllowedPosXtoKeepAwayFromMapHorizontalBounds,
250 std::max(fCamMinAllowedPosXtoKeepAwayFromMapHorizontalBounds, player.getObject3D()->getPosVec().getX()));
251
252 fCamPosYTarget = std::min(
253 fCamMaxAllowedPosYtoKeepAwayFromMapVerticalBounds,
254 std::max(fCamMinAllowedPosYtoKeepAwayFromMapVerticalBounds, player.getObject3D()->getPosVec().getY()));
255 }
256
257 fCamPosXTarget += m_fShakeFactorX;
258 fCamPosYTarget += m_fShakeFactorY;
259
260 PureVector vecCamPos{
261 PFL::smooth(
262 cam.getPosVec().getX() + m_fShakeFactorX, fCamPosXTarget, GAME_CAM_SPEED_X * fFps),
263 PFL::smooth(
264 cam.getPosVec().getY() + m_fShakeFactorY,
265 fCamPosYTarget,
266 /* if we are not following xhair, we want an eased vertical camera movement because it looks nice */
267 (bCamFollowsXHair ? (GAME_CAM_SPEED_X * fFps) : (GAME_CAM_SPEED_Y * fFps))),
268 GAME_CAM_Z
269 };
270
271 cam.getPosVec() = vecCamPos;
272 cam.getTargetVec().Set(
273 bCamTiltingAllowed ? ((vecCamPos.getX() + fCamPosXTarget) / 2.f) : vecCamPos.getX(),
274 bCamTiltingAllowed ? ((vecCamPos.getY() + fCamPosYTarget) / 2.f) : vecCamPos.getY(),
275 player.getObject3D()->getPosVec().getZ()
276 );
277
278 // we can always reset Up like this, since Up vector doesn't have effect on camera pitch/yaw, only on roll!
279 cam.getUpVec().Set(0.f, 1.f, 0.f);
280}
