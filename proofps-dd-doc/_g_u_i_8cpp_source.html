303void proofps_dd::GUI::resetMenuState(bool bExitingFromGameSession)
304{
305 if (bExitingFromGameSession)
306 {
307 showBgWithLogo();
308 if (m_pPge->getConfigProfiles().getVars()[CVAR_GUI_MAINMENU].getAsBool())
309 {
310 m_currentMenu = MenuState::Main;
311 }
312 else
313 {
314 m_currentMenu = MenuState::Exiting;
315 m_pPge->getPure().getWindow().Close();
316 }
317 }
318 else
319 {
320 if (m_pPge->getConfigProfiles().getVars()[CVAR_GUI_MAINMENU].getAsBool())
321 {
322 m_currentMenu = MenuState::Main;
323 showBgWithLogo();
324 }
325 else
326 {
327 m_currentMenu = MenuState::None;
328 hideBgWithLogo();
329 }
330 }

1158{
1159 if ((data->EventChar >= '0') && (data->EventChar <= '9') || (data->EventChar == '.'))
1160 {

1162 }

1165
1166static bool isValidIPv4(const char* IPAddress)
1167{
1168 // this is a low-effort validity checker, later when IPv6 will be supported I will simply use arpa/inet.h or winsock2.h and call
1169 // the appropriate converter function to check for validity.
1170 int a, b, c, d;
1171 if (sscanf_s(IPAddress, "%d.%d.%d.%d", &a, &b, &c, &d) == 4)
1172 {
1173 return (a >= 0) && (b >= 0) && (c >= 0) && (d >= 0) && (a < 256) && (b < 256) && (c < 256) && (d < 256);
1174 }
1175 return false;
1176}

2643void proofps_dd::GUI::ImGuiTextTableCurrentCellShortenedFit(const std::string& text, size_t nAppendLastNChars)
2644{
2645 if (text.empty())
2646 {
2647 return;
2648 }
2649
2650 const auto fFullTextRequiredWidth = ImGui::CalcTextSize(text.c_str()).x;
2651 if (fFullTextRequiredWidth == 0.f)
2652 {
2653 assert(false); // should never happen since text is not empty

2655 }
2656

2658 const int nFirstNCharsFitInCell = static_cast<int>((std::floor(std::min(1.f, fAvailWidthInCell / fFullTextRequiredWidth) * text.length())));
2659 if (nFirstNCharsFitInCell <= 0)
2660 {
2661 // can happen if column width is too small
2662 ImGui::TextUnformatted("#"); // even if this '#' does not fit, it should be partially rendered!

2667 {
2668 ImGui::TextUnformatted(text.c_str());
2669 }
2670 else
2671 {
2672 if (text.length() < nAppendLastNChars)
2673 {
2674 ImGui::TextUnformatted("#"); // even if this '#' does not fit, it should be partially rendered!
2675 return;

2679 const auto fShortenedDottedTextEndRequiredWidth = ImGui::CalcTextSize(sShortenedTextDottedEnd.c_str()).x;
2680 if (fShortenedDottedTextEndRequiredWidth == 0.f)
2681 {
2682 assert(false); // should never happen since text is not empty
2683 return;
2684 }
2685 if (fShortenedDottedTextEndRequiredWidth >= fFullTextRequiredWidth)
2686 {
2687 // might happen if text end is something like " ..." or " " (only spaces)
2688 ImGui::TextUnformatted(text.c_str());
2689 return;

2692 {
2693 // can happen if column width is too small
2694 ImGui::TextUnformatted("###"); // even if this '###' does not fit, it should be partially rendered!
2695 return;
2696 }

2699 (std::floor(
2700 std::min(1.f, (fAvailWidthInCell - fShortenedDottedTextEndRequiredWidth) / (fFullTextRequiredWidth)) * text.length())
2701 ));
