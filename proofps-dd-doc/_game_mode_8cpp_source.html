100bool proofps_dd::GameMode::renamePlayer(const std::string& sOldName, const std::string& sNewName)
101{
102 if (sOldName.empty() || sNewName.empty())
103 {
104 return false;
105 }
106
107 FragTableRow* pPlayerToRename = nullptr;
108
109 // loop over the all players and check for name collision, at the same time save ptr to target player
110 for (auto& player : m_players)
111 {
112 if (player.m_sName == sOldName)
113 {
114 pPlayerToRename = &player; // will rename this
115 }
116 else if (player.m_sName == sNewName)
117 {
118 return false; // name collision
119 }
120 }
121
122 if (pPlayerToRename)
123 {
124 pPlayerToRename->m_sName = sNewName;
125 return true;
126 }
127
128 // no such player found
129 return false;
130}

210bool proofps_dd::DeathMatchMode::serverCheckAndUpdateWinningConditions(pge_network::PgeINetwork& network)
211{
212 assert(network.isServer());
213
214 if (m_bWon)
215 {
216 // once it is won, it stays won until next restart()
217 return m_bWon;
218 }
219
220 if (getTimeLimitSecs() > 0)
221 {
222 if (getTimeRemainingMillisecs() == 0)
223 {
224 m_bWon = true;
225 m_timeWin = std::chrono::steady_clock::now();
226 serverSendGameSessionStateToClients(network);
227 return true;
228 }
229 }
230
231 if (getFragLimit() > 0)
232 {
233 // assume m_players is sorted, since add/updatePlayer() use insertion sort
234 if ((m_players.size() > 0) && (m_players.begin()->m_nFrags >= static_cast<int>(getFragLimit())))
235 {
236 m_bWon = true;
237 m_timeWin = std::chrono::steady_clock::now();
238 serverSendGameSessionStateToClients(network);
239 return true;
240 }
241 }
242
243 return false;
244}

273unsigned int proofps_dd::DeathMatchMode::getTimeRemainingMillisecs() const
274{
275 if (m_bWon || (getResetTime().time_since_epoch().count() == 0) || (getTimeLimitSecs() == 0))
276 {
277 return 0;
278 }
279
280 const auto nMillisecondsElapsedSinceReset = std::chrono::duration_cast<std::chrono::milliseconds>(
281 std::chrono::steady_clock::now() - getResetTime()).count();
282 const long long nTimeLimitMilliseconds = getTimeLimitSecs() * 1000;
283
284 if (nTimeLimitMilliseconds <= nMillisecondsElapsedSinceReset)
285 {
286 return 0;
287 }
288
289 return static_cast<unsigned int>(nTimeLimitMilliseconds - nMillisecondsElapsedSinceReset);
290}

292void proofps_dd::DeathMatchMode::clientUpdateTimeRemainingMillisecs(const unsigned int& nRemMillisecs, pge_network::PgeINetwork& network)
293{
294 assert(!network.isServer());
295
296 const unsigned int nRemSecs = static_cast<unsigned int>(std::floor(nRemMillisecs / 1000));
297 if (nRemSecs > m_nTimeLimitSecs)
298 {
299 // should not happen, but should log as error
300 getConsole().EOLn("GameMode::%s(): SHOULD NOT HAPPEN: nRemSecs > m_nTimeLimitSecs: %u > %u!", __func__, nRemSecs, m_nTimeLimitSecs);
301 }
302
303 m_timeReset = std::chrono::steady_clock::now() - std::chrono::seconds(m_nTimeLimitSecs - std::min(nRemSecs, m_nTimeLimitSecs));
304}

316bool proofps_dd::DeathMatchMode::addPlayer(const Player& player, pge_network::PgeINetwork& network)
317{
318 bool bRet = true;
319 auto it = m_players.cbegin();
320 while (bRet &&
321 (it != m_players.cend()) &&
322 (comparePlayers(it->m_nFrags, player.getFrags(), it->m_nDeaths, player.getDeaths()) <= 0))
323 {
324 bRet = it->m_sName != player.getName();
325 ++it;
326 }
327
328 // even though we not yet found a player with same name, we need to check remaining players, before inserting this one ...
329 auto itCheck = it;
330 while (bRet &&
331 (itCheck != m_players.cend()))
332 {
333 bRet = itCheck->m_sName != player.getName();
334 ++itCheck;
335 }
336
337 if (bRet)
338 {
339 m_players.insert(it, proofps_dd::FragTableRow{ player.getName(), player.getFrags(), player.getDeaths(), player.getServerSideConnectionHandle() });
340
341 if (network.isServer() && (player.getServerSideConnectionHandle() != pge_network::ServerConnHandle))
342 {
343 // we automatically inform the new player about the current game goal/win state since maybe they are connecting when game is already won;
344 // inform them ONLY IF GAME IS WON, since otherwise they are good already, and sending them "NOT WON" now would cause them to zero out already updated frag table data!
345 if (isGameWon())
346 {
347 serverSendGameSessionStateToClient(network, player.getServerSideConnectionHandle());
348 }
349 }
350 }
351
352 if (network.isServer())
353 {
354 serverCheckAndUpdateWinningConditions(network); // to make sure winning time is updated if game has just been won!
355 }
356 return bRet;
357}

359bool proofps_dd::DeathMatchMode::updatePlayer(const Player& player, pge_network::PgeINetwork& network)
360{
361 const auto itFound = std::find_if(
362 std::begin(m_players),
363 std::end(m_players),
364 [&player](const auto& row) { return row.m_sName == player.getName(); });
365 if (itFound == m_players.end())
366 {
367 // this player doesn't even exist in the frag table
368 return false;
369 }
370
371 if ((player.getFrags() == itFound->m_nFrags) && (player.getDeaths() == itFound->m_nDeaths))
372 {
373 // nothing to update
374 return true;
375 }
376
377 // In case of players having equal nFrags and nDeaths, we need to apply different rules based on how values are being updated:
378 // - if new nFrags > old nFrags, player has to be placed behind other players having equal nFrags, since they had it earlier;
379 // - if new nFrags < old nFrags, player has to be placed in front of other players having equals nFrags, since player had more earlier.
380
381 const bool bPutBehindEquals = player.getFrags() > itFound->m_nFrags;
382 auto it = m_players.begin();
383 if (bPutBehindEquals)
384 {
385 while ((it != m_players.end())
386 && (comparePlayers(it->m_nFrags, player.getFrags(), it->m_nDeaths, player.getDeaths()) <= 0))
387 {
388 ++it;
389 }
390 }
391 else
392 {
393 while ((it != m_players.end())
394 && (comparePlayers(it->m_nFrags, player.getFrags(), it->m_nDeaths, player.getDeaths()) < 0))
395 {
396 ++it;
397 }
398 }
399
400 if (itFound != it)
401 {
402 // move 'itFound' before 'it' (proofps_dd::FragTableRow is not copied, only internal pointers are updated)
403 m_players.splice(it, m_players, itFound);
404 }
405 itFound->m_nFrags = player.getFrags();
406 itFound->m_nDeaths = player.getDeaths();
407
408 if (network.isServer())
409 {
410 serverCheckAndUpdateWinningConditions(network); // to make sure winning time is updated if game has just been won!
411 }
412
413 return true;
414}

433int proofps_dd::DeathMatchMode::comparePlayers(int p1frags, int p2frags, int p1deaths, int p2deaths)
434{
435 if (p1frags == p2frags)
436 {
437 return p1deaths - p2deaths;
438 }
439 else
440 {
441 return p2frags - p1frags;
