118void proofps_dd::XHair::showInCenter()
119{
120 assert(m_pObjXHair); // otherwise ctor would had already thrown
121
122 // this is to get rid of all mouse move messages that were probably queued up in the meantime (e.g. during map loading), otherwise
123 // there is no use of setting cursor pos to center if enqueued messages will reposition it when PURE runs the window's processMessages().
124 m_pge.getPure().getWindow().ProcessMessages();
125
126 // getInput().getMouse().SetCursorPos() is not triggering any mouse move event and nulls out pending raw input events as well!
127 m_pge.getInput().getMouse().SetCursorPos(
128 m_pge.getPure().getWindow().getX() + m_pge.getPure().getWindow().getWidth() / 2,
129 m_pge.getPure().getWindow().getY() + m_pge.getPure().getWindow().getHeight() / 2);
130
131 m_pObjXHair->getPosVec().Set(0, 0, 0); // reposition to viewport center so it won't appear at random places
132 m_pObjXHair->Show();
133 m_prevXHairPos.x = 0.f;
134 m_prevXHairPos.y = 0.f;
135 m_bVisible = true;
136}

157void proofps_dd::XHair::updateUnprojectedCoords(PureCamera& cam)
158{
159 // TODO: explicit call to this function would NOT be required if we did maintain a dirty flag and explicit setX(), setY() functions, which
160 // would set the dirty flag, and the getUnprojectedCoords() would invoke updateUnprojectedCoords() if dirty flag is set!
161
162 assert(m_pObjXHair); // ctor would throw otherwise
163
164 /*
165 Originally, I was unsure about if we really needed to unproject the 2D xhair's position.
166 Because of matrix multiplication and inversion, maybe this way is too expensive.
167
168 I see 2 other ways of doing this:
169 - keep the xhair 2D, and anytime mouse is moved, we should also change 2 variables: fCamPosOffsetX and fCamPosOffsetY
170 based on the dx and dy variables in InputHandling::clientMouseWhenConnectedToServer(). Camera will have this position offset applied relative to
171 player's position. This looks to be the easiest solution.
172 However, we still need to unproject for other feature, such as displaying name over hovered player object because 2D xhair needs to be 3D to be
173 collision-tested with player objects.
174
175 - change the xhair to 3D, so in InputHandling::clientMouseWhenConnectedToServer() the dx and dy variable changes will be applied to xhair's 3D position.
176 With this method, pick/select can be implemented in a different way: no need to unproject, we just need collision
177 logic to find out which player object collides with our xhair object!
178 However, this way InputHandling::clientMouseWhenConnectedToServer() is more complicated because dx and dy should be multiplied by some factor related
179 to camera FovX and/or FovY, since the linear 2D movement should be non-linear in 3D so it will actually feel linear on the screen!
180 However, we still need to project xhair to screen for other feature, such as displaying player name over hovered player object because we need to know
181 where exactly render the text above the 3D xhair on the screen.
182
183 So after all, I'm still keeping the 2D xhair, and unproject it, once per frame.
184 This unprojected coordinate can be used for multiple things:
185 - calculate 3D position between player position and xhair position so camera can be placed there;
186 - collide against player objects so I will know whose name should be displayed above xhair's 2D position.
187 */
188 if (!cam.project2dTo3d(
189 static_cast<TPureUInt>(roundf(m_pObjXHair->getPosVec().getX()) + cam.getViewport().size.width / 2),
190 static_cast<TPureUInt>(roundf(m_pObjXHair->getPosVec().getY()) + cam.getViewport().size.height / 2),
191 /* in v0.2.5 this is player's Z (-1.2f as per GAME_PLAYERS_POS_Z) mapped to depth buffer: 0.9747f,
192 0.f Z is 0.981f depth, but
193 I was using 0.96f in earlier versions so it was a bit aligned on Z-axis, the camera was less moving away to screen corners,
194 but better to use the mapped player Z since the aim of unprojecting xhair is to collide it with player objects */
195 0.9747f,
196 m_vecUnprojected))
197 {
198 //getConsole().EOLn("XHair::%s(): project2dTo3d() failed!", __func__);
199 }
200 else
201 {
202 m_pObjDebugCube->getPosVec() = m_vecUnprojected;
203 //getConsole().EOLn("XHair obj X: %f, Y: %f, m_vecUnprojected X: %f, Y: %f",
204 // m_pObjXHair->getPosVec().getX(), m_pObjXHair->getPosVec().getY(),
205 // m_vecUnprojected.getX(), m_vecUnprojected.getY());
206 }
207
208 // this code snippet here is just for finding out which screen depth to use:
209 //PureVector vecProjected;
210 //if (cam.project3dTo2d(0.f, 0.f, 0.f /* GAME_PLAYERS_POS_Z as defined in Maps */, vecProjected))
211 //{
212 // getConsole().EOLn("XHair::%s(): vecProjected X: %f, Y: %f, Z (depth): %f!", __func__, vecProjected.getX(), vecProjected.getY(), vecProjected.getZ());
213 //}
214}

236void proofps_dd::XHair::startBlinking()
237{
238 // idea is that game calls either xhair show() or hide(), while GUI can also invoke startBlinking, so game does not deal with blinking,
239 // this blinking is allowed only if xhair is visible anyway, a hidden xhair wont be shown by startBlinking().
240 m_timeStartedBlinking = std::chrono::steady_clock::now();
241 m_bBlinking = m_bVisible; // visibility state always overrides blinking

244void proofps_dd::XHair::stopBlinking()
245{
246 m_bBlinking = false;
247 // during blinking, game maybe invoked hide() for whatever reason, later game may invoke stopBlinking() also for whatever reason, anyway we should enable
248 // xhair rendering only if game still did NOT invoke hide() in the meantime
249 assert(m_pObjXHair); // ctor would throw otherwise
250 m_pObjXHair->SetRenderingAllowed(m_bVisible); // visibility state always overrides after-blinking true visibility state
251
252 for (int i = 0; i < 4; i++)
253 {
254 assert(m_vHLightRects[i]); // ctor would throw otherwise
255 m_vHLightRects[i]->SetRenderingAllowed(m_bVisible);

331void proofps_dd::XHair::updateVisuals()
332{
333 // expected to be invoked every frame, by the game itself, after updating XHair object position by user input
334
335 if (!m_bVisible)
336 {
337 stopBlinking();
338 return;
339 }
340
341 if (m_bBlinking)
342 {
343 constexpr auto nBlinkPeriodMillisecs = 200;
344 const auto nElapsedTimeSinceBlinkingStartMillisecs = std::chrono::duration_cast<std::chrono::milliseconds>(std::chrono::steady_clock::now() - m_timeStartedBlinking).count();
345 assert(m_pObjXHair); // ctor would throw otherwise
346 m_pObjXHair->SetRenderingAllowed(
347 ((nElapsedTimeSinceBlinkingStartMillisecs / nBlinkPeriodMillisecs) % 2) == 1);
348 }
349
350 updateHighlight();
351}

360void proofps_dd::XHair::adjustHighlightSize(const bool& bCanShowHighlightPerXHairMovement)
361{
362 const bool bHighlightVisibleFinal = bCanShowHighlightPerXHairMovement && m_pObjXHair->isRenderingAllowed();
363 for (int i = 0; i < 4; i++)
364 {
365 assert(m_vHLightRects[i]); // ctor would throw otherwise
366 m_vHLightRects[i]->SetRenderingAllowed(bHighlightVisibleFinal);
367 }
368
369 if (!bHighlightVisibleFinal)
370 {
371 return;
372 }
373
374 constexpr float fDistanceFromXHair = 10.f;
375 const float fHLlineThickness = 10.f * m_fBaseScaling;
376
377 const float fHLupperHeight = std::max(0.f,
378 (m_pge.getPure().getWindow().getClientHeight() / 2.f -
379 m_pObjXHair->getPosVec().getY() -
380 m_pObjXHair->getScaledSizeVec().getY() / 2.f -
381 fDistanceFromXHair));
382 m_vHLightRects[static_cast<int>(HighlightRect::Upper)]->SetScaling(PureVector(fHLlineThickness * m_fRelativeScaleFactor, fHLupperHeight, 1.f));
383 //getConsole().EOLn("XHair::%s(): fHLupperHeight: %f!", __func__, fHLupperHeight);
384 m_vHLightRects[static_cast<int>(HighlightRect::Upper)]->getPosVec().SetX(m_pObjXHair->getPosVec().getX());
385 m_vHLightRects[static_cast<int>(HighlightRect::Upper)]->getPosVec().SetY(
386 m_pObjXHair->getPosVec().getY() +
387 fHLupperHeight / 2.f +
388 m_pObjXHair->getScaledSizeVec().getY() / 2.f +
389 fDistanceFromXHair);
390
391 const float fHLlowerHeight = std::max(0.f,
392 (m_pge.getPure().getWindow().getClientHeight() / 2.f +
393 m_pObjXHair->getPosVec().getY() +
394 m_pObjXHair->getScaledSizeVec().getY() / 2.f +
395 fDistanceFromXHair));
396 m_vHLightRects[static_cast<int>(HighlightRect::Lower)]->SetScaling(PureVector(fHLlineThickness * m_fRelativeScaleFactor, fHLlowerHeight, 1.f));
397 //getConsole().EOLn("XHair::%s(): fHLlowerHeight: %f!", __func__, fHLlowerHeight);
398 m_vHLightRects[static_cast<int>(HighlightRect::Lower)]->getPosVec().SetX(m_pObjXHair->getPosVec().getX());
399 m_vHLightRects[static_cast<int>(HighlightRect::Lower)]->getPosVec().SetY(
400 m_pObjXHair->getPosVec().getY() -
401 fHLlowerHeight / 2.f -
402 m_pObjXHair->getScaledSizeVec().getY() / 2.f -

407 m_pObjXHair->getPosVec().getX() +

434void proofps_dd::XHair::updateHighlight()
435{
436 // expected to be invoked every frame
437
438 static std::chrono::time_point<std::chrono::steady_clock> timeXHairLastMoved;
439 bool bCanShowHighlightPerXHairMovement = true;
440 if ((m_pObjXHair->getPosVec().getX() != m_prevXHairPos.x)
441 ||
442 (m_pObjXHair->getPosVec().getY() != m_prevXHairPos.y))
443 {
444 timeXHairLastMoved = std::chrono::steady_clock::now();
445
446 m_prevXHairPos.x = m_pObjXHair->getPosVec().getX();
447 m_prevXHairPos.y = m_pObjXHair->getPosVec().getY();
448 }
449 else
450 {
451 constexpr float fXHairPostMoveTimeoutMillisecs = 1000;
452 if ((std::chrono::duration_cast<std::chrono::milliseconds>(std::chrono::steady_clock::now() - timeXHairLastMoved)).count() >= fXHairPostMoveTimeoutMillisecs)
453 {
454 bCanShowHighlightPerXHairMovement = false;
455 for (int i = 0; i < 4; i++)
456 {
457 assert(m_vHLightRects[i]); // ctor would throw otherwise
458 m_vHLightRects[i]->SetRenderingAllowed(false);
459 }
460 }
461 }
462
463 // must keep adjusting highlight here, since it is needed even if xhair not being moved, still its scale can change due to shooting, etc.
464 adjustHighlightSize(bCanShowHighlightPerXHairMovement);
465}
